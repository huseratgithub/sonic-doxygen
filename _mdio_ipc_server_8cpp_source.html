101sai_status_t MdioIpcServer::syncd_ipc_cmd_mdio_common(char *resp, int argc, char *argv[])
102{
103 int ret = 0;
104 uint32_t mdio_addr = 0, reg_addr = 0, val = 0;
105
106 if (argc < 3)
107 {
108 return SAI_STATUS_INVALID_PARAMETER;
109 }
110
111 mdio_addr = (uint32_t)strtoul(argv[1], NULL, 0);
112 reg_addr = (uint32_t)strtoul(argv[2], NULL, 0);
113
114 if (m_switchRid == SAI_NULL_OBJECT_ID)
115 {
116 SWSS_LOG_ERROR("mdio switch id not initialized");
117 return SAI_STATUS_FAILURE;
118 }
119
120 if (argc > 3)
121 {
122 val = (uint32_t)strtoul(argv[3], NULL, 0);
123 ret = m_vendorSai->switchMdioWrite(m_switchRid, mdio_addr, reg_addr, 1, &val);
124 sprintf(resp, "%d\n", ret);
125 }
126 else
127 {
128 ret = m_vendorSai->switchMdioRead(m_switchRid, mdio_addr, reg_addr, 1, &val);
129 sprintf(resp, "%d 0x%x\n", ret, val);
130 }
131
132 return (ret == 0) ? SAI_STATUS_SUCCESS : SAI_STATUS_FAILURE;
133}

186int MdioIpcServer::syncd_ipc_task_main()
187{
188 SWSS_LOG_ENTER();
189
190 int i;
191 int fd;
192 int len;
193 int ret;
194 int sock_srv;
195 int sock_cli;
196 int sock_max;
197 syncd_mdio_ipc_conn_t conn[CONN_MAX];
198 struct sockaddr_un addr;
199 char path[64];
200 fd_set rfds;
201 char cmd[SYNCD_IPC_BUFF_SIZE], resp[SYNCD_IPC_BUFF_SIZE], *argv[64], *save;
202 int argc = 0;
203
204 strcpy(path, SYNCD_IPC_SOCK_SYNCD);
205 fd = open(path, O_DIRECTORY);
206 if (fd < 0)
207 {
208 SWSS_LOG_ERROR("Unable to open the directory %s for IPC\n", path);
209 return errno;
210 }
211
212 sock_srv = socket(AF_UNIX, SOCK_STREAM, 0);
213 if (sock_srv < 0)
214 {
215 SWSS_LOG_ERROR("socket() returns %d", errno);
216 return errno;
217 }
218
219 /***************************************/
220 /* Set up the UNIX socket address */
221 /* by using AF_UNIX for the family and */
222 /* giving it a file path to bind to. */
223 /* */
224 /* Delete the file so the bind will */
225 /* succeed, then bind to that file. */
226 /***************************************/
227 memset(&addr, 0, sizeof(addr));
228 addr.sun_family = AF_UNIX;
229 sprintf(addr.sun_path, "%s/%s.srv", path, SYNCD_IPC_SOCK_FILE);
230 unlink(addr.sun_path);
231 if (bind(sock_srv, (struct sockaddr *)&addr, sizeof(addr)) < 0)
232 {
233 SWSS_LOG_ERROR("bind() returns %d", errno);
234 close(sock_srv);
235 return errno;
236 }
237
238 /* Listen for the upcoming client sockets */
239 if (listen(sock_srv, CONN_MAX) < 0)
240 {
241 SWSS_LOG_ERROR("listen() returns %d", errno);
242 unlink(addr.sun_path);
243 close(sock_srv);
244 return errno;
245 }
246
247 SWSS_LOG_NOTICE("IPC service is online\n");
248
249 memset(conn, 0, sizeof(conn));
250 while (m_taskAlive)
251 {
252 time_t now;
253 struct timeval timeout;
254
255 /* garbage collection */
256 now = time(NULL);
257 for (i = 0; i < CONN_MAX; ++i)
258 {
259 if ((conn[i].fd > 0) && (conn[i].timeout < now))
260 {
261 SWSS_LOG_NOTICE("socket %d: connection timeout\n", conn[i].fd);
262 close(conn[i].fd);
263 conn[i].fd = 0;
264 conn[i].timeout = 0;
265 }
266 }
267
268 /* reset read file descriptors */
269 FD_ZERO(&rfds);
270 FD_SET(sock_srv, &rfds);
271 sock_max = sock_srv;
272 for (i = 0; i < CONN_MAX; ++i)
273 {
274 if (conn[i].fd <= 0)
275 {
276 continue;
277 }
278 FD_SET(conn[i].fd, &rfds);
279 if (sock_max < conn[i].fd)
280 {
281 sock_max = conn[i].fd;
282 }
283 }
284
285 /* monitor the socket descriptors */
286 timeout.tv_sec = 1;
287 timeout.tv_usec = 0;
288 ret = select(sock_max + 1, &rfds, NULL, NULL, &timeout);
289 if (ret == 0)
290 {
291 continue;
292 }
293 else if (ret < 0)
294 {
295 if (errno == EINTR)
296 {
297 continue;
298 }
299 else
300 {
301 SWSS_LOG_ERROR("select() returns %d", errno);
302 break;
303 }
304 }
305
306 /* Accept the new connection */
307 now = time(NULL);
308 if (FD_ISSET(sock_srv, &rfds))
309 {
310 sock_cli = accept(sock_srv, NULL, NULL);
311 if (sock_cli <= 0)
312 {
313 SWSS_LOG_ERROR("accept() returns %d", errno);
314 continue;
315 }
316
317 for (i = 0; i < CONN_MAX; ++i)
318 {
319 if (conn[i].fd <= 0)
320 {
321 break;
322 }
323 }
324 if (i < CONN_MAX)
325 {
326 conn[i].fd = sock_cli;
327 conn[i].timeout = now + CONN_TIMEOUT;
328 }
329 else
330 {
331 SWSS_LOG_ERROR("too many connections!");
332 close(sock_cli);
333 }
334 }
335
336 /* Handle the client requests */
337 for (i = 0; i < CONN_MAX; ++i)
338 {
339 sai_status_t rc = SAI_STATUS_NOT_SUPPORTED;
340
341 sock_cli = conn[i].fd;
342 if ((sock_cli <= 0) || !FD_ISSET(sock_cli, &rfds))
343 {
344 continue;
345 }
346
347 /* get the command message */
348 len = (int)recv(sock_cli, (void *)cmd, sizeof(cmd) - 1, 0);
349 if (len <= 0)
350 {
351 close(sock_cli);
352 conn[i].fd = 0;
353 conn[i].timeout = 0;
354 continue;
355 }
356 cmd[len] = 0;
357
358 /* tokenize the command string */
359 argc = 0;
360 std::string str(cmd);
361 boost::algorithm::trim(str);
362 boost::algorithm::to_lower(str);
363 std::vector<char>v(str.size()+1);
364 memcpy( &v.front(), str.c_str(), str.size() + 1 );
365 argv[argc++] = strtok_r(v.data(), " \t\r\n", &save);
366 while (argc < COUNTOF(argv))
367 {
368 argv[argc] = strtok_r(NULL, " \t\r\n", &save);
369 if (argv[argc] == NULL)
370 break;
371 ++argc;
372 }
373
374 /* command dispatch */
375 resp[0] = 0;
376 rc = SAI_STATUS_NOT_SUPPORTED;
377 if (argv[0] == NULL)
378 {
379 rc = SAI_STATUS_NOT_SUPPORTED;
380 }
381 else if (strcmp("mdio", argv[0]) == 0)
382 {
383 rc = MdioIpcServer::syncd_ipc_cmd_mdio(resp, argc, argv);
384 }
385 else if (strcmp("mdio-cl22", argv[0]) == 0)
386 {
387 rc = MdioIpcServer::syncd_ipc_cmd_mdio_cl22(resp, argc, argv);
388 }
389
390 /* build the error message */
391 if (rc != SAI_STATUS_SUCCESS)
392 {
393 sprintf(resp, "%d\n", rc);
394 }
395
396 /* send out the response */
397 len = (int)strlen(resp);
398 if (send(sock_cli, resp, len, 0) < len)
399 {
400 SWSS_LOG_ERROR("send() returns %d", errno);
401 }
402
403 /* update the connection timeout counter */
404 conn[i].timeout = time(NULL) + CONN_TIMEOUT;
405 }
406 }
407
408 /* close socket descriptors */
409 for (i = 0; i < CONN_MAX; ++i)
410 {
411 if (conn[i].fd <= 0)
412 {
413 continue;
414 }
415 close(conn[i].fd);
416 }
417 close(sock_srv);
418 unlink(addr.sun_path);
419 return errno;
420}
