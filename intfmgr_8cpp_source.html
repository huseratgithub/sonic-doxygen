79void IntfMgr::setIntfIp(const string &alias, const string &opCmd,
80 const IpPrefix &ipPrefix)
81{
82 stringstream cmd;
83 string res;
84 string ipPrefixStr = ipPrefix.to_string();
85 string broadcastIpStr = ipPrefix.getBroadcastIp().to_string();
86 int prefixLen = ipPrefix.getMaskLength();
87
88 if (ipPrefix.isV4())
89 {
90 (prefixLen < 31) ?
91 (cmd << IP_CMD << " address " << shellquote(opCmd) << " " << shellquote(ipPrefixStr) << " broadcast " << shellquote(broadcastIpStr) <<" dev " << shellquote(alias)) :
92 (cmd << IP_CMD << " address " << shellquote(opCmd) << " " << shellquote(ipPrefixStr) << " dev " << shellquote(alias));
93 }
94 else
95 {
96 string metric = "";
97 // Kernel adds connected route with default metric of 256. But the metric is not
98 // communicated to frr unless the ip address is added with explicit metric
99 // In voq system, We need the static route to the remote neighbor and connected
100 // route to have the same metric to enable BGP to choose paths from routes learned
101 // via eBGP and iBGP over the internal inband port be part of same ecmp group.
102 // For v4 both the metrics (connected and static) are default 0 so we do not need
103 // to set the metric explicitly.
104 if(mySwitchType == "voq")
105 {
106 metric = " metric 256";
107 }
108
109 (prefixLen < 127) ?
110 (cmd << IP_CMD << " -6 address " << shellquote(opCmd) << " " << shellquote(ipPrefixStr) << " broadcast " << shellquote(broadcastIpStr) <<
111 " dev " << shellquote(alias) << metric) :
112 (cmd << IP_CMD << " -6 address " << shellquote(opCmd) << " " << shellquote(ipPrefixStr) << " dev " << shellquote(alias) << metric);
113 }
114
115 int ret = swss::exec(cmd.str(), res);
116 if (ret)
117 {
118 if (!ipPrefix.isV4() && opCmd == "add")
119 {
120 SWSS_LOG_NOTICE("Failed to assign IPv6 on interface %s with return code %d, trying to enable IPv6 and retry", alias.c_str(), ret);
121 if (!enableIpv6Flag(alias))
122 {
123 SWSS_LOG_ERROR("Failed to enable IPv6 on interface %s", alias.c_str());
124 return;
125 }
126 ret = swss::exec(cmd.str(), res);
127 }
128
129 if (ret)
130 {
131 SWSS_LOG_ERROR("Command '%s' failed with rc %d", cmd.str().c_str(), ret);
132 }
133 }
134}

535bool IntfMgr::setIntfGratArp(const string &alias, const string &grat_arp)
536{
537 /*
538 * Enable gratuitous ARP by accepting unsolicited ARP replies and untracked neighbor advertisements
539 */
540 stringstream cmd;
541 string res;
542 string garp_enabled;
543 int rc;
544
545 if (grat_arp == "enabled")
546 {
547 garp_enabled = "1";
548 }
549 else if (grat_arp == "disabled")
550 {
551 garp_enabled = "0";
552 }
553 else
554 {
555 SWSS_LOG_ERROR("GARP state is invalid: \"%s\"", grat_arp.c_str());
556 return false;
557 }
558
559 cmd << ECHO_CMD << " " << garp_enabled << " > /proc/sys/net/ipv4/conf/" << alias << "/arp_accept";
560 EXEC_WITH_ERROR_THROW(cmd.str(), res);
561 SWSS_LOG_INFO("ARP accept set to \"%s\" on interface \"%s\"", grat_arp.c_str(), alias.c_str());
562
563 cmd.clear();
564 cmd.str(std::string());
565
566 // `accept_untracked_na` is not available in all kernels, so check for it before trying to set it
567 cmd << "test -f /proc/sys/net/ipv6/conf/" << alias << "/accept_untracked_na";
568 rc = swss::exec(cmd.str(), res);
569
570 if (rc == 0) {
571 cmd.clear();
572 cmd.str(std::string());
573 cmd << ECHO_CMD << " " << garp_enabled << " > /proc/sys/net/ipv6/conf/" << alias << "/accept_untracked_na";
574 EXEC_WITH_ERROR_THROW(cmd.str(), res);
575 SWSS_LOG_INFO("`accept_untracked_na` set to \"%s\" on interface \"%s\"", grat_arp.c_str(), alias.c_str());
576 }
577
578 return true;
579}

708 const string& op)
709{
710 SWSS_LOG_ENTER();
711
712 string alias(keys[0]);
713 string vlanId;
714 string parentAlias;
715 size_t found = alias.find(VLAN_SUB_INTERFACE_SEPARATOR);
716 if (found != string::npos)
717 {
718 subIntf subIf(alias);
719 // alias holds the complete sub interface name
720 // while parentAlias holds the parent port name
721 /*Check if subinterface is valid and sub interface name length is < 15(IFNAMSIZ)*/
722 if (!subIf.isValid())
723 {
724 SWSS_LOG_ERROR("Invalid subnitf: %s", alias.c_str());
725 return true;
726 }
727 parentAlias = subIf.parentIntf();
728 int subIntfId = subIf.subIntfIdx();
729 /*If long name format, subinterface Id is vlanid */
730 if (!subIf.isShortName())
731 {
732 vlanId = std::to_string(subIntfId);
733 FieldValueTuple vlanTuple("vlan", vlanId);
734 data.push_back(vlanTuple);
735 }
736 }
737 bool is_lo = !alias.compare(0, strlen(LOOPBACK_PREFIX), LOOPBACK_PREFIX);
738 string mac = "";
739 string vrf_name = "";
740 string mtu = "";
741 string adminStatus = "";
742 string nat_zone = "";
743 string proxy_arp = "";
744 string grat_arp = "";
745 string mpls = "";
746 string ipv6_link_local_mode = "";
747 string loopback_action = "";
748
749 for (auto idx : data)
750 {
751 const auto &field = fvField(idx);
752 const auto &value = fvValue(idx);
753
754 if (field == "vnet_name" || field == "vrf_name")
755 {
756 vrf_name = value;
757 }
758 else if (field == "mac_addr")
759 {
760 mac = value;
761 }
762 else if (field == "admin_status")
763 {
764 adminStatus = value;
765 }
766 else if (field == "proxy_arp")
767 {
768 proxy_arp = value;
769 }
770 else if (field == "grat_arp")
771 {
772 grat_arp = value;
773 }
774 else if (field == "mpls")
775 {
776 mpls = value;
777 }
778 else if (field == "nat_zone")
779 {
780 nat_zone = value;
781 }
782 else if (field == "ipv6_use_link_local_only")
783 {
784 ipv6_link_local_mode = value;
785 }
786 else if (field == "vlan")
787 {
788 vlanId = value;
789 }
790 else if (field == "loopback_action")
791 {
792 loopback_action = value;
793 }
794 }
795
796 if (op == SET_COMMAND)
797 {
798 if (!isIntfStateOk(parentAlias.empty() ? alias : parentAlias))
799 {
800 SWSS_LOG_DEBUG("Interface is not ready, skipping %s", alias.c_str());
801 return false;
802 }
803
804 if (!vrf_name.empty() && !isIntfStateOk(vrf_name))
805 {
806 SWSS_LOG_DEBUG("VRF is not ready, skipping %s", vrf_name.c_str());
807 return false;
808 }
809
810 /* if to change vrf then skip */
811 if (isIntfChangeVrf(alias, vrf_name))
812 {
813 SWSS_LOG_ERROR("%s can not change to %s directly, skipping", alias.c_str(), vrf_name.c_str());
814 return true;
815 }
816
817 if (is_lo)
818 {
819 if (m_loopbackIntfList.find(alias) == m_loopbackIntfList.end())
820 {
821 addLoopbackIntf(alias);
822 m_loopbackIntfList.insert(alias);
823 SWSS_LOG_INFO("Added %s loopback interface", alias.c_str());
824 }
825 }
826 else
827 {
828 /* Set nat zone */
829 if (!nat_zone.empty())
830 {
831 FieldValueTuple fvTuple("nat_zone", nat_zone);
832 data.push_back(fvTuple);
833 }
834
835 /* Set loopback action */
836 if (!loopback_action.empty())
837 {
838 FieldValueTuple fvTuple("loopback_action", loopback_action);
839 data.push_back(fvTuple);
840 }
841
842 /* Set mpls */
843 if (!setIntfMpls(alias, mpls))
844 {
845 SWSS_LOG_ERROR("Failed to set MPLS to \"%s\" for the \"%s\" interface", mpls.c_str(), alias.c_str());
846 return false;
847 }
848 if (!mpls.empty())
849 {
850 FieldValueTuple fvTuple("mpls", mpls);
851 data.push_back(fvTuple);
852 }
853
854 /* Set ipv6 mode */
855 if (!ipv6_link_local_mode.empty())
856 {
857 if ((ipv6_link_local_mode == "enable") && (m_ipv6LinkLocalModeList.find(alias) == m_ipv6LinkLocalModeList.end()))
858 {
859 m_ipv6LinkLocalModeList.insert(alias);
860 SWSS_LOG_INFO("Inserted ipv6 link local mode list for %s", alias.c_str());
861 }
862 else if ((ipv6_link_local_mode == "disable") && (m_ipv6LinkLocalModeList.find(alias) != m_ipv6LinkLocalModeList.end()))
863 {
864 m_ipv6LinkLocalModeList.erase(alias);
865 delIpv6LinkLocalNeigh(alias);
866 SWSS_LOG_INFO("Erased ipv6 link local mode list for %s", alias.c_str());
867 }
868 FieldValueTuple fvTuple("ipv6_use_link_local_only", ipv6_link_local_mode);
869 data.push_back(fvTuple);
870 }
871 }
872
873 if (!parentAlias.empty())
874 {
875 subIntf subIf(alias);
876 if (m_subIntfList.find(alias) == m_subIntfList.end())
877 {
878 if (vlanId == "0" || vlanId.empty())
879 {
880 SWSS_LOG_INFO("Vlan ID not configured for sub interface %s", alias.c_str());
881 return false;
882 }
883 try
884 {
885 addHostSubIntf(parentAlias, alias, vlanId);
886 }
887 catch (const std::runtime_error &e)
888 {
889 SWSS_LOG_NOTICE("Sub interface ip link add failure. Runtime error: %s", e.what());
890 return false;
891 }
892
893 m_subIntfList[alias].vlanId = vlanId;
894 }
895
896 if (!mtu.empty())
897 {
898 string subintf_mtu;
899 try
900 {
901 string parentMtu = getIntfMtu(subIf.parentIntf());
902 subintf_mtu = setHostSubIntfMtu(alias, mtu, parentMtu);
903 FieldValueTuple fvTuple("mtu", mtu);
904 std::remove(data.begin(), data.end(), fvTuple);
905 FieldValueTuple newMtuFvTuple("mtu", subintf_mtu);
906 data.push_back(newMtuFvTuple);
907 }
908 catch (const std::runtime_error &e)
909 {
910 SWSS_LOG_NOTICE("Sub interface ip link set mtu failure. Runtime error: %s", e.what());
911 return false;
912 }
913 m_subIntfList[alias].mtu = mtu;
914 }
915 else
916 {
917 FieldValueTuple fvTuple("mtu", MTU_INHERITANCE);
918 data.push_back(fvTuple);
919 m_subIntfList[alias].mtu = MTU_INHERITANCE;
920 }
921
922 if (adminStatus.empty())
923 {
924 adminStatus = "up";
925 FieldValueTuple fvTuple("admin_status", adminStatus);
926 data.push_back(fvTuple);
927 }
928 try
929 {
930 string parentAdmin = getIntfAdminStatus(subIf.parentIntf());
931 string subintf_admin = setHostSubIntfAdminStatus(alias, adminStatus, parentAdmin);
932 m_subIntfList[alias].currAdminStatus = subintf_admin;
933 FieldValueTuple fvTuple("admin_status", adminStatus);
934 std::remove(data.begin(), data.end(), fvTuple);
935 FieldValueTuple newAdminFvTuple("admin_status", subintf_admin);
936 data.push_back(newAdminFvTuple);
937 }
938 catch (const std::runtime_error &e)
939 {
940 SWSS_LOG_NOTICE("Sub interface ip link set admin status %s failure. Runtime error: %s", adminStatus.c_str(), e.what());
941 return false;
942 }
943 m_subIntfList[alias].adminStatus = adminStatus;
944
945 // set STATE_DB port state
946 setSubIntfStateOk(alias);
947 }
948
949 if (!vrf_name.empty())
950 {
951 setIntfVrf(alias, vrf_name);
952 }
953
954 /*Set the mac of interface*/
955 if (!mac.empty())
956 {
957 setIntfMac(alias, mac);
958 }
959 else
960 {
961 FieldValueTuple fvTuple("mac_addr", MacAddress().to_string());
962 data.push_back(fvTuple);
963 }
964
965 if (!proxy_arp.empty())
966 {
967 if (!setIntfProxyArp(alias, proxy_arp))
968 {
969 SWSS_LOG_ERROR("Failed to set proxy ARP to \"%s\" state for the \"%s\" interface", proxy_arp.c_str(), alias.c_str());
970 return false;
971 }
972
973 if (!alias.compare(0, strlen(VLAN_PREFIX), VLAN_PREFIX))
974 {
975 FieldValueTuple fvTuple("proxy_arp", proxy_arp);
976 data.push_back(fvTuple);
977 }
978 }
979
980 if (!grat_arp.empty())
981 {
982 if (!setIntfGratArp(alias, grat_arp))
983 {
984 SWSS_LOG_ERROR("Failed to set ARP accept to \"%s\" state for the \"%s\" interface", grat_arp.c_str(), alias.c_str());
985 return false;
986 }
987
988 if (!alias.compare(0, strlen(VLAN_PREFIX), VLAN_PREFIX))
989 {
990 FieldValueTuple fvTuple("grat_arp", grat_arp);
991 data.push_back(fvTuple);
992 }
993 }
994
995 m_appIntfTableProducer.set(alias, data);
996 m_stateIntfTable.hset(alias, "vrf", vrf_name);
997 }
998 else if (op == DEL_COMMAND)
999 {
1000 /* make sure all ip addresses associated with interface are removed, otherwise these ip address would
1001 be set with global vrf and it may cause ip address conflict. */
1002 if (getIntfIpCount(alias))
1003 {
1004 return false;
1005 }
1006
1007 setIntfVrf(alias, "");
1008
1009 if (is_lo)
1010 {
1011 delLoopbackIntf(alias);
1012 m_loopbackIntfList.erase(alias);
1013 }
1014
1015 if (!parentAlias.empty())
1016 {
1017 removeHostSubIntf(alias);
1018 m_subIntfList.erase(alias);
1019
1020 removeSubIntfState(alias);
1021 }
1022
1023 if (m_ipv6LinkLocalModeList.find(alias) != m_ipv6LinkLocalModeList.end())
1024 {
1025 m_ipv6LinkLocalModeList.erase(alias);
1026 delIpv6LinkLocalNeigh(alias);
1027 SWSS_LOG_INFO("Erased ipv6 link local mode list for %s", alias.c_str());
1028 }
1029
1030 m_appIntfTableProducer.del(alias);
1031 m_stateIntfTable.del(alias);
1032 }
1033 else
1034 {
1035 SWSS_LOG_ERROR("Unknown operation: %s", op.c_str());
1036 }
1037
1038 return true;
1039}

1041bool IntfMgr::doIntfAddrTask(const vector<string>& keys,
1042 const vector<FieldValueTuple>& data,
1043 const string& op)
1044{
1045 SWSS_LOG_ENTER();
1046
1047 string alias(keys[0]);
1048 IpPrefix ip_prefix(keys[1]);
1049 string appKey = keys[0] + ":" + keys[1];
1050
1051 if (op == SET_COMMAND)
1052 {
1053 /*
1054 * Don't proceed if port/LAG/VLAN/subport and intfGeneral is not ready yet.
1055 * The pending task will be checked periodically and retried.
1056 */
1057 if (!isIntfStateOk(alias) || !isIntfCreated(alias))
1058 {
1059 SWSS_LOG_DEBUG("Interface is not ready, skipping %s", alias.c_str());
1060 return false;
1061 }
1062
1063 setIntfIp(alias, "add", ip_prefix);
1064
1065 std::vector<FieldValueTuple> fvVector;
1066 FieldValueTuple f("family", ip_prefix.isV4() ? IPV4_NAME : IPV6_NAME);
1067
1068 // Don't send ipv4 link local config to AppDB and Orchagent
1069 if ((ip_prefix.isV4() == false) || (ip_prefix.getIp().getAddrScope() != IpAddress::AddrScope::LINK_SCOPE))
1070 {
1071 FieldValueTuple s("scope", "global");
1072 fvVector.push_back(s);
1073 fvVector.push_back(f);
1074 m_appIntfTableProducer.set(appKey, fvVector);
1075 m_stateIntfTable.hset(keys[0] + state_db_key_delimiter + keys[1], "state", "ok");
1076 }
1077 }
1078 else if (op == DEL_COMMAND)
1079 {
1080 setIntfIp(alias, "del", ip_prefix);
1081
1082 // Don't send ipv4 link local config to AppDB and Orchagent
1083 if ((ip_prefix.isV4() == false) || (ip_prefix.getIp().getAddrScope() != IpAddress::AddrScope::LINK_SCOPE))
1084 {
1085 m_appIntfTableProducer.del(appKey);
1086 m_stateIntfTable.del(keys[0] + state_db_key_delimiter + keys[1]);
1087 }
1088 }
1089 else
1090 {
1091 SWSS_LOG_ERROR("Unknown operation: %s", op.c_str());
1092 }
1093
1094 return true;
1095}

1097void IntfMgr::doTask(Consumer &consumer)
1098{
1099 SWSS_LOG_ENTER();
1100
1101 string table_name = consumer.getTableName();
1102
1103 auto it = consumer.m_toSync.begin();
1104 while (it != consumer.m_toSync.end())
1105 {
1106 KeyOpFieldsValuesTuple t = it->second;
1107 if ((table_name == STATE_PORT_TABLE_NAME) || (table_name == STATE_LAG_TABLE_NAME))
1108 {
1109 doPortTableTask(kfvKey(t), kfvFieldsValues(t), kfvOp(t));
1110 }
1111 else
1112 {
1113 vector<string> keys = tokenize(kfvKey(t), config_db_key_delimiter);
1114 const vector<FieldValueTuple>& data = kfvFieldsValues(t);
1115 string op = kfvOp(t);
1116
1117 if (keys.size() == 1)
1118 {
1119 if((table_name == CFG_VOQ_INBAND_INTERFACE_TABLE_NAME) &&
1120 (op == SET_COMMAND))
1121 {
1122 //No further processing needed. Just relay to orchagent
1123 m_appIntfTableProducer.set(keys[0], data);
1124 m_stateIntfTable.hset(keys[0], "vrf", "");
1125
1126 it = consumer.m_toSync.erase(it);
1127 continue;
1128 }
1129 if (!doIntfGeneralTask(keys, data, op))
1130 {
1131 it++;
1132 continue;
1133 }
1134 else
1135 {
1136 //Entry programmed, remove it from pending list if present
1137 m_pendingReplayIntfList.erase(keys[0]);
1138 }
1139 }
1140 else if (keys.size() == 2)
1141 {
1142 if (!doIntfAddrTask(keys, data, op))
1143 {
1144 it++;
1145 continue;
1146 }
1147 else
1148 {
1149 //Entry programmed, remove it from pending list if present
1150 m_pendingReplayIntfList.erase(keys[0] + config_db_key_delimiter + keys[1] );
1151 }
1152 }
1153 else
1154 {
1155 SWSS_LOG_ERROR("Invalid key %s", kfvKey(t).c_str());
1156 }
1157 }
1158
1159 it = consumer.m_toSync.erase(it);
1160 }
1161
1162 if (!m_replayDone && WarmStart::isWarmStart() && m_pendingReplayIntfList.empty() )
1163 {
1164 setWarmReplayDoneState();
1165 }
1166}
