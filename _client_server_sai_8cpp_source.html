38sai_status_t ClientServerSai::initialize(
39 _In_ uint64_t flags,
40 _In_ const sai_service_method_table_t *service_method_table)
41{
42 MUTEX();
43 SWSS_LOG_ENTER();
44
45 if (m_apiInitialized)
46 {
47 SWSS_LOG_ERROR("%s: api already initialized", __PRETTY_FUNCTION__);
48
49 return SAI_STATUS_FAILURE;
50 }
51
52 if (flags != 0)
53 {
54 SWSS_LOG_ERROR("invalid flags passed to SAI API initialize");
55
56 return SAI_STATUS_INVALID_PARAMETER;
57 }
58
59 if ((service_method_table == NULL) ||
60 (service_method_table->profile_get_next_value == NULL) ||
61 (service_method_table->profile_get_value == NULL))
62 {
63 SWSS_LOG_ERROR("invalid service_method_table handle passed to SAI API initialize");
64
65 return SAI_STATUS_INVALID_PARAMETER;
66 }
67
68 memcpy(&m_service_method_table, service_method_table, sizeof(m_service_method_table));
69
70 const char* enableClient = service_method_table->profile_get_value(0, SAI_REDIS_KEY_ENABLE_CLIENT);
71
72 if (enableClient && strcmp(enableClient, "true") == 0)
73 {
74 SWSS_LOG_NOTICE("acting as a sairedis CLIENT");
75
76 m_sai = std::make_shared<ClientSai>();
77 }
78 else
79 {
80 SWSS_LOG_NOTICE("acting as a sairedis SERVER");
81
82 m_sai = std::make_shared<ServerSai>();
83 }
84
85 auto status = m_sai->initialize(flags, service_method_table);
86
87 SWSS_LOG_NOTICE("init client/server sai: %s", sai_serialize_status(status).c_str());
88
89 if (status == SAI_STATUS_SUCCESS)
90 {
91 m_apiInitialized = true;
92 }
93
94 return status;
95}
