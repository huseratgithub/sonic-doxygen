88bool VNetVrfObject::createObj(vector<sai_attribute_t>& attrs)
89{

91
92 sai_status_t status = sai_virtual_router_api->create_virtual_router(&router_id,
93 gSwitchId,
94 static_cast<uint32_t>(attrs.size()),
95 attrs.data());
96 if (status != SAI_STATUS_SUCCESS)
97 {
98 SWSS_LOG_ERROR("Failed to create virtual router name: %s, rv: %d",
99 vnet_name_.c_str(), status);
100 throw std::runtime_error("Failed to create VR object");
101 }
102 gFlowCounterRouteOrch->onAddVR(router_id);
103 return true;
104 };
105
106 /*
107 * Create ingress and egress VRF based on VR_VALID
108 */
109
110 for (auto vr_type : vr_cntxt)
111 {
112 sai_object_id_t router_id = gVirtualRouterId;
113 if (vr_type != VR_TYPE::VR_INVALID)
114 {
115 if (getScope() != "default")
116 {
117 l_fn(router_id);
118 }
119 SWSS_LOG_DEBUG("VNET vr_type %d router id %" PRIx64 " ", static_cast<int>(vr_type), router_id);
120 vr_ids_.insert(std::pair<VR_TYPE, sai_object_id_t>(vr_type, router_id));
121 }
122 }
123
124 SWSS_LOG_INFO("VNET '%s' router object created ", vnet_name_.c_str());
125 return true;

421bool VNetOrch::addOperation(const Request& request)
422{
423 SWSS_LOG_ENTER();
424
425 sai_attribute_t attr;
426 vector<sai_attribute_t> attrs;
427 set<string> peer_list = {};
428 bool peer = false, create = false, advertise_prefix = false;
429 uint32_t vni=0;
430 string tunnel;
431 string scope;
432
433 for (const auto& name: request.getAttrFieldNames())
434 {
435 if (name == "src_mac")
436 {
437 const auto& mac = request.getAttrMacAddress("src_mac");
438 attr.id = SAI_VIRTUAL_ROUTER_ATTR_SRC_MAC_ADDRESS;
439 memcpy(attr.value.mac, mac.getMac(), sizeof(sai_mac_t));
440 attrs.push_back(attr);
441 }
442 else if (name == "peer_list")
443 {
444 peer_list = request.getAttrSet("peer_list");
445 peer = true;
446 }
447 else if (name == "vni")
448 {
449 vni = static_cast<sai_uint32_t>(request.getAttrUint("vni"));
450 }
451 else if (name == "vxlan_tunnel")
452 {
453 tunnel = request.getAttrString("vxlan_tunnel");
454 }
455 else if (name == "scope")
456 {
457 scope = request.getAttrString("scope");
458 }
459 else if (name == "advertise_prefix")
460 {
461 advertise_prefix = request.getAttrBool("advertise_prefix");
462 }
463 else
464 {
465 SWSS_LOG_INFO("Unknown attribute: %s", name.c_str());
466 continue;
467 }
468 }
469
470 const std::string& vnet_name = request.getKeyString(0);
471 SWSS_LOG_INFO("VNET '%s' add request", vnet_name.c_str());
472
473 try
474 {
475 VNetObject_T obj;
476 auto it = vnet_table_.find(vnet_name);
477 if (isVnetExecVrf())
478 {
479 VxlanTunnelOrch* vxlan_orch = gDirectory.get<VxlanTunnelOrch*>();
480
481 if (!vxlan_orch->isTunnelExists(tunnel))
482 {
483 SWSS_LOG_WARN("Vxlan tunnel '%s' doesn't exist", tunnel.c_str());
484 return false;
485 }
486
487 if (it == std::end(vnet_table_))
488 {
489 VNetInfo vnet_info = { tunnel, vni, peer_list, scope, advertise_prefix };
490 obj = createObject<VNetVrfObject>(vnet_name, vnet_info, attrs);
491 create = true;
492
493 VNetVrfObject *vrf_obj = dynamic_cast<VNetVrfObject*>(obj.get());
494 if (!vxlan_orch->createVxlanTunnelMap(tunnel, TUNNEL_MAP_T_VIRTUAL_ROUTER, vni,
495 vrf_obj->getEncapMapId(), vrf_obj->getDecapMapId(), VXLAN_ENCAP_TTL))
496 {
497 SWSS_LOG_ERROR("VNET '%s', tunnel '%s', map create failed",
498 vnet_name.c_str(), tunnel.c_str());
499 return false;
500 }
501
502 SWSS_LOG_NOTICE("VNET '%s' was added ", vnet_name.c_str());
503 }
504 else
505 {
506 SWSS_LOG_NOTICE("VNET '%s' already exists ", vnet_name.c_str());
507 }
508 }
509
510 if (create)
511 {
512 vnet_table_[vnet_name] = std::move(obj);
513 }
514 else if (peer)
515 {
516 it->second->setPeerList(peer_list);
517 }
518 else if (!attrs.empty())
519 {
520 if(!it->second->updateObj(attrs))
521 {
522 return true;
523 }
524 }
525
526 }
527 catch(std::runtime_error& _)
528 {
529 SWSS_LOG_ERROR("VNET add operation error for %s: error %s ", vnet_name.c_str(), _.what());
530 return false;
531 }
532
533 SWSS_LOG_INFO("VNET '%s' added/updated ", vnet_name.c_str());
534 return true;
535}

735bool VNetRouteOrch::addNextHopGroup(const string& vnet, const NextHopGroupKey &nexthops, VNetVrfObject *vrf_obj)
736{
737 SWSS_LOG_ENTER();
738
739 assert(!hasNextHopGroup(vnet, nexthops));
740
741 if (!gRouteOrch->checkNextHopGroupCount())
742 {
743 SWSS_LOG_ERROR("Reached maximum number of next hop groups. Failed to create new next hop group.");
744 return false;
745 }
746
747 vector<sai_object_id_t> next_hop_ids;
748 set<NextHopKey> next_hop_set = nexthops.getNextHops();
749 std::map<sai_object_id_t, NextHopKey> nhopgroup_members_set;
750 std::map<NextHopKey, uint32_t> nh_seq_id_in_nhgrp;
751 uint32_t seq_id = 0;
752
753 for (auto it : next_hop_set)
754 {
755 nh_seq_id_in_nhgrp[it] = ++seq_id;
756 if (nexthop_info_[vnet].find(it.ip_address) != nexthop_info_[vnet].end() && nexthop_info_[vnet][it.ip_address].bfd_state != SAI_BFD_SESSION_STATE_UP)
757 {
758 continue;
759 }
760 sai_object_id_t next_hop_id = vrf_obj->getTunnelNextHop(it);
761 next_hop_ids.push_back(next_hop_id);
762 nhopgroup_members_set[next_hop_id] = it;
763 }
764
765 sai_attribute_t nhg_attr;
766 vector<sai_attribute_t> nhg_attrs;
767
768 nhg_attr.id = SAI_NEXT_HOP_GROUP_ATTR_TYPE;
769 nhg_attr.value.s32 = gSwitchOrch->checkOrderedEcmpEnable() ? SAI_NEXT_HOP_GROUP_TYPE_DYNAMIC_ORDERED_ECMP : SAI_NEXT_HOP_GROUP_TYPE_ECMP;
770 nhg_attrs.push_back(nhg_attr);
771
772 sai_object_id_t next_hop_group_id;
773 sai_status_t status = sai_next_hop_group_api->create_next_hop_group(&next_hop_group_id,
774 gSwitchId,
775 (uint32_t)nhg_attrs.size(),
776 nhg_attrs.data());
777
778 if (status != SAI_STATUS_SUCCESS)
779 {
780 SWSS_LOG_ERROR("Failed to create next hop group %s, rv:%d",
781 nexthops.to_string().c_str(), status);
782 return false;
783 }
784
785 gRouteOrch->increaseNextHopGroupCount();
786 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
787 SWSS_LOG_NOTICE("Create next hop group %s", nexthops.to_string().c_str());
788
789 NextHopGroupInfo next_hop_group_entry;
790 next_hop_group_entry.next_hop_group_id = next_hop_group_id;
791
792 for (auto nhid: next_hop_ids)
793 {
794 // Create a next hop group member
795 vector<sai_attribute_t> nhgm_attrs;
796
797 sai_attribute_t nhgm_attr;
798 nhgm_attr.id = SAI_NEXT_HOP_GROUP_MEMBER_ATTR_NEXT_HOP_GROUP_ID;
799 nhgm_attr.value.oid = next_hop_group_id;
800 nhgm_attrs.push_back(nhgm_attr);
801
802 nhgm_attr.id = SAI_NEXT_HOP_GROUP_MEMBER_ATTR_NEXT_HOP_ID;
803 nhgm_attr.value.oid = nhid;
804 nhgm_attrs.push_back(nhgm_attr);
805
806 if (gSwitchOrch->checkOrderedEcmpEnable())
807 {
808 nhgm_attr.id = SAI_NEXT_HOP_GROUP_MEMBER_ATTR_SEQUENCE_ID;
809 nhgm_attr.value.u32 = nh_seq_id_in_nhgrp[nhopgroup_members_set.find(nhid)->second];
810 nhgm_attrs.push_back(nhgm_attr);
811 }
812
813 sai_object_id_t next_hop_group_member_id;
814 status = sai_next_hop_group_api->create_next_hop_group_member(&next_hop_group_member_id,
815 gSwitchId,
816 (uint32_t)nhgm_attrs.size(),
817 nhgm_attrs.data());
818
819 if (status != SAI_STATUS_SUCCESS)
820 {
821 SWSS_LOG_ERROR("Failed to create next hop group %" PRIx64 " member %" PRIx64 ": %d\n",
822 next_hop_group_id, next_hop_group_member_id, status);
823 return false;
824 }
825
826 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP_MEMBER);
827
828 // Save the membership into next hop structure
829 next_hop_group_entry.active_members[nhopgroup_members_set.find(nhid)->second] =
830 next_hop_group_member_id;
831 }
832
833 /*
834 * Initialize the next hop group structure with ref_count as 0. This
835 * count will increase once the route is successfully syncd.
836 */
837 next_hop_group_entry.ref_count = 0;
838 syncd_nexthop_groups_[vnet][nexthops] = next_hop_group_entry;
839
840 return true;
841}

843bool VNetRouteOrch::removeNextHopGroup(const string& vnet, const NextHopGroupKey &nexthops, VNetVrfObject *vrf_obj)
844{
845 SWSS_LOG_ENTER();
846
847 sai_object_id_t next_hop_group_id;
848 auto next_hop_group_entry = syncd_nexthop_groups_[vnet].find(nexthops);
849 sai_status_t status;
850
851 assert(next_hop_group_entry != syncd_nexthop_groups_[vnet].end());
852
853 if (next_hop_group_entry->second.ref_count != 0)
854 {
855 return true;
856 }
857
858 next_hop_group_id = next_hop_group_entry->second.next_hop_group_id;
859 SWSS_LOG_NOTICE("Delete next hop group %s", nexthops.to_string().c_str());
860
861 for (auto nhop = next_hop_group_entry->second.active_members.begin();
862 nhop != next_hop_group_entry->second.active_members.end();)
863 {
864 NextHopKey nexthop = nhop->first;
865
866 status = sai_next_hop_group_api->remove_next_hop_group_member(nhop->second);
867 if (status != SAI_STATUS_SUCCESS)
868 {
869 SWSS_LOG_ERROR("Failed to remove next hop group member %" PRIx64 ", rv:%d",
870 nhop->second, status);
871 return false;
872 }
873
874 vrf_obj->removeTunnelNextHop(nexthop);
875
876 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP_MEMBER);
877 nhop = next_hop_group_entry->second.active_members.erase(nhop);
878 }
879
880 status = sai_next_hop_group_api->remove_next_hop_group(next_hop_group_id);
881 if (status != SAI_STATUS_SUCCESS)
882 {
883 SWSS_LOG_ERROR("Failed to remove next hop group %" PRIx64 ", rv:%d", next_hop_group_id, status);
884 return false;
885 }
886
887 gRouteOrch->decreaseNextHopGroupCount();
888 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
889
890 syncd_nexthop_groups_[vnet].erase(nexthops);
891
892 return true;
893}

895template<>
896bool VNetRouteOrch::doRouteTask<VNetVrfObject>(const string& vnet, IpPrefix& ipPrefix,
897 NextHopGroupKey& nexthops, string& op, string& profile,
898 const map<NextHopKey, IpAddress>& monitors)
899{
900 SWSS_LOG_ENTER();
901
902 if (!vnet_orch_->isVnetExists(vnet))
903 {
904 SWSS_LOG_WARN("VNET %s doesn't exist for prefix %s, op %s",
905 vnet.c_str(), ipPrefix.to_string().c_str(), op.c_str());
906 return (op == DEL_COMMAND)?true:false;
907 }
908
909 set<sai_object_id_t> vr_set;
910 auto& peer_list = vnet_orch_->getPeerList(vnet);
911
912 auto l_fn = [&] (const string& vnet) {
913 auto *vnet_obj = vnet_orch_->getTypePtr<VNetVrfObject>(vnet);
914 sai_object_id_t vr_id = vnet_obj->getVRidIngress();
915 vr_set.insert(vr_id);
916 };
917
918 l_fn(vnet);
919 for (auto peer : peer_list)
920 {
921 if (!vnet_orch_->isVnetExists(peer))
922 {
923 SWSS_LOG_INFO("Peer VNET %s not yet created", peer.c_str());
924 return false;
925 }
926 l_fn(peer);
927 }
928
929 auto *vrf_obj = vnet_orch_->getTypePtr<VNetVrfObject>(vnet);
930 sai_ip_prefix_t pfx;
931 copy(pfx, ipPrefix);
932
933 if (op == SET_COMMAND)
934 {
935 sai_object_id_t nh_id;
936 if (!hasNextHopGroup(vnet, nexthops))
937 {
938 setEndpointMonitor(vnet, monitors, nexthops);
939 if (nexthops.getSize() == 1)
940 {
941 NextHopKey nexthop(nexthops.to_string(), true);
942 NextHopGroupInfo next_hop_group_entry;
943 next_hop_group_entry.next_hop_group_id = vrf_obj->getTunnelNextHop(nexthop);
944 next_hop_group_entry.ref_count = 0;
945 if (nexthop_info_[vnet].find(nexthop.ip_address) == nexthop_info_[vnet].end() || nexthop_info_[vnet][nexthop.ip_address].bfd_state == SAI_BFD_SESSION_STATE_UP)
946 {
947 next_hop_group_entry.active_members[nexthop] = SAI_NULL_OBJECT_ID;
948 }
949 syncd_nexthop_groups_[vnet][nexthops] = next_hop_group_entry;
950 }
951 else
952 {
953 if (!addNextHopGroup(vnet, nexthops, vrf_obj))
954 {
955 delEndpointMonitor(vnet, nexthops);
956 SWSS_LOG_ERROR("Failed to create next hop group %s", nexthops.to_string().c_str());
957 return false;
958 }
959 }
960 }
961 nh_id = syncd_nexthop_groups_[vnet][nexthops].next_hop_group_id;
962
963 auto it_route = syncd_tunnel_routes_[vnet].find(ipPrefix);
964 for (auto vr_id : vr_set)
965 {
966 bool route_status = true;
967
968 // Remove route if the nexthop group has no active endpoint
969 if (syncd_nexthop_groups_[vnet][nexthops].active_members.empty())
970 {
971 if (it_route != syncd_tunnel_routes_[vnet].end())
972 {
973 NextHopGroupKey nhg = it_route->second;
974 // Remove route when updating from a nhg with active member to another nhg without
975 if (!syncd_nexthop_groups_[vnet][nhg].active_members.empty())
976 {
977 del_route(vr_id, pfx);
978 }
979 }
980 }
981 else
982 {
983 if (it_route == syncd_tunnel_routes_[vnet].end())
984 {
985 route_status = add_route(vr_id, pfx, nh_id);
986 }
987 else
988 {
989 NextHopGroupKey nhg = it_route->second;
990 if (syncd_nexthop_groups_[vnet][nhg].active_members.empty())
991 {
992 route_status = add_route(vr_id, pfx, nh_id);
993 }
994 else
995 {
996 route_status = update_route(vr_id, pfx, nh_id);
997 }
998 }
999 }
1000
1001 if (!route_status)
1002 {
1003 SWSS_LOG_ERROR("Route add/update failed for %s, vr_id '0x%" PRIx64, ipPrefix.to_string().c_str(), vr_id);
1004 /* Clean up the newly created next hop group entry */
1005 if (nexthops.getSize() > 1)
1006 {
1007 removeNextHopGroup(vnet, nexthops, vrf_obj);
1008 }
1009 return false;
1010 }
1011 }
1012
1013 if (it_route != syncd_tunnel_routes_[vnet].end() && it_route->second != nexthops)
1014 {
1015 // In case of updating an existing route, decrease the reference count for the previous nexthop group if not same as new nhg
1016 NextHopGroupKey nhg = it_route->second;
1017 if (--syncd_nexthop_groups_[vnet][nhg].ref_count == 0)
1018 {
1019 if (nhg.getSize() > 1)
1020 {
1021 removeNextHopGroup(vnet, nhg, vrf_obj);
1022 }
1023 else
1024 {
1025 syncd_nexthop_groups_[vnet].erase(nhg);
1026 NextHopKey nexthop(nhg.to_string(), true);
1027 vrf_obj->removeTunnelNextHop(nexthop);
1028 }
1029 delEndpointMonitor(vnet, nhg);
1030 }
1031 else
1032 {
1033 syncd_nexthop_groups_[vnet][nhg].tunnel_routes.erase(ipPrefix);
1034 }
1035 vrf_obj->removeRoute(ipPrefix);
1036 vrf_obj->removeProfile(ipPrefix);
1037 }
1038 if (it_route == syncd_tunnel_routes_[vnet].end() || (it_route != syncd_tunnel_routes_[vnet].end() && it_route->second != nexthops))
1039 {
1040 syncd_nexthop_groups_[vnet][nexthops].tunnel_routes.insert(ipPrefix);
1041
1042 syncd_tunnel_routes_[vnet][ipPrefix] = nexthops;
1043 syncd_nexthop_groups_[vnet][nexthops].ref_count++;
1044 vrf_obj->addRoute(ipPrefix, nexthops);
1045 }
1046 if (!profile.empty())
1047 {
1048 vrf_obj->addProfile(ipPrefix, profile);
1049 }
1050
1051 postRouteState(vnet, ipPrefix, nexthops, profile);
1052 }
1053 else if (op == DEL_COMMAND)
1054 {
1055 auto it_route = syncd_tunnel_routes_[vnet].find(ipPrefix);
1056 if (it_route == syncd_tunnel_routes_[vnet].end())
1057 {
1058 SWSS_LOG_INFO("Failed to find tunnel route entry, prefix %s\n",
1059 ipPrefix.to_string().c_str());
1060 return true;
1061 }
1062 NextHopGroupKey nhg = it_route->second;
1063
1064 for (auto vr_id : vr_set)
1065 {
1066 // If an nhg has no active member, the route should already be removed
1067 if (!syncd_nexthop_groups_[vnet][nhg].active_members.empty())
1068 {
1069 if (!del_route(vr_id, pfx))
1070 {
1071 SWSS_LOG_ERROR("Route del failed for %s, vr_id '0x%" PRIx64, ipPrefix.to_string().c_str(), vr_id);
1072 return false;
1073 }
1074 }
1075 }
1076
1077 if(--syncd_nexthop_groups_[vnet][nhg].ref_count == 0)
1078 {
1079 if (nhg.getSize() > 1)
1080 {
1081 removeNextHopGroup(vnet, nhg, vrf_obj);
1082 }
1083 else
1084 {
1085 syncd_nexthop_groups_[vnet].erase(nhg);
1086 NextHopKey nexthop(nhg.to_string(), true);
1087 vrf_obj->removeTunnelNextHop(nexthop);
1088 }
1089 delEndpointMonitor(vnet, nhg);
1090 }
1091 else
1092 {
1093 syncd_nexthop_groups_[vnet][nhg].tunnel_routes.erase(ipPrefix);
1094 }
1095
1096 syncd_tunnel_routes_[vnet].erase(ipPrefix);
1097 if (syncd_tunnel_routes_[vnet].empty())
1098 {
1099 syncd_tunnel_routes_.erase(vnet);
1100 }
1101
1102 vrf_obj->removeRoute(ipPrefix);
1103 vrf_obj->removeProfile(ipPrefix);
1104
1105 removeRouteState(vnet, ipPrefix);
1106 }
1107
1108 return true;
1109}

1204 return true;
1205 }
1206

1208
1209 Port port;
1210 if (is_subnet && (!gPortsOrch->getPort(nh.ifname, port) || (port.m_rif_id == SAI_NULL_OBJECT_ID)))
1211 {
1212 SWSS_LOG_WARN("Port/RIF %s doesn't exist", nh.ifname.c_str());
1213 return false;
1214 }
1215
1216 set<sai_object_id_t> vr_set;
1217 auto& peer_list = vnet_orch_->getPeerList(vnet);
1218 auto vr_id = vrf_obj->getVRidIngress();
1219
1220 /*
1221 * If RIF doesn't belong to this VRF, and if it is a replicated subnet
1222 * route for the peering VRF, Only install in ingress VRF.
1223 */
1224
1225 if (!is_subnet)
1226 {
1227 vr_set = vrf_obj->getVRids();
1228 }
1229 else if (vr_id == port.m_vr_id)
1230 {
1231 vr_set = vrf_obj->getVRids();
1232 }
1233 else
1234 {
1235 vr_set.insert(vr_id);
1236 }
1237
1238 auto l_fn = [&] (const string& vnet) {
1239 auto *vnet_obj = vnet_orch_->getTypePtr<VNetVrfObject>(vnet);
1240 sai_object_id_t vr_id = vnet_obj->getVRidIngress();
1241 vr_set.insert(vr_id);
1242 };
1243
1244 for (auto peer : peer_list)
1245 {
1246 if (!vnet_orch_->isVnetExists(peer))
1247 {
1248 SWSS_LOG_INFO("Peer VNET %s not yet created", peer.c_str());
1249 return false;

1251 l_fn(peer);
1252 }
1253
1254 sai_ip_prefix_t pfx;
1255 copy(pfx, ipPrefix);
1256 sai_object_id_t nh_id=SAI_NULL_OBJECT_ID;
1257
1258 if (is_subnet)
1259 {
1260 nh_id = port.m_rif_id;
1261 }
1262 else if (nh.ips.getSize() == 1)
1263 {
1264 NextHopKey nexthop(nh.ips.to_string(), nh.ifname);
1265 if (gNeighOrch->hasNextHop(nexthop))
1266 {
1267 nh_id = gNeighOrch->getNextHopId(nexthop);
1268 }
1269 else
1270 {
1271 SWSS_LOG_INFO("Failed to get next hop %s for %s",
1272 nexthop.to_string().c_str(), ipPrefix.to_string().c_str());
1273 return false;
1274 }
1275 }
1276 else
1277 {
1278 // FIXME - Handle ECMP routes
1279 SWSS_LOG_WARN("VNET ECMP NHs not implemented for '%s'", ipPrefix.to_string().c_str());
1280 return true;
1281 }
1282
1283 for (auto vr_id : vr_set)
1284 {
1285 if (vr_id == SAI_NULL_OBJECT_ID)
1286 {
1287 continue;
1288 }
1289 if (op == SET_COMMAND && !add_route(vr_id, pfx, nh_id))
1290 {
1291 SWSS_LOG_INFO("Route add failed for %s", ipPrefix.to_string().c_str());
1292 break;
1293 }
1294 else if (op == DEL_COMMAND && !del_route(vr_id, pfx))
1295 {
1296 SWSS_LOG_INFO("Route del failed for %s", ipPrefix.to_string().c_str());
1297 break;
1298 }
1299 }
1300
1301 if (op == SET_COMMAND)
1302 {
1303 vrf_obj->addRoute(ipPrefix, nh);
1304 }
1305 else
1306 {
1307 vrf_obj->removeRoute(ipPrefix);
1308 }
1309

1313bool VNetRouteOrch::handleRoutes(const Request& request)
1314{
1315 SWSS_LOG_ENTER();
1316
1317 IpAddresses ip_addresses;
1318 string ifname = "";
1319
1320 for (const auto& name: request.getAttrFieldNames())
1321 {
1322 if (name == "ifname")
1323 {
1324 ifname = request.getAttrString(name);
1325 }
1326 else if (name == "nexthop")
1327 {
1328 auto ipstr = request.getAttrString(name);
1329 ip_addresses = IpAddresses(ipstr);
1330 }
1331 else
1332 {
1333 SWSS_LOG_INFO("Unknown attribute: %s", name.c_str());
1334 continue;
1335 }
1336 }
1337
1338 const std::string& vnet_name = request.getKeyString(0);
1339 auto ip_pfx = request.getKeyIpPrefix(1);
1340 auto op = request.getOperation();
1341 nextHop nh = { ip_addresses, ifname };
1342
1343 SWSS_LOG_INFO("VNET-RT '%s' op '%s' for ip %s", vnet_name.c_str(),
1344 op.c_str(), ip_pfx.to_string().c_str());
1345
1346 if (op == SET_COMMAND)
1347 {
1348 addRoute(vnet_name, ip_pfx, nh);
1349 }
1350 else
1351 {
1352 delRoute(ip_pfx);
1353 }
1354
1355 if (vnet_orch_->isVnetExecVrf())
1356 {
1357 return doRouteTask<VNetVrfObject>(vnet_name, ip_pfx, nh, op);
1358 }
1359
1360 return true;
1361}

1363void VNetRouteOrch::attach(Observer* observer, const IpAddress& dstAddr)
1364{
1365 SWSS_LOG_ENTER();
1366
1367 auto insert_result = next_hop_observers_.emplace(dstAddr, VNetNextHopObserverEntry());
1368 auto observerEntry = insert_result.first;
1369 /* Create a new observer entry if no current observer is observing this
1370 * IP address */
1371 if (insert_result.second)
1372 {
1373 /* Find the prefixes that cover the destination IP */
1374 for (auto route : syncd_routes_)
1375 {
1376 if (route.first.isAddressInSubnet(dstAddr))
1377 {
1378 SWSS_LOG_INFO("Prefix %s covers destination address",
1379 route.first.to_string().c_str());
1380
1381 observerEntry->second.routeTable.emplace(
1382 route.first,
1383 route.second
1384 );
1385 }
1386 }
1387 }
1388
1389 observerEntry->second.observers.push_back(observer);
1390
1391 auto bestRoute = observerEntry->second.routeTable.rbegin();
1392 if (bestRoute != observerEntry->second.routeTable.rend())
1393 {
1394 SWSS_LOG_NOTICE("Attached next hop observer of route %s for destination IP %s",
1395 bestRoute->first.to_string().c_str(),
1396 dstAddr.to_string().c_str());
1397 for (auto vnetEntry : bestRoute->second)
1398 {
1399 VNetNextHopUpdate update =
1400 {
1401 SET_COMMAND,
1402 vnetEntry.first, // vnet name
1403 dstAddr, // destination
1404 bestRoute->first, // prefix
1405 vnetEntry.second // nexthop
1406 };
1407 observer->update(SUBJECT_TYPE_NEXTHOP_CHANGE, reinterpret_cast<void*>(&update));
1408 }
1409 }

1412void VNetRouteOrch::detach(Observer* observer, const IpAddress& dstAddr)
1413{
1414 SWSS_LOG_ENTER();
1415 auto observerEntry = next_hop_observers_.find(dstAddr);
1416
1417 if (observerEntry == next_hop_observers_.end())
1418 {
1419 SWSS_LOG_ERROR("Failed to detach observer for %s. Entry not found.", dstAddr.to_string().c_str());
1420 assert(false);
1421 return;
1422 }
1423
1424 auto iter = std::find(
1425 observerEntry->second.observers.begin(),
1426 observerEntry->second.observers.end(),
1427 observer);
1428 if (iter == observerEntry->second.observers.end())
1429 {
1430 SWSS_LOG_ERROR("Failed to detach observer for %s. Observer not found.", dstAddr.to_string().c_str());
1431 assert(false);
1432 return;
1433 }
1434
1435 auto bestRoute = observerEntry->second.routeTable.rbegin();
1436 if (bestRoute != observerEntry->second.routeTable.rend())
1437 {
1438 for (auto vnetEntry : bestRoute->second)
1439 {
1440 VNetNextHopUpdate update =
1441 {
1442 DEL_COMMAND,
1443 vnetEntry.first, // vnet name
1444 dstAddr, // destination
1445 bestRoute->first, // prefix
1446 vnetEntry.second // nexthop
1447 };
1448 observer->update(SUBJECT_TYPE_NEXTHOP_CHANGE, reinterpret_cast<void*>(&update));
1449 }
1450 }
1451 next_hop_observers_.erase(observerEntry);
1452}

1454void VNetRouteOrch::addRoute(const std::string& vnet, const IpPrefix& ipPrefix, const nextHop& nh)
1455{
1456 SWSS_LOG_ENTER();
1457 for (auto& next_hop_observer : next_hop_observers_)
1458 {
1459 if (ipPrefix.isAddressInSubnet(next_hop_observer.first))
1460 {
1461 auto route_insert_result = next_hop_observer.second.routeTable.emplace(ipPrefix, VNetEntry());
1462
1463 auto vnet_result_result = route_insert_result.first->second.emplace(vnet, nh);
1464 if (!vnet_result_result.second)
1465 {
1466 if (vnet_result_result.first->second.ips == nh.ips
1467 && vnet_result_result.first->second.ifname == nh.ifname)
1468 {
1469 continue;
1470 }
1471 vnet_result_result.first->second = nh;
1472 }
1473
1474 // If the inserted route is the best route. (Table should not be empty. Because we inserted a new entry above)
1475 if (route_insert_result.first == --next_hop_observer.second.routeTable.end())
1476 {
1477 VNetNextHopUpdate update =
1478 {
1479 SET_COMMAND,
1480 vnet, // vnet name
1481 next_hop_observer.first, // destination
1482 ipPrefix, // prefix
1483 nh // nexthop
1484 };
1485 for (auto& observer : next_hop_observer.second.observers)
1486 {
1487 observer->update(SUBJECT_TYPE_NEXTHOP_CHANGE, reinterpret_cast<void*>(&update));
1488 }
1489 }
1490 }
1491 }
1492 syncd_routes_.emplace(ipPrefix, VNetEntry()).first->second[vnet] = nh;
1493}

1495void VNetRouteOrch::delRoute(const IpPrefix& ipPrefix)
1496{
1497 SWSS_LOG_ENTER();
1498
1499 auto route_itr = syncd_routes_.find(ipPrefix);
1500 if (route_itr == syncd_routes_.end())
1501 {
1502 SWSS_LOG_ERROR("Failed to find route %s.", ipPrefix.to_string().c_str());
1503 assert(false);
1504 return;
1505 }
1506 auto next_hop_observer = next_hop_observers_.begin();
1507 while(next_hop_observer != next_hop_observers_.end())
1508 {
1509 if (ipPrefix.isAddressInSubnet(next_hop_observer->first))
1510 {
1511 auto itr = next_hop_observer->second.routeTable.find(ipPrefix);
1512 if ( itr == next_hop_observer->second.routeTable.end())
1513 {
1514 SWSS_LOG_ERROR(
1515 "Failed to find any ip(%s) belong to this route(%s).",
1516 next_hop_observer->first.to_string().c_str(),
1517 ipPrefix.to_string().c_str());
1518 assert(false);
1519 continue;
1520 }
1521 if (itr->second.empty())
1522 {
1523 continue;
1524 }
1525 for (auto& observer : next_hop_observer->second.observers)
1526 {
1527 VNetNextHopUpdate update = {
1528 DEL_COMMAND,
1529 itr->second.rbegin()->first, // vnet name
1530 next_hop_observer->first, // destination
1531 itr->first, // prefix
1532 itr->second.rbegin()->second // nexthop
1533 };
1534 observer->update(SUBJECT_TYPE_NEXTHOP_CHANGE, reinterpret_cast<void*>(&update));
1535 }
1536 next_hop_observer->second.routeTable.erase(itr);
1537 if (next_hop_observer->second.routeTable.empty())
1538 {
1539 next_hop_observer = next_hop_observers_.erase(next_hop_observer);
1540 continue;
1541 }
1542 }
1543 next_hop_observer++;
1544 }
1545 syncd_routes_.erase(route_itr);
1546}

1724void VNetRouteOrch::updateVnetTunnel(const BfdUpdate& update)
1725{
1726 SWSS_LOG_ENTER();
1727
1728 auto key = update.peer;
1729 sai_bfd_session_state_t state = update.state;
1730
1731 size_t found_vrf = key.find(state_db_key_delimiter);
1732 if (found_vrf == string::npos)
1733 {
1734 SWSS_LOG_ERROR("Failed to parse key %s, no vrf is given", key.c_str());
1735 return;
1736 }
1737
1738 size_t found_ifname = key.find(state_db_key_delimiter, found_vrf + 1);
1739 if (found_ifname == string::npos)
1740 {
1741 SWSS_LOG_ERROR("Failed to parse key %s, no ifname is given", key.c_str());
1742 return;
1743 }
1744
1745 string vrf_name = key.substr(0, found_vrf);
1746 string alias = key.substr(found_vrf + 1, found_ifname - found_vrf - 1);
1747 IpAddress peer_address(key.substr(found_ifname + 1));
1748
1749 if (alias != "default" || vrf_name != "default")
1750 {
1751 return;
1752 }
1753
1754 auto it_peer = bfd_sessions_.find(peer_address);
1755
1756 if (it_peer == bfd_sessions_.end()) {
1757 SWSS_LOG_INFO("No endpoint for BFD peer %s", peer_address.to_string().c_str());
1758 return;
1759 }
1760
1761 BfdSessionInfo& bfd_info = it_peer->second;
1762 bfd_info.bfd_state = state;
1763
1764 string vnet = bfd_info.vnet;
1765 NextHopKey endpoint = bfd_info.endpoint;
1766 auto *vrf_obj = vnet_orch_->getTypePtr<VNetVrfObject>(vnet);
1767
1768 if (syncd_nexthop_groups_.find(vnet) == syncd_nexthop_groups_.end())
1769 {
1770 SWSS_LOG_ERROR("Vnet %s not found", vnet.c_str());
1771 return;
1772 }
1773
1774 nexthop_info_[vnet][endpoint.ip_address].bfd_state = state;
1775
1776 for (auto& nhg_info_pair : syncd_nexthop_groups_[vnet])
1777 {
1778 NextHopGroupKey nexthops = nhg_info_pair.first;
1779 NextHopGroupInfo& nhg_info = nhg_info_pair.second;
1780
1781 std::set<NextHopKey> next_hop_set = nexthops.getNextHops();
1782 uint32_t seq_id = 0;
1783 uint32_t nh_seq_id = 0;
1784 for (auto nh: next_hop_set)
1785 {
1786 seq_id++;
1787 if (nh == endpoint)
1788 {
1789 nh_seq_id = seq_id;
1790 break;
1791 }
1792 }
1793
1794 if (!nh_seq_id)
1795 {
1796 continue;
1797 }
1798
1799 if (state == SAI_BFD_SESSION_STATE_UP)
1800 {
1801 sai_object_id_t next_hop_group_member_id = SAI_NULL_OBJECT_ID;
1802 if (nexthops.getSize() > 1)
1803 {
1804 // Create a next hop group member
1805 vector<sai_attribute_t> nhgm_attrs;
1806
1807 sai_attribute_t nhgm_attr;
1808 nhgm_attr.id = SAI_NEXT_HOP_GROUP_MEMBER_ATTR_NEXT_HOP_GROUP_ID;
1809 nhgm_attr.value.oid = nhg_info.next_hop_group_id;
1810 nhgm_attrs.push_back(nhgm_attr);
1811
1812 nhgm_attr.id = SAI_NEXT_HOP_GROUP_MEMBER_ATTR_NEXT_HOP_ID;
1813 nhgm_attr.value.oid = vrf_obj->getTunnelNextHop(endpoint);
1814 nhgm_attrs.push_back(nhgm_attr);
1815
1816 if (gSwitchOrch->checkOrderedEcmpEnable())
1817 {
1818 nhgm_attr.id = SAI_NEXT_HOP_GROUP_MEMBER_ATTR_SEQUENCE_ID;
1819 nhgm_attr.value.u32 = nh_seq_id;
1820 nhgm_attrs.push_back(nhgm_attr);
1821 }
1822
1823 sai_status_t status = sai_next_hop_group_api->create_next_hop_group_member(&next_hop_group_member_id,
1824 gSwitchId,
1825 (uint32_t)nhgm_attrs.size(),
1826 nhgm_attrs.data());
1827
1828 if (status != SAI_STATUS_SUCCESS)
1829 {
1830 SWSS_LOG_ERROR("Failed to add next hop member to group %" PRIx64 ": %d\n",
1831 nhg_info.next_hop_group_id, status);
1832 task_process_status handle_status = handleSaiCreateStatus(SAI_API_NEXT_HOP_GROUP, status);
1833 if (handle_status != task_success)
1834 {
1835 continue;
1836 }
1837 }
1838
1839 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP_MEMBER);
1840 }
1841
1842 // Re-create routes when it was temporarily removed
1843 if (nhg_info.active_members.empty())
1844 {
1845 nhg_info.active_members[endpoint] = next_hop_group_member_id;
1846 if (vnet_orch_->isVnetExecVrf())
1847 {
1848 for (auto ip_pfx : syncd_nexthop_groups_[vnet][nexthops].tunnel_routes)
1849 {
1850 string op = SET_COMMAND;
1851 updateTunnelRoute(vnet, ip_pfx, nexthops, op);
1852 }
1853 }
1854 }
1855 else
1856 {
1857 nhg_info.active_members[endpoint] = next_hop_group_member_id;
1858 }
1859 }
1860 else
1861 {
1862 if (nexthops.getSize() > 1 && nhg_info.active_members.find(endpoint) != nhg_info.active_members.end())
1863 {
1864 sai_object_id_t nexthop_id = nhg_info.active_members[endpoint];
1865 sai_status_t status = sai_next_hop_group_api->remove_next_hop_group_member(nexthop_id);
1866
1867 if (status != SAI_STATUS_SUCCESS)
1868 {
1869 SWSS_LOG_ERROR("Failed to remove next hop member %" PRIx64 " from group %" PRIx64 ": %d\n",
1870 nexthop_id, nhg_info.next_hop_group_id, status);
1871 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_NEXT_HOP_GROUP, status);
1872 if (handle_status != task_success)
1873 {
1874 continue;
1875 }
1876 }
1877
1878 vrf_obj->removeTunnelNextHop(endpoint);
1879
1880 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP_MEMBER);
1881 }
1882
1883 if (nhg_info.active_members.find(endpoint) != nhg_info.active_members.end())
1884 {
1885 nhg_info.active_members.erase(endpoint);
1886
1887 // Remove routes when nexthop group has no active endpoint
1888 if (nhg_info.active_members.empty())
1889 {
1890 if (vnet_orch_->isVnetExecVrf())
1891 {
1892 for (auto ip_pfx : syncd_nexthop_groups_[vnet][nexthops].tunnel_routes)
1893 {
1894 string op = DEL_COMMAND;
1895 updateTunnelRoute(vnet, ip_pfx, nexthops, op);
1896 }
1897 }
1898 }
1899 }
1900 }
1901
1902 // Post configured in State DB
1903 for (auto ip_pfx : syncd_nexthop_groups_[vnet][nexthops].tunnel_routes)
1904 {
1905 string profile = vrf_obj->getProfile(ip_pfx);
1906 postRouteState(vnet, ip_pfx, nexthops, profile);
1907 }
1908 }
1909}

1911bool VNetRouteOrch::handleTunnel(const Request& request)
1912{
1913 SWSS_LOG_ENTER();
1914
1915 vector<IpAddress> ip_list;
1916 vector<string> mac_list;
1917 vector<string> vni_list;
1918 vector<IpAddress> monitor_list;
1919 string profile = "";
1920
1921 for (const auto& name: request.getAttrFieldNames())
1922 {
1923 if (name == "endpoint")
1924 {
1925 ip_list = request.getAttrIPList(name);
1926 }
1927 else if (name == "vni")
1928 {
1929 string vni_str = request.getAttrString(name);
1930 vni_list = tokenize(vni_str, ',');
1931 }
1932 else if (name == "mac_address")
1933 {
1934 string mac_str = request.getAttrString(name);
1935 mac_list = tokenize(mac_str, ',');
1936 }
1937 else if (name == "endpoint_monitor")
1938 {
1939 monitor_list = request.getAttrIPList(name);
1940 }
1941 else if (name == "profile")
1942 {
1943 profile = request.getAttrString(name);
1944 }
1945 else
1946 {
1947 SWSS_LOG_INFO("Unknown attribute: %s", name.c_str());
1948 continue;
1949 }
1950 }
1951
1952 if (vni_list.size() > 1 && vni_list.size() != ip_list.size())
1953 {
1954 SWSS_LOG_ERROR("VNI size of %zu does not match endpoint size of %zu", vni_list.size(), ip_list.size());
1955 return false;
1956 }
1957
1958 if (!mac_list.empty() && mac_list.size() != ip_list.size())
1959 {
1960 SWSS_LOG_ERROR("MAC address size of %zu does not match endpoint size of %zu", mac_list.size(), ip_list.size());
1961 return false;
1962 }
1963
1964 if (!monitor_list.empty() && monitor_list.size() != ip_list.size())
1965 {
1966 SWSS_LOG_ERROR("Peer monitor size of %zu does not match endpoint size of %zu", monitor_list.size(), ip_list.size());
1967 return false;
1968 }
1969
1970 const std::string& vnet_name = request.getKeyString(0);
1971 auto ip_pfx = request.getKeyIpPrefix(1);
1972 auto op = request.getOperation();
1973
1974 SWSS_LOG_INFO("VNET-RT '%s' op '%s' for pfx %s", vnet_name.c_str(),
1975 op.c_str(), ip_pfx.to_string().c_str());
1976
1977 NextHopGroupKey nhg("", true);
1978 map<NextHopKey, IpAddress> monitors;
1979 for (size_t idx_ip = 0; idx_ip < ip_list.size(); idx_ip++)
1980 {
1981 IpAddress ip = ip_list[idx_ip];
1982 MacAddress mac;
1983 uint32_t vni = 0;
1984 if (vni_list.size() == 1 && vni_list[0] != "")
1985 {
1986 vni = (uint32_t)stoul(vni_list[0]);
1987 }
1988 else if (vni_list.size() > 1 && vni_list[idx_ip] != "")
1989 {
1990 vni = (uint32_t)stoul(vni_list[idx_ip]);
1991 }
1992
1993 if (!mac_list.empty() && mac_list[idx_ip] != "")
1994 {
1995 mac = MacAddress(mac_list[idx_ip]);
1996 }
1997
