229std::unordered_map<sai_uint32_t, sai_object_id_t> SaiSwitch::saiGetHardwareLaneMap() const
230{
231 SWSS_LOG_ENTER();
232
233 std::unordered_map<sai_uint32_t, sai_object_id_t> map;
234
235 const std::vector<sai_object_id_t> portList = saiGetPortList();
236
237 /*
238 * NOTE: Currently we don't support port breakout this will need to be
239 * addressed in future.
240 */
241
242 for (const auto &port_rid: portList)
243 {
244 sai_uint32_t lanes[MAX_LANES_PER_PORT];
245
246 memset(lanes, 0, sizeof(lanes));
247
248 sai_attribute_t attr;
249
250 attr.id = SAI_PORT_ATTR_HW_LANE_LIST;
251 attr.value.u32list.count = MAX_LANES_PER_PORT;
252 attr.value.u32list.list = lanes;
253
254 sai_status_t status = m_vendorSai->get(SAI_OBJECT_TYPE_PORT, port_rid, 1, &attr);
255
256 if (status != SAI_STATUS_SUCCESS)
257 {
258 SWSS_LOG_THROW("failed to get hardware lane list port RID %s: %s",
259 sai_serialize_object_id(port_rid).c_str(),
260 sai_serialize_status(status).c_str());
261 }
262
263 sai_int32_t laneCount = attr.value.u32list.count;
264
265 for (int j = 0; j < laneCount; j++)
266 {
267 map[lanes[j]] = port_rid;
268 }
269 }
270
271 return map;

288void SaiSwitch::helperCheckLaneMap()
289{
290 SWSS_LOG_ENTER();
291
292 auto redisLaneMap = m_client->getLaneMap(m_switch_vid);
293
294 auto laneMap = saiGetHardwareLaneMap();
295
296 if (redisLaneMap.size() == 0)
297 {
298 SWSS_LOG_INFO("no lanes defined in redis, seems like it is first syncd start");
299
300 m_client->saveLaneMap(m_switch_vid, laneMap);
301
302 redisLaneMap = laneMap; // copy
303 }
304
305 if (laneMap.size() != redisLaneMap.size())
306 {
307 SWSS_LOG_THROW("lanes map size differ: %lu vs %lu", laneMap.size(), redisLaneMap.size());
308 }
309
310 for (auto kv: laneMap)
311 {
312 sai_uint32_t lane = kv.first;
313 sai_object_id_t portId = kv.second;
314
315 if (redisLaneMap.find(lane) == redisLaneMap.end())
316 {
317 SWSS_LOG_THROW("lane %u not found in redis", lane);
318 }
319
320 if (redisLaneMap[lane] != portId)
321 {
322 // if this happens, we need to remap VIDTORID and RIDTOVID
323 SWSS_LOG_THROW("FIXME: lane port id differs: %s vs %s, port ids must be remapped",
324 sai_serialize_object_id(portId).c_str(),
325 sai_serialize_object_id(redisLaneMap[lane]).c_str());
326 }
327 }

442sai_object_id_t SaiSwitch::helperGetSwitchAttrOid(
443 _In_ sai_attr_id_t attr_id)
444{
445 SWSS_LOG_ENTER();
446
447 sai_attribute_t attr;
448
449 /*
450 * Get original value from the ASIC.
451 */
452
453 auto meta = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_SWITCH, attr_id);
454
455 if (meta == NULL)
456 {
457 SWSS_LOG_THROW("can't get switch attribute %d metadata", attr_id);
458 }
459
460 if (meta->attrvaluetype != SAI_ATTR_VALUE_TYPE_OBJECT_ID)
461 {
462 SWSS_LOG_THROW("atribute %s is not OID attribute", meta->attridname);
463 }
464
465 attr.id = attr_id;
466
467 sai_status_t status = m_vendorSai->get(SAI_OBJECT_TYPE_SWITCH, m_switch_rid, 1, &attr);
468
469 if (status != SAI_STATUS_SUCCESS)
470 {
471 SWSS_LOG_WARN("failed to get %s: %s",
472 meta->attridname,
473 sai_serialize_status(status).c_str());
474
475 return SAI_NULL_OBJECT_ID;
476 }
477
478 SWSS_LOG_INFO("%s RID %s",
479 meta->attridname,
480 sai_serialize_object_id(attr.value.oid).c_str());
481
482 sai_object_id_t rid = attr.value.oid;
483
484 sai_object_id_t redis_rid = SAI_NULL_OBJECT_ID;
485
486 if (rid == SAI_NULL_OBJECT_ID)
487 {
488 return rid;
489 }
490
491 /*
492 * Get value value of the same attribute from redis.
493 */
494
495 auto ptr_redis_rid_str = m_client->getSwitchHiddenAttribute(m_switch_vid, meta->attridname);
496
497 if (ptr_redis_rid_str == NULL)
498 {
499 /*
500 * Redis value of this attribute is not present yet, save it!
501 */
502
503 redisSaveInternalOids(rid);
504
505 SWSS_LOG_INFO("redis %s id is not defined yet in redis", meta->attridname);
506
507 m_client->saveSwitchHiddenAttribute(m_switch_vid, meta->attridname, rid);
508
509 m_default_rid_map[attr_id] = rid;
510
511 return rid;
512 }
513
514 sai_deserialize_object_id(*ptr_redis_rid_str, redis_rid);
515
516 if (rid != redis_rid)
517 {
518 /*
519 * In that case we need to remap VIDTORID and RIDTOVID. This is
520 * required since if previous value will be used in redis maps, and it
521 * will be invalid when that value will be used to call SAI api.
522 */
523
524 SWSS_LOG_THROW("FIXME: %s RID differs: %s (asic) vs %s (redis), ids must be remapped v2r/r2v",
525 meta->attridname,
526 sai_serialize_object_id(rid).c_str(),
527 sai_serialize_object_id(redis_rid).c_str());
528 }
529
530 m_default_rid_map[attr_id] = rid;
531
532 return rid;
533}

568bool SaiSwitch::isNonRemovableRid(
569 _In_ sai_object_id_t rid) const
570{
571 SWSS_LOG_ENTER();
572
573 if (rid == SAI_NULL_OBJECT_ID)
574 {
575 SWSS_LOG_THROW("NULL rid passed");
576 }
577
578 /*
579 * Check for SAI_SWITCH_ATTR_DEFAULT_* oids like cpu, default virtual
580 * router. Those objects can't be removed if user ask for it.
581 */
582
583 /* Here we are checking for isSwitchObjectDefaultRid first then ColdBootDiscoveredRid
584 * as it is possible we can discover switch Internal OID as part of warm-boot also especially
585 * when we are doing SAI upgrade as part of warm-boot.*/
586
587 if (isSwitchObjectDefaultRid(rid))
588 {
589 return true;
590 }
591
592 if (!isColdBootDiscoveredRid(rid))
593 {
594 /*
595 * This object was not discovered on cold boot so it can be removed.
596 */
597
598 return false;
599 }
600
601 sai_object_type_t ot = m_vendorSai->objectTypeQuery(rid);
602
603 /*
604 * List of objects after init (mlnx 2700):
605 *
606 * PORT: 33 // prevent
607 * VIRTUAL_ROUTER: 1 // default
608 * STP: 1 // default
609 * HOSTIF_TRAP_GROUP: 1 // default
610 * QUEUE: 528 // prevent
611 * SCHEDULER_GROUP: 512 // prevent
612 * INGRESS_PRIORITY_GROUP: 256 // prevent
613 * HASH: 2 // prevent
614 * SWITCH: 1 // prevent
615 * VLAN: 1 // default
616 * VLAN_MEMBER: 32 // can be removed
617 * STP_PORT: 32 // can be removed (cpu don't belong to stp)
618 * BRIDGE: 1 // default
619 * BRIDGE_PORT: 33 // can be removed but cpu bridge port can't
620 */
621
622 switch (ot)
623 {
624 case SAI_OBJECT_TYPE_VLAN_MEMBER:
625 case SAI_OBJECT_TYPE_STP_PORT:
626 case SAI_OBJECT_TYPE_BRIDGE_PORT:
627
628 /*
629 * Those objects were discovered during cold boot, but they can
630 * still be removed since switch allows that.
631 */
632
633 return false;
634
635 case SAI_OBJECT_TYPE_PORT:
636 case SAI_OBJECT_TYPE_QUEUE:
637 case SAI_OBJECT_TYPE_INGRESS_PRIORITY_GROUP:
638 case SAI_OBJECT_TYPE_SCHEDULER_GROUP:
639 case SAI_OBJECT_TYPE_HASH:
640
641 /*
642 * TODO: Some vendors support removing of those objects then we
643 * need to came up with different approach. Probably SaiSwitch
644 * will need to decide whether it's possible to remove object.
645 */
646
647 return true;
648
649 default:
650 break;
651 }
652
653 SWSS_LOG_WARN("can't determine wheter object %s RID %s can be removed, FIXME",
654 sai_serialize_object_type(ot).c_str(),
655 sai_serialize_object_id(rid).c_str());
656
657 return true;
658}

759void SaiSwitch::helperSaveDiscoveredObjectsToRedis()
760{
761 SWSS_LOG_ENTER();
762
763 SWSS_LOG_TIMER("save discovered objects to redis");
764
765 /*
766 * There is a problem:
767 *
768 * After switch creation, on the switch objects are created internally like
769 * VLAN members, queues, SGs etc. Some of those objects are removable.
770 * User can decide that he don't want VLAN members and he will remove them.
771 * Those objects will be removed from ASIC view in redis as well.
772 *
773 * Now after hard reinit, syncd will pick up what is in the db and it will
774 * try to recreate ASIC state. First it will create switch, and this
775 * switch will create those VLAN members again inside ASIC and it will try
776 * to put them back to the DB, since we need to keep track of all default
777 * objects.
778 *
779 * We need a way to decide whether we need to put those objects to DB or
780 * not. Since we are performing syncd hard reinit and recreating switch
781 * that there was something in the DB already. And basing on that we can
782 * deduce that we don't need to put again all our discovered objects to the
783 * DB, since some of those objects could be removed at the beginning.
784 *
785 * Hard reinit is performed before taking any action from the redis queue.
786 * But when user will decide to create switch, table consumer will put that
787 * switch to the DB right away before calling SaiSwitch constructor. But
788 * that switch will be the only object in the ASIC table, so out simple
789 * deduction here could be checking if there is more than object in the db.
790 * If there are at least 2, then we don't need to save discovered oids.
791 *
792 * We will use at least number 32, since that should be at least number of
793 * ports that previously discovered. We could also query for PORTs. We can
794 * also deduce this using HIDDEN key objects, if they are defined then that
795 * would mean we already put objects to db at the first place.
796 *
797 * PS. This is not the best way to solve this problem, but works.
798 *
799 * TODO: Some of those objects could be removed, like vlan members etc, we
800 * could actually put those objects back, but only those objects which we
801 * would consider non removable, and this is hard to determine now. A
802 * method getNonRemovableObjects would be nice, then we could put those
803 * objects to the view every time, and not put only discovered objects that
804 * reflect removable objects like vlan member.
805 *
806 * This must be per switch.
807 */
808
809 const int ObjectsTreshold = 32;
810
811 size_t keys = m_client->getAsicObjectsSize(m_switch_vid);
812
813 bool manyObjectsPresent = keys > ObjectsTreshold;
814
815 SWSS_LOG_NOTICE("objects in ASIC state table present: %zu", keys);
816
817 if (manyObjectsPresent)
818 {
819 SWSS_LOG_NOTICE("will NOT put discovered objects into db");
820
821 // TODO or just check here and remove directly form ASIC_VIEW ?
822 return;
823 }
824
825 SWSS_LOG_NOTICE("putting ALL discovered objects to redis");
826
827 for (sai_object_id_t rid: m_discovered_rids)
828 {
829 /*
830 * We also could thing of optimizing this since it's one call to redis
831 * per rid, and probably this should be ATOMIC.
832 *
833 * NOTE: We are also storing read only object's here, like default
834 * virtual router, CPU, default trap group, etc.
835 */
836
837 redisSetDummyAsicStateForRealObjectId(rid);
838 }
839
840 /*
841 * If we are here, this is probably COLD boot, since any previous boot
842 * would put lots of objects into redis DB (ports, queues, scheduler_groups
843 * etc), and since this is cold boot, we can put those discovered objects
844 * to cold boot objects map to redis DB. This will become handy when doing
845 * warm boot and figuring out which object is default created and which is
846 * user created, since after warm boot user could previously assign buffer
847 * profile on ingress priority group and this buffer profile will be
848 * discovered by sai discovery logic.
849 *
850 * Question is here whether we should put VID here or RID. And after cold
851 * boot when hard reinit logic happens, we need to remap them, also note
852 * that some object could be removed like VLAN members and they will not
853 * have existing corresponding OID.
854 */
855
856 redisSaveColdBootDiscoveredVids();
857}

1003void SaiSwitch::collectPortRelatedObjects(
1004 _In_ sai_object_id_t portRid)
1005{
1006 SWSS_LOG_ENTER();
1007
1008 std::set<sai_object_id_t> related;
1009
1010 sai_attr_id_t attrs[] = {
1011 SAI_PORT_ATTR_QOS_QUEUE_LIST,
1012 SAI_PORT_ATTR_QOS_SCHEDULER_GROUP_LIST,
1013 SAI_PORT_ATTR_INGRESS_PRIORITY_GROUP_LIST
1014 };
1015
1016 for (size_t i = 0; i < sizeof(attrs)/sizeof(sai_attr_id_t); i++)
1017 {
1018 std::vector<sai_object_id_t> objlist;
1019
1020 objlist.resize(MAX_OBJLIST_LEN);
1021
1022 sai_attribute_t attr;
1023
1024 attr.id = attrs[i];
1025
1026 attr.value.objlist.count = MAX_OBJLIST_LEN;
1027 attr.value.objlist.list = objlist.data();
1028
1029 // since we have those objects already discovered we don't need to
1030 // query SAI to get related objects, but lets do that since user could
1031 // add/remove some of related objects (this should also be tracked in
1032 // SaiSwitch class internally
1033
1034 auto status = m_vendorSai->get(SAI_OBJECT_TYPE_PORT, portRid, 1, &attr);
1035
1036 if (status != SAI_STATUS_SUCCESS)
1037 {
1038 SWSS_LOG_THROW("failed to obtain related obejcts for port rid %s: %s, attr id: %d",
1039 sai_serialize_object_id(portRid).c_str(),
1040 sai_serialize_status(status).c_str(),
1041 attr.id);
1042 }
1043
1044 objlist.resize(attr.value.objlist.count);
1045
1046 related.insert(objlist.begin(), objlist.end());
1047 }
1048
1049 // treat port serdes as related object
1050
1051 sai_attribute_t attr;
1052
1053 attr.id = SAI_PORT_ATTR_PORT_SERDES_ID;
1054
1055 auto status = m_vendorSai->get(SAI_OBJECT_TYPE_PORT, portRid, 1, &attr);
1056
1057 if (status == SAI_STATUS_SUCCESS && attr.value.oid != SAI_NULL_OBJECT_ID)
1058 {
1059 // some platforms not support PORT SERDES, so get it only on success
1060
1061 related.insert(attr.value.oid);
1062 }
1063
1064 SWSS_LOG_NOTICE("obtained %zu port %s related RIDs",
1065 related.size(),
1066 sai_serialize_object_id(portRid).c_str());
1067
1068 m_portRelatedObjects[portRid] = related;
1069}

1071void SaiSwitch::postPortRemove(
1072 _In_ sai_object_id_t portRid)
1073{
1074 SWSS_LOG_ENTER();
1075
1076 if (m_portRelatedObjects.find(portRid) == m_portRelatedObjects.end())
1077 {
1078 SWSS_LOG_THROW("no port related objects populated for port RID %s",
1079 sai_serialize_object_id(portRid).c_str());
1080 }
1081
1082 /*
1083 * Port was successfully removed from vendor SAI,
1084 * we need to remove queues, ipgs and sg from:
1085 *
1086 * - redis ASIC DB
1087 * - discovered existing objects in saiswitch class
1088 * - local vid2rid map
1089 * - redis RIDTOVID map
1090 *
1091 * - also remove LANES mapping
1092 */
1093
1094 for (auto rid: m_portRelatedObjects.at(portRid))
1095 {
1096 // remove from existing objects
1097
1098 if (isDiscoveredRid(rid))
1099 {
1100 removeExistingObjectReference(rid);
1101 }
1102
1103 // remove from RID2VID and VID2RID map in redis
1104
1105 auto vid = m_client->getVidForRid(rid);
1106
1107 if (vid == SAI_NULL_OBJECT_ID)
1108 {
1109 SWSS_LOG_THROW("expected rid %s to be present in RIDTOVID",
1110 sai_serialize_object_id(rid).c_str());
1111 }
1112
1113 // TODO should this remove rid,vid and object be as db op?
1114
1115 m_translator->eraseRidAndVid(rid, vid);
1116
1117 // remove from ASIC DB
1118
1119 m_client->removeAsicObject(vid);
1120 }
1121
1122 int removed = m_client->removePortFromLanesMap(m_switch_vid, portRid);
1123
1124 SWSS_LOG_NOTICE("removed %u lanes from redis lane map for port RID %s",
1125 removed,
1126 sai_serialize_object_id(portRid).c_str());
1127
1128 if (removed == 0)
1129 {
1130 SWSS_LOG_THROW("NO LANES found in redis lane map for given port RID %s",
1131 sai_serialize_object_id(portRid).c_str());
1132 }
1133
1134 SWSS_LOG_NOTICE("post port remove actions succeeded");
1135}

1137void SaiSwitch::checkWarmBootDiscoveredRids()

1177

1187 // entries for them

1189 redisSetDummyAsicStateForRealObjectId(rid);
1190 }
1191 }
1192
1193 if (m_warmBootNewDiscoveredVids.size())
1194 {
1195 SWSS_LOG_NOTICE("discovered %zu new RIDs on WARM BOOT, new firmware? or bug", m_warmBootNewDiscoveredVids.size());
1196 }

1198 SWSS_LOG_NOTICE("all discovered RIDs are present in current RID2VID map for switch VID %s",
1199 sai_serialize_object_id(m_switch_vid).c_str());
