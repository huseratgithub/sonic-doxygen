78bool TeamdCtlMgr::try_add_lag(const std::string & lag_name)
79{
80 if (m_lags_to_add.find(lag_name) == m_lags_to_add.end())
81 {
82 m_lags_to_add[lag_name] = 0;
83 }
84
85 int attempt = m_lags_to_add[lag_name];
86
87 auto tdc = teamdctl_alloc();
88 if (!tdc)
89 {
90 SWSS_LOG_ERROR("Can't allocate memory for teamdctl handler. LAG='%s'. attempt=%d", lag_name.c_str(), attempt);
91 m_lags_to_add[lag_name]++;
92 return false;
93 }
94
95 teamdctl_set_log_fn(tdc, &teamdctl_log_function);
96
97 int err = teamdctl_connect(tdc, lag_name.c_str(), nullptr, "usock");
98 if (err)
99 {
100 if (attempt != 0)
101 {
102 SWSS_LOG_WARN("Can't connect to teamd LAG='%s', error='%s'. attempt=%d", lag_name.c_str(), strerror(-err), attempt);
103 }
104 teamdctl_free(tdc);
105 m_lags_to_add[lag_name]++;
106 return false;
107 }
108
109 m_handlers.emplace(lag_name, tdc);
110 m_lags_to_add.erase(lag_name);
111 SWSS_LOG_NOTICE("The LAG '%s' has been added.", lag_name.c_str());
112
113 return true;
114}

122bool TeamdCtlMgr::remove_lag(const std::string & lag_name)
123{

182TeamdCtlDump TeamdCtlMgr::get_dump(const std::string & lag_name, bool to_retry)
183{
184 TeamdCtlDump res = { false, "" };
185 if (has_key(lag_name))
186 {
187 auto tdc = m_handlers[lag_name];
188 char * dump;
189 int r = teamdctl_state_get_raw_direct(tdc, &dump);
190 if (r == 0)
191 {
192 res = { true, std::string(dump) };
193
194 // If this lag interface errored last time, remove the entry
195 if (m_lags_err_retry.find(lag_name) != m_lags_err_retry.end())
196 {
197 SWSS_LOG_NOTICE("The LAG '%s' had errored in get_dump earlier, removing it", lag_name.c_str());
198 m_lags_err_retry.erase(lag_name);
199 }
200 }
201 else
202 {
203 // In case of failure and retry flag is set, check if it fails for MAX_RETRY times.
204 if (to_retry)
205 {
206 if (m_lags_err_retry.find(lag_name) != m_lags_err_retry.end())
207 {
208 if (m_lags_err_retry[lag_name] == MAX_RETRY)
209 {
210 SWSS_LOG_ERROR("Can't get dump for LAG '%s'. Skipping", lag_name.c_str());
211 m_lags_err_retry.erase(lag_name);
212 }
213 else
214 m_lags_err_retry[lag_name]++;
215 }
216 else
217 {
218
219 // This time a different lag interface errored out.
220 m_lags_err_retry[lag_name] = 1;
221 }
222 }
223 else
224 {
225 // No need to retry if the flag is not set.
226 SWSS_LOG_ERROR("Can't get dump for LAG '%s'. Skipping", lag_name.c_str());
227 }
228 }
229 }
230 else
231 {
232 SWSS_LOG_ERROR("Can't update state. LAG not found. LAG='%s'", lag_name.c_str());
233 }
234
