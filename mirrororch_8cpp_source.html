376task_process_status MirrorOrch::createEntry(const string& key, const vector<FieldValueTuple>& data)
377{
378 SWSS_LOG_ENTER();
379
380 auto session = m_syncdMirrors.find(key);
381 if (session != m_syncdMirrors.end())
382 {
383 SWSS_LOG_NOTICE("Failed to create session %s: object already exists", key.c_str());
384 return task_process_status::task_duplicated;
385 }
386
387 string platform = getenv("platform") ? getenv("platform") : "";
388 MirrorEntry entry(platform);
389
390 for (auto i : data)
391 {
392 try {
393 if (fvField(i) == MIRROR_SESSION_SRC_IP)
394 {
395 entry.srcIp = fvValue(i);
396 if (!entry.srcIp.isV4())
397 {
398 SWSS_LOG_ERROR("Unsupported version of sessions %s source IP address", key.c_str());
399 return task_process_status::task_invalid_entry;
400 }
401 }
402 else if (fvField(i) == MIRROR_SESSION_DST_IP)
403 {
404 entry.dstIp = fvValue(i);
405 if (!entry.dstIp.isV4())
406 {
407 SWSS_LOG_ERROR("Unsupported version of sessions %s destination IP address", key.c_str());
408 return task_process_status::task_invalid_entry;
409 }
410 }
411 else if (fvField(i) == MIRROR_SESSION_GRE_TYPE)
412 {
413 entry.greType = to_uint<uint16_t>(fvValue(i));
414 }
415 else if (fvField(i) == MIRROR_SESSION_DSCP)
416 {
417 entry.dscp = to_uint<uint8_t>(fvValue(i), MIRROR_SESSION_DSCP_MIN, MIRROR_SESSION_DSCP_MAX);
418 }
419 else if (fvField(i) == MIRROR_SESSION_TTL)
420 {
421 entry.ttl = to_uint<uint8_t>(fvValue(i));
422 }
423 else if (fvField(i) == MIRROR_SESSION_QUEUE)
424 {
425 entry.queue = to_uint<uint8_t>(fvValue(i));
426 if (entry.queue >= m_maxNumTC)
427 {
428 SWSS_LOG_ERROR("Failed to get valid queue %s", fvValue(i).c_str());
429 return task_process_status::task_invalid_entry;
430 }
431 }
432 else if (fvField(i) == MIRROR_SESSION_POLICER)
433 {
434 if (!m_policerOrch->policerExists(fvValue(i)))
435 {
436 SWSS_LOG_ERROR("Failed to get policer %s",
437 fvValue(i).c_str());
438 return task_process_status::task_need_retry;
439 }
440
441 m_policerOrch->increaseRefCount(fvValue(i));
442 entry.policer = fvValue(i);
443 }
444 else if (fvField(i) == MIRROR_SESSION_SRC_PORT)
445 {
446 if (!validateSrcPortList(fvValue(i)))
447 {
448 SWSS_LOG_ERROR("Failed to get valid source port list %s", fvValue(i).c_str());
449 return task_process_status::task_invalid_entry;
450 }
451 entry.src_port = fvValue(i);
452 }
453 else if (fvField(i) == MIRROR_SESSION_DST_PORT)
454 {
455 if (!validateDstPort(fvValue(i)))
456 {
457 SWSS_LOG_ERROR("Failed to get valid destination port %s", fvValue(i).c_str());
458 return task_process_status::task_invalid_entry;
459 }
460 entry.dst_port = fvValue(i);
461 }
462 else if (fvField(i) == MIRROR_SESSION_DIRECTION)
463 {
464 if (!(fvValue(i) == MIRROR_RX_DIRECTION || fvValue(i) == MIRROR_TX_DIRECTION
465 || fvValue(i) == MIRROR_BOTH_DIRECTION))
466 {
467 SWSS_LOG_ERROR("Failed to get valid direction %s", fvValue(i).c_str());
468 return task_process_status::task_invalid_entry;
469 }
470 entry.direction = fvValue(i);
471 }
472 else if (fvField(i) == MIRROR_SESSION_TYPE)
473 {
474 entry.type = fvValue(i);
475 }
476 else
477 {
478 SWSS_LOG_ERROR("Failed to parse session %s configuration. Unknown attribute %s", key.c_str(), fvField(i).c_str());
479 return task_process_status::task_invalid_entry;
480 }
481 }
482 catch (const exception& e)
483 {
484 SWSS_LOG_ERROR("Failed to parse session %s attribute %s error: %s.", key.c_str(), fvField(i).c_str(), e.what());
485 return task_process_status::task_invalid_entry;
486 }
487 catch (...)
488 {
489 SWSS_LOG_ERROR("Failed to parse session %s attribute %s. Unknown error has been occurred", key.c_str(), fvField(i).c_str());
490 return task_process_status::task_failed;
491 }
492 }
493
494 if (!isHwResourcesAvailable())
495 {
496 SWSS_LOG_ERROR("Failed to create session %s: HW resources are not available", key.c_str());
497 return task_process_status::task_failed;
498 }
499
500 m_syncdMirrors.emplace(key, entry);
501 setSessionState(key, entry);
502
503 if (entry.type == MIRROR_SESSION_SPAN && !entry.dst_port.empty())
504 {
505 auto &session1 = m_syncdMirrors.find(key)->second;
506 activateSession(key, session1);
507 }
508 else
509 {
510 // Attach the destination IP to the routeOrch
511 m_routeOrch->attach(this, entry.dstIp);
512 }
513
514 SWSS_LOG_NOTICE("Created mirror session %s", key.c_str());
515
516 return task_process_status::task_success;
517}

519task_process_status MirrorOrch::deleteEntry(const string& name)
520{
521 SWSS_LOG_ENTER();
522
523 auto sessionIter = m_syncdMirrors.find(name);
524 if (sessionIter == m_syncdMirrors.end())
525 {
526 SWSS_LOG_ERROR("Failed to remove non-existent mirror session %s",
527 name.c_str());
528 return task_process_status::task_invalid_entry;
529 }
530
531 auto& session = sessionIter->second;
532
533 if (session.refCount)
534 {
535 SWSS_LOG_WARN("Failed to remove still referenced mirror session %s, retry...",
536 name.c_str());
537 return task_process_status::task_need_retry;
538 }
539
540 if (session.status)
541 {
542 if (!deactivateSession(name, session))
543 {
544 SWSS_LOG_ERROR("Failed to remove mirror session %s", name.c_str());
545 return task_process_status::task_failed;
546 }
547 }
548
549 if (session.type != MIRROR_SESSION_SPAN)
550 {
551 m_routeOrch->detach(this, session.dstIp);
552 }
553
554 if (!session.policer.empty())
555 {
556 m_policerOrch->decreaseRefCount(session.policer);
557 }
558
559 removeSessionState(name);
560
561 m_syncdMirrors.erase(sessionIter);
562
563 SWSS_LOG_NOTICE("Removed mirror session %s", name.c_str());
564
565 return task_process_status::task_success;

641bool MirrorOrch::getNeighborInfo(const string& name, MirrorEntry& session)
642{
643 SWSS_LOG_ENTER();
644
645 // 1) If session destination IP is directly connected, and the neighbor
646 // information is retrieved successfully, then continue.
647 // 2) If session has next hop, and the next hop's neighbor information is
648 // retrieved successfully, then continue.
649 // 3) Otherwise, return false.
650 if (!m_neighOrch->getNeighborEntry(session.dstIp,
651 session.neighborInfo.neighbor, session.neighborInfo.mac) &&
652 (session.nexthopInfo.nexthop.ip_address.isZero() ||
653 !m_neighOrch->getNeighborEntry(session.nexthopInfo.nexthop,
654 session.neighborInfo.neighbor, session.neighborInfo.mac)))
655 {
656 return false;
657 }
658
659 SWSS_LOG_NOTICE("Mirror session %s neighbor is %s",
660 name.c_str(), session.neighborInfo.neighbor.alias.c_str());
661
662 // Get mirror session monitor port information
663 m_portsOrch->getPort(session.neighborInfo.neighbor.alias,
664 session.neighborInfo.port);
665
666 switch (session.neighborInfo.port.m_type)
667 {
668 case Port::PHY:
669 {
670 session.neighborInfo.portId = session.neighborInfo.port.m_port_id;
671 return true;
672 }
673 case Port::LAG:
674 {
675 if (session.neighborInfo.port.m_members.empty())
676 {
677 return false;
678 }
679
680 // Recover the LAG member monitor port picked before warm reboot
681 // to minimalize the data plane changes across warm reboot.
682 if (m_recoverySessionMap.find(name) != m_recoverySessionMap.end())
683 {
684 string alias = tokenize(m_recoverySessionMap[name],
685 state_db_key_delimiter, 1)[0];
686 Port member;
687 m_portsOrch->getPort(alias, member);
688
689 SWSS_LOG_NOTICE("Recover mirror session %s with LAG member port %s",
690 name.c_str(), alias.c_str());
691 session.neighborInfo.portId = member.m_port_id;
692 }
693 else
694 {
695 // Get the first member of the LAG
696 Port member;
697 string first_member_alias = *session.neighborInfo.port.m_members.begin();
698 m_portsOrch->getPort(first_member_alias, member);
699
700 session.neighborInfo.portId = member.m_port_id;
701 }
702
703 return true;
704 }
705 case Port::VLAN:
706 {
707 SWSS_LOG_NOTICE("Get mirror session destination IP neighbor VLAN %d",
708 session.neighborInfo.port.m_vlan_info.vlan_id);
709
710 // Recover the VLAN member monitor port picked before warm reboot
711 // since the FDB entries are not yet learned on the hardware
712 if (m_recoverySessionMap.find(name) != m_recoverySessionMap.end())
713 {
714 string alias = tokenize(m_recoverySessionMap[name],
715 state_db_key_delimiter, 1)[0];
716 Port member;
717 m_portsOrch->getPort(alias, member);
718
719 SWSS_LOG_NOTICE("Recover mirror session %s with VLAN member port %s",
720 name.c_str(), alias.c_str());
721 session.neighborInfo.portId = member.m_port_id;
722 }
723 else
724 {
725 Port member;
726 if (!m_fdbOrch->getPort(session.neighborInfo.mac,
727 session.neighborInfo.port.m_vlan_info.vlan_id, member))
728 {
729 SWSS_LOG_NOTICE("Waiting to get FDB entry MAC %s under VLAN %s",
730 session.neighborInfo.mac.to_string().c_str(),
731 session.neighborInfo.port.m_alias.c_str());
732 return false;
733 }
734 else
735 {
736 // Update monitor port
737 session.neighborInfo.portId = member.m_port_id;
738 }
739 }
740
741 return true;
742 }
743 case Port::SYSTEM:
744 {
745 return true;
746 }
747 default:
748 {
749 return false;
750 }
751 }

754bool MirrorOrch::updateSession(const string& name, MirrorEntry& session)
755{
756 SWSS_LOG_ENTER();
757
758 bool ret = true;
759 MirrorEntry old_session(session);
760
761 // Get neighbor information
762 if (getNeighborInfo(name, session))
763 {
764 // Update corresponding attributes
765 if (session.status)
766 {
767 if (old_session.neighborInfo.port.m_type !=
768 session.neighborInfo.port.m_type &&
769 (old_session.neighborInfo.port.m_type == Port::VLAN ||
770 session.neighborInfo.port.m_type == Port::VLAN))
771 {
772 ret &= updateSessionType(name, session);
773 }
774
775 if (old_session.neighborInfo.mac !=
776 session.neighborInfo.mac)
777 {
778 ret &= updateSessionDstMac(name, session);
779 }
780
781 if (old_session.neighborInfo.portId !=
782 session.neighborInfo.portId)
783 {
784 ret &= updateSessionDstPort(name, session);
785 }
786 }
787 // Activate mirror session
788 else
789 {
790 ret &= activateSession(name, session);
791 }
792 }
793 // Deactivate mirror session and wait for update
794 else
795 {
796 if (session.status)
797 {
798 ret &= deactivateSession(name, session);
799 }
800 }
801
802 return ret;

903bool MirrorOrch::activateSession(const string& name, MirrorEntry& session)
904{
905 SWSS_LOG_ENTER();
906
907 sai_status_t status;
908 sai_attribute_t attr;
909 vector<sai_attribute_t> attrs;
910
911 assert(!session.status);
912
913 // Some platforms don't support SAI_MIRROR_SESSION_ATTR_TC and only
914 // support global mirror session traffic class.
915 if (session.queue != 0)
916 {
917 attr.id = SAI_MIRROR_SESSION_ATTR_TC;
918 attr.value.u8 = session.queue;
919 attrs.push_back(attr);
920 }
921
922 if (session.type == MIRROR_SESSION_SPAN)
923 {
924 Port dst_port;
925 if (!m_portsOrch->getPort(session.dst_port, dst_port))
926 {
927 SWSS_LOG_ERROR("Failed to locate Port/LAG %s", session.dst_port.c_str());
928 return false;
929 }
930
931 attr.id = SAI_MIRROR_SESSION_ATTR_MONITOR_PORT;
932 attr.value.oid = dst_port.m_port_id;
933 attrs.push_back(attr);
934
935 attr.id = SAI_MIRROR_SESSION_ATTR_TYPE;
936 attr.value.oid = SAI_MIRROR_SESSION_TYPE_LOCAL;
937 attrs.push_back(attr);
938 }
939 else
940 {
941 attr.id = SAI_MIRROR_SESSION_ATTR_MONITOR_PORT;
942 // Set monitor port to recirc port in voq switch.
943 if (gMySwitchType == "voq")
944 {
945 Port recirc_port;
946 if (!m_portsOrch->getRecircPort(recirc_port, "Rec"))
947 {
948 SWSS_LOG_ERROR("Failed to get recirc port");
949 return false;
950 }
951 attr.value.oid = recirc_port.m_port_id;
952 }
953 else
954 {
955 attr.value.oid = session.neighborInfo.portId;
956 }
957 attrs.push_back(attr);
958
959 attr.id = SAI_MIRROR_SESSION_ATTR_TYPE;
960 attr.value.s32 = SAI_MIRROR_SESSION_TYPE_ENHANCED_REMOTE;
961 attrs.push_back(attr);
962
963 // Add the VLAN header when the packet is sent out from a VLAN
964 if (session.neighborInfo.port.m_type == Port::VLAN)
965 {
966 attr.id = SAI_MIRROR_SESSION_ATTR_VLAN_HEADER_VALID;
967 attr.value.booldata = true;
968 attrs.push_back(attr);
969
970 attr.id = SAI_MIRROR_SESSION_ATTR_VLAN_TPID;
971 attr.value.u16 = ETH_P_8021Q;
972 attrs.push_back(attr);
973
974 attr.id = SAI_MIRROR_SESSION_ATTR_VLAN_ID;
975 attr.value.u16 = session.neighborInfo.port.m_vlan_info.vlan_id;
976 attrs.push_back(attr);
977
978 attr.id = SAI_MIRROR_SESSION_ATTR_VLAN_PRI;
979 attr.value.u8 = MIRROR_SESSION_DEFAULT_VLAN_PRI;
980 attrs.push_back(attr);
981
982 attr.id = SAI_MIRROR_SESSION_ATTR_VLAN_CFI;
983 attr.value.u8 = MIRROR_SESSION_DEFAULT_VLAN_CFI;
984 attrs.push_back(attr);
985 }
986
987 attr.id = SAI_MIRROR_SESSION_ATTR_ERSPAN_ENCAPSULATION_TYPE;
988 attr.value.s32 = SAI_ERSPAN_ENCAPSULATION_TYPE_MIRROR_L3_GRE_TUNNEL;
989 attrs.push_back(attr);
990
991 attr.id = SAI_MIRROR_SESSION_ATTR_IPHDR_VERSION;
992 attr.value.u8 = MIRROR_SESSION_DEFAULT_IP_HDR_VER;
993 attrs.push_back(attr);
994
995 // TOS value format is the following:
996 // DSCP 6 bits | ECN 2 bits
997 attr.id = SAI_MIRROR_SESSION_ATTR_TOS;
998 attr.value.u16 = (uint16_t)(session.dscp << MIRROR_SESSION_DSCP_SHIFT);
999 attrs.push_back(attr);
1000
1001 attr.id = SAI_MIRROR_SESSION_ATTR_TTL;
1002 attr.value.u8 = session.ttl;
1003 attrs.push_back(attr);
1004
1005 attr.id = SAI_MIRROR_SESSION_ATTR_SRC_IP_ADDRESS;
1006 copy(attr.value.ipaddr, session.srcIp);
1007 attrs.push_back(attr);
1008
1009 attr.id = SAI_MIRROR_SESSION_ATTR_DST_IP_ADDRESS;
1010 copy(attr.value.ipaddr, session.dstIp);
1011 attrs.push_back(attr);
1012
1013 attr.id = SAI_MIRROR_SESSION_ATTR_SRC_MAC_ADDRESS;
1014 memcpy(attr.value.mac, gMacAddress.getMac(), sizeof(sai_mac_t));
1015 attrs.push_back(attr);
1016
1017 attr.id = SAI_MIRROR_SESSION_ATTR_DST_MAC_ADDRESS;
1018 // Use router mac as mirror dst mac in voq switch.
1019 if ((gMySwitchType == "voq") && (session.type == MIRROR_SESSION_ERSPAN))
1020 {
1021 memcpy(attr.value.mac, gMacAddress.getMac(), sizeof(sai_mac_t));
1022 }
1023 else
1024 {
1025 memcpy(attr.value.mac, session.neighborInfo.mac.getMac(), sizeof(sai_mac_t));
1026 }
1027 attrs.push_back(attr);
1028
1029 attr.id = SAI_MIRROR_SESSION_ATTR_GRE_PROTOCOL_TYPE;
1030 attr.value.u16 = session.greType;
1031 attrs.push_back(attr);
1032 }
1033
1034 if (!session.policer.empty())
1035 {
1036 sai_object_id_t oid = SAI_NULL_OBJECT_ID;
1037 if (!m_policerOrch->getPolicerOid(session.policer, oid))
1038 {
1039 SWSS_LOG_ERROR("Failed to get policer %s", session.policer.c_str());
1040 return false;
1041 }
1042
1043 attr.id = SAI_MIRROR_SESSION_ATTR_POLICER;
1044 attr.value.oid = oid;
1045 attrs.push_back(attr);
1046 }
1047
1048 status = sai_mirror_api->
1049 create_mirror_session(&session.sessionId, gSwitchId, (uint32_t)attrs.size(), attrs.data());
1050 if (status != SAI_STATUS_SUCCESS)
1051 {
1052 SWSS_LOG_ERROR("Failed to activate mirroring session %s", name.c_str());
1053 session.status = false;
1054
1055 task_process_status handle_status = handleSaiCreateStatus(SAI_API_MIRROR, status);
1056 if (handle_status != task_success)
1057 {
1058 return parseHandleSaiStatusFailure(handle_status);
1059 }
1060 }
1061
1062 session.status = true;
1063
1064 if (!session.src_port.empty() && !session.direction.empty())
1065 {
1066 status = configurePortMirrorSession(name, session, true);
1067 if (status == false)
1068 {
1069 SWSS_LOG_ERROR("Failed to activate port mirror session %s", name.c_str());
1070 session.status = false;
1071 return false;
1072 }
1073 }
1074
1075 setSessionState(name, session);
1076
1077 MirrorSessionUpdate update = { name, true };
1078 notify(SUBJECT_TYPE_MIRROR_SESSION_CHANGE, static_cast<void *>(&update));
1079
1080 SWSS_LOG_NOTICE("Activated mirror session %s", name.c_str());
1081
1082 return true;
1083}

1272// The function is called when SUBJECT_TYPE_NEXTHOP_CHANGE is received
1273// This function will handle the case when the session's destination IP's
1274// next hop changes.
1275void MirrorOrch::updateNextHop(const NextHopUpdate& update)
1276{
1277 SWSS_LOG_ENTER();
1278
1279 for (auto it = m_syncdMirrors.begin(); it != m_syncdMirrors.end(); it++)
1280 {
1281 const auto& name = it->first;
1282 auto& session = it->second;
1283
1284 // Check if mirror session's destination IP is the update's destination IP
1285 if (session.dstIp != update.destination)
1286 {
1287 continue;
1288 }
1289
1290 session.nexthopInfo.prefix = update.prefix;
1291
1292 setSessionState(name, session, MIRROR_SESSION_ROUTE_PREFIX);
1293
1294 // This is the ECMP scenario that the new next hop group contains the previous
1295 // next hop. There is no need to update this session's monitor port.
1296 if (update.nexthopGroup != NextHopGroupKey() &&
1297 update.nexthopGroup.getNextHops().count(session.nexthopInfo.nexthop))
1298
1299 {
1300 continue;
1301 }
1302
1303 SWSS_LOG_NOTICE("Updating mirror session %s with route %s",
1304 name.c_str(), update.prefix.to_string().c_str());
1305
1306 if (update.nexthopGroup != NextHopGroupKey())
1307 {
1308 SWSS_LOG_NOTICE(" next hop IPs: %s", update.nexthopGroup.to_string().c_str());
1309
1310 // Recover the session based on the state database information
1311 if (m_recoverySessionMap.find(name) != m_recoverySessionMap.end())
1312 {
1313 NextHopKey nexthop = NextHopKey(tokenize(m_recoverySessionMap[name],
1314 state_db_key_delimiter, 1)[1]);
1315
1316 // Check if recovered next hop IP is within the update's next hop IPs
1317 if (update.nexthopGroup.getNextHops().count(nexthop))
1318 {
1319 SWSS_LOG_NOTICE("Recover mirror session %s with next hop %s",
1320 name.c_str(), nexthop.to_string().c_str());
1321 session.nexthopInfo.nexthop = nexthop;
1322 }
1323 else
1324 {
1325 // Correct the next hop IP
1326 SWSS_LOG_NOTICE("Correct mirror session %s next hop from %s to %s",
1327 name.c_str(), session.nexthopInfo.nexthop.to_string().c_str(),
1328 nexthop.to_string().c_str());
1329 session.nexthopInfo.nexthop = *update.nexthopGroup.getNextHops().begin();
1330 }
1331 }
1332 else
1333 {
1334 // Pick the first one from the next hop group
1335 session.nexthopInfo.nexthop = *update.nexthopGroup.getNextHops().begin();
1336 }
1337 }
1338 else
1339 {
1340 string alias = "";
1341 session.nexthopInfo.nexthop = NextHopKey("0.0.0.0", alias);
1342 }
1343
1344 // Update State DB Nexthop
1345 setSessionState(name, session, MIRROR_SESSION_NEXT_HOP_IP);
1346
1347 SWSS_LOG_NOTICE("Updated mirror session state db %s nexthop to %s",
1348 name.c_str(), session.nexthopInfo.nexthop.to_string().c_str());
1349
1350
1351 // Resolve the neighbor of the new next hop
1352 updateSession(name, session);
1353 }
1354}

1438{

1440
1441 for (auto it = m_syncdMirrors.begin(); it != m_syncdMirrors.end(); it++)
1442 {
1443 const auto& name = it->first;
1444 auto& session = it->second;
1445
1446 // Check the following conditions:
1447 // 1) Session is active
1448 // 2) LAG is part of mirror session source ports.
1449 // 3) Member port is not part of session source ports.
1450 // if the above condition matches then set/unset mirror configuration to new member port.
1451 if (session.status &&
1452 !session.src_port.empty() &&
1453 session.src_port.find(update.lag.m_alias.c_str()) != std::string::npos &&
1454 !checkPortExistsInSrcPortList(update.member.m_alias, session.src_port))

1459 }

1461 {
1462 setUnsetPortMirror(update.member, false, update.add, session.sessionId);
1463 }
1464 }

1467 // 1) the neighbor is LAG
1468 // 2) the neighbor LAG matches the update LAG
1469 if (session.neighborInfo.port.m_type != Port::LAG ||
1470 session.neighborInfo.port != update.lag)
1471 {
1472 continue;
1473 }
1474
1475 if (update.add)
1476 {
1477 // Activate mirror session if it was deactivated due to the reason
1478 // that previously there was no member in the LAG. If the mirror
1479 // session is already activated, no further action is needed.
1480 if (!session.status)
1481 {
1482 assert(!update.lag.m_members.empty());
1483 const string& member_name = *update.lag.m_members.begin();

1485 m_portsOrch->getPort(member_name, member);
1486
1487 session.neighborInfo.portId = member.m_port_id;
1488 activateSession(name, session);
1489 }
1490 }
1491 else
1492 {
1493 // If LAG is empty, deactivate session
1494 if (update.lag.m_members.empty())
1495 {

1497 {
1498 deactivateSession(name, session);
1499 }

1503 else

1508
1509 session.neighborInfo.portId = member.m_port_id;

1511 updateSessionDstPort(name, session);
