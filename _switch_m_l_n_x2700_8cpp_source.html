278sai_status_t SwitchMLNX2700::refresh_bridge_port_list(
279 _In_ const sai_attr_metadata_t *meta,
280 _In_ sai_object_id_t bridge_id)
281{
282 SWSS_LOG_ENTER();
283
284 // XXX possible issues with vxlan and lag.
285
286 auto &all_bridge_ports = m_objectHash.at(SAI_OBJECT_TYPE_BRIDGE_PORT);
287
288 sai_attribute_t attr;
289
290 auto me_port_list = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_BRIDGE, SAI_BRIDGE_ATTR_PORT_LIST);
291 auto m_port_id = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_BRIDGE_PORT, SAI_BRIDGE_PORT_ATTR_PORT_ID);
292 auto m_bridge_id = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_BRIDGE_PORT, SAI_BRIDGE_PORT_ATTR_BRIDGE_ID);
293
294 /*
295 * First get all port's that belong to this bridge id.
296 */
297
298 std::map<sai_object_id_t, SwitchState::AttrHash> bridge_port_list_on_bridge_id;
299
300 for (const auto &bp: all_bridge_ports)
301 {
302 auto it = bp.second.find(m_bridge_id->attridname);
303
304 if (it == bp.second.end())
305 {
306 continue;
307 }
308
309 if (bridge_id == it->second->getAttr()->value.oid)
310 {
311 /*
312 * This bridge port belongs to currently processing bridge ID.
313 */
314
315 sai_object_id_t bridge_port;
316
317 sai_deserialize_object_id(bp.first, bridge_port);
318
319 bridge_port_list_on_bridge_id[bridge_port] = bp.second;
320 }
321 }
322
323 /*
324 * Now sort those bridge port id's by port id to be consistent.
325 */
326
327 std::vector<sai_object_id_t> bridge_port_list;
328
329 for (const auto &p: m_port_list)
330 {
331 for (const auto &bp: bridge_port_list_on_bridge_id)
332 {
333 auto it = bp.second.find(m_port_id->attridname);
334
335 if (it == bp.second.end())
336 {
337 SWSS_LOG_THROW("bridge port is missing %s, not supported yet, FIXME", m_port_id->attridname);
338 }
339
340 if (p == it->second->getAttr()->value.oid)
341 {
342 bridge_port_list.push_back(bp.first);
343 }
344 }
345 }
346
347 if (bridge_port_list_on_bridge_id.size() != bridge_port_list.size())
348 {
349 SWSS_LOG_THROW("filter by port id failed size on lists is different: %zu vs %zu",
350 bridge_port_list_on_bridge_id.size(),
351 bridge_port_list.size());
352 }
353
354 /* default 1q router at the end */
355
356 bridge_port_list.push_back(m_default_bridge_port_1q_router);
357
358 /*
359SAI_BRIDGE_PORT_ATTR_BRIDGE_ID: oid:0x100100000039
360SAI_BRIDGE_PORT_ATTR_FDB_LEARNING_MODE: SAI_BRIDGE_PORT_FDB_LEARNING_MODE_HW
361SAI_BRIDGE_PORT_ATTR_PORT_ID: oid:0x1010000000001
362SAI_BRIDGE_PORT_ATTR_TYPE: SAI_BRIDGE_PORT_TYPE_PORT
363*/
364
365 uint32_t bridge_port_list_count = (uint32_t)bridge_port_list.size();
366
367 SWSS_LOG_NOTICE("recalculated %s: %u", me_port_list->attridname, bridge_port_list_count);
368
369 attr.id = SAI_BRIDGE_ATTR_PORT_LIST;
370 attr.value.objlist.count = bridge_port_list_count;
371 attr.value.objlist.list = bridge_port_list.data();
372
373 return set(SAI_OBJECT_TYPE_BRIDGE, bridge_id, &attr);
374}
