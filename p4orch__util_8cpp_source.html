34std::string verifyAttrs(const std::vector<swss::FieldValueTuple> &targets,
35 const std::vector<swss::FieldValueTuple> &exp, const std::vector<swss::FieldValueTuple> &opt,
36 bool allow_unknown)
37{
38 std::map<std::string, std::string> exp_map;
39 for (const auto &fv : exp)
40 {
41 exp_map[fvField(fv)] = fvValue(fv);
42 }
43 std::map<std::string, std::string> opt_map;
44 for (const auto &fv : opt)
45 {
46 opt_map[fvField(fv)] = fvValue(fv);
47 }
48
49 std::set<std::string> fields;
50 for (const auto &fv : targets)
51 {
52 fields.insert(fvField(fv));
53 bool found = false;
54 if (exp_map.count(fvField(fv)))
55 {
56 found = true;
57 if (fvValue(fv) != exp_map.at(fvField(fv)))
58 {
59 return fvField(fv) + " value mismatch, exp " + exp_map.at(fvField(fv)) + " got " + fvValue(fv);
60 }
61 }
62 if (opt_map.count(fvField(fv)))
63 {
64 found = true;
65 if (fvValue(fv) != opt_map.at(fvField(fv)))
66 {
67 return fvField(fv) + " value mismatch, exp " + opt_map.at(fvField(fv)) + " got " + fvValue(fv);
68 }
69 }
70 if (!found && !allow_unknown)
71 {
72 return std::string("Unexpected field ") + fvField(fv);
73 }
74 }
75 for (const auto &it : exp_map)
76 {
77 if (!fields.count(it.first))
78 {
79 return std::string("Missing field ") + it.first;
80 }
81 }
82 return "";
83}
