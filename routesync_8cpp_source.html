347void RouteSync::onEvpnRouteMsg(struct nlmsghdr *h, int len)
348{
349 struct rtmsg *rtm;
350 struct rtattr *tb[RTA_MAX + 1] = {0};
351 void *dest = NULL;
352 char anyaddr[16] = {0};
353 char dstaddr[16] = {0};
354 int dst_len = 0;
355 char buf[MAX_ADDR_SIZE];
356 char destipprefix[IFNAMSIZ + MAX_ADDR_SIZE + 2] = {0};
357 int nlmsg_type = h->nlmsg_type;
358 unsigned int vrf_index;
359
360 rtm = (struct rtmsg *)NLMSG_DATA(h);
361
362 /* Parse attributes and extract fields of interest. */
363 netlink_parse_rtattr(tb, RTA_MAX, RTM_RTA(rtm), len);
364
365 if (tb[RTA_DST])
366 {
367 dest = RTA_DATA(tb[RTA_DST]);
368 }
369 else
370 {
371 dest = anyaddr;
372 }
373
374 if (rtm->rtm_family == AF_INET)
375 {
376 if (rtm->rtm_dst_len > IPV4_MAX_BITLEN)
377 {
378 return;
379 }
380 memcpy(dstaddr, dest, IPV4_MAX_BYTE);
381 dst_len = rtm->rtm_dst_len;
382 }
383 else if (rtm->rtm_family == AF_INET6)
384 {
385 if (rtm->rtm_dst_len > IPV6_MAX_BITLEN)
386 {
387 return;
388 }
389 memcpy(dstaddr, dest, IPV6_MAX_BYTE);
390 dst_len = rtm->rtm_dst_len;
391 }
392
393 SWSS_LOG_DEBUG("Rx MsgType:%d Family:%d Prefix:%s/%d", h->nlmsg_type, rtm->rtm_family,
394 inet_ntop(rtm->rtm_family, dstaddr, buf, MAX_ADDR_SIZE), dst_len);
395
396 /* Table corresponding to route. */
397 if (tb[RTA_TABLE])
398 {
399 vrf_index = *(int *)RTA_DATA(tb[RTA_TABLE]);
400 }
401 else
402 {
403 vrf_index = rtm->rtm_table;
404 }
405
406 if (vrf_index)
407 {
408 if (!getIfName(vrf_index, destipprefix, IFNAMSIZ))
409 {
410 SWSS_LOG_ERROR("Fail to get the VRF name (ifindex %u)", vrf_index);
411 return;
412 }
413 /*
414 * Now vrf device name is required to start with VRF_PREFIX,
415 * it is difficult to split vrf_name:ipv6_addr.
416 */
417 if (memcmp(destipprefix, VRF_PREFIX, strlen(VRF_PREFIX)))
418 {
419 SWSS_LOG_ERROR("Invalid VRF name %s (ifindex %u)", destipprefix, vrf_index);
420 return;
421 }
422 destipprefix[strlen(destipprefix)] = ':';
423 }
424
425 if((rtm->rtm_family == AF_INET && dst_len == IPV4_MAX_BITLEN)
426 || (rtm->rtm_family == AF_INET6 && dst_len == IPV6_MAX_BITLEN))
427 {
428 snprintf(destipprefix + strlen(destipprefix), sizeof(destipprefix) - strlen(destipprefix), "%s",
429 inet_ntop(rtm->rtm_family, dstaddr, buf, MAX_ADDR_SIZE));
430 }
431 else
432 {
433 snprintf(destipprefix + strlen(destipprefix), sizeof(destipprefix) - strlen(destipprefix), "%s/%u",
434 inet_ntop(rtm->rtm_family, dstaddr, buf, MAX_ADDR_SIZE), dst_len);
435 }
436
437 SWSS_LOG_INFO("Receive route message dest ip prefix: %s Op:%s",
438 destipprefix,
439 nlmsg_type == RTM_NEWROUTE ? "add":"del");
440
441 /*
442 * Upon arrival of a delete msg we could either push the change right away,
443 * or we could opt to defer it if we are going through a warm-reboot cycle.
444 */
445 bool warmRestartInProgress = m_warmStartHelper.inProgress();
446
447 if (nlmsg_type == RTM_DELROUTE)
448 {
449 if (!warmRestartInProgress)
450 {
451 m_routeTable.del(destipprefix);
452 return;
453 }
454 else
455 {
456 SWSS_LOG_INFO("Warm-Restart mode: Receiving delete msg: %s",
457 destipprefix);
458
459 vector<FieldValueTuple> fvVector;
460 const KeyOpFieldsValuesTuple kfv = std::make_tuple(destipprefix,
461 DEL_COMMAND,
462 fvVector);
463 m_warmStartHelper.insertRefreshMap(kfv);
464 return;
465 }
466 }
467 else if (nlmsg_type != RTM_NEWROUTE)
468 {
469 return;
470 }
471
472 switch (rtm->rtm_type)
473 {
474 case RTN_BLACKHOLE:
475 case RTN_UNREACHABLE:
476 case RTN_PROHIBIT:
477 {
478 SWSS_LOG_ERROR("RTN_BLACKHOLE route not expected (%s)", destipprefix);
479 return;
480 }
481 case RTN_UNICAST:
482 break;
483
484 case RTN_MULTICAST:
485 case RTN_BROADCAST:
486 case RTN_LOCAL:
487 SWSS_LOG_NOTICE("BUM routes aren't supported yet (%s)", destipprefix);
488 return;
489
490 default:
491 return;
492 }
493
494 /* Get nexthop lists */
495 string nexthops;
496 string vni_list;
497 string mac_list;
498 string intf_list;
499 bool ret;
500
501 ret = getEvpnNextHop(h, len, tb, nexthops, vni_list, mac_list, intf_list);
502 if (ret == false)
503 {
504 SWSS_LOG_NOTICE("EVPN Route issue with RouteTable msg: %s vtep:%s vni:%s mac:%s intf:%s",
505 destipprefix, nexthops.c_str(), vni_list.c_str(), mac_list.c_str(), intf_list.c_str());
506 return;
507 }
508
509 if (nexthops.empty() || mac_list.empty())
510 {
511 SWSS_LOG_NOTICE("EVPN IP Prefix: %s nexthop or rmac is empty", destipprefix);
512 return;
513 }
514
515 vector<FieldValueTuple> fvVector;
516 FieldValueTuple nh("nexthop", nexthops);
517 FieldValueTuple intf("ifname", intf_list);
518 FieldValueTuple vni("vni_label", vni_list);
519 FieldValueTuple mac("router_mac", mac_list);
520
521 fvVector.push_back(nh);
522 fvVector.push_back(intf);
523 fvVector.push_back(vni);
524 fvVector.push_back(mac);
525
526 if (!warmRestartInProgress)
527 {
528 m_routeTable.set(destipprefix, fvVector);
529 SWSS_LOG_DEBUG("RouteTable set msg: %s vtep:%s vni:%s mac:%s intf:%s",
530 destipprefix, nexthops.c_str(), vni_list.c_str(), mac_list.c_str(), intf_list.c_str());
531 }
532
533 /*
534 * During routing-stack restarting scenarios route-updates will be temporarily
535 * put on hold by warm-reboot logic.
536 */
537 else
538 {
539 SWSS_LOG_INFO("Warm-Restart mode: RouteTable set msg: %s vtep:%s vni:%s mac:%s",
540 destipprefix, nexthops.c_str(), vni_list.c_str(), mac_list.c_str());
541
542 const KeyOpFieldsValuesTuple kfv = std::make_tuple(destipprefix,
543 SET_COMMAND,
544 fvVector);
545 m_warmStartHelper.insertRefreshMap(kfv);
546 }
547 return;
548}

570void RouteSync::onMsg(int nlmsg_type, struct nl_object *obj)
571{
572 struct rtnl_route *route_obj = (struct rtnl_route *)obj;
573
574 /* Supports IPv4 or IPv6 address, otherwise return immediately */
575 auto family = rtnl_route_get_family(route_obj);
576 /* Check for Label route. */
577 if (family == AF_MPLS)
578 {
579 onLabelRouteMsg(nlmsg_type, obj);
580 return;
581 }
582 if (family != AF_INET && family != AF_INET6)
583 {
584 SWSS_LOG_INFO("Unknown route family support (object: %s)", nl_object_get_type(obj));
585 return;
586 }
587
588 /* Get the index of the master device */
589 unsigned int master_index = rtnl_route_get_table(route_obj);
590 char master_name[IFNAMSIZ] = {0};
591
592 /* if the table_id is not set in the route obj then route is for default vrf. */
593 if (master_index)
594 {
595 /* Get the name of the master device */
596 getIfName(master_index, master_name, IFNAMSIZ);
597
598 /* If the master device name starts with VNET_PREFIX, it is a VNET route.
599 The VNET name is exactly the name of the associated master device. */
600 if (string(master_name).find(VNET_PREFIX) == 0)
601 {
602 onVnetRouteMsg(nlmsg_type, obj, string(master_name));
603 }
604 /* Otherwise, it is a regular route (include VRF route). */
605 else
606 {
607 onRouteMsg(nlmsg_type, obj, master_name);
608 }
609 }
610 else
611 {
612 onRouteMsg(nlmsg_type, obj, NULL);
613 }
614}

699 break;

701 case RTN_MULTICAST:
702 case RTN_BROADCAST:
703 case RTN_LOCAL:
704 SWSS_LOG_INFO("BUM routes aren't supported yet (%s)", destipprefix);
705 return;

707 default:
708 return;
709 }
710
711 struct nl_list_head *nhs = rtnl_route_get_nexthops(route_obj);
712 if (!nhs)
713 {
714 SWSS_LOG_INFO("Nexthop list is empty for %s", destipprefix);
715 return;
716 }
717
718 /* Get nexthop lists */
719 string gw_list;
720 string intf_list;
721 string mpls_list;

745 fvVector.push_back(intf);
746 if (!mpls_list.empty())

872/*
873 * Handle vnet route
874 * @arg nlmsg_type Netlink message type
875 * @arg obj Netlink object
876 * @arg vnet Vnet name
877 */
878void RouteSync::onVnetRouteMsg(int nlmsg_type, struct nl_object *obj, string vnet)
879{
880 struct rtnl_route *route_obj = (struct rtnl_route *)obj;
881
882 /* Get the destination IP prefix */
883 struct nl_addr *dip = rtnl_route_get_dst(route_obj);
884 char destipprefix[MAX_ADDR_SIZE + 1] = {0};
885 nl_addr2str(dip, destipprefix, MAX_ADDR_SIZE);
886
887 string vnet_dip = vnet + string(":") + destipprefix;
888 SWSS_LOG_DEBUG("Receive new vnet route message %s", vnet_dip.c_str());
889
890 /* Ignore IPv6 link-local and mc addresses as Vnet routes */
891 auto family = rtnl_route_get_family(route_obj);
892 if (family == AF_INET6 &&
893 (IN6_IS_ADDR_LINKLOCAL(nl_addr_get_binary_addr(dip)) || IN6_IS_ADDR_MULTICAST(nl_addr_get_binary_addr(dip))))
894 {
895 SWSS_LOG_INFO("Ignore linklocal vnet routes %d for %s", nlmsg_type, vnet_dip.c_str());
896 return;
897 }
898
899 if (nlmsg_type == RTM_DELROUTE)
900 {
901 /* Duplicated delete as we do not know if it is a VXLAN tunnel route*/
902 m_vnet_routeTable.del(vnet_dip);
903 m_vnet_tunnelTable.del(vnet_dip);
904 return;
905 }
906 else if (nlmsg_type != RTM_NEWROUTE)
907 {
908 SWSS_LOG_INFO("Unknown message-type: %d for %s", nlmsg_type, vnet_dip.c_str());
909 return;
910 }
911
912 switch (rtnl_route_get_type(route_obj))
913 {
914 case RTN_UNICAST:
915 break;
916
917 /* We may support blackhole in the future */
918 case RTN_BLACKHOLE:
919 SWSS_LOG_INFO("Blackhole route is supported yet (%s)", vnet_dip.c_str());
920 return;
921
922 case RTN_MULTICAST:
923 case RTN_BROADCAST:
924 case RTN_LOCAL:
925 SWSS_LOG_INFO("BUM routes aren't supported yet (%s)", vnet_dip.c_str());
926 return;
927
928 default:
929 return;
930 }
931
932 struct nl_list_head *nhs = rtnl_route_get_nexthops(route_obj);
933 if (!nhs)
934 {
935 SWSS_LOG_INFO("Nexthop list is empty for %s", vnet_dip.c_str());
936 return;
937 }
938
939 /* Get nexthop lists */
940 string nexthops = getNextHopGw(route_obj);
941 string ifnames = getNextHopIf(route_obj);
942
943 /* If the the first interface name starts with VXLAN_IF_NAME_PREFIX,
944 the route is a VXLAN tunnel route. */
945 if (ifnames.find(VXLAN_IF_NAME_PREFIX) == 0)
946 {
947 vector<FieldValueTuple> fvVector;
948 FieldValueTuple ep("endpoint", nexthops);
949 fvVector.push_back(ep);
950
951 m_vnet_tunnelTable.set(vnet_dip, fvVector);
952 SWSS_LOG_DEBUG("%s set msg: %s %s",
953 APP_VNET_RT_TUNNEL_TABLE_NAME, vnet_dip.c_str(), nexthops.c_str());
954 return;
955 }
956 /* Regular VNET route */
957 else
958 {
959 vector<FieldValueTuple> fvVector;
960 FieldValueTuple idx("ifname", ifnames);
961 fvVector.push_back(idx);
962
963 /* If the route has at least one next hop gateway, e.g., nexthops does not only have ',' */
964 if (nexthops.length() + 1 > (unsigned int)rtnl_route_get_nnexthops(route_obj))
965 {
966 FieldValueTuple nh("nexthop", nexthops);
967 fvVector.push_back(nh);
968 SWSS_LOG_DEBUG("%s set msg: %s %s %s",
969 APP_VNET_RT_TABLE_NAME, vnet_dip.c_str(), ifnames.c_str(), nexthops.c_str());
970 }
971 else
972 {
973 SWSS_LOG_DEBUG("%s set msg: %s %s",
974 APP_VNET_RT_TABLE_NAME, vnet_dip.c_str(), ifnames.c_str());
975 }
976
977 m_vnet_routeTable.set(vnet_dip, fvVector);
978 }
979}

984 * @arg if_name String to store interface name
985 * @arg name_len Length of destination string, including terminating zero byte

989bool RouteSync::getIfName(int if_index, char *if_name, size_t name_len)
990{

992 {
993 return false;
994 }
995
996 memset(if_name, 0, name_len);
997
998 /* Cannot get interface name. Possibly the interface gets re-created. */

1001 /* Trying to refill cache */

1003 if (!rtnl_link_i2name(m_link_cache, if_index, if_name, name_len))
1004 {
1005 return false;
1006 }
1007 }
1008
1009 return true;

1012/*
1013 * getNextHopList() - parses next hop list attached to route_obj
1014 * @arg route_obj (input) Netlink route object
1015 * @arg gw_list (output) comma-separated list of NH IP gateways
1016 * @arg mpls_list (output) comma-separated list of NH MPLS info
1017 * @arg intf_list (output) comma-separated list of NH interfaces
1018 *
1019 * Return void
1020 */
1021void RouteSync::getNextHopList(struct rtnl_route *route_obj, string& gw_list,
1022 string& mpls_list, string& intf_list)
1023{
1024 bool mpls_found = false;
1025
1026 for (int i = 0; i < rtnl_route_get_nnexthops(route_obj); i++)
1027 {
1028 struct rtnl_nexthop *nexthop = rtnl_route_nexthop_n(route_obj, i);
1029 struct nl_addr *addr = NULL;
1030
1031 /* RTA_GATEWAY is NH gateway info for IP routes only */
1032 if ((addr = rtnl_route_nh_get_gateway(nexthop)))
1033 {
1034 char gw_ip[MAX_ADDR_SIZE + 1] = {0};
1035 nl_addr2str(addr, gw_ip, MAX_ADDR_SIZE);
1036 gw_list += gw_ip;
1037
1038 /* LWTUNNEL_ENCAP_MPLS RTA_DST is MPLS NH label stack for IP routes only */
1039 if ((addr = rtnl_route_nh_get_encap_mpls_dst(nexthop)))
1040 {
1041 char labelstack[MAX_ADDR_SIZE + 1] = {0};
1042 nl_addr2str(addr, labelstack, MAX_ADDR_SIZE);
1043 mpls_list += string("push");
1044 mpls_list += labelstack;
1045 mpls_found = true;
1046 }
1047 /* Filler for proper parsing in routeorch */
1048 else
1049 {
1050 mpls_list += string("na");
1051 }
1052 }
1053 /* RTA_VIA is NH gateway info for MPLS routes only */
1054 else if ((addr = rtnl_route_nh_get_via(nexthop)))
1055 {
1056 char gw_ip[MAX_ADDR_SIZE + 1] = {0};
1057 nl_addr2str(addr, gw_ip, MAX_ADDR_SIZE);
1058 gw_list += gw_ip;
1059
1060 /* RTA_NEWDST is MPLS NH label stack for MPLS routes only */
1061 if ((addr = rtnl_route_nh_get_newdst(nexthop)))
1062 {
1063 char labelstack[MAX_ADDR_SIZE + 1] = {0};
1064 nl_addr2str(addr, labelstack, MAX_ADDR_SIZE);
1065 mpls_list += string("swap");
1066 mpls_list += labelstack;
1067 mpls_found = true;
1068 }
1069 /* Filler for proper parsing in routeorch */
1070 else
1071 {
1072 mpls_list += string("na");
1073 }
1074 }
1075 else
1076 {
1077 if (rtnl_route_get_family(route_obj) == AF_INET6)
1078 {
1079 gw_list += "::";
1080 }
1081 /* for MPLS route, use IPv4 as default gateway. */
1082 else
1083 {
1084 gw_list += "0.0.0.0";
1085 }
1086 mpls_list += string("na");
1087 }
1088
1089 /* Get the ID of next hop interface */
1090 unsigned if_index = rtnl_route_nh_get_ifindex(nexthop);
1091 char if_name[IFNAMSIZ] = "0";
1092 if (getIfName(if_index, if_name, IFNAMSIZ))
1093 {
1094 intf_list += if_name;
1095 }
1096 /* If we cannot get the interface name */
1097 else
1098 {
1099 intf_list += "unknown";
1100 }
1101
1102 if (i + 1 < rtnl_route_get_nnexthops(route_obj))
1103 {
1104 gw_list += NHG_DELIMITER;
1105 mpls_list += NHG_DELIMITER;
1106 intf_list += NHG_DELIMITER;
1107 }
1108 }
1109
1110 if (!mpls_found)
1111 {
1112 mpls_list.clear();
1113 }
1114}
