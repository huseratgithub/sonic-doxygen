146 {
147 gearboxPhyMap = gearbox.loadPhyMap(tmpGearboxTable);
148 SWSS_LOG_DEBUG("BOX: gearboxPhyMap size = %d.", (int) gearboxPhyMap.size());
149
150 for (auto it = gearboxPhyMap.begin(); it != gearboxPhyMap.end(); ++it)

152 SWSS_LOG_NOTICE("BOX: Initialize PHY %d.", it->first);
153
154 if (initSaiPhyApi(&it->second) != SAI_STATUS_SUCCESS)
155 {
156 SWSS_LOG_ERROR("BOX: Failed to initialize PHY %d.", it->first);
157 }
158 else
159 {
160 SWSS_LOG_NOTICE("BOX: Created new PHY phy_id:%d phy_oid:%s.", it->second.phy_id, it->second.phy_oid.c_str());
161 tmpGearboxTable->hset("phy:"+to_string(it->second.phy_id), "phy_oid", it->second.phy_oid.c_str());
162 tmpGearboxTable->hset("phy:"+to_string(it->second.phy_id), "firmware_major_version", it->second.firmware_major_version.c_str());
163 }
164 }
165 }
166 delete tmpGearboxTable;

187bool getSystemPortConfigList(DBConnector *cfgDb, DBConnector *appDb, vector<sai_system_port_config_t> &sysportcfglist)
188{
189 Table cfgDeviceMetaDataTable(cfgDb, CFG_DEVICE_METADATA_TABLE_NAME);
190 Table cfgSystemPortTable(cfgDb, CFG_SYSTEM_PORT_TABLE_NAME);
191 Table appSystemPortTable(appDb, APP_SYSTEM_PORT_TABLE_NAME);
192
193 if (gMySwitchType != "voq")
194 {
195 //Non VOQ switch. Nothing to read
196 return true;
197 }
198
199 string value;
200 if (!cfgDeviceMetaDataTable.hget("localhost", "switch_id", value))
201 {
202 //VOQ switch id is not configured.
203 SWSS_LOG_ERROR("VOQ switch id is not configured");
204 return false;
205 }
206
207 if (value.size())
208 gVoqMySwitchId = stoi(value);
209
210 if (gVoqMySwitchId < 0)
211 {
212 SWSS_LOG_ERROR("Invalid VOQ switch id %d configured", gVoqMySwitchId);
213 return false;
214 }
215
216 if (!cfgDeviceMetaDataTable.hget("localhost", "max_cores", value))
217 {
218 //VOQ max cores is not configured.
219 SWSS_LOG_ERROR("VOQ max cores is not configured");
220 return false;
221 }
222
223 if (value.size())
224 gVoqMaxCores = stoi(value);
225
226 if (gVoqMaxCores == 0)
227 {
228 SWSS_LOG_ERROR("Invalid VOQ max cores %d configured", gVoqMaxCores);
229 return false;
230 }
231
232 if (!cfgDeviceMetaDataTable.hget("localhost", "hostname", value))
233 {
234 // hostname is not configured.
235 SWSS_LOG_ERROR("Host name is not configured");
236 return false;
237 }
238 gMyHostName = value;
239
240 if (!gMyHostName.size())
241 {
242 SWSS_LOG_ERROR("Invalid host name %s configured", gMyHostName.c_str());
243 return false;
244 }
245
246 if (!cfgDeviceMetaDataTable.hget("localhost", "asic_name", value))
247 {
248 // asic_name is not configured.
249 SWSS_LOG_ERROR("Asic name is not configured");
250 return false;
251 }
252 gMyAsicName = value;
253
254 if (!gMyAsicName.size())
255 {
256 SWSS_LOG_ERROR("Invalid asic name %s configured", gMyAsicName.c_str());
257 return false;
258 }
259
260 vector<string> spKeys;
261 cfgSystemPortTable.getKeys(spKeys);
262
263 //Retrieve system port configurations
264 vector<FieldValueTuple> spFv;
265 sai_system_port_config_t sysport;
266 for (auto &k : spKeys)
267 {
268 cfgSystemPortTable.get(k, spFv);
269
270 for (auto &fv : spFv)
271 {
272 if (fv.first == "switch_id")
273 {
274 sysport.attached_switch_id = stoi(fv.second);
275 continue;
276 }
277 if (fv.first == "core_index")
278 {
279 sysport.attached_core_index = stoi(fv.second);
280 continue;
281 }
282 if (fv.first == "core_port_index")
283 {
284 sysport.attached_core_port_index = stoi(fv.second);
285 continue;
286 }
287 if (fv.first == "speed")
288 {
289 sysport.speed = stoi(fv.second);
290 continue;
291 }
292 if (fv.first == "system_port_id")
293 {
294 sysport.port_id = stoi(fv.second);
295 continue;
296 }
297 if (fv.first == "num_voq")
298 {
299 sysport.num_voq = stoi(fv.second);
300 continue;
301 }
302 }
303 //Add to system port config list
304 sysportcfglist.push_back(sysport);
305
306 //Also push to APP DB
307 appSystemPortTable.set(k, spFv);
308 }
309
310 SWSS_LOG_NOTICE("Created System Port config list for %d system ports", (int32_t) sysportcfglist.size());
311
312 return true;
313}
