75void ZeroMQSelectableChannel::zmqPollThread()
76{
77 SWSS_LOG_ENTER();
78
79 SWSS_LOG_NOTICE("begin");
80
81 while (m_runThread)
82 {
83 zmq_pollitem_t items [1] = { };
84
85 items[0].socket = m_socket;
86 items[0].events = ZMQ_POLLIN;
87
88 m_allowZmqPoll = false;
89
90 int rc = zmq_poll(items, 1, ZMQ_POLL_TIMEOUT);
91
92 if (m_runThread == false)
93 {
94 SWSS_LOG_NOTICE("ending pool thread, since run is false");
95 break;
96 }
97
98 if (rc <= 0 && zmq_errno() == ETERM)
99 {
100 SWSS_LOG_NOTICE("zmq_poll ETERM");
101 break;
102 }
103
104 if (rc == 0)
105 {
106 SWSS_LOG_DEBUG("zmq_poll: no events, continue");
107 continue;
108 }
109
110 // TODO we should have loop here in case we get multiple events since
111 // zmq poll will only signal events once, but in our case we don't
112 // expect multiple events, since we want to send/receive
113
114 int zmq_events = 0;
115 size_t zmq_events_len = sizeof(zmq_events);
116
117 rc = zmq_getsockopt(m_socket, ZMQ_EVENTS, &zmq_events, &zmq_events_len);
118
119 if (rc != 0)
120 {
121 SWSS_LOG_ERROR("zmq_getsockopt FAILED, zmq_errno: %d", zmq_errno());
122 break;
123 }
124
125 if (rc == 0 && zmq_events & ZMQ_POLLIN)
126 {
127 m_selectableEvent.notify(); // will release epoll
128
129 while (m_runThread && !m_allowZmqPoll)
130 {
131 usleep(10); // could be increased or replaced by spin lock
132
133 //SWSS_LOG_NOTICE("m_allowZmqPoll == false");
134 }
135 }
136 else
137 {
138 // should not happen, we only except ZMQ_POLLIN events
139
140 SWSS_LOG_ERROR("unknown condition: rc: %d, zmq_events: %d, bug?", rc, zmq_events);
141 break;
142 }
143 }
144
145 SWSS_LOG_NOTICE("end");
146}

226uint64_t ZeroMQSelectableChannel::readData()
227{
228 SWSS_LOG_ENTER();
