64{
65 auto port_info = m_sflowPortInfoMap.find(port_id);

77 m_sflowRateSampleMap[rate] = new_session;

79 else
80 {
81 new_session = session->second;

83

87 {
88 return false;
89 }
90 }
91 port_info->second.m_sample_id = new_session.m_sample_id;
92
93 m_sflowRateSampleMap[rate].ref_count++;
94 m_sflowRateSampleMap[old_rate].ref_count--;
95 if (m_sflowRateSampleMap[old_rate].ref_count == 0)

98 {
99 SWSS_LOG_ERROR("Failed to clean old session %" PRIx64 " with rate %d",
100 m_sflowRateSampleMap[old_rate].m_sample_id, old_rate);
101 }
102 else
103 {

105 }
106 }
