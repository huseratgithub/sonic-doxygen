74void MclagLink::getBridgePortIdToAttrPortIdMap(std::map<std::string, std:: string> *oid_map)
75{
76 std::string bridge_port_id;
77 size_t pos1 = 0;
78
79 auto keys = p_asic_db->keys("ASIC_STATE:SAI_OBJECT_TYPE_BRIDGE_PORT:*");
80
81 for (auto& key : keys)
82 {
83 pos1 = key.find("oid:", 0);
84 bridge_port_id = key.substr(pos1);
85
86 auto hash = p_asic_db->hgetall(key);
87 auto attr_port_id = hash.find("SAI_BRIDGE_PORT_ATTR_PORT_ID");
88 if (attr_port_id == hash.end())
89 {
90 attr_port_id = hash.find("SAI_BRIDGE_PORT_ATTR_TUNNEL_ID");
91 if (attr_port_id == hash.end())
92 continue;
93 }
94
95 oid_map->insert(pair<string, string>(bridge_port_id, attr_port_id->second));
96 }
97
98 return;
99}

190void MclagLink::setPortIsolate(char *msg)
191{
192 static const unordered_set<string> supported {
193 BRCM_PLATFORM_SUBSTRING,
194 BFN_PLATFORM_SUBSTRING,
195 CTC_PLATFORM_SUBSTRING
196 };
197
198 const char *platform = getenv("platform");
199 if (platform != nullptr && supported.find(string(platform)) != supported.end())
200 {
201 mclag_sub_option_hdr_t *op_hdr = NULL;
202 string isolate_src_port;
203 string isolate_dst_port;
204 char * cur = NULL;
205 vector<FieldValueTuple> fvts;
206
207 cur = msg;
208
209
210 /*get isolate src port infor*/
211 op_hdr = reinterpret_cast<mclag_sub_option_hdr_t *>(static_cast<void *>(cur));
212 cur = cur + MCLAG_SUB_OPTION_HDR_LEN;
213 isolate_src_port.insert(0, (const char*)cur, op_hdr->op_len);
214
215 cur = cur + op_hdr->op_len;
216
217 /*get isolate dst ports infor*/
218 op_hdr = reinterpret_cast<mclag_sub_option_hdr_t *>(static_cast<void *>(cur));
219 cur = cur + MCLAG_SUB_OPTION_HDR_LEN;
220
221
222 if (op_hdr->op_len == 0)
223 {
224 /*Destination port can be empty when all remote MLAG interfaces
225 * goes down down or ICCP session goes down. Do not delete the
226 * isolation group when all remote interfaces are down. Just remove
227 * all destination ports from the group
228 */
229
230 if (is_iccp_up)
231 {
232 fvts.emplace_back("DESCRIPTION", "Isolation group for MCLAG");
233 fvts.emplace_back("TYPE", "bridge-port");
234 fvts.emplace_back("PORTS", isolate_src_port);
235 fvts.emplace_back("MEMBERS", isolate_dst_port);
236 p_iso_grp_tbl->set("MCLAG_ISO_GRP", fvts);
237 SWSS_LOG_NOTICE("Delete all isolation group destination ports");
238 }
239 else
240 {
241 p_iso_grp_tbl->del("MCLAG_ISO_GRP");
242 SWSS_LOG_NOTICE("Isolation group deleted");
243 }
244 }
245 else
246 {
247 string temp;
248
249 isolate_dst_port.insert(0, (const char*)cur, op_hdr->op_len);
250 istringstream dst_ss(isolate_dst_port);
251
252 isolate_dst_port.clear();
253 while (getline(dst_ss, temp, ','))
254 {
255 if (0 == temp.find("Ethernet"))
256 {
257 continue;
258 }
259 if (isolate_dst_port.length())
260 {
261 isolate_dst_port = isolate_dst_port + ',' + temp;
262 }
263 else
264 {
265 isolate_dst_port = temp;
266 }
267 }
268
269 fvts.emplace_back("DESCRIPTION", "Isolation group for MCLAG");
270 fvts.emplace_back("TYPE", "bridge-port");
271 fvts.emplace_back("PORTS", isolate_src_port);
272 fvts.emplace_back("MEMBERS", isolate_dst_port);
273
274 p_iso_grp_tbl->set("MCLAG_ISO_GRP", fvts);
275 SWSS_LOG_NOTICE("Isolation group created with ports %s and members %s",
276 isolate_src_port.c_str(),
277 isolate_dst_port.c_str());
278 }
279 }
280 else
281 {
282 mclag_sub_option_hdr_t *op_hdr = NULL;
283 string isolate_src_port;
284 string isolate_dst_port;
285 char * cur = NULL;
286 string acl_name = "mclag";
287 string acl_rule_name = "mclag:mclag";
288 vector<FieldValueTuple> acl_attrs;
289 vector<FieldValueTuple> acl_rule_attrs;
290 std::string acl_key = std::string("") + APP_ACL_TABLE_TABLE_NAME + ":" + acl_name;
291 std::string acl_rule_key = std::string("") + APP_ACL_RULE_TABLE_NAME + ":" + acl_rule_name;
292 static int acl_table_is_added = 0;
293
294 cur = msg;
295
296 /*get isolate src port infor*/
297 op_hdr = reinterpret_cast<mclag_sub_option_hdr_t *>(static_cast<void *>(cur));
298 cur = cur + MCLAG_SUB_OPTION_HDR_LEN;
299 isolate_src_port.insert(0, (const char*)cur, op_hdr->op_len);
300
301 cur = cur + op_hdr->op_len;
302
303 /*get isolate dst ports infor*/
304 op_hdr = reinterpret_cast<mclag_sub_option_hdr_t *>(static_cast<void *>(cur));
305 cur = cur + MCLAG_SUB_OPTION_HDR_LEN;
306 isolate_dst_port.insert(0, (const char*)cur, op_hdr->op_len);
307
308 if (op_hdr->op_len == 0)
309 {
310 /* If dst port is NULL, delete the acl table 'mclag' */
311 p_acl_table_tbl->del(acl_name);
312 acl_table_is_added = 0;
313 SWSS_LOG_NOTICE("set port isolate, src port: %s, dst port is NULL",
314 isolate_src_port.c_str());
315 return;
316 }
317
318 SWSS_LOG_NOTICE("set port isolate, src port: %s, dst port: %s",
319 isolate_src_port.c_str(), isolate_dst_port.c_str());
320
321 if (acl_table_is_added == 0)
322 {
323 /*First create ACL table*/
324 FieldValueTuple desc_attr("policy_desc", "Mclag egress port isolate acl");
325 acl_attrs.push_back(desc_attr);
326
327 FieldValueTuple type_attr("type", "L3");
328 acl_attrs.push_back(type_attr);
329
330 FieldValueTuple port_attr("ports", isolate_src_port);
331 acl_attrs.push_back(port_attr);
332
333 p_acl_table_tbl->set(acl_name, acl_attrs);
334
335 acl_table_is_added = 1;
336 /*End create ACL table*/
337 }
338
339 /*Then create ACL rule table*/
340 FieldValueTuple ip_type_attr("IP_TYPE", "ANY");
341 acl_rule_attrs.push_back(ip_type_attr);
342
343 string temp;
344 istringstream dst_ss(isolate_dst_port);
345
346 isolate_dst_port.clear();
347 while (getline(dst_ss, temp, ','))
348 {
349 if (0 == temp.find("PortChannel"))
350 {
351 continue;
352 }
353
354 if (isolate_dst_port.length())
355 {
356 isolate_dst_port = isolate_dst_port + ',' + temp;
357 }
358 else
359 {
360 isolate_dst_port = temp;
361 }
362 }
363
364 FieldValueTuple out_port_attr("OUT_PORTS", isolate_dst_port);
365 acl_rule_attrs.push_back(out_port_attr);
366
367 FieldValueTuple packet_attr("PACKET_ACTION", "DROP");
368 acl_rule_attrs.push_back(packet_attr);
369
370 p_acl_rule_tbl->set(acl_rule_name, acl_rule_attrs);
371 /*End create ACL rule table*/
372 }
373
374 return;
375}

462void MclagLink::setFdbEntry(char *msg, int msg_len)
463{
464 struct mclag_fdb_info * fdb_info = NULL;
465 struct mclag_fdb fdb;
466 string fdb_key;
467 char key[64] = { 0 };
468 char *cur = NULL;
469 short count = 0;
470 int index = 0;
471
472 cur = msg;
473 count = (short)(msg_len/sizeof(struct mclag_fdb_info));
474
475 for (index =0; index < count; index ++)
476 {
477 memset(key, 0, 64);
478
479 fdb_info = reinterpret_cast<struct mclag_fdb_info *>(static_cast<void *>(cur + index * sizeof(struct mclag_fdb_info)));
480
481 fdb.mac = MacAddress::to_string(fdb_info->mac);
482 fdb.port_name = fdb_info->port_name;
483 fdb.vid = fdb_info->vid;
484 if (fdb_info->type == MCLAG_FDB_TYPE_STATIC)
485 fdb.type = "static";
486 else if (fdb_info->type == MCLAG_FDB_TYPE_DYNAMIC)
487 fdb.type = "dynamic";
488 else if (fdb_info->type == MCLAG_FDB_TYPE_DYNAMIC_LOCAL)
489 fdb.type = "dynamic_local";
490
491 snprintf(key, 64, "%s%d:%s", "Vlan", fdb_info->vid, fdb.mac.c_str());
492 fdb_key = key;
493
494 SWSS_LOG_DEBUG("Received MAC key: %s, op_type: %d, mac type: %s , port: %s",
495 fdb_key.c_str(), fdb_info->op_type, fdb.type.c_str(), fdb.port_name.c_str());
496
497 if (fdb_info->op_type == MCLAG_FDB_OPER_ADD)
498 {
499 vector<FieldValueTuple> attrs;
500
501 /*set port attr*/
502 FieldValueTuple port_attr("port", fdb.port_name);
503 attrs.push_back(port_attr);
504
505 /*set type attr*/
506 FieldValueTuple type_attr("type", fdb.type);
507 attrs.push_back(type_attr);
508 p_fdb_tbl->set(fdb_key, attrs);
509 SWSS_LOG_NOTICE("add fdb entry into ASIC_DB:key =%s, type =%s", fdb_key.c_str(), fdb.type.c_str());
510 }
511 else if (fdb_info->op_type == MCLAG_FDB_OPER_DEL)
512 {
513 p_fdb_tbl->del(fdb_key);
514 SWSS_LOG_NOTICE("del fdb entry from ASIC_DB:key =%s", fdb_key.c_str());
515 }
516 }
517 return;
518}

520void MclagLink::mclagsyncdSendFdbEntries(std::deque<KeyOpFieldsValuesTuple> &entries)
521{
522 size_t infor_len = sizeof(mclag_msg_hdr_t);
523 struct mclag_fdb_info info;
524 mclag_msg_hdr_t * msg_head = NULL;
525 int count = 0;
526 ssize_t write = 0;
527
528 char *infor_start = m_messageBuffer_send;
529
530 /* Nothing popped */
531 if (entries.empty())
532 {
533 return;
534 }
535
536 for (auto entry: entries)
537 {
538 memset(&info, 0, sizeof(struct mclag_fdb_info));
539 count++;
540 std::string key = kfvKey(entry);
541 std::string op = kfvOp(entry);
542
543 std::size_t delimiter = key.find_first_of(":");
544 auto vlan_name = key.substr(0, delimiter);
545 const auto mac_address_str = key.substr(delimiter+1);
546 uint8_t mac_address[ETHER_ADDR_LEN];
547
548 MacAddress::parseMacString(mac_address_str, mac_address);
549
550 info.vid = (unsigned int) stoi(vlan_name.substr(4));
551 memcpy(info.mac, mac_address , ETHER_ADDR_LEN);
552
553 if (op == "SET")
554 info.op_type = MCLAG_FDB_OPER_ADD;
555 else
556 info.op_type = MCLAG_FDB_OPER_DEL;
557
558 for (auto i : kfvFieldsValues(entry))
559 {
560 if (fvField(i) == "port")
561 {
562 memcpy(info.port_name, fvValue(i).c_str(), fvValue(i).length());
563 }
564 if (fvField(i) == "type")
565 {
566 if (fvValue(i) == "dynamic")
567 info.type = MCLAG_FDB_TYPE_DYNAMIC;
568 else if (fvValue(i) == "static")
569 info.type = MCLAG_FDB_TYPE_STATIC;
570 else
571 SWSS_LOG_ERROR("MCLAGSYNCD STATE FDB updates key=%s, invalid MAC type %s\n", key.c_str(), fvValue(i).c_str());
572 }
573 }
574 SWSS_LOG_NOTICE("MCLAGSYNCD STATE FDB updates key=%s, operation=%s, type: %d, port: %s \n",
575 key.c_str(), op.c_str(), info.type, info.port_name);
576
577 if (MCLAG_MAX_SEND_MSG_LEN - infor_len < sizeof(struct mclag_fdb_info))
578 {
579 msg_head = reinterpret_cast<mclag_msg_hdr_t *>(static_cast<void *>(infor_start));
580 msg_head->version = 1;
581 msg_head->msg_len = (unsigned short)infor_len;
582 msg_head ->msg_type = MCLAG_SYNCD_MSG_TYPE_FDB_OPERATION;
583
584 SWSS_LOG_DEBUG("mclagsycnd buffer full send msg to iccpd, msg_len =%d, msg_type =%d count : %d",
585 msg_head->msg_len, msg_head->msg_type, count);
586 write = ::write(m_connection_socket, infor_start, msg_head->msg_len);
587
588 if (write <= 0)
589 {
590 SWSS_LOG_ERROR("mclagsycnd update FDB to ICCPD Buffer full, write to m_connection_socket failed");
591 }
592
593 infor_len = sizeof(mclag_msg_hdr_t);
594 count = 0;
595 }
596 memcpy((char*)(infor_start + infor_len), (char*)&info, sizeof(struct mclag_fdb_info));
597 infor_len = infor_len + sizeof(struct mclag_fdb_info);
598 }
599
600
601 if (infor_len <= sizeof(mclag_msg_hdr_t)) /*no fdb entry need notifying iccpd*/
602 return;
603
604 msg_head = reinterpret_cast<mclag_msg_hdr_t *>(static_cast<void *>(infor_start));
605
606 msg_head->version = 1;
607 msg_head->msg_len = (unsigned short)infor_len;
608 msg_head ->msg_type = MCLAG_SYNCD_MSG_TYPE_FDB_OPERATION;
609
610 SWSS_LOG_DEBUG("mclagsycnd send msg to iccpd, msg_len =%d, msg_type =%d count : %d",
611 msg_head->msg_len, msg_head->msg_type, count);
612 write = ::write(m_connection_socket, infor_start, msg_head->msg_len);
613
614 if (write <= 0)
615 {
616 SWSS_LOG_ERROR("mclagsycnd update FDB to ICCPD, write to m_connection_socket failed");
617 }
618
619 return;
620}

623void MclagLink::processMclagDomainCfg(std::deque<KeyOpFieldsValuesTuple> &entries)
624{
625 char *infor_start = getSendMsgBuffer();
626 size_t infor_len = sizeof(mclag_msg_hdr_t);
627 uint8_t system_mac[ETHER_ADDR_LEN];
628 int add_cfg_dependent_selectables = 0;
629
630 int count = 0;
631 ssize_t write = 0;
632
633 struct mclag_domain_cfg_info cfg_info;
634 mclag_msg_hdr_t *cfg_msg_hdr = NULL;
635
636 /* Nothing popped */
637 if (entries.empty())
638 {
639 return;
640 }
641
642 MacAddress::parseMacString(m_system_mac, system_mac);
643
644 for (auto entry: entries)
645 {
646
647 std::string domain_id_str = kfvKey(entry);
648 std::string op = kfvOp(entry);
649 int entryExists = 0;
650 int attrBmap = MCLAG_CFG_ATTR_NONE;

703 }
704 }
705 if (fvField(i) == "keepalive_interval")

878 infor_len = infor_len + sizeof(struct mclag_domain_cfg_info);

1743 m_pos(0),

1745 m_server_up(false),
1746 m_select(select)
1747{
1748 struct sockaddr_in addr = {};
1749 int true_val = 1;
1750
1751 m_server_socket = socket(PF_INET, SOCK_STREAM, IPPROTO_TCP);
1752 if (m_server_socket < 0)
1753 throw system_error(errno, system_category());
1754
1755 if (setsockopt(m_server_socket, SOL_SOCKET, SO_REUSEADDR, &true_val,
1756 sizeof(true_val)) < 0)
1757 {
1758 close(m_server_socket);
1759 throw system_error(errno, system_category());
1760 }
1761
1762 if (setsockopt(m_server_socket, SOL_SOCKET, SO_KEEPALIVE, &true_val,
1763 sizeof(true_val)) < 0)
1764 {
1765 close(m_server_socket);
1766 throw system_error(errno, system_category());
1767 }
1768
1769 addr.sin_family = AF_INET;
1770 addr.sin_port = htons((unsigned short int)port);
1771 addr.sin_addr.s_addr = htonl(MCLAG_DEFAULT_IP);
1772
1773 if (bind(m_server_socket, (struct sockaddr *)&addr, sizeof(addr)) < 0)
1774 {
1775 close(m_server_socket);
1776 throw system_error(errno, system_category());
1777 }
1778
1779 if (listen(m_server_socket, 2) != 0)
1780 {
1781 close(m_server_socket);
1782 throw system_error(errno, system_category());
1783 }

1785 m_server_up = true;

1879

1881 if (read == 0)

1883 if (read < 0)
1884 throw system_error(errno, system_category());

1886

1888 {
1889 hdr = reinterpret_cast<mclag_msg_hdr_t *>(static_cast<void *>(m_messageBuffer + start));

1892 break;
1893

1895 if (left < msg_len)
1896 break;
1897
1898 if (!mclag_msg_ok(hdr, left))
1899 throw system_error(make_error_code(errc::bad_message), "Malformed MCLAG message received");

1902
1903 switch (hdr->msg_type)

1924 case MCLAG_MSG_TYPE_SET_TRAFFIC_DIST_ENABLE:

1926 mclagsyncdSetTrafficDisable(msg, hdr->msg_type);
