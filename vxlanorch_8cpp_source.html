270static sai_object_id_t
271create_tunnel(
272 struct tunnel_ids_t* ids,
273 sai_ip_address_t *src_ip,
274 sai_ip_address_t *dst_ip,
275 sai_object_id_t underlay_rif,
276 bool p2p,
277 sai_uint8_t encap_ttl=0)
278{
279 sai_attribute_t attr;
280 std::vector<sai_attribute_t> tunnel_attrs;
281
282 attr.id = SAI_TUNNEL_ATTR_TYPE;
283 attr.value.s32 = SAI_TUNNEL_TYPE_VXLAN;
284 tunnel_attrs.push_back(attr);
285
286 attr.id = SAI_TUNNEL_ATTR_UNDERLAY_INTERFACE;
287 attr.value.oid = underlay_rif;
288 tunnel_attrs.push_back(attr);
289
290 sai_object_id_t map_list[TUNNEL_MAP_T_MAX_MAPPER+1];
291 uint8_t num_map=0;
292
293 for (int i=TUNNEL_MAP_T_VLAN;i<TUNNEL_MAP_T_MAX_MAPPER;i++)
294 {
295 if (ids->tunnel_decap_id[i] != SAI_NULL_OBJECT_ID)
296 {
297 map_list[num_map] = ids->tunnel_decap_id[i];
298 SWSS_LOG_INFO("create_tunnel:maplist[%d]=0x%" PRIx64 "",num_map,map_list[num_map]);
299 num_map++;
300 }
301 }
302
303 attr.id = SAI_TUNNEL_ATTR_DECAP_MAPPERS;
304 attr.value.objlist.count = num_map;
305 attr.value.objlist.list = map_list;
306 tunnel_attrs.push_back(attr);
307
308 sai_object_id_t emap_list[TUNNEL_MAP_T_MAX_MAPPER+1];
309 uint8_t num_emap=0;
310
311 for (int i=TUNNEL_MAP_T_VLAN;i<TUNNEL_MAP_T_MAX_MAPPER;i++)
312 {
313 if (ids->tunnel_encap_id[i] != SAI_NULL_OBJECT_ID)
314 {
315 emap_list[num_emap] = ids->tunnel_encap_id[i];
316 SWSS_LOG_NOTICE("create_tunnel:encapmaplist[%d]=0x%" PRIx64 "",num_emap,emap_list[num_emap]);
317 num_emap++;
318 }
319 }
320
321 attr.id = SAI_TUNNEL_ATTR_ENCAP_MAPPERS;
322 attr.value.objlist.count = num_emap;
323 attr.value.objlist.list = emap_list;
324 tunnel_attrs.push_back(attr);
325
326 // source ip
327 if (src_ip != nullptr)
328 {
329 attr.id = SAI_TUNNEL_ATTR_ENCAP_SRC_IP;
330 attr.value.ipaddr = *src_ip;
331 tunnel_attrs.push_back(attr);
332 }
333
334 // dest ip
335 if ((dst_ip != nullptr) && p2p)
336 {
337 attr.id = SAI_TUNNEL_ATTR_PEER_MODE;
338 attr.value.s32 = SAI_TUNNEL_PEER_MODE_P2P;
339 tunnel_attrs.push_back(attr);
340 attr.id = SAI_TUNNEL_ATTR_ENCAP_DST_IP;
341 attr.value.ipaddr = *dst_ip;
342 tunnel_attrs.push_back(attr);
343 }
344 else
345 {
346 attr.id = SAI_TUNNEL_ATTR_PEER_MODE;
347 attr.value.s32 = SAI_TUNNEL_PEER_MODE_P2MP;
348 tunnel_attrs.push_back(attr);
349 }
350
351 if (encap_ttl != 0)
352 {
353 attr.id = SAI_TUNNEL_ATTR_ENCAP_TTL_MODE;
354 attr.value.s32 = SAI_TUNNEL_TTL_MODE_PIPE_MODEL;
355 tunnel_attrs.push_back(attr);
356
357 attr.id = SAI_TUNNEL_ATTR_ENCAP_TTL_VAL;
358 attr.value.u8 = encap_ttl;
359 tunnel_attrs.push_back(attr);
360 }
361
362 sai_object_id_t tunnel_id;
363 sai_status_t status = sai_tunnel_api->create_tunnel(
364 &tunnel_id,
365 gSwitchId,
366 static_cast<uint32_t>(tunnel_attrs.size()),
367 tunnel_attrs.data()
368 );
369 if (status != SAI_STATUS_SUCCESS)
370 {
371 throw std::runtime_error("Can't create a tunnel object");
372 }
373
374 return tunnel_id;
375}

1187 }
1188 else
1189 {
1190 is_dip_tunnel_supported = false;
1191 for (uint32_t idx = 0; idx < values.count; idx++)

1204 m_asic_db = shared_ptr<DBConnector>(new DBConnector("ASIC_DB", 0));
1205 try
