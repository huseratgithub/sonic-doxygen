20SwitchState::SwitchState(
21 _In_ sai_object_id_t switch_id,
22 _In_ std::shared_ptr<SwitchConfig> config):

34 }
35
36 for (int i = SAI_OBJECT_TYPE_NULL; i < (int)SAI_OBJECT_TYPE_EXTENSIONS_MAX; ++i)

46 /*
47 * Create switch by default, it will require special treat on

51 m_objectHash[SAI_OBJECT_TYPE_SWITCH][sai_serialize_object_id(switch_id)] = {};
52
53 if (m_switchConfig->m_useTapDevice)
54 {
55 m_linkCallbackIndex = NetMsgRegistrar::getInstance().registerCallback(
56 std::bind(&SwitchState::asyncOnLinkMsg, this, std::placeholders::_1, std::placeholders::_2));
57 }
58
59 if (m_switchConfig->m_resourceLimiter)
60 {
61 SWSS_LOG_NOTICE("resource limiter is SET on switch %s",
62 sai_serialize_object_id(switch_id).c_str());
63 }

167void SwitchState::asyncOnLinkMsg(
168 _In_ int nlmsg_type,
169 _In_ struct nl_object *obj)
170{
171 SWSS_LOG_ENTER();
172
173 switch (nlmsg_type)
174 {
175 case RTM_NEWLINK:
176 case RTM_DELLINK:
177 break;
178
179 default:
180
181 SWSS_LOG_WARN("unsupported nlmsg_type: %d", nlmsg_type);
182 return;
183 }
184
185 struct rtnl_link *link = (struct rtnl_link *)obj;
186
187 int if_index = rtnl_link_get_ifindex(link);
188 unsigned int if_flags = rtnl_link_get_flags(link); // IFF_LOWER_UP and IFF_RUNNING
189 const char* if_name = rtnl_link_get_name(link);
190
191 SWSS_LOG_NOTICE("received %s ifname: %s, ifflags: 0x%x, ifindex: %d",
192 (nlmsg_type == RTM_NEWLINK ? "RTM_NEWLINK" : "RTM_DELLINK"),
193 if_name,
194 if_flags,
195 if_index);
196
197 auto payload = std::make_shared<EventPayloadNetLinkMsg>(m_switch_id, nlmsg_type, if_index, if_flags, if_name);
198
199 m_switchConfig->m_eventQueue->enqueue(std::make_shared<Event>(EVENT_TYPE_NET_LINK_MSG, payload));
200}

202sai_status_t SwitchState::getStatsExt(
203 _In_ sai_object_type_t object_type,
204 _In_ sai_object_id_t object_id,
205 _In_ uint32_t number_of_counters,
206 _In_ const sai_stat_id_t* counter_ids,
207 _In_ sai_stats_mode_t mode,
208 _Out_ uint64_t *counters)
209{
210 SWSS_LOG_ENTER();
211
212 bool perform_set = false;
213
214 auto info = sai_metadata_get_object_type_info(object_type);
215
216 bool enabled = false;
217
218 auto meta = m_meta.lock();
219
220 if (meta)
221 {
222 enabled = meta->meta_unittests_enabled();
223 }
224 else
225 {
226 SWSS_LOG_WARN("meta pointer expired");
227 }
228
229 if (enabled && (number_of_counters & VS_COUNTERS_COUNT_MSB ))
230 {
231 number_of_counters &= ~VS_COUNTERS_COUNT_MSB;
232
233 SWSS_LOG_NOTICE("unittests are enabled and counters count MSB is set to 1, performing SET on %s counters (%s)",
234 sai_serialize_object_id(object_id).c_str(),
235 info->statenum->name);
236
237 perform_set = true;
238 }
239
240 auto str_object_id = sai_serialize_object_id(object_id);
241
242 auto mapit = m_countersMap.find(str_object_id);
243
244 if (mapit == m_countersMap.end())
245 m_countersMap[str_object_id] = { };
246
247 auto& localcounters = m_countersMap[str_object_id];
248
249 for (uint32_t i = 0; i < number_of_counters; ++i)
250 {
251 int32_t id = counter_ids[i];
252
253 if (perform_set)
254 {
255 localcounters[ id ] = counters[i];
256 }
257 else
258 {
259 auto it = localcounters.find(id);
260
261 if (it == localcounters.end())
262 {
263 // if counter is not found on list, just return 0
264 counters[i] = 0;
265 }
266 else
267 {
268 counters[i] = it->second;
269 }
270
271 if (mode == SAI_STATS_MODE_READ_AND_CLEAR)
272 {
273 localcounters[ id ] = 0;
274 }
275 }
276 }
277
278 return SAI_STATUS_SUCCESS;
279}

281sai_status_t SwitchState::queryStatsCapability(
282 _In_ sai_object_id_t switchId,
283 _In_ sai_object_type_t objectType,
284 _Inout_ sai_stat_capability_list_t *stats_capability)
285{
286 SWSS_LOG_ENTER();
287
288 auto info = sai_metadata_get_object_type_info(objectType);
289
290 if (stats_capability->count == 0 || stats_capability->list == nullptr)
291 {
292 stats_capability->count = (uint32_t)info->statenum->valuescount;
293 return SAI_STATUS_BUFFER_OVERFLOW;
294 }
295
296 SWSS_LOG_NOTICE("query counter capability for object ID %s of counter type %s",
297 sai_serialize_object_id(switchId).c_str(),
298 info->statenum->name);
299
300 auto statenumlist = info->statenum->values;
301
302 for (uint32_t i = 0; i < stats_capability->count; i++)
303 {
304 stats_capability->list[i].stat_enum = statenumlist[i];
305 stats_capability->list[i].stat_modes = SAI_STATS_MODE_READ_AND_CLEAR;
306 }
307 return SAI_STATUS_SUCCESS;
308}
