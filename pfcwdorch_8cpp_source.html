62template <typename DropHandler, typename ForwardHandler>
63void PfcWdOrch<DropHandler, ForwardHandler>::doTask(Consumer& consumer)
64{
65 SWSS_LOG_ENTER();
66
67 if (!gPortsOrch->allPortsReady())
68 {
69 return;
70 }
71
72 if ((consumer.getDbName() == "CONFIG_DB") && (consumer.getTableName() == CFG_PFC_WD_TABLE_NAME))
73 {
74 auto it = consumer.m_toSync.begin();
75 while (it != consumer.m_toSync.end())
76 {
77 KeyOpFieldsValuesTuple t = it->second;
78
79 string key = kfvKey(t);
80 string op = kfvOp(t);
81
82 task_process_status task_status = task_process_status::task_ignore;
83 if (op == SET_COMMAND)
84 {
85 task_status = createEntry(key, kfvFieldsValues(t));
86 }
87 else if (op == DEL_COMMAND)
88 {
89 task_status = deleteEntry(key);
90 }
91 else
92 {
93 task_status = task_process_status::task_invalid_entry;
94 SWSS_LOG_ERROR("Unknown operation type %s\n", op.c_str());
95 }
96 switch (task_status)
97 {
98 case task_process_status::task_success:
99 consumer.m_toSync.erase(it++);
100 break;
101 case task_process_status::task_need_retry:
102 SWSS_LOG_INFO("Failed to process PFC watchdog %s task, retry it", op.c_str());
103 ++it;
104 break;
105 case task_process_status::task_invalid_entry:
106 SWSS_LOG_ERROR("Failed to process PFC watchdog %s task, invalid entry", op.c_str());
107 consumer.m_toSync.erase(it++);
108 break;
109 default:
110 SWSS_LOG_ERROR("Invalid task status %d", task_status);
111 consumer.m_toSync.erase(it++);
112 break;
113 }
114 }
115 }

182template <typename DropHandler, typename ForwardHandler>
183task_process_status PfcWdOrch<DropHandler, ForwardHandler>::createEntry(const string& key,
184 const vector<FieldValueTuple>& data)
185{
186 SWSS_LOG_ENTER();
187
188 uint32_t detectionTime = 0;
189 uint32_t restorationTime = 0;
190 // According to requirements, drop action is default
191 PfcWdAction action = PfcWdAction::PFC_WD_ACTION_DROP;
192 Port port;
193 if (!gPortsOrch->getPort(key, port))
194 {
195 SWSS_LOG_ERROR("Invalid port interface %s", key.c_str());
196 return task_process_status::task_invalid_entry;
197 }
198
199 if (port.m_type != Port::PHY)
200 {
201 SWSS_LOG_ERROR("Interface %s is not physical port", key.c_str());
202 return task_process_status::task_invalid_entry;
203 }
204
205 for (auto i : data)
206 {
207 const auto &field = fvField(i);
208 const auto &value = fvValue(i);
209
210 try
211 {
212 if (field == PFC_WD_DETECTION_TIME)
213 {
214 detectionTime = to_uint<uint32_t>(
215 value,
216 PFC_WD_DETECTION_TIME_MIN,
217 PFC_WD_DETECTION_TIME_MAX);
218 }
219 else if (field == PFC_WD_RESTORATION_TIME)
220 {
221 restorationTime = to_uint<uint32_t>(value,
222 PFC_WD_RESTORATION_TIME_MIN,
223 PFC_WD_RESTORATION_TIME_MAX);
224 }
225 else if (field == PFC_WD_ACTION)
226 {
227 action = deserializeAction(value);
228 if (action == PfcWdAction::PFC_WD_ACTION_UNKNOWN)
229 {
230 SWSS_LOG_ERROR("Invalid PFC Watchdog action %s", value.c_str());
231 return task_process_status::task_invalid_entry;
232 }
233 if ((m_platform == CISCO_8000_PLATFORM_SUBSTRING) && (action == PfcWdAction::PFC_WD_ACTION_FORWARD)) {
234 SWSS_LOG_ERROR("Unsupported action %s for platform %s", value.c_str(), m_platform.c_str());
235 return task_process_status::task_invalid_entry;
236 }
237 if(m_platform == BRCM_PLATFORM_SUBSTRING)
238 {
239 if(gSwitchOrch->checkPfcDlrInitEnable())
240 {
241 if(getPfcDlrPacketAction() == PfcWdAction::PFC_WD_ACTION_UNKNOWN)
242 {
243 sai_attribute_t attr;
244 attr.id = SAI_SWITCH_ATTR_PFC_DLR_PACKET_ACTION;
245 attr.value.u32 = (sai_uint32_t)action;
246
247 sai_status_t status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
248 if(status != SAI_STATUS_SUCCESS)
249 {
250 SWSS_LOG_ERROR("Failed to set switch level PFC DLR packet action rv : %d", status);
251 return task_process_status::task_invalid_entry;
252 }
253 setPfcDlrPacketAction(action);
254 }
255 else
256 {
257 if(getPfcDlrPacketAction() != action)
258 {
259 string DlrPacketAction = serializeAction(getPfcDlrPacketAction());
260 SWSS_LOG_ERROR("Invalid PFC Watchdog action %s as switch level action %s is set",
261 value.c_str(), DlrPacketAction.c_str());
262 return task_process_status::task_invalid_entry;
263 }
264 }
265 }
266 }
267 }
268 else
269 {
270 SWSS_LOG_ERROR(
271 "Failed to parse PFC Watchdog %s configuration. Unknown attribute %s.\n",
272 key.c_str(),
273 field.c_str());
274 return task_process_status::task_invalid_entry;
275 }
276 }
277 catch (const exception& e)
278 {
279 SWSS_LOG_ERROR(
280 "Failed to parse PFC Watchdog %s attribute %s error: %s.",
281 key.c_str(),
282 field.c_str(),
283 e.what());
284 return task_process_status::task_invalid_entry;
285 }
286 catch (...)
287 {
288 SWSS_LOG_ERROR(
289 "Failed to parse PFC Watchdog %s attribute %s. Unknown error has been occurred",
290 key.c_str(),
291 field.c_str());
292 return task_process_status::task_invalid_entry;
293 }
294 }
295
296 // Validation
297 if (detectionTime == 0)
298 {
299 SWSS_LOG_ERROR("%s missing", PFC_WD_DETECTION_TIME);
300 return task_process_status::task_invalid_entry;
301 }
302
303 if (!startWdOnPort(port, detectionTime, restorationTime, action))
304 {
305 SWSS_LOG_ERROR("Failed to start PFC Watchdog on port %s", port.m_alias.c_str());
306 return task_process_status::task_need_retry;
307 }
308
309 SWSS_LOG_NOTICE("Started PFC Watchdog on port %s", port.m_alias.c_str());
310 return task_process_status::task_success;
311}

331template <typename DropHandler, typename ForwardHandler>
332task_process_status PfcWdSwOrch<DropHandler, ForwardHandler>::createEntry(const string& key,
333 const vector<FieldValueTuple>& data)
334{
335 SWSS_LOG_ENTER();
336
337 if (key == PFC_WD_GLOBAL)
338 {
339 for (auto valuePair: data)
340 {
341 const auto &field = fvField(valuePair);
342 const auto &value = fvValue(valuePair);
343
344 if (field == POLL_INTERVAL_FIELD)
345 {
346 vector<FieldValueTuple> fieldValues;
347 fieldValues.emplace_back(POLL_INTERVAL_FIELD, value);
348 m_flexCounterGroupTable->set(PFC_WD_FLEX_COUNTER_GROUP, fieldValues);
349 }
350 else if (field == BIG_RED_SWITCH_FIELD)
351 {
352 SWSS_LOG_NOTICE("Receive brs mode set, %s", value.c_str());
353 setBigRedSwitchMode(value);
354 }
355 }
356 }
357 else
358 {
359 return PfcWdOrch<DropHandler, ForwardHandler>::createEntry(key, data);
360 }
361
362 return task_process_status::task_success;

417template <typename DropHandler, typename ForwardHandler>
418void PfcWdSwOrch<DropHandler, ForwardHandler>::enableBigRedSwitchMode()
419{
420 SWSS_LOG_ENTER();
421
422 m_bigRedSwitchFlag = true;
423 // Write to database that each queue enables BIG_RED_SWITCH
424 auto allPorts = gPortsOrch->getAllPorts();
425
426 for (auto &it: allPorts)
427 {
428 Port port = it.second;
429 uint8_t pfcMask = 0;
430
431 if (port.m_type != Port::PHY)
432 {
433 SWSS_LOG_INFO("Skip non-phy port %s", port.m_alias.c_str());
434 continue;
435 }
436
437 if (!gPortsOrch->getPortPfcWatchdogStatus(port.m_port_id, &pfcMask))
438 {
439 SWSS_LOG_ERROR("Failed to get PFC watchdog mask on port %s", port.m_alias.c_str());
440 return;
441 }
442
443 for (uint8_t i = 0; i < PFC_WD_TC_MAX; i++)
444 {
445 sai_object_id_t queueId = port.m_queue_ids[i];
446 if ((pfcMask & (1 << i)) == 0 && m_entryMap.find(queueId) == m_entryMap.end())
447 {
448 continue;
449 }
450
451 string queueIdStr = sai_serialize_object_id(queueId);
452
453 vector<FieldValueTuple> countersFieldValues;
454 countersFieldValues.emplace_back("BIG_RED_SWITCH_MODE", "enable");
455 this->getCountersTable()->set(queueIdStr, countersFieldValues);
456 }
457 }
458
459 // Disable pfcwdaction handler on each queue if exists.
460 for (auto & entry: m_entryMap)
461 {
462 if (entry.second.handler != nullptr)
463 {
464 entry.second.handler->commitCounters();
465 entry.second.handler = nullptr;
466 }
467 }
468
469 // Create pfcwdaction handler on all the ports.
470 for (auto & it: allPorts)
471 {
472 Port port = it.second;
473 uint8_t pfcMask = 0;
474
475 if (port.m_type != Port::PHY)
476 {
477 SWSS_LOG_INFO("Skip non-phy port %s", port.m_alias.c_str());
478 continue;
479 }
480
481 if (!gPortsOrch->getPortPfcWatchdogStatus(port.m_port_id, &pfcMask))
482 {
483 SWSS_LOG_ERROR("Failed to get PFC watchdog mask on port %s", port.m_alias.c_str());
484 return;
485 }
486
487 for (uint8_t i = 0; i < PFC_WD_TC_MAX; i++)
488 {
489 if ((pfcMask & (1 << i)) == 0)
490 {
491 continue;
492 }
493
494 sai_object_id_t queueId = port.m_queue_ids[i];
495 string queueIdStr = sai_serialize_object_id(queueId);
496
497 auto entry = m_brsEntryMap.emplace(queueId, PfcWdQueueEntry(PfcWdAction::PFC_WD_ACTION_DROP, port.m_port_id, i, port.m_alias)).first;
498
499 if (entry->second.handler== nullptr)
500 {
501 SWSS_LOG_NOTICE(
502 "PFC Watchdog BIG_RED_SWITCH mode enabled on port %s, queue index %d, queue id 0x%" PRIx64 " and port id 0x%" PRIx64 ".",
503 entry->second.portAlias.c_str(),
504 entry->second.index,
505 entry->first,
506 entry->second.portId);
507
508 entry->second.handler = make_shared<DropHandler>(
509 entry->second.portId,
510 entry->first,
511 entry->second.index,
512 this->getCountersTable());
513 entry->second.handler->initCounters();
514 }
515 }
516 }

519template <typename DropHandler, typename ForwardHandler>
520bool PfcWdSwOrch<DropHandler, ForwardHandler>::registerInWdDb(const Port& port,
521 uint32_t detectionTime, uint32_t restorationTime, PfcWdAction action)
522{
523 SWSS_LOG_ENTER();
524
525 uint8_t pfcMask = 0;
526
527 if (!gPortsOrch->getPortPfcWatchdogStatus(port.m_port_id, &pfcMask))
528 {
529 SWSS_LOG_ERROR("Failed to get PFC mask on port %s", port.m_alias.c_str());
530 return false;
531 }
532
533 set<uint8_t> losslessTc;
534 for (uint8_t i = 0; i < PFC_WD_TC_MAX; i++)
535 {
536 if ((pfcMask & (1 << i)) == 0)
537 {
538 continue;
539 }
540
541 losslessTc.insert(i);
542 }
543 if (losslessTc.empty())
544 {
545 SWSS_LOG_NOTICE("No lossless TC found on port %s", port.m_alias.c_str());
546 return false;
547 }
548
549 if (!c_portStatIds.empty())
550 {
551 string key = getFlexCounterTableKey(sai_serialize_object_id(port.m_port_id));
552 vector<FieldValueTuple> fieldValues;
553 // Only register lossless tc counters in database.
554 string str = counterIdsToStr(c_portStatIds, &sai_serialize_port_stat);
555 string filteredStr = filterPfcCounters(str, losslessTc);
556 fieldValues.emplace_back(PORT_COUNTER_ID_LIST, filteredStr);
557
558 m_flexCounterTable->set(key, fieldValues);
559 }
560
561 for (auto i : losslessTc)
562 {
563 sai_object_id_t queueId = port.m_queue_ids[i];
564 string queueIdStr = sai_serialize_object_id(queueId);
565
566 // Store detection and restoration time for plugins
567 vector<FieldValueTuple> countersFieldValues;
568 countersFieldValues.emplace_back("PFC_WD_DETECTION_TIME", to_string(detectionTime * 1000));
569 // Restoration time is optional
570 countersFieldValues.emplace_back("PFC_WD_RESTORATION_TIME",
571 restorationTime == 0 ?
572 "" :
573 to_string(restorationTime * 1000));
574 countersFieldValues.emplace_back("PFC_WD_ACTION", this->serializeAction(action));
575
576 this->getCountersTable()->set(queueIdStr, countersFieldValues);
577
578 // We register our queues in PFC_WD table so that syncd will know that it must poll them
579 vector<FieldValueTuple> queueFieldValues;
580
581 if (!c_queueStatIds.empty())
582 {
583 string str = counterIdsToStr(c_queueStatIds, sai_serialize_queue_stat);
584 queueFieldValues.emplace_back(QUEUE_COUNTER_ID_LIST, str);
585 }
586
587 if (!c_queueAttrIds.empty())
588 {
589 string str = counterIdsToStr(c_queueAttrIds, sai_serialize_queue_attr);
590 queueFieldValues.emplace_back(QUEUE_ATTR_ID_LIST, str);
591 }
592
593 // Create internal entry
594 m_entryMap.emplace(queueId, PfcWdQueueEntry(action, port.m_port_id, i, port.m_alias));
595
596 string key = getFlexCounterTableKey(queueIdStr);
597 m_flexCounterTable->set(key, queueFieldValues);
598
599 // Initialize PFC WD related counters
600 PfcWdActionHandler::initWdCounters(
601 this->getCountersTable(),
602 sai_serialize_object_id(queueId));
603 }
604
605 // We do NOT need to create ACL table group here. It will be
606 // done when ACL tables are bound to ports
607 return true;
608}

610template <typename DropHandler, typename ForwardHandler>
611string PfcWdSwOrch<DropHandler, ForwardHandler>::filterPfcCounters(string counters, set<uint8_t>& losslessTc)
612{
613 SWSS_LOG_ENTER();
614
615 istringstream is(counters);
616 string counter;
617 string filterCounters;
618
619 while (getline(is, counter, ','))
620 {
621 size_t index = 0;
622 index = counter.find(SAI_PORT_STAT_PFC_PREFIX);
623 if (index != 0)
624 {
625 filterCounters = filterCounters + counter + ",";
626 }
627 else
628 {
629 uint8_t tc = (uint8_t)atoi(counter.substr(index + sizeof(SAI_PORT_STAT_PFC_PREFIX) - 1, 1).c_str());
630 if (losslessTc.count(tc))
631 {
632 filterCounters = filterCounters + counter + ",";
633 }
634 }
635 }
636
637 if (!filterCounters.empty())
638 {
639 filterCounters.pop_back();
640 }
641
642 return filterCounters;
643}

702{

792template <typename DropHandler, typename ForwardHandler>
793void PfcWdSwOrch<DropHandler, ForwardHandler>::doTask(Consumer& consumer)
794{
795 PfcWdOrch<DropHandler, ForwardHandler>::doTask(consumer);
796
797 if (!gPortsOrch->allPortsReady())
798 {
799 return;
800 }
801
802 if ((consumer.getDbName() == "APPL_DB") && (consumer.getTableName() == APP_PFC_WD_TABLE_NAME))
803 {
804 auto it = consumer.m_toSync.begin();
805 while (it != consumer.m_toSync.end())
806 {
807 KeyOpFieldsValuesTuple &t = it->second;
808
809 string &key = kfvKey(t);
810 Port port;
811 if (!gPortsOrch->getPort(key, port))
812 {
813 SWSS_LOG_ERROR("Invalid port interface %s", key.c_str());
814 it = consumer.m_toSync.erase(it);
815 continue;
816 }
817 if (port.m_type != Port::PHY)
818 {
819 SWSS_LOG_ERROR("Interface %s is not physical port", key.c_str());
820 it = consumer.m_toSync.erase(it);
821 continue;
822 }
823
824 vector<FieldValueTuple> &fvTuples = kfvFieldsValues(t);
825 for (const auto &fv : fvTuples)
826 {
827 int qIdx = -1;
828 string q = fvField(fv);
829 try
830 {
831 qIdx = stoi(q);
832 }
833 catch (const std::invalid_argument &e)
834 {
835 SWSS_LOG_ERROR("Invalid argument %s to %s()", q.c_str(), e.what());
836 continue;
837 }
838 catch (const std::out_of_range &e)
839 {
840 SWSS_LOG_ERROR("Out of range argument %s to %s()", q.c_str(), e.what());
841 continue;
842 }
843
844 if ((qIdx < 0) || (static_cast<unsigned int>(qIdx) >= port.m_queue_ids.size()))
845 {
846 SWSS_LOG_ERROR("Invalid queue index %d on port %s", qIdx, key.c_str());
847 continue;
848 }
849
850 string status = fvValue(fv);
851 if (status != PFC_WD_IN_STORM)
852 {
853 SWSS_LOG_ERROR("Port %s queue %s not in %s", key.c_str(), q.c_str(), PFC_WD_IN_STORM);
854 continue;
855 }
856
857 SWSS_LOG_INFO("Port %s queue %s in status %s ", key.c_str(), q.c_str(), status.c_str());
858 if (!startWdActionOnQueue(PFC_WD_IN_STORM, port.m_queue_ids[qIdx]))
859 {
860 SWSS_LOG_ERROR("Failed to start PFC watchdog %s action on port %s queue %d", PFC_WD_IN_STORM, key.c_str(), qIdx);
861 continue;
862 }
863 }
864
865 it = consumer.m_toSync.erase(it);
866 }
867 }
868}

870template <typename DropHandler, typename ForwardHandler>
871void PfcWdSwOrch<DropHandler, ForwardHandler>::doTask()
872{
873 SWSS_LOG_ENTER();
874
875 // In the warm-reboot case with ongoing PFC storm,
876 // we care about dependency.
877 // PFC watchdog should be started on a port queue before
878 // a storm action can be taken in effect. The PFC watchdog
879 // configuration is stored in CONFIG_DB CFG_PFC_WD_TABLE_NAME,
880 // while the ongoing storming port queue is recorded
881 // in APPL_DB APP_PFC_WD_TABLE_NAME. We thus invoke the Executor
882 // in this order.
883 // In the cold-boot case, APP_PFC_WD_TABLE_NAME will not
884 // be populated. No dependency is introduced in this case.
885 auto *cfg_exec = this->getExecutor(CFG_PFC_WD_TABLE_NAME);
886 cfg_exec->drain();
887
888 auto *appl_exec = this->getExecutor(APP_PFC_WD_TABLE_NAME);
889 appl_exec->drain();
890
891 for (const auto &it : this->m_consumerMap)
892 {
893 auto *exec = it.second.get();
894
895 if ((exec == cfg_exec) || (exec == appl_exec))
896 {
897 continue;
898 }
899 exec->drain();
900 }
901}

958template <typename DropHandler, typename ForwardHandler>
959bool PfcWdSwOrch<DropHandler, ForwardHandler>::startWdActionOnQueue(const string &event, sai_object_id_t queueId)
960{
961 auto entry = m_entryMap.find(queueId);
962 if (entry == m_entryMap.end())
963 {
964 SWSS_LOG_ERROR("Queue 0x%" PRIx64 " is not registered", queueId);
965 return false;
966 }
967
968 SWSS_LOG_NOTICE("Receive notification, %s", event.c_str());
969
970 if (m_bigRedSwitchFlag)
971 {
972 SWSS_LOG_NOTICE("Big_RED_SWITCH mode is on, ignore syncd pfc watchdog notification");
973 }
974 else if (event == "storm")
975 {
976 if (entry->second.action == PfcWdAction::PFC_WD_ACTION_ALERT)
977 {
978 if (entry->second.handler == nullptr)
979 {
980 report_pfc_storm(entry->first, &entry->second);
981
982 entry->second.handler = make_shared<PfcWdActionHandler>(
983 entry->second.portId,
984 entry->first,
985 entry->second.index,
986 this->getCountersTable());
987 entry->second.handler->initCounters();
988 // Log storm event to APPL_DB for warm-reboot purpose
989 string key = m_applTable->getTableName() + m_applTable->getTableNameSeparator() + entry->second.portAlias;
990 m_applDb->hset(key, to_string(entry->second.index), PFC_WD_IN_STORM);
991 }
992 }
993 else if (entry->second.action == PfcWdAction::PFC_WD_ACTION_DROP)
994 {
995 if (entry->second.handler == nullptr)
996 {
997 report_pfc_storm(entry->first, &entry->second);
998
999 entry->second.handler = make_shared<DropHandler>(
1000 entry->second.portId,
1001 entry->first,
1002 entry->second.index,
1003 this->getCountersTable());
1004 entry->second.handler->initCounters();
1005 // Log storm event to APPL_DB for warm-reboot purpose
1006 string key = m_applTable->getTableName() + m_applTable->getTableNameSeparator() + entry->second.portAlias;
1007 m_applDb->hset(key, to_string(entry->second.index), PFC_WD_IN_STORM);
1008 }
1009 }
1010 else if (entry->second.action == PfcWdAction::PFC_WD_ACTION_FORWARD)
1011 {
1012 if (entry->second.handler == nullptr)
1013 {
1014 report_pfc_storm(entry->first, &entry->second);
1015
1016 entry->second.handler = make_shared<ForwardHandler>(
1017 entry->second.portId,
1018 entry->first,
1019 entry->second.index,
1020 this->getCountersTable());
1021 entry->second.handler->initCounters();
1022 // Log storm event to APPL_DB for warm-reboot purpose
1023 string key = m_applTable->getTableName() + m_applTable->getTableNameSeparator() + entry->second.portAlias;
1024 m_applDb->hset(key, to_string(entry->second.index), PFC_WD_IN_STORM);
1025 }
1026 }
1027 else
1028 {
1029 SWSS_LOG_ERROR("Unknown PFC WD action");
1030 return false;
1031 }
1032 }
1033 else if (event == "restore")
1034 {
1035 if (entry->second.handler != nullptr)
1036 {
1037 SWSS_LOG_NOTICE(
1038 "PFC Watchdog storm restored on port %s, queue index %d, queue id 0x%" PRIx64 " and port id 0x%" PRIx64 ".",
1039 entry->second.portAlias.c_str(),
1040 entry->second.index,
1041 entry->first,
1042 entry->second.portId);
1043
1044 entry->second.handler->commitCounters();
1045 entry->second.handler = nullptr;
1046 // Remove storm status in APPL_DB for warm-reboot purpose
1047 string key = m_applTable->getTableName() + m_applTable->getTableNameSeparator() + entry->second.portAlias;
1048 m_applDb->hdel(key, to_string(entry->second.index));
1049 }
1050 }
1051 else
1052 {
1053 SWSS_LOG_ERROR("Received unknown event from plugin, %s", event.c_str());
1054 return false;
1055 }
1056
1057 return true;
1058}

1060template <typename DropHandler, typename ForwardHandler>
1061bool PfcWdSwOrch<DropHandler, ForwardHandler>::bake()
1062{
1063 // clean all *_last and *_LEFT fields in COUNTERS_TABLE
1064 // to allow warm-reboot pfc detect & restore state machine to enter the same init state as cold-reboot
1065 vector<string> cKeys;
1066 this->getCountersTable()->getKeys(cKeys);
1067 for (const auto &key : cKeys)
1068 {
1069 vector<FieldValueTuple> fvTuples;
1070 this->getCountersTable()->get(key, fvTuples);
1071 vector<string> wLasts;
1072 for (const auto &fv : fvTuples)
1073 {
1074 if ((fvField(fv).find("_last") != string::npos) || (fvField(fv).find("_LEFT") != string::npos))
1075 {
1076 wLasts.push_back(fvField(fv));
1077 }
1078 }
1079 if (!wLasts.empty())
1080 {
1081 this->getCountersDb()->hdel(
1082 this->getCountersTable()->getTableName()
1083 + this->getCountersTable()->getTableNameSeparator()
1084 + key,
1085 wLasts);
1086 }
1087 }
1088
1089 Orch::bake();
1090
1091 Consumer *consumer = dynamic_cast<Consumer *>(this->getExecutor(APP_PFC_WD_TABLE_NAME));
1092 if (consumer == NULL)
1093 {
1094 SWSS_LOG_ERROR("No consumer %s in Orch", APP_PFC_WD_TABLE_NAME);
1095 return false;
1096 }
1097
1098 size_t refilled = consumer->refillToSync(m_applTable.get());
1099 SWSS_LOG_NOTICE("Add warm input PFC watchdog State: %s, %zd", APP_PFC_WD_TABLE_NAME, refilled);
1100
1101 return true;
1102}
