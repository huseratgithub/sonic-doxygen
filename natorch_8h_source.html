41struct NatEntryValue
42{
43 IpAddress translated_ip; // Translated IP address
44 string nat_type; // Nat Type - SNAT or DNAT
45 string entry_type; // Entry type - Static or Dynamic
46 time_t activeTime; // Timestamp in secs when the entry was last seen as active
47 time_t ageOutTime; // Timestamp in secs when the entry expires
48 bool addedToHw; // Boolean to represent added to hardware
49
50 bool operator<(const NatEntryValue& other) const
51 {
52 return tie(translated_ip, nat_type, entry_type) < tie(other.translated_ip, other.nat_type, other.entry_type);
53 }
54};
