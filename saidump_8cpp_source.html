81 SWSS_LOG_NOTICE("Dumping temp view");
82 options.dumpTempView = true;
83 break;
84
85 case 'h':
86 printUsage();
87 exit(EXIT_SUCCESS);
88
89 case '?':
90 SWSS_LOG_WARN("unknown option %c", optopt);
91 printUsage();

96 exit(EXIT_FAILURE);

178 std::map<sai_object_id_t, const sai_object_type_info_t*> oidtypemap;

180
181 std::cout << "digraph \"SAI Object Dependency Graph\" {" << std::endl;
182 std::cout << "size = \"30,12\"; ratio = fill;" << std::endl;
183 std::cout << "node [ style = filled ];" << std::endl;
184
185 // build object type map first
186
187 std::set<sai_object_type_t> definedtypes;
188
189 std::map<sai_object_type_t, int> usagemap;
190
191 for (const auto& key: td)
192 {
193 sai_object_meta_key_t meta_key;
194 sai_deserialize_object_meta_key(key.first, meta_key);
195
196 auto info = sai_metadata_get_object_type_info(meta_key.objecttype);
197
198 typemap[info->objecttype] = info;
199
200 if (!info->isnonobjectid)
201 oidtypemap[meta_key.objectkey.key.object_id] = info;
202
203 if (definedtypes.find(meta_key.objecttype) != definedtypes.end())
204 continue;
205
206 definedtypes.insert(meta_key.objecttype);
207 }
208
209 std::set<std::string> definedlinks;
210
211 std::set<sai_object_type_t> ref;
212 std::set<sai_object_type_t> attrref;
213
214#define SAI_OBJECT_TYPE_PREFIX_LEN (sizeof("SAI_OBJECT_TYPE_") - 1)
215
216 for (const auto& key: td)
217 {
218 sai_object_meta_key_t meta_key;
219 sai_deserialize_object_meta_key(key.first, meta_key);
220
221 auto info = sai_metadata_get_object_type_info(meta_key.objecttype);
222
223 // process non object id objects if any
224 for (size_t j = 0; j < info->structmemberscount; ++j)
225 {
226 const sai_struct_member_info_t *m = info->structmembers[j];
227
228 if (m->membervaluetype == SAI_ATTR_VALUE_TYPE_OBJECT_ID)
229 {
230 sai_object_id_t member_oid = m->getoid(&meta_key);
231
232 auto member_info = oidtypemap.at(member_oid);
233
234 if (member_info->objecttype == SAI_OBJECT_TYPE_SWITCH)
235 {
236 // skip link of SWITCH to non object id object types, since
237 // all of them contain switch_id
238 continue;
239 }
240
241 std::stringstream ss;
242
243 ss << std::string(member_info->objecttypename + SAI_OBJECT_TYPE_PREFIX_LEN) << " -> "
244 << std::string(info->objecttypename + SAI_OBJECT_TYPE_PREFIX_LEN)
245 << "[ color=\"" << GV_ARROW_COLOR << "\", style = dashed, penwidth = 2 ]";
246
247 std::string link = ss.str();
248
249 if (definedlinks.find(link) != definedlinks.end())
250 continue;
251
252 definedlinks.insert(link);
253
254 std::cout << link << std::endl;
255 }
256 }
257
258 // process attributes for this object
259
260 for (const auto&field: key.second)
261 {
262 const sai_attr_metadata_t *meta;
263 sai_deserialize_attr_id(field.first, &meta);
264
265 if (!meta->isoidattribute || meta->isreadonly)
266 {
267 // skip non oid attributes and read only attributes
268 continue;
269 }
270
271 sai_attribute_t attr;
272
273 sai_deserialize_attr_value(field.second, *meta, attr, false);
274
275 sai_object_list_t list = { 0, NULL };
276
277 switch (meta->attrvaluetype)
278 {
279 case SAI_ATTR_VALUE_TYPE_OBJECT_ID:
280 list.count = 1;
281 list.list = &attr.value.oid;
282 break;
283
284 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_ID:
285 if (attr.value.aclfield.enable)
286 {
287 list.count = 1;
288 list.list = &attr.value.aclfield.data.oid;
289 }
290 break;
291
292 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_ID:
293 if (attr.value.aclaction.enable)
294 {
295 list.count = 1;
296 list.list = &attr.value.aclaction.parameter.oid;
297 }
298 break;
299
300 case SAI_ATTR_VALUE_TYPE_OBJECT_LIST:
301 list = attr.value.objlist;
302 break;
303
304 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_LIST:
305 if (attr.value.aclfield.enable)
306 list = attr.value.aclfield.data.objlist;
307 break;
308
309 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_LIST:
310 if (attr.value.aclaction.enable)
311 list = attr.value.aclaction.parameter.objlist;
312 break;
313
314 default:
315 SWSS_LOG_THROW("attr value type: %d is not supported, FIXME", meta->attrvaluetype);
316 }
317
318 for (uint32_t i = 0; i < list.count; ++i)
319 {
320 sai_object_id_t oid = list.list[i];
321
322 if (oid == SAI_NULL_OBJECT_ID)
323 continue;
324
325 // this object type is not root, can be in the middle or leaf
326 ref.insert(info->objecttype);
327
328 auto attr_oid_info = oidtypemap.at(oid);
329
330 std::stringstream ss;
331
332 attrref.insert(attr_oid_info->objecttype);
333
334 ss << std::string(attr_oid_info->objecttypename + SAI_OBJECT_TYPE_PREFIX_LEN) << " -> "
335 << std::string(info->objecttypename + SAI_OBJECT_TYPE_PREFIX_LEN)
336 << "[ color = \"" << GV_ARROW_COLOR << "\" ]";
337
338 std::string link = ss.str();
339
340 if (definedlinks.find(link) != definedlinks.end())
341 continue;
342
343 definedlinks.insert(link);
344
345 std::cout << link << std::endl;
346 }
347
348 sai_deserialize_free_attribute_value(meta->attrvaluetype, attr);
349 }
350 }
351
352 for (auto t: typemap)
353 {
354 auto ot = t.first;
355 auto info = t.second;
356
357 auto name = std::string(info->objecttypename + SAI_OBJECT_TYPE_PREFIX_LEN);
358
359 if (info->isnonobjectid)
360 {
361 /* non object id leafs */
362
363 std::cout << name << " [ color = plum, shape = rect ];\n";
364 continue;
365 }
366
367 if (ref.find(ot) != ref.end() && attrref.find(ot) != attrref.end())
368 {
369 /* middle nodes */
370
371 std::cout << name << " [ color =\"" << GV_NODE_COLOR << "\" ];\n";
372 continue;
373 }
374
375 if (ref.find(ot) != ref.end() && attrref.find(ot) == attrref.end())
376 {
377 /* leafs */
378
379 std::cout << name << " [ color = palegreen, shape = rect ];\n";
380 continue;
381 }
382
383 if (ref.find(ot) == ref.end() && attrref.find(ot) != attrref.end())
384 {
385 /* roots */
386
387 std::cout << name << " [ color = \"" << GV_ROOT_COLOR << "\" ];\n";
388 continue;
389 }
390
391 /* objects which are there but not referenced nowhere for example STP */
392
393 std::cout << name << " [ color = \"" << GV_ROOT_COLOR << "\", shape = rect ];\n";
394 }
395
396 std::cout << "SWITCH -> PORT [ dir = none, color = red, peripheries = 2, penwidth = 2, style = dashed ];" << std::endl;
397 std::cout << "SWITCH [ color = orange, fillcolor = orange, shape = parallelogram, peripheries = 2 ];" << std::endl;
398 std::cout << "PORT [ color = gold, shape = diamond, peripheries = 2 ];" << std::endl;
399 std::cout << "}" << std::endl;
