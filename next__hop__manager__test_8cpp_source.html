152// Verifies whether the attribute list is the same as expected for SAI next
153// hop's create_next_hop().
154// Returns true if they match; otherwise, false.
155bool MatchCreateNextHopArgAttrList(const sai_attribute_t *attr_list,
156 const std::unordered_map<sai_attr_id_t, sai_attribute_value_t> &expected_attr_list)
157{
158 if (attr_list == nullptr)
159 {
160 return false;
161 }
162
163 // Sanity check for expected_attr_list.
164 const auto end = expected_attr_list.end();
165 if (expected_attr_list.size() != 3 || expected_attr_list.find(SAI_NEXT_HOP_ATTR_TYPE) == end ||
166 expected_attr_list.find(SAI_NEXT_HOP_ATTR_IP) == end ||
167 (expected_attr_list.find(SAI_NEXT_HOP_ATTR_ROUTER_INTERFACE_ID) == end &&
168 expected_attr_list.find(SAI_NEXT_HOP_ATTR_TUNNEL_ID) == end))
169 {
170 return false;
171 }
172
173 for (int i = 0; i < 3; ++i)
174 {
175 switch (attr_list[i].id)
176 {
177 case SAI_NEXT_HOP_ATTR_TYPE:
178 if (attr_list[i].value.s32 != expected_attr_list.at(SAI_NEXT_HOP_ATTR_TYPE).s32)
179 return false;
180 break;
181 case SAI_NEXT_HOP_ATTR_IP: {
182 auto construct_ip_addr = [](const sai_ip_address_t &sai_ip_address) -> swss::IpAddress {
183 swss::ip_addr_t ipaddr;
184 if (sai_ip_address.addr_family == SAI_IP_ADDR_FAMILY_IPV4)
185 {
186 ipaddr.family = AF_INET;
187 ipaddr.ip_addr.ipv4_addr = sai_ip_address.addr.ip4;
188 }
189 else
190 {
191 ipaddr.family = AF_INET6;
192 memcpy(&ipaddr.ip_addr.ipv6_addr, &sai_ip_address.addr.ip6, sizeof(ipaddr.ip_addr.ipv6_addr));
193 }
194
195 return swss::IpAddress(ipaddr);
196 };
197
198 auto ipaddr = construct_ip_addr(attr_list[i].value.ipaddr);
199 auto expected_ipaddr = construct_ip_addr(expected_attr_list.at(SAI_NEXT_HOP_ATTR_IP).ipaddr);
200 if (ipaddr != expected_ipaddr)
201 {
202 return false;
203 }
204 break;
205 }
206 case SAI_NEXT_HOP_ATTR_ROUTER_INTERFACE_ID:
207 if (attr_list[i].value.oid != expected_attr_list.at(SAI_NEXT_HOP_ATTR_ROUTER_INTERFACE_ID).oid)
208 {
209 return false;
210 }
211 break;
212 case SAI_NEXT_HOP_ATTR_TUNNEL_ID:
213 if (attr_list[i].value.oid != expected_attr_list.at(SAI_NEXT_HOP_ATTR_TUNNEL_ID).oid)
214 {
215 return false;
216 }
217 break;
218 default:
219 // Invalid attribute ID in next hop's attribute list.
220 return false;
221 }
222 }
223
224 return true;
225}
