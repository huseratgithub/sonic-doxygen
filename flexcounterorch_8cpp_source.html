83void FlexCounterOrch::doTask(Consumer &consumer)
84{
85 SWSS_LOG_ENTER();
86
87 VxlanTunnelOrch* vxlan_tunnel_orch = gDirectory.get<VxlanTunnelOrch*>();
88 if (gPortsOrch && !gPortsOrch->allPortsReady())
89 {
90 return;
91 }
92
93 if (gFabricPortsOrch && !gFabricPortsOrch->allPortsReady())
94 {
95 return;
96 }
97
98 auto it = consumer.m_toSync.begin();
99 while (it != consumer.m_toSync.end())
100 {
101 KeyOpFieldsValuesTuple t = it->second;
102
103 string key = kfvKey(t);
104 string op = kfvOp(t);
105 auto data = kfvFieldsValues(t);
106
107 if (!flexCounterGroupMap.count(key))
108 {
109 SWSS_LOG_NOTICE("Invalid flex counter group input, %s", key.c_str());
110 consumer.m_toSync.erase(it++);
111 continue;
112 }
113
114 if (op == SET_COMMAND)
115 {
116 auto itDelay = std::find(std::begin(data), std::end(data), FieldValueTuple(FLEX_COUNTER_DELAY_STATUS_FIELD, "true"));
117
118 if (itDelay != data.end())
119 {
120 consumer.m_toSync.erase(it++);
121 continue;
122 }
123 for (auto valuePair:data)
124 {
125 const auto &field = fvField(valuePair);
126 const auto &value = fvValue(valuePair);
127
128 if (field == POLL_INTERVAL_FIELD)
129 {
130 vector<FieldValueTuple> fieldValues;
131 fieldValues.emplace_back(POLL_INTERVAL_FIELD, value);
132 m_flexCounterGroupTable->set(flexCounterGroupMap[key], fieldValues);
133 if (gPortsOrch && gPortsOrch->isGearboxEnabled())
134 {
135 if (key == PORT_KEY || key.rfind("MACSEC", 0) == 0)
136 {
137 m_gbflexCounterGroupTable->set(flexCounterGroupMap[key], fieldValues);
138 }
139 }
140 }
141 else if(field == FLEX_COUNTER_STATUS_FIELD)
142 {
143 // Currently, the counters are disabled for polling by default
144 // The queue maps will be generated as soon as counters are enabled for polling
145 // Counter polling is enabled by pushing the COUNTER_ID_LIST/ATTR_ID_LIST, which contains
146 // the list of SAI stats/attributes of polling interest, to the FLEX_COUNTER_DB under the
147 // additional condition that the polling interval at that time is set nonzero positive,
148 // which is automatically satisfied upon the creation of the orch object that requires
149 // the syncd flex counter polling service
150 // This postponement is introduced by design to accelerate the initialization process
151 if(gPortsOrch && (value == "enable"))
152 {
153 if(key == PORT_KEY)
154 {
155 gPortsOrch->generatePortCounterMap();
156 m_port_counter_enabled = true;
157 }
158 else if(key == PORT_BUFFER_DROP_KEY)
159 {
160 gPortsOrch->generatePortBufferDropCounterMap();
161 m_port_buffer_drop_counter_enabled = true;
162 }
163 else if(key == QUEUE_KEY)
164 {
165 gPortsOrch->generateQueueMap(getQueueConfigurations());
166 m_queue_enabled = true;
167 gPortsOrch->addQueueFlexCounters(getQueueConfigurations());
168 }
169 else if(key == QUEUE_WATERMARK)
170 {
171 gPortsOrch->generateQueueMap(getQueueConfigurations());
172 m_queue_watermark_enabled = true;
173 gPortsOrch->addQueueWatermarkFlexCounters(getQueueConfigurations());
174 }
175 else if(key == PG_DROP_KEY)
176 {
177 gPortsOrch->generatePriorityGroupMap(getPgConfigurations());
178 m_pg_enabled = true;
179 gPortsOrch->addPriorityGroupFlexCounters(getPgConfigurations());
180 }
181 else if(key == PG_WATERMARK_KEY)
182 {
183 gPortsOrch->generatePriorityGroupMap(getPgConfigurations());
184 m_pg_watermark_enabled = true;
185 gPortsOrch->addPriorityGroupWatermarkFlexCounters(getPgConfigurations());
186 }
187 }
188 if(gIntfsOrch && (key == RIF_KEY) && (value == "enable"))
189 {
190 gIntfsOrch->generateInterfaceMap();
191 }
192 if (gBufferOrch && (key == BUFFER_POOL_WATERMARK_KEY) && (value == "enable"))
193 {
194 gBufferOrch->generateBufferPoolWatermarkCounterIdList();
195 }
196 if (gFabricPortsOrch)
197 {
198 gFabricPortsOrch->generateQueueStats();
199 }
200 if (vxlan_tunnel_orch && (key== TUNNEL_KEY) && (value == "enable"))
201 {
202 vxlan_tunnel_orch->generateTunnelCounterMap();
203 }
204 if (gCoppOrch && (key == FLOW_CNT_TRAP_KEY))
205 {
206 if (value == "enable")
207 {
208 m_hostif_trap_counter_enabled = true;
209 gCoppOrch->generateHostIfTrapCounterIdList();
210 }
211 else if (value == "disable")
212 {
213 gCoppOrch->clearHostIfTrapCounterIdList();
214 m_hostif_trap_counter_enabled = false;
215 }
216 }
217 if (gFlowCounterRouteOrch && gFlowCounterRouteOrch->getRouteFlowCounterSupported() && key == FLOW_CNT_ROUTE_KEY)
218 {
219 if (value == "enable" && !m_route_flow_counter_enabled)
220 {
221 m_route_flow_counter_enabled = true;
222 gFlowCounterRouteOrch->generateRouteFlowStats();
223 }
224 else if (value == "disable" && m_route_flow_counter_enabled)
225 {
226 gFlowCounterRouteOrch->clearRouteFlowStats();
227 m_route_flow_counter_enabled = false;
228 }
229 }
230 vector<FieldValueTuple> fieldValues;
231 fieldValues.emplace_back(FLEX_COUNTER_STATUS_FIELD, value);
232 m_flexCounterGroupTable->set(flexCounterGroupMap[key], fieldValues);
233
234 if (gPortsOrch && gPortsOrch->isGearboxEnabled())
235 {
236 if (key == PORT_KEY || key.rfind("MACSEC", 0) == 0)
237 {
238 m_gbflexCounterGroupTable->set(flexCounterGroupMap[key], fieldValues);
239 }
240 }
241 }
242 else if(field == FLEX_COUNTER_DELAY_STATUS_FIELD)
243 {
244 // This field is ignored since it is being used before getting into this loop.
245 // If it is exist and the value is 'true' we need to skip the iteration in order to delay the counter creation.
246 // The field will clear out and counter will be created when enable_counters script is called.
247 }
248 else
249 {
250 SWSS_LOG_NOTICE("Unsupported field %s", field.c_str());
251 }
252 }
253 }
254
255 consumer.m_toSync.erase(it++);
256 }

289bool FlexCounterOrch::bake()
290{
291 /*
292 * bake is called during warmreboot reconciling procedure.
293 * By default, it should fetch items from the tables the sub agents listen to,
294 * and then push them into m_toSync of each sub agent.
295 * The motivation is to make sub agents handle the saved entries first and then handle the upcoming entries.
296 */
297
298 std::deque<KeyOpFieldsValuesTuple> entries;
299 vector<string> keys;
300 m_flexCounterConfigTable.getKeys(keys);
301 for (const auto &key: keys)
302 {
303 if (!flexCounterGroupMap.count(key))
304 {
305 SWSS_LOG_NOTICE("FlexCounterOrch: Invalid flex counter group intput %s is skipped during reconciling", key.c_str());
306 continue;
307 }
308
309 if (key == BUFFER_POOL_WATERMARK_KEY)
310 {
311 SWSS_LOG_NOTICE("FlexCounterOrch: Do not handle any FLEX_COUNTER table for %s update during reconciling",
312 BUFFER_POOL_WATERMARK_KEY);
313 continue;
314 }
315
316 KeyOpFieldsValuesTuple kco;
317
318 kfvKey(kco) = key;
319 kfvOp(kco) = SET_COMMAND;
320
321 if (!m_flexCounterConfigTable.get(key, kfvFieldsValues(kco)))
322 {
323 continue;
324 }
325 entries.push_back(kco);
326 }
327 Consumer* consumer = dynamic_cast<Consumer *>(getExecutor(CFG_FLEX_COUNTER_TABLE_NAME));
328 return consumer->addToSync(entries);
329}

331map<string, FlexCounterQueueStates> FlexCounterOrch::getQueueConfigurations()
332{
333 SWSS_LOG_ENTER();
334
335 map<string, FlexCounterQueueStates> queuesStateVector;
336 std::vector<std::string> portQueueKeys;
337 m_bufferQueueConfigTable.getKeys(portQueueKeys);
338
339 for (const auto& portQueueKey : portQueueKeys)
340 {
341 auto toks = tokenize(portQueueKey, '|');
342 if (toks.size() != 2)
343 {
344 SWSS_LOG_ERROR("Invalid BUFFER_QUEUE key: [%s]", portQueueKey.c_str());
345 continue;
346 }
347
348 auto configPortNames = tokenize(toks[0], ',');
349 auto configPortQueues = toks[1];
350 toks = tokenize(configPortQueues, '-');
351
352 for (const auto& configPortName : configPortNames)
353 {
354 uint32_t maxQueueNumber = gPortsOrch->getNumberOfPortSupportedQueueCounters(configPortName);
355 uint32_t maxQueueIndex = maxQueueNumber - 1;
356 uint32_t minQueueIndex = 0;
357
358 if (!queuesStateVector.count(configPortName))
359 {
360 FlexCounterQueueStates flexCounterQueueState(maxQueueNumber);
361 queuesStateVector.insert(make_pair(configPortName, flexCounterQueueState));
362 }
363
364 try {
365 auto startIndex = to_uint<uint32_t>(toks[0], minQueueIndex, maxQueueIndex);
366 if (toks.size() > 1)
367 {
368 auto endIndex = to_uint<uint32_t>(toks[1], minQueueIndex, maxQueueIndex);
369 queuesStateVector.at(configPortName).enableQueueCounters(startIndex, endIndex);
370 }
371 else
372 {
373 queuesStateVector.at(configPortName).enableQueueCounter(startIndex);
374 }
375 } catch (std::invalid_argument const& e) {
376 SWSS_LOG_ERROR("Invalid queue index [%s] for port [%s]", configPortQueues.c_str(), configPortName.c_str());
377 continue;
378 }
379 }
380 }
381
382 return queuesStateVector;
383}
