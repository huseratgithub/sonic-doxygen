125bool Srv6Orch::removeSrv6Nexthops(const NextHopGroupKey &nhg)
126{
127 SWSS_LOG_ENTER();
128
129 for (auto &sr_nh : nhg.getNextHops())
130 {
131 string srv6_source, segname;
132 sai_status_t status = SAI_STATUS_SUCCESS;
133 srv6_source = sr_nh.srv6_source;
134 segname = sr_nh.srv6_segment;
135
136 SWSS_LOG_NOTICE("SRV6 Nexthop %s refcount %d", sr_nh.to_string(false,true).c_str(), m_neighOrch->getNextHopRefCount(sr_nh));
137 if (m_neighOrch->getNextHopRefCount(sr_nh) == 0)

143 return false;
144 }

154 m_neighOrch->updateSrv6Nexthop(sr_nh, 0);
155 srv6_nexthop_table_.erase(sr_nh);

157 /* Delete NH from the tunnel map */
158 SWSS_LOG_INFO("Delete NH %s from tunnel map",
159 sr_nh.to_string(false, true).c_str());
160 srv6TunnelUpdateNexthops(srv6_source, sr_nh, false);
161 }

164 if (tunnel_nhs == 0)

167 if (status != SAI_STATUS_SUCCESS)

169 SWSS_LOG_ERROR("Failed to remove SRV6 tunnel object for source %s", srv6_source.c_str());
170 return false;
171 }
172 srv6_tunnel_table_.erase(srv6_source);
173 }
174 else
175 {
176 SWSS_LOG_INFO("Nexthops referencing this tunnel object %s: %zu", srv6_source.c_str(),tunnel_nhs);
177 }
178 }
179 return true;

182bool Srv6Orch::createSrv6Nexthop(const NextHopKey &nh)
183{
184 SWSS_LOG_ENTER();
185 string srv6_segment = nh.srv6_segment;
186 string srv6_source = nh.srv6_source;
187
188 if (srv6NexthopExists(nh))
189 {
190 SWSS_LOG_INFO("SRV6 nexthop already created for %s", nh.to_string(false,true).c_str());
191 return true;
192 }
193 sai_object_id_t srv6_object_id = sid_table_[srv6_segment].sid_object_id;
194 sai_object_id_t srv6_tunnel_id = srv6_tunnel_table_[srv6_source].tunnel_object_id;
195
196 if (srv6_object_id == SAI_NULL_OBJECT_ID)
197 {
198 SWSS_LOG_ERROR("segment object doesn't exist for segment %s", srv6_segment.c_str());
199 return false;
200 }
201
202 if (srv6_tunnel_id == SAI_NULL_OBJECT_ID)
203 {
204 SWSS_LOG_ERROR("tunnel object doesn't exist for source %s", srv6_source.c_str());
205 return false;
206 }
207 SWSS_LOG_INFO("Create srv6 nh for tunnel src %s with seg %s", srv6_source.c_str(), srv6_segment.c_str());
208 vector<sai_attribute_t> nh_attrs;
209 sai_object_id_t nexthop_id;
210 sai_attribute_t attr;
211 sai_status_t status;
212
213 attr.id = SAI_NEXT_HOP_ATTR_TYPE;
214 attr.value.s32 = SAI_NEXT_HOP_TYPE_SRV6_SIDLIST;
215 nh_attrs.push_back(attr);

217 attr.id = SAI_NEXT_HOP_ATTR_SRV6_SIDLIST_ID;
218 attr.value.oid = srv6_object_id;

222 attr.value.oid = srv6_tunnel_id;
223 nh_attrs.push_back(attr);
224

226 (uint32_t)nh_attrs.size(),
227 nh_attrs.data());
228 if (status != SAI_STATUS_SUCCESS)

230 SWSS_LOG_ERROR("Failed to create srv6 nexthop for %s", nh.to_string(false,true).c_str());
231 return false;
232 }
233 m_neighOrch->updateSrv6Nexthop(nh, nexthop_id);
234 srv6_nexthop_table_[nh] = nexthop_id;
235 sid_table_[srv6_segment].nexthops.insert(nh);
236 srv6TunnelUpdateNexthops(srv6_source, nh, true);
237 return true;

270bool Srv6Orch::createUpdateSidList(const string sid_name, const string sid_list)
271{
272 SWSS_LOG_ENTER();
273 bool exists = (sid_table_.find(sid_name) != sid_table_.end());
274 sai_segment_list_t segment_list;
275 vector<string>sid_ips = tokenize(sid_list, SID_LIST_DELIMITER);
276 sai_object_id_t segment_oid;
277 segment_list.count = (uint32_t)sid_ips.size();
278 if (segment_list.count == 0)
279 {
280 SWSS_LOG_ERROR("segment list count is zero, skip");
281 return true;
282 }
283 SWSS_LOG_INFO("Segment count %d", segment_list.count);
284 segment_list.list = new sai_ip6_t[segment_list.count];
285 uint32_t index = 0;
286
287 for (string ip_str : sid_ips)
288 {
289 IpPrefix ip(ip_str);
290 SWSS_LOG_INFO("Segment %s, count %d", ip.to_string().c_str(), segment_list.count);
291 memcpy(segment_list.list[index++], ip.getIp().getV6Addr(), 16);
292 }
293 sai_attribute_t attr;
294 sai_status_t status;
295 if (!exists)
296 {
297 /* Create sidlist object with list of ipv6 prefixes */
298 SWSS_LOG_INFO("Create SID list");
299 vector<sai_attribute_t> attributes;
300 attr.id = SAI_SRV6_SIDLIST_ATTR_SEGMENT_LIST;
301 attr.value.segmentlist.list = segment_list.list;
302 attr.value.segmentlist.count = segment_list.count;
303 attributes.push_back(attr);
304
305 attr.id = SAI_SRV6_SIDLIST_ATTR_TYPE;
306 attr.value.s32 = SAI_SRV6_SIDLIST_TYPE_ENCAPS_RED;
307 attributes.push_back(attr);
308 status = sai_srv6_api->create_srv6_sidlist(&segment_oid, gSwitchId, (uint32_t) attributes.size(), attributes.data());
309 if (status != SAI_STATUS_SUCCESS)
310 {
311 SWSS_LOG_ERROR("Failed to create srv6 sidlist object, rv %d", status);
312 return false;
313 }
314 sid_table_[sid_name].sid_object_id = segment_oid;
315 }
316 else
317 {
318 SWSS_LOG_INFO("Set SID list");
319
320 /* Update sidlist object with new set of ipv6 addresses */
321 attr.id = SAI_SRV6_SIDLIST_ATTR_SEGMENT_LIST;
322 attr.value.segmentlist.list = segment_list.list;
323 attr.value.segmentlist.count = segment_list.count;
324 segment_oid = (sid_table_.find(sid_name)->second).sid_object_id;
325 status = sai_srv6_api->set_srv6_sidlist_attribute(segment_oid, &attr);
326 if (status != SAI_STATUS_SUCCESS)
327 {
328 SWSS_LOG_ERROR("Failed to set srv6 sidlist object with new segments, rv %d", status);
329 return false;
330 }
331 }
332 delete segment_list.list;
333 return true;
334}

471
472 SWSS_LOG_INFO("MySid: sid %s, action %s, vrf %s, block %d, node %d, func %d, arg %d dt_vrf %s",

474 my_sid_entry.function_len, my_sid_entry.args_len, dt_vrf.c_str());

500 return false;
501 }
502 vrf_attr.id = SAI_MY_SID_ENTRY_ATTR_VRF;
503 vrf_attr.value.oid = dt_vrf_id;
504 attributes.push_back(vrf_attr);
505 vrf_update = true;
506 }
507 attr.id = SAI_MY_SID_ENTRY_ATTR_ENDPOINT_BEHAVIOR;
508 attr.value.s32 = end_behavior;
509 attributes.push_back(attr);
510
511 attr.id = SAI_MY_SID_ENTRY_ATTR_ENDPOINT_BEHAVIOR_FLAVOR;
512 attr.value.s32 = end_flavor;
513 attributes.push_back(attr);
514

519 if (status != SAI_STATUS_SUCCESS)

537 }
538 SWSS_LOG_INFO("Store keystring %s in cache", key_string.c_str());
539 if(vrf_update)

542 srv6_my_sid_table_[key_string].endVrfString = dt_vrf;
543 }
544 srv6_my_sid_table_[key_string].endBehavior = end_behavior;
545 srv6_my_sid_table_[key_string].entry = my_sid_entry;
