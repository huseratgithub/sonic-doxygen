132bool SaiSwitchAsic::isNonRemovableRid(
133 _In_ sai_object_id_t rid) const
134{
135 SWSS_LOG_ENTER();
136
137 if (rid == SAI_NULL_OBJECT_ID)
138 {
139 SWSS_LOG_THROW("NULL rid passed");
140 }
141
142 /*
143 * Check for SAI_SWITCH_ATTR_DEFAULT_* oids like cpu, default virtual
144 * router. Those objects can't be removed if user ask for it.
145 */
146
147 /*
148 * Here we are checking for isSwitchObjectDefaultRid first then
149 * ColdBootDiscoveredRid as it is possible we can discover switch Internal
150 * OID as part of warm-boot also especially when we are doing SAI upgrade
151 * as part of warm-boot.
152 * */
153
154 if (isSwitchObjectDefaultRid(rid))
155 {
156 return true;
157 }
158
159 if (!isColdBootDiscoveredRid(rid))
160 {
161 /*
162 * This object was not discovered on cold boot so it can be removed.
163 */
164
165 return false;
166 }
167
168 auto vid = m_rid2vid.at(rid);
169
170 auto ot = syncd::VidManager::objectTypeQuery(vid);
171
172 switch (ot)
173 {
174 case SAI_OBJECT_TYPE_VLAN_MEMBER:
175 case SAI_OBJECT_TYPE_STP_PORT:
176 case SAI_OBJECT_TYPE_BRIDGE_PORT:
177
178 /*
179 * Those objects were discovered during cold boot, but they can
180 * still be removed since switch allows that.
181 */
182
183 return false;
184
185 case SAI_OBJECT_TYPE_PORT:
186 case SAI_OBJECT_TYPE_QUEUE:
187 case SAI_OBJECT_TYPE_INGRESS_PRIORITY_GROUP:
188 case SAI_OBJECT_TYPE_SCHEDULER_GROUP:
189 case SAI_OBJECT_TYPE_HASH:
190
191 /*
192 * TODO: Some vendors support removing of those objects then we
193 * need to came up with different approach. Probably SaiSwitch
194 * will need to decide whether it's possible to remove object.
195 */
196
197 return true;
198
199 default:
200 break;
201 }
202
203 SWSS_LOG_WARN("can't determine wheter object %s RID %s can be removed, FIXME",
204 sai_serialize_object_type(ot).c_str(),
205 sai_serialize_object_id(rid).c_str());
206
207 return true;
208}
