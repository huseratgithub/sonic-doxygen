55std::string AttrKeyMap::constructKey(
56 _In_ sai_object_id_t switchId,
57 _In_ const sai_object_meta_key_t& metaKey,
58 _In_ uint32_t attrCount,
59 _In_ const sai_attribute_t* attrList)
60{
61 SWSS_LOG_ENTER();
62
63 if (switchId == SAI_NULL_OBJECT_ID)
64 {
65 SWSS_LOG_THROW("switchId is NULL for %s",
66 sai_serialize_object_meta_key(metaKey).c_str());
67 }
68
69 // Use map to make sure that keys will be always sorted by attr id.
70
71 std::map<int32_t, std::string> keys;
72
73 for (uint32_t idx = 0; idx < attrCount; ++idx)
74 {
75 const auto& attr = attrList[idx];
76
77 auto* md = sai_metadata_get_attr_metadata(metaKey.objecttype, attr.id);
78
79 if (!md)
80 {
81 SWSS_LOG_THROW("failed to get metadata for object type: %s and attr id: %d",
82 sai_serialize_object_id(metaKey.objecttype).c_str(),
83 attr.id);
84 }
85
86 const auto& value = attr.value;
87
88 if (!SAI_HAS_FLAG_KEY(md->flags))
89 {
90 continue;
91 }
92
93 std::string name = md->attridname + std::string(":");
94
95 switch (md->attrvaluetype)
96 {
97 case SAI_ATTR_VALUE_TYPE_UINT32_LIST: // only for port lanes
98
99 // NOTE: this list should be sorted
100
101 for (uint32_t i = 0; i < value.u32list.count; ++i)
102 {
103 name += std::to_string(value.u32list.list[i]);
104
105 if (i != value.u32list.count - 1)
106 {
107 name += ",";
108 }
109 }
110
111 break;
112
113 case SAI_ATTR_VALUE_TYPE_INT32:
114 name += std::to_string(value.s32); // if enum then get enum name?
115 break;
116
117 case SAI_ATTR_VALUE_TYPE_UINT32:
118 name += std::to_string(value.u32);
119 break;
120
121 case SAI_ATTR_VALUE_TYPE_UINT8:
122 name += std::to_string(value.u8);
123 break;
124
125 case SAI_ATTR_VALUE_TYPE_UINT16:
126 name += std::to_string(value.u16);
127 break;
128
129 case SAI_ATTR_VALUE_TYPE_OBJECT_ID:
130 name += sai_serialize_object_id(value.oid);
131 break;
132
133 default:
134
135 // NOTE: only primitive types should be considered as keys
136 SWSS_LOG_THROW("FATAL: attribute %s marked as key, but have invalid serialization type, FIXME",
137 md->attridname);
138 }
139
140 keys[md->attrid] = name;
141 }
142
143 // switch ID is added, since same key pattern is allowed on different switch objects
144
145 std::string key = sai_serialize_object_id(switchId) + ";";
146
147 for (auto& k: keys)
148 {
149 key += k.second + ";";
150 }
151
152 SWSS_LOG_DEBUG("constructed key: %s", key.c_str());
153
154 return key;
155}
