98void P4Orch::doTask(Consumer &consumer)
99{
100 SWSS_LOG_ENTER();
101
102 if (!gPortsOrch->allPortsReady())
103 {
104 return;
105 }
106
107 const std::string table_name = consumer.getTableName();
108 if (table_name != APP_P4RT_TABLE_NAME)
109 {
110 SWSS_LOG_ERROR("Incorrect table name %s (expected %s)", table_name.c_str(), APP_P4RT_TABLE_NAME);
111 return;
112 }
113
114 auto it = consumer.m_toSync.begin();
115 while (it != consumer.m_toSync.end())
116 {
117 const swss::KeyOpFieldsValuesTuple key_op_fvs_tuple = it->second;
118 const std::string key = kfvKey(key_op_fvs_tuple);
119 it = consumer.m_toSync.erase(it);
120 std::string table_name;
121 std::string key_content;
122 parseP4RTKey(key, &table_name, &key_content);
123 if (table_name.empty())
124 {
125 auto status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
126 << "Table name cannot be empty, but was empty in key: " << key;
127 SWSS_LOG_ERROR("%s", status.message().c_str());
128 m_publisher.publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
129 status);
130 continue;
131 }
132 if (m_p4TableToManagerMap.find(table_name) != m_p4TableToManagerMap.end())
133 {
134 m_p4TableToManagerMap[table_name]->enqueue(table_name, key_op_fvs_tuple);
135 }
136 else
137 {
138 if (table_name.rfind(p4orch::kTablePrefixEXT, 0) != std::string::npos)
139 {
140 m_p4TableToManagerMap[APP_P4RT_EXT_TABLES_MANAGER]->enqueue(table_name, key_op_fvs_tuple);
141 }
142 else
143 {
144 auto status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
145 << "Failed to find P4Orch Manager for " << table_name << " P4RT DB table";
146 SWSS_LOG_ERROR("%s", status.message().c_str());
147 m_publisher.publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
148 status);
149 }
150 }
151 }
152
153 for (const auto &manager : m_p4ManagerPrecedence)
154 {
155 manager->drain();
156 }
157}

183void P4Orch::handlePortStatusChangeNotification(const std::string &op, const std::string &data)
184{
185 if (op == "port_state_change")
186 {
187 uint32_t count;
188 sai_port_oper_status_notification_t *port_oper_status = nullptr;
189 sai_deserialize_port_oper_status_ntf(data, count, &port_oper_status);
190
191 for (uint32_t i = 0; i < count; i++)
192 {
193 sai_object_id_t id = port_oper_status[i].port_id;
194 sai_port_oper_status_t status = port_oper_status[i].port_state;
195
196 Port port;
197 if (!gPortsOrch->getPort(id, port))
198 {
199 SWSS_LOG_ERROR("Failed to get port object for port id 0x%" PRIx64, id);
200 continue;
201 }
202
203 // Update port oper-status in local map
204 m_wcmpManager->updatePortOperStatusMap(port.m_alias, status);
205
206 if (status == SAI_PORT_OPER_STATUS_UP)
207 {
208 m_wcmpManager->restorePrunedNextHops(port.m_alias);
209 }
210 else
211 {
212 m_wcmpManager->pruneNextHops(port.m_alias);
213 }
214 }
215
216 sai_deserialize_free_port_oper_status_ntf(count, port_oper_status);
217 }
