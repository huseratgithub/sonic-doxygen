109 Orch(cfgDb, tableNames),

112 m_cfgTunnelTable(cfgDb, CFG_TUNNEL_TABLE_NAME)

115 m_cfgPeerTable.getKeys(peer_keys);

120 m_cfgPeerTable.get(i, fvs);

134 std::vector<string> tunnel_keys;
135 m_cfgTunnelTable.getKeys(tunnel_keys);
136

155 std::string res;

159void TunnelMgr::doTask(Consumer &consumer)
160{
161 SWSS_LOG_ENTER();
162
163 string table_name = consumer.getTableName();
164
165 auto it = consumer.m_toSync.begin();
166 while (it != consumer.m_toSync.end())
167 {
168 bool task_result = true;
169
170 KeyOpFieldsValuesTuple t = it->second;
171 const std::string & op = kfvOp(t);
172
173 if (op == SET_COMMAND)
174 {
175 if (table_name == CFG_LOOPBACK_INTERFACE_TABLE_NAME)
176 {
177 task_result = doLpbkIntfTask(t);
178 }
179 else if (table_name == CFG_TUNNEL_TABLE_NAME)
180 {
181 task_result = doTunnelTask(t);
182 }
183 else if (table_name == APP_TUNNEL_ROUTE_TABLE_NAME)
184 {
185 task_result = doTunnelRouteTask(t);
186 }
187 }
188 else if (op == DEL_COMMAND)
189 {
190 if (table_name == CFG_TUNNEL_TABLE_NAME)
191 {
192 task_result = doTunnelTask(t);
193 }
194 else if (table_name == APP_TUNNEL_ROUTE_TABLE_NAME)
195 {
196 task_result = doTunnelRouteTask(t);
197 }
198 }
199 else
200 {
201 SWSS_LOG_ERROR("Unknown operation: '%s'", op.c_str());
202 }
203
204 if (task_result == true)
205 {
206 it = consumer.m_toSync.erase(it);
207 }
208 else
209 {
210 ++it;
211 }
212 }
213
214 if (!replayDone && m_tunnelReplay.empty() && WarmStart::isWarmStart())
215 {
216 finalizeWarmReboot();
217 }
218}

220bool TunnelMgr::doTunnelTask(const KeyOpFieldsValuesTuple & t)
221{
222 SWSS_LOG_ENTER();
223
224 const std::string & tunnelName = kfvKey(t);
225 const std::string & op = kfvOp(t);
226 TunnelInfo tunInfo;
227
228 for (auto fieldValue : kfvFieldsValues(t))
229 {
230 const std::string & field = fvField(fieldValue);
231 const std::string & value = fvValue(fieldValue);
232 if (field == "dst_ip")
233 {
234 tunInfo.dst_ip = value;
235 }
236 else if (field == "tunnel_type")
237 {
238 tunInfo.type = value;
239 }
240 }
241
242 if (op == SET_COMMAND)
243 {
244 if (tunInfo.type == IPINIP)
245 {
246 tunInfo.remote_ip = m_peerIp;
247
248 if (!m_peerIp.empty() && !configIpTunnel(tunInfo))
249 {
250 return false;
251 }
252 else if (m_peerIp.empty())
253 {
254 SWSS_LOG_NOTICE("Peer/Remote IP not configured");
255 }
256
257 /* If the tunnel is already in hardware (i.e. present in the replay),
258 * don't try to create it again since it will cause an OA crash
259 * (warmboot case)
260 */
261 if (m_tunnelReplay.find(tunnelName) == m_tunnelReplay.end())
262 {
263 m_appIpInIpTunnelTable.set(tunnelName, kfvFieldsValues(t));
264 }
265 }
266 m_tunnelReplay.erase(tunnelName);
267 m_tunnelCache[tunnelName] = tunInfo;
268 }
269 else
270 {
271 auto it = m_tunnelCache.find(tunnelName);
272
273 if (it == m_tunnelCache.end())
274 {
275 SWSS_LOG_ERROR("Tunnel %s not found", tunnelName.c_str());
276 return true;
277 }
278
279 tunInfo = it->second;
280 if (tunInfo.type == IPINIP)
281 {
282 m_appIpInIpTunnelTable.del(tunnelName);
283 }
284 else
285 {
286 SWSS_LOG_WARN("Tunnel %s type %s is not handled", tunnelName.c_str(), tunInfo.type.c_str());
287 }
288 m_tunnelCache.erase(tunnelName);
289 }
290
291 SWSS_LOG_NOTICE("Tunnel %s task, op %s", tunnelName.c_str(), op.c_str());
292 return true;
293}

295bool TunnelMgr::doLpbkIntfTask(const KeyOpFieldsValuesTuple & t)
296{
297 SWSS_LOG_ENTER();
298
299 vector<string> keys = tokenize(kfvKey(t), config_db_key_delimiter);
300
301 /* Skip entry with just interface name. Need to handle only IP prefix*/
302 if (keys.size() == 1)
303 {
304 return true;
305 }
306
307 string alias(keys[0]);
308 IpPrefix ipPrefix(keys[1]);
309
310 m_intfCache[alias] = ipPrefix;
311
312 if (alias == LOOPBACK_SRC && !m_tunnelCache.empty())
313 {
314 int ret = 0;
315 std::string res;
316 ret = cmdIpTunnelIfAddress(ipPrefix.to_string(), res);
317 if (ret != 0)
318 {
319 SWSS_LOG_WARN("Failed to assign IP addr for tun if %s, res %s",
320 ipPrefix.to_string().c_str(), res.c_str());
321 }
322 }
323
324 SWSS_LOG_NOTICE("Loopback intf %s saved %s", alias.c_str(), ipPrefix.to_string().c_str());
325 return true;
326}
