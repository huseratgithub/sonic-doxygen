42bool write_db_data(vector<KeyOpFieldsValuesTuple> &db_items)
43{
44 DBConnector db("APPL_DB", 0, false);
45 RedisPipeline pipeline(&db); // dtor of RedisPipeline will automatically flush data
46 unordered_map<string, ProducerStateTable> table_map;
47
48 for (auto &db_item : db_items)
49 {
50 dump_db_item(db_item);
51
52 string key = kfvKey(db_item);
53 size_t pos = key.find(name_delimiter);
54 if ((string::npos == pos) || ((key.size() - 1) == pos))
55 {
56 SWSS_LOG_ERROR("Invalid formatted hash:%s\n", key.c_str());
57 return false;
58 }
59 string table_name = key.substr(0, pos);
60 string key_name = key.substr(pos + 1);
61 auto ret = table_map.emplace(std::piecewise_construct, std::forward_as_tuple(table_name), std::forward_as_tuple(&pipeline, table_name, true));
62
63 if (kfvOp(db_item) == SET_COMMAND)
64 ret.first->second.set(key_name, kfvFieldsValues(db_item), SET_COMMAND);
65 else if (kfvOp(db_item) == DEL_COMMAND)
66 ret.first->second.del(key_name, DEL_COMMAND);
67 else
68 {
69 SWSS_LOG_ERROR("Invalid operation: %s\n", kfvOp(db_item).c_str());
70 return false;
71 }
72 }
73
74 return true;
