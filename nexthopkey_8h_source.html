160 std::string parseMplsNextHop(const std::string& str)
161 {
162 // parseMplsNextHop initializes MPLS-related member data of the NextHopKey
163 // based on content of the input param str.
164 // label_stack may be updated.
165 std::string ip_str;
166 auto keys = tokenize(str, LABELSTACK_DELIMITER);
167 if (keys.size() == 1)
168 {
169 // No MPLS info to parse
170 ip_str = str;
171 }
172 else if (keys.size() == 2)
173 {
174 // Expected MPLS format = "<outsegtype><labelstack>+<non-mpls-str>"
175 // key[0] = <outsegtype> = "swap" | "push"
176 // key[0] = <labelstack> = "<label0>/<label1>/../<labelN>"
177 // key[1] = <non-mpls-str> = returned to caller and not parsed here
178 // Example = "push10100/10101+10.0.0.3@Ethernet4"
179 label_stack = LabelStack(keys[0]);
180 ip_str = keys[1];
181 }
182 else
183 {
184 // Malformed string
185 std::string err = "Error converting " + str + " to MPLS NextHop";
186 throw std::invalid_argument(err);
187 }
188 return ip_str;
189 }
