28 bool operator < (const RoutePattern &other) const
29 {
30 // We don't compare the vrf id here because:
31 // 1. vrf id could be SAI_NULL_OBJECT_ID if the VRF name is not resolved, two pattern with different VRF name and vrf_id=SAI_NULL_OBJECT_ID
32 // and same prefix will be treat as same route pattern, which is not expected
33 // 2. vrf name must be different
34 auto vrf_name_compare = vrf_name.compare(other.vrf_name);
35 if (vrf_name_compare < 0)
36 {
37 return true;
38 }
39 else if (vrf_name_compare == 0 && ip_prefix < other.ip_prefix)
40 {
41 return true;
42 }
43 else
44 {
45 return false;
46 }
47 }
48
49 bool is_match(sai_object_id_t vrf, IpPrefix prefix) const
50 {
51 // No need compare VRF name here because:
52 // 1. If the VRF is not resolved, the vrf_id shall be SAI_NULL_OBJECT_ID, it cannot match any input vrf_id
53 // 2. If the VRF is resolved, different vrf must have different vrf id
54 if (vrf_id != vrf)
55 {
56 return false;
57 }
58
59 if (!exact_match)
60 {
61 return (ip_prefix.getMaskLength() <= prefix.getMaskLength() && ip_prefix.isAddressInSubnet(prefix.getIp()));
62 }
63 else
64 {
65 return prefix == ip_prefix;
66 }
67 }
68
69 bool is_overlap_with(const RoutePattern &other) const
70 {
71 if (this == &other)
72 {
73 return false;
74 }
75
76 if (vrf_name != other.vrf_name)
77 {
78 return false;
79 }
80
81 if (vrf_name != other.vrf_name)
82 {
83 return false;
84 }
85
86 return is_match(other.vrf_id, other.ip_prefix) || other.is_match(vrf_id, ip_prefix);
87 }
88
89 std::string to_string() const
90 {
91 std::ostringstream oss;
92 oss << "RoutePattern(vrf_id=" << vrf_id << ",ip_prefix=" << ip_prefix.to_string() << ")";
93 return oss.str();
94 }
