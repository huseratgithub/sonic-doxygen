568void MuxNbrHandler::update(NextHopKey nh, sai_object_id_t tunnelId, bool add, MuxState state)
569{
570 SWSS_LOG_INFO("Neigh %s on %s, add %d, state %d",
571 nh.ip_address.to_string().c_str(), nh.alias.c_str(), add, state);
572
573 IpPrefix pfx = nh.ip_address.to_string();
574
575 if (add)
576 {
577 if (!nh.alias.empty() && nh.alias != alias_)
578 {
579 alias_ = nh.alias;
580 }
581
582 if (neighbors_.find(nh.ip_address) != neighbors_.end())
583 {
584 return;
585 }
586
587 switch (state)
588 {
589 case MuxState::MUX_STATE_INIT:
590 neighbors_[nh.ip_address] = SAI_NULL_OBJECT_ID;
591 break;
592 case MuxState::MUX_STATE_ACTIVE:
593 neighbors_[nh.ip_address] = gNeighOrch->getLocalNextHopId(nh);
594 gNeighOrch->enableNeighbor(nh);
595 break;
596 case MuxState::MUX_STATE_STANDBY:
597 neighbors_[nh.ip_address] = tunnelId;
598 gNeighOrch->disableNeighbor(nh);
599 updateTunnelRoute(nh, true);
600 create_route(pfx, tunnelId);
601 break;
602 default:
603 SWSS_LOG_NOTICE("State '%s' not handled for nbr %s update",
604 muxStateValToString.at(state).c_str(), nh.ip_address.to_string().c_str());
605 break;
606 }
607 }
608 else
609 {
610 /* if current state is standby, remove the tunnel route */
611 if (state == MuxState::MUX_STATE_STANDBY)
612 {
613 remove_route(pfx);
614 updateTunnelRoute(nh, false);
615 }
616 neighbors_.erase(nh.ip_address);
617 }
618}

620bool MuxNbrHandler::enable(bool update_rt)
621{
622 NeighborEntry neigh;
623
624 auto it = neighbors_.begin();
625 while (it != neighbors_.end())
626 {
627 SWSS_LOG_INFO("Enabling neigh %s on %s", it->first.to_string().c_str(), alias_.c_str());
628
629 neigh = NeighborEntry(it->first, alias_);
630 if (!gNeighOrch->enableNeighbor(neigh))
631 {
632 SWSS_LOG_INFO("Enabling neigh failed for %s", neigh.ip_address.to_string().c_str());
633 return false;
634 }
635
636 /* Update NH to point to learned neighbor */
637 it->second = gNeighOrch->getLocalNextHopId(neigh);
638
639 /* Reprogram route */
640 NextHopKey nh_key = NextHopKey(it->first, alias_);
641 uint32_t num_routes = 0;
642 if (!gRouteOrch->updateNextHopRoutes(nh_key, num_routes))
643 {
644 SWSS_LOG_INFO("Update route failed for NH %s", nh_key.ip_address.to_string().c_str());
645 return false;
646 }
647
648 /* Increment ref count for new NHs */
649 gNeighOrch->increaseNextHopRefCount(nh_key, num_routes);
650
651 /*
652 * Invalidate current nexthop group and update with new NH
653 * Ref count update is not required for tunnel NH IDs (nh_removed)
654 */
655 uint32_t nh_removed, nh_added;
656 if (!gRouteOrch->invalidnexthopinNextHopGroup(nh_key, nh_removed))
657 {
658 SWSS_LOG_ERROR("Removing existing NH failed for %s", nh_key.ip_address.to_string().c_str());
659 return false;
660 }
661
662 if (!gRouteOrch->validnexthopinNextHopGroup(nh_key, nh_added))
663 {
664 SWSS_LOG_ERROR("Adding NH failed for %s", nh_key.ip_address.to_string().c_str());
665 return false;
666 }
667
668 /* Increment ref count for ECMP NH members */
669 gNeighOrch->increaseNextHopRefCount(nh_key, nh_added);
670
671 IpPrefix pfx = it->first.to_string();
672 if (update_rt)
673 {
674 if (remove_route(pfx) != SAI_STATUS_SUCCESS)
675 {
676 return false;
677 }
678 updateTunnelRoute(nh_key, false);
679 }
680
681 it++;
682 }
683
684 return true;
685}

687bool MuxNbrHandler::disable(sai_object_id_t tnh)
688{
689 NeighborEntry neigh;
690
691 auto it = neighbors_.begin();
692 while (it != neighbors_.end())
693 {
694 SWSS_LOG_INFO("Disabling neigh %s on %s", it->first.to_string().c_str(), alias_.c_str());
695
696 /* Update NH to point to Tunnel nexhtop */
697 it->second = tnh;
698
699 /* Reprogram route */
700 NextHopKey nh_key = NextHopKey(it->first, alias_);
701 uint32_t num_routes = 0;
702 if (!gRouteOrch->updateNextHopRoutes(nh_key, num_routes))
703 {
704 SWSS_LOG_INFO("Update route failed for NH %s", nh_key.ip_address.to_string().c_str());
705 return false;
706 }
707
708 /* Decrement ref count for old NHs */
709 gNeighOrch->decreaseNextHopRefCount(nh_key, num_routes);
710
711 /* Invalidate current nexthop group and update with new NH */
712 uint32_t nh_removed, nh_added;
713 if (!gRouteOrch->invalidnexthopinNextHopGroup(nh_key, nh_removed))
714 {
715 SWSS_LOG_ERROR("Removing existing NH failed for %s", nh_key.ip_address.to_string().c_str());
716 return false;
717 }
718
719 /* Decrement ref count for ECMP NH members */
720 gNeighOrch->decreaseNextHopRefCount(nh_key, nh_removed);
721
722 if (!gRouteOrch->validnexthopinNextHopGroup(nh_key, nh_added))
723 {
724 SWSS_LOG_ERROR("Adding NH failed for %s", nh_key.ip_address.to_string().c_str());
725 return false;
726 }
727
728 neigh = NeighborEntry(it->first, alias_);
729 if (!gNeighOrch->disableNeighbor(neigh))
730 {
731 SWSS_LOG_INFO("Disabling neigh failed for %s", neigh.ip_address.to_string().c_str());
732 return false;
733 }
734
735 updateTunnelRoute(nh_key, true);
736
737 IpPrefix pfx = it->first.to_string();
738 if (create_route(pfx, it->second) != SAI_STATUS_SUCCESS)
739 {
740 return false;
741 }
742
743 it++;
744 }
745
746 return true;
747}

1053void MuxOrch::updateFdb(const FdbUpdate& update)
1054{
1055 if (!update.add)
1056 {
1057 /*
1058 * For Mac aging, flush events, skip updating mux neighbors.
1059 * Instead, wait for neighbor update events
1060 */
1061 return;
1062 }
1063
1064 NeighborEntry neigh;
1065 MacAddress mac;
1066 MuxCable* ptr;
1067 for (auto nh = mux_nexthop_tb_.begin(); nh != mux_nexthop_tb_.end(); ++nh)
1068 {
1069 auto res = neigh_orch_->getNeighborEntry(nh->first, neigh, mac);
1070 if (!res || update.entry.mac != mac)
1071 {
1072 continue;
1073 }
1074
1075 if (nh->second != update.entry.port_name)
1076 {
1077 if (!nh->second.empty() && isMuxExists(nh->second))
1078 {
1079 ptr = getMuxCable(nh->second);
1080 if (ptr->isIpInSubnet(nh->first.ip_address))
1081 {
1082 continue;
1083 }
1084 nh->second = update.entry.port_name;
1085 ptr->updateNeighbor(nh->first, false);
1086 }
1087
1088 if (isMuxExists(update.entry.port_name))
1089 {
1090 ptr = getMuxCable(update.entry.port_name);
1091 ptr->updateNeighbor(nh->first, true);
1092 }
1093 }

1097void MuxOrch::updateNeighbor(const NeighborUpdate& update)
1098{
1099 if (mux_cable_tb_.empty())
1100 {
1101 return;
1102 }
1103
1104 auto standalone_tunnel_neigh_it = standalone_tunnel_neighbors_.find(update.entry.ip_address);
1105 // Handling zero MAC neighbor updates
1106 if (!update.mac)
1107 {
1108 /* For neighbors that were previously resolvable but are now unresolvable,
1109 * we expect such neighbor entries to be deleted prior to a zero MAC update
1110 * arriving for that same neighbor.
1111 */
1112
1113 if (update.add)
1114 {
1115 if (standalone_tunnel_neigh_it == standalone_tunnel_neighbors_.end())
1116 {
1117 createStandaloneTunnelRoute(update.entry.ip_address);
1118 }
1119 /* If the MAC address in the neighbor entry is zero but the neighbor IP
1120 * is already present in standalone_tunnel_neighbors_, assume we have already
1121 * added a tunnel route for it and exit early
1122 */
1123 return;
1124 }
1125 }
1126 /* If the update operation for a neighbor contains a non-zero MAC, we must
1127 * make sure to remove any existing tunnel routes to prevent conflicts.
1128 * This block also covers the case of neighbor deletion.
1129 */
1130 if (standalone_tunnel_neigh_it != standalone_tunnel_neighbors_.end())
1131 {
1132 removeStandaloneTunnelRoute(update.entry.ip_address);
1133 }
1134
1135 for (auto it = mux_cable_tb_.begin(); it != mux_cable_tb_.end(); it++)
1136 {
1137 MuxCable* ptr = it->second.get();
1138 if (ptr->isIpInSubnet(update.entry.ip_address))
1139 {
1140 ptr->updateNeighbor(update.entry, update.add);
1141 return;
1142 }
1143 }
1144
1145 string port, old_port;
1146 if (update.add && !getMuxPort(update.mac, update.entry.alias, port))
1147 {
1148 return;
1149 }
1150 else if (update.add)
1151 {
1152 /* Check if the neighbor already exists */
1153 old_port = getNexthopMuxName(update.entry);
1154
1155 /* if new port from FDB is empty or same as existing port, return and
1156 * no further handling is required
1157 */
1158 if (port.empty() || old_port == port)
1159 {
1160 addNexthop(update.entry, old_port);
1161 return;
1162 }
1163
1164 addNexthop(update.entry);
1165 }
1166 else
1167 {
1168 auto it = mux_nexthop_tb_.find(update.entry);
1169 if (it != mux_nexthop_tb_.end())
1170 {
1171 port = it->second;
1172 removeNexthop(update.entry);
1173 }
1174 }
1175
1176 MuxCable* ptr;
1177 if (!old_port.empty() && old_port != port && isMuxExists(old_port))
1178 {
1179 ptr = getMuxCable(old_port);
1180 ptr->updateNeighbor(update.entry, false);
1181 addNexthop(update.entry);
1182 }
1183
1184 if (!port.empty() && isMuxExists(port))
1185 {
1186 ptr = getMuxCable(port);
1187 ptr->updateNeighbor(update.entry, update.add);
1188 }
1189}

1273bool MuxOrch::handleMuxCfg(const Request& request)
1274{
1275 SWSS_LOG_ENTER();
1276
1277 auto srv_ip = request.getAttrIpPrefix("server_ipv4");
1278 auto srv_ip6 = request.getAttrIpPrefix("server_ipv6");
1279
1280 MuxCableType cable_type = MuxCableType::ACTIVE_STANDBY;
1281 std::set<IpAddress> skip_neighbors;
1282
1283 const auto& port_name = request.getKeyString(0);
1284 auto op = request.getOperation();
1285
1286 for (const auto &name : request.getAttrFieldNames())
1287 {
1288 if (name == "soc_ipv4")
1289 {
1290 auto soc_ip = request.getAttrIpPrefix("soc_ipv4");
1291 SWSS_LOG_NOTICE("%s: %s was added to ignored neighbor list", port_name.c_str(), soc_ip.getIp().to_string().c_str());
1292 skip_neighbors.insert(soc_ip.getIp());
1293 }
1294 else if (name == "soc_ipv6")
1295 {
1296 auto soc_ip6 = request.getAttrIpPrefix("soc_ipv6");
1297 SWSS_LOG_NOTICE("%s: %s was added to ignored neighbor list", port_name.c_str(), soc_ip6.getIp().to_string().c_str());
1298 skip_neighbors.insert(soc_ip6.getIp());
1299 }
1300 else if (name == "cable_type")
1301 {
1302 auto cable_type_str = request.getAttrString("cable_type");
1303 if (cable_type_str == "active-active")
1304 {
1305 cable_type = MuxCableType::ACTIVE_ACTIVE;
1306 }
1307 }
1308 }
1309
1310 if (op == SET_COMMAND)
1311 {
1312 if(isMuxExists(port_name))
1313 {
1314 SWSS_LOG_INFO("Mux for port '%s' already exists", port_name.c_str());
1315 return true;
1316 }
1317
1318 if (mux_peer_switch_.isZero())
1319 {
1320 SWSS_LOG_INFO("Mux Peer switch addr not yet configured, port '%s'", port_name.c_str());
1321 return false;
1322 }
1323
1324 mux_cable_tb_[port_name] = std::make_unique<MuxCable>
1325 (MuxCable(port_name, srv_ip, srv_ip6, mux_peer_switch_, cable_type));
1326 addSkipNeighbors(skip_neighbors);
1327
1328 SWSS_LOG_NOTICE("Mux entry for port '%s' was added, cable type %d", port_name.c_str(), cable_type);
1329 }
1330 else
1331 {
1332 if(!isMuxExists(port_name))
1333 {
1334 SWSS_LOG_ERROR("Mux for port '%s' does not exists", port_name.c_str());
1335 return true;
1336 }
1337
1338 removeSkipNeighbors(skip_neighbors);
1339 mux_cable_tb_.erase(port_name);
1340
1341 SWSS_LOG_NOTICE("Mux cable for port '%s' was removed", port_name.c_str());
1342 }
1343
1344 return true;
