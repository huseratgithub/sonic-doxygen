383 // For those object type who support per object stats mode, e.g. buffer pool.
384 virtual void addObject(
385 _In_ sai_object_id_t vid,
386 _In_ sai_object_id_t rid,
387 _In_ const std::vector<std::string> &idStrings,
388 _In_ const std::string &per_object_stats_mode) override
389 {
390 SWSS_LOG_ENTER();
391 sai_stats_mode_t instance_stats_mode = SAI_STATS_MODE_READ_AND_CLEAR;
392 sai_stats_mode_t effective_stats_mode;
393 // TODO: use if const expression when c++17 is supported
394 if (HasStatsMode<CounterIdsType>::value)
395 {
396 if (per_object_stats_mode == STATS_MODE_READ_AND_CLEAR)
397 {
398 instance_stats_mode = SAI_STATS_MODE_READ_AND_CLEAR;
399 }
400 else if (per_object_stats_mode == STATS_MODE_READ)
401 {
402 instance_stats_mode = SAI_STATS_MODE_READ;
403 }
404 else
405 {
406 SWSS_LOG_WARN("Stats mode %s not supported for flex counter. Using STATS_MODE_READ_AND_CLEAR", per_object_stats_mode.c_str());
407 }
408
409 effective_stats_mode = (m_groupStatsMode == SAI_STATS_MODE_READ_AND_CLEAR ||
410 instance_stats_mode == SAI_STATS_MODE_READ_AND_CLEAR) ? SAI_STATS_MODE_READ_AND_CLEAR : SAI_STATS_MODE_READ;
411 }
412 else
413 {
414 effective_stats_mode = m_groupStatsMode;
415 }
416
417 std::vector<StatType> counter_ids;
418 for (const auto &str : idStrings)
419 {
420 StatType stat;
421 deserializeStat(str.c_str(), &stat);
422 counter_ids.push_back(stat);
423 }
424
425 updateSupportedCounters(rid, counter_ids, effective_stats_mode);
426
427 std::vector<StatType> supportedIds;
428 for (auto &counter : counter_ids)
429 {
430 if (isCounterSupported(counter))
431 {
432 supportedIds.push_back(counter);
433 }
434 }
435
436 if (supportedIds.empty())
437 {
438 SWSS_LOG_NOTICE("%s %s does not has supported counters", m_name.c_str(), sai_serialize_object_id(rid).c_str());
439 return;
440 }
441
442 if (double_confirm_supported_counters)
443 {
444 std::vector<uint64_t> stats(supportedIds.size());
445 if (!collectData(rid, supportedIds, effective_stats_mode, false, stats))
446 {
447 SWSS_LOG_ERROR("%s RID %s can't provide the statistic", m_name.c_str(), sai_serialize_object_id(rid).c_str());
448 return;
449 }
450 }
451
452 // Perform a remove and re-add to simplify the logic here
453 removeObject(vid, false);
454
455 bool supportBulk;
456 // TODO: use if const expression when cpp17 is supported
457 if (HasStatsMode<CounterIdsType>::value)
458 {
459 supportBulk = false;
460 }
461 else
462 {
463 supportBulk = checkBulkCapability(vid, rid, supportedIds);
464 }
465
466 if (!supportBulk)
467 {
468 auto counter_data = std::make_shared<CounterIds<StatType>>(rid, supportedIds);
469 // TODO: use if const expression when cpp17 is supported
470 if (HasStatsMode<CounterIdsType>::value)
471 {
472 counter_data->setStatsMode(instance_stats_mode);
473 }
474 m_objectIdsMap.emplace(vid, counter_data);
475 }
476 else
477 {
478 std::sort(supportedIds.begin(), supportedIds.end());
479 auto bulkContext = getBulkStatsContext(supportedIds);
480 addBulkStatsContext(vid, rid, supportedIds, *bulkContext.get());
481 }
482 }

676 void bulkCollectData(
677 _In_ swss::Table &countersTable,
678 _Inout_ BulkContextType &ctx)
679 {
680 SWSS_LOG_ENTER();
681 auto statsMode = m_groupStatsMode == SAI_STATS_MODE_READ ? SAI_STATS_MODE_BULK_READ : SAI_STATS_MODE_BULK_READ_AND_CLEAR;
682 sai_status_t status = m_vendorSai->bulkGetStats(
683 SAI_NULL_OBJECT_ID,
684 m_objectType,
685 static_cast<uint32_t>(ctx.object_keys.size()),
686 ctx.object_keys.data(),
687 static_cast<uint32_t>(ctx.counter_ids.size()),
688 reinterpret_cast<const sai_stat_id_t *>(ctx.counter_ids.data()),
689 statsMode,
690 ctx.object_statuses.data(),
691 ctx.counters.data());
692 if (SAI_STATUS_SUCCESS != status)
693 {
694 SWSS_LOG_WARN("Failed to bulk get stats for %s: %u", m_name.c_str(), status);
695 }
696
697 std::vector<swss::FieldValueTuple> values;
698 for (size_t i = 0; i < ctx.object_keys.size(); i++)
699 {
700 if (SAI_STATUS_SUCCESS != ctx.object_statuses[i])
701 {
702 SWSS_LOG_ERROR("Failed to get stats of %s 0x%" PRIx64 ": %d", m_name.c_str(), ctx.object_keys[i].key.object_id, ctx.object_statuses[i]);
703 continue;
704 }
705 const auto &vid = ctx.object_vids[i];
706
707 for (size_t j = 0; j < ctx.counter_ids.size(); j++)
708 {
709 values.emplace_back(serializeStat(ctx.counter_ids[j]), std::to_string(ctx.counters[i * ctx.counter_ids.size() + j]));
710 }
711 countersTable.set(sai_serialize_object_id(vid), values, "");
712 values.clear();
713 }

752 SWSS_LOG_ENTER();
753 bool found = false;
754 for (auto iter = m_bulkContexts.begin(); iter != m_bulkContexts.end(); iter++)
755 {
756 auto &ctx = *iter->second.get();
757 auto vid_iter = std::find(ctx.object_vids.begin(), ctx.object_vids.end(), vid);
758 if (vid_iter == ctx.object_vids.end())
759 {
760 continue;
761 }
762 found = true;
763 auto index = std::distance(ctx.object_vids.begin(), vid_iter);
764 ctx.object_vids.erase(vid_iter);
765 if (ctx.object_vids.empty())
766 {
767 m_bulkContexts.erase(iter);
768 }
769 else
770 {
771 auto key_iter = ctx.object_keys.begin();
772 std::advance(key_iter, index);
773 ctx.object_keys.erase(key_iter);
774 ctx.counters.resize(ctx.counter_ids.size() * ctx.object_keys.size());
775 ctx.object_statuses.pop_back();
776 }
777 break;
778 }
779
780 return found;
781 }

805 void updateSupportedCounters(
806 _In_ sai_object_id_t rid,
807 _In_ const std::vector<StatType>& counter_ids,
808 _In_ sai_stats_mode_t stats_mode)
809 {
810 SWSS_LOG_ENTER();
811 if (!m_supportedCounters.empty() && !always_check_supported_counters)
812 {
813 SWSS_LOG_NOTICE("Ignore checking of supported counters");
814 return;
815 }
816
817 if (always_check_supported_counters)
818 {
819 m_supportedCounters.clear();
820 }
821
822 if (!use_sai_stats_capa_query || querySupportedCounters(rid, stats_mode) != SAI_STATUS_SUCCESS)
823 {
824 /* Fallback to legacy approach */
825 getSupportedCounters(rid, counter_ids, stats_mode);
826 }
827 }
828
829 sai_status_t querySupportedCounters(
830 _In_ sai_object_id_t rid,
831 _In_ sai_stats_mode_t stats_mode)
832 {
833 SWSS_LOG_ENTER();
834 sai_stat_capability_list_t stats_capability;
835 stats_capability.count = 0;
836 stats_capability.list = nullptr;
837
838 /* First call is to check the size needed to allocate */
839 sai_status_t status = m_vendorSai->queryStatsCapability(
840 rid,
841 m_objectType,
842 &stats_capability);
843
844 /* Second call is for query statistics capability */
845 if (status == SAI_STATUS_BUFFER_OVERFLOW)
846 {
847 std::vector<sai_stat_capability_t> statCapabilityList(stats_capability.count);
848 stats_capability.list = statCapabilityList.data();
849 status = m_vendorSai->queryStatsCapability(
850 rid,
851 m_objectType,
852 &stats_capability);
853
854 if (status != SAI_STATUS_SUCCESS)
855 {
856 SWSS_LOG_INFO("Unable to get %s supported counters for %s",
857 m_name.c_str(),
858 sai_serialize_object_id(rid).c_str());
859 }
860 else
861 {
862 for (auto statCapability: statCapabilityList)
863 {
864 auto currentStatModes = statCapability.stat_modes;
865 if (!(currentStatModes & stats_mode))
866 {
867 continue;
868 }
869
870 StatType counter = static_cast<StatType>(statCapability.stat_enum);
871 m_supportedCounters.insert(counter);
872 }
873 }
874 }
875 return status;
876 }

878 void getSupportedCounters(
879 _In_ sai_object_id_t rid,
880 _In_ const std::vector<StatType>& counter_ids,
881 _In_ sai_stats_mode_t stats_mode)
882 {
883 SWSS_LOG_ENTER();
884 std::vector<uint64_t> values(1);
885
886 for (const auto &counter : counter_ids)
887 {
888 if (isCounterSupported(counter))
889 {
890 continue;
891 }
892
893 std::vector<StatType> tmp_counter_ids {counter};
894 if (!collectData(rid, tmp_counter_ids, stats_mode, false, values))
895 {
896 continue;
897 }
898
899 m_supportedCounters.insert(counter);
900 }
901 }

909 std::map<std::vector<StatType>, std::shared_ptr<BulkContextType>> m_bulkContexts;
910};

928 void addObject(
929 _In_ sai_object_id_t vid,
930 _In_ sai_object_id_t rid,
931 _In_ const std::vector<std::string> &idStrings,
932 _In_ const std::string &per_object_stats_mode) override

935

937

940 AttrType attr;
941 deserializeAttr(str, attr);
942 attrIds.push_back(attr);
943 }
944

946 if (it != Base::m_objectIdsMap.end())
947 {
948 it->second->counter_ids = attrIds;

956 void collectData(
957 _In_ swss::Table &countersTable) override
958 {
959 SWSS_LOG_ENTER();
960
961 for (const auto &kv : Base::m_objectIdsMap)
962 {
963 const auto &vid = kv.first;
964 const auto &rid = kv.second->rid;
965 const auto &attrIds = kv.second->counter_ids;
966
967 std::vector<sai_attribute_t> attrs(attrIds.size());
968 for (size_t i = 0; i < attrIds.size(); i++)
969 {
970 attrs[i].id = attrIds[i];
971 }
972
973 // Get attr
974 sai_status_t status = Base::m_vendorSai->get(
975 Base::m_objectType,
976 rid,
977 static_cast<uint32_t>(attrIds.size()),
978 attrs.data());
979
980 if (status != SAI_STATUS_SUCCESS)
981 {
982 SWSS_LOG_ERROR("Failed to get attr of %s 0x%" PRIx64 ": %d",
983 sai_serialize_object_type(Base::m_objectType).c_str(), vid, status);
984 continue;
985 }
986
987 std::vector<swss::FieldValueTuple> values;
988 for (size_t i = 0; i != attrIds.size(); i++)
989 {
990 auto meta = sai_metadata_get_attr_metadata(Base::m_objectType, attrs[i].id);
991 if (!meta)
992 {
993 SWSS_LOG_THROW("Failed to get metadata for %s", sai_serialize_object_type(Base::m_objectType).c_str());
994 }
995 values.emplace_back(meta->attridname, sai_serialize_attr_value(*meta, attrs[i]));
996 }
997 countersTable.set(sai_serialize_object_id(vid), values, "");
998 }
999 }

1391void FlexCounter::flexCounterThreadRunFunction()
1392{
1393 SWSS_LOG_ENTER();
1394
1395 swss::DBConnector db(m_dbCounters, 0);
1396 swss::RedisPipeline pipeline(&db);
1397 swss::Table countersTable(&pipeline, COUNTERS_TABLE, true);
1398
1399 while (m_runFlexCounterThread)
1400 {
1401 MUTEX;
1402
1403 if (m_enable && !allIdsEmpty() && (m_pollInterval > 0))
1404 {
1405 auto start = std::chrono::steady_clock::now();
1406
1407 collectCounters(countersTable);
1408
1409 runPlugins(db);
1410
1411 auto finish = std::chrono::steady_clock::now();
1412
1413 uint32_t delay = static_cast<uint32_t>(
1414 std::chrono::duration_cast<std::chrono::milliseconds>(finish - start).count());
1415
1416 uint32_t correction = delay % m_pollInterval;
1417 correction = m_pollInterval - correction;
1418 MUTEX_UNLOCK; // explicit unlock
1419
1420 SWSS_LOG_DEBUG("End of flex counter thread FC %s, took %d ms", m_instanceId.c_str(), delay);
1421
1422 std::unique_lock<std::mutex> lk(m_mtxSleep);
1423
1424 m_cvSleep.wait_for(lk, std::chrono::milliseconds(correction));
1425
1426 continue;
1427 }
1428
1429 MUTEX_UNLOCK; // explicit unlock
1430
1431 // nothing to collect, wait until notified
1432 waitPoll();
1433 }
1434}
