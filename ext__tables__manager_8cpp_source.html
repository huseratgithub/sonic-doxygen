146ReturnCode ExtTablesManager::validateP4ExtTableAppDbEntry(P4ExtTableAppDbEntry &app_db_entry)
147{
148 // Perform generic APP DB entry validations. Operation specific validations
149 // will be done by the respective request process methods.
150 ReturnCode status;
151
152 TableInfo *table;
153 table = getTableInfo(app_db_entry.table_name);
154 if (table == nullptr)
155 {
156 SWSS_LOG_ERROR("Not a valid extension table %s", app_db_entry.table_name.c_str());
157 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
158 << "Not a valid extension table " << app_db_entry.table_name.c_str();
159 }
160
161 if (table->action_ref_tables.empty())
162 {
163 return ReturnCode();
164 }
165
166 ActionInfo *action;
167 for (auto app_db_action_it = app_db_entry.action_params.begin();
168 app_db_action_it != app_db_entry.action_params.end(); app_db_action_it++)

172 if (action == nullptr)

175 << "Not a valid action " << action_name.c_str()
176 << " in extension table " << app_db_entry.table_name.c_str();
177 }
178
179 if (!action->refers_to)
180 {
181 continue;
182 }
183
184 status = validateActionParamsCrossRef(app_db_entry, action);

195ReturnCodeOr<P4ExtTableAppDbEntry> ExtTablesManager::deserializeP4ExtTableEntry(
196 const std::string &table_name,
197 const std::string &key, const std::vector<swss::FieldValueTuple> &attributes)
198{
199 std::string action_name;
200
201 SWSS_LOG_ENTER();
202
203 P4ExtTableAppDbEntry app_db_entry_or = {};
204 app_db_entry_or.table_name = table_name;
205 app_db_entry_or.table_key = key;
206
207 action_name = "";
208 for (const auto &it : attributes)
209 {
210 auto field = fvField(it);
211 auto value = fvValue(it);
212
213 if (field == p4orch::kAction)
214 {
215 action_name = value;
216 continue;
217 }
218
219 const auto &tokenized_fields = tokenize(field, p4orch::kFieldDelimiter);
220 if (tokenized_fields.size() <= 1)
221 {
222 SWSS_LOG_ERROR("Unknown extension entry field");
223 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
224 << "Unknown extension entry field " << QuotedVar(field);
225 }
226
227 const auto &prefix = tokenized_fields[0];
228 if (prefix == p4orch::kActionParamPrefix)
229 {
230 const auto &param_name = tokenized_fields[1];
231 app_db_entry_or.action_params[action_name][param_name] = value;
232 continue;
233 }
234 else
235 {
236 SWSS_LOG_ERROR("Unexpected extension entry field");
237 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
238 << "Unexpected extension entry field " << QuotedVar(field);
239 }
240 }
241
242 return app_db_entry_or;
243}

247 std::string &ext_table_entry_attr)

254 {
255 TableInfo *table;
256 table = getTableInfo(app_db_entry.table_name);
257 if (!table)

259 SWSS_LOG_ERROR("extension entry for invalid table %s", app_db_entry.table_name.c_str());

261 << "extension entry for invalid table " << app_db_entry.table_name.c_str();
262 }

264 nlohmann::json j = nlohmann::json::parse(app_db_entry.table_key);

268 std::size_t pos;
269
270 match = it.key();
271 value = it.value();
272
273 prefix = p4orch::kMatchPrefix;
274 pos = match.rfind(prefix);
275 if (pos != std::string::npos)
276 {
277 match.erase(0, prefix.length());
278 }
279 else
280 {
281 SWSS_LOG_ERROR("Failed to encode match fields for sai call");
282 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to encode match fields for sai call";
283 }
284
285 prefix = p4orch::kFieldDelimiter;
286 pos = match.rfind(prefix);
287 if (pos != std::string::npos)
288 {
289 match.erase(0, prefix.length());
290 }
291 else
292 {
293 SWSS_LOG_ERROR("Failed to encode match fields for sai call");
294 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to encode match fields for sai call";
295 }
296
297 auto match_defn_it = table->match_fields.find(match);
298 if (match_defn_it == table->match_fields.end())
299 {
300 SWSS_LOG_ERROR("extension entry for invalid match field %s", match.c_str());
301 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
302 << "extension entry for invalid match field " << match.c_str();
303 }
304
305 sai_metadata_j = nlohmann::json::object({});
306 sai_metadata_j["sai_attr_value_type"] = match_defn_it->second.datatype;
307
308 sai_j = nlohmann::json::object({});
309 sai_j[match]["value"] = value;
310 sai_j[match]["sai_metadata"] = sai_metadata_j;
311
312 sai_array_j.push_back(sai_j);
313 }
314
315 for (auto app_db_action_it = app_db_entry.action_params.begin();
316 app_db_action_it != app_db_entry.action_params.end(); app_db_action_it++)
317 {
318 sai_j = nlohmann::json::object({});
319 auto action_dep_object_it = app_db_entry.action_dep_objects.find(app_db_action_it->first);
320 if (action_dep_object_it == app_db_entry.action_dep_objects.end())
321 {
322 auto action_defn_it = table->action_fields.find(app_db_action_it->first);
323 for (auto app_db_param_it = app_db_action_it->second.begin();
324 app_db_param_it != app_db_action_it->second.end(); app_db_param_it++)
325 {
326 nlohmann::json params_j = nlohmann::json::object({});
327 if (action_defn_it != table->action_fields.end())
328 {
329 auto param_defn_it = action_defn_it->second.params.find(app_db_param_it->first);
330 if (param_defn_it != action_defn_it->second.params.end())
331 {
332 sai_metadata_j = nlohmann::json::object({});
333 sai_metadata_j["sai_attr_value_type"] = param_defn_it->second.datatype;
334
335 params_j[app_db_param_it->first]["sai_metadata"] = sai_metadata_j;
336 }
337 }
338 params_j[app_db_param_it->first]["value"] = app_db_param_it->second;
339 sai_j[app_db_action_it->first].push_back(params_j);
340 }
341 }
342 else
343 {
344 auto action_dep_object = action_dep_object_it->second;
345
346 sai_metadata_j = nlohmann::json::object({});
347 sai_metadata_j["sai_attr_value_type"] = "SAI_ATTR_VALUE_TYPE_OBJECT_ID";
348
349 sai_j[app_db_action_it->first]["sai_metadata"] = sai_metadata_j;
350 sai_j[app_db_action_it->first]["value"] = action_dep_object.oid;
351 }
352
353 sai_array_j.push_back(sai_j);
354 }
355 }
356 catch (std::exception &ex)
357 {
358 SWSS_LOG_ERROR("Failed to encode table %s entry for sai call", app_db_entry.table_name.c_str());
359 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to encode table entry for sai call";
360 }
361
362 sai_entry_j = nlohmann::json::object({});
363 sai_entry_j.push_back(nlohmann::json::object_t::value_type("attributes", sai_array_j));
364 SWSS_LOG_ERROR("table: %s, sai entry: %s", app_db_entry.table_name.c_str(), sai_entry_j.dump().c_str());
365 ext_table_entry_attr = sai_entry_j.dump();
366
367 return ReturnCode();

599 << table_name.c_str() << " , entry " << table_key.c_str();
600 }
601
602
603 auto ext_table_key = KeyGenerator::generateExtTableKey(table_name, table_key);
604 status = getSaiObject(ext_table_key, object_type, key);
605 if (!status.ok())
606 {
607 SWSS_LOG_ERROR("Invalid formation of a key %s", ext_table_key.c_str());
608 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Invalid formation of a key";
609 }
610
611 uint32_t ref_count;
612 if (!m_p4OidMapper->getRefCount(object_type, key, &ref_count))
613 {
614 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Failed to get reference count for " << QuotedVar(key));
615 }
616 if (ref_count > 0)
617 {
618 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
619 << "extension entry " << QuotedVar(key)
620 << " referenced by other objects (ref_count = " << ref_count);
621 }
622 m_p4OidMapper->eraseOID(object_type, key);
623
624 for (auto action_dep_object_it = ext_table_entry->action_dep_objects.begin();
625 action_dep_object_it != ext_table_entry->action_dep_objects.end(); action_dep_object_it++)
626 {
627 auto action_dep_object = action_dep_object_it->second;
628 m_p4OidMapper->decreaseRefCount(action_dep_object.sai_object, action_dep_object.key);
629 }
630
631 if (ext_table_entry->sai_counter_oid != SAI_NULL_OBJECT_ID)
632 {
633 m_countersTable->del(ext_table_entry->db_key);
634 removeGenericCounter(ext_table_entry->sai_counter_oid);
635 }
636
637 m_extTables[table_name].erase(table_key);
638
639 return ReturnCode();
640}

697void ExtTablesManager::drain()
698{
699 SWSS_LOG_ENTER();
700 std::string table_prefix = "EXT_";
701
702 if (gP4Orch->tablesinfo) {
703 for (auto table_it = gP4Orch->tablesinfo->m_tablePrecedenceMap.begin();
704 table_it != gP4Orch->tablesinfo->m_tablePrecedenceMap.end(); ++table_it)
705 {
706 auto table_name = table_prefix + table_it->second;
707 boost::algorithm::to_upper(table_name);
708 auto it_m = m_entriesTables.find(table_name);
709 if (it_m == m_entriesTables.end())
710 {
711 continue;
712 }
713
714 for (const auto &key_op_fvs_tuple : it_m->second)
715 {
716 std::string table_name;
717 std::string table_key;
718
719 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &table_key);
720 const std::vector<swss::FieldValueTuple> &attributes = kfvFieldsValues(key_op_fvs_tuple);
721
722 if (table_name.rfind(table_prefix, 0) == std::string::npos)
723 {
724 SWSS_LOG_ERROR("Table %s is without prefix %s", table_name.c_str(), table_prefix.c_str());
725 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
726 StatusCode::SWSS_RC_INVALID_PARAM, /*replace=*/true);
727 continue;
728 }
729 table_name = table_name.substr(table_prefix.length());
730 boost::algorithm::to_lower(table_name);
731
732 ReturnCode status;
733 auto app_db_entry_or = deserializeP4ExtTableEntry(table_name, table_key, attributes);
734 if (!app_db_entry_or.ok())
735 {
736 status = app_db_entry_or.status();
737 SWSS_LOG_ERROR("Unable to deserialize APP DB entry with key %s: %s",
738 QuotedVar(kfvKey(key_op_fvs_tuple)).c_str(), status.message().c_str());
739 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
740 status, /*replace=*/true);
741 continue;
742 }
743
744 auto &app_db_entry = *app_db_entry_or;
745 status = validateP4ExtTableAppDbEntry(app_db_entry);
746 if (!status.ok())
747 {
748 SWSS_LOG_ERROR("Validation failed for extension APP DB entry with key %s: %s",
749 QuotedVar(kfvKey(key_op_fvs_tuple)).c_str(), status.message().c_str());
750 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
751 status, /*replace=*/true);
752 continue;
753 }
754
755 const std::string &operation = kfvOp(key_op_fvs_tuple);
756 if (operation == SET_COMMAND)
757 {
758 auto *ext_table_entry = getP4ExtTableEntry(app_db_entry.table_name, app_db_entry.table_key);
759 if (ext_table_entry == nullptr)
760 {
761 // Create extension entry
762 app_db_entry.db_key = kfvKey(key_op_fvs_tuple);
763 status = processAddRequest(app_db_entry);
764 }
765 else
766 {
767 // Modify existing extension entry
768 status = processUpdateRequest(app_db_entry, ext_table_entry);
769 }
770 }
771 else if (operation == DEL_COMMAND)
772 {
773 // Delete extension entry
774 status = processDeleteRequest(app_db_entry);
775 }
776 else
777 {
778 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
779 << "Unknown operation type " << QuotedVar(operation);
780 SWSS_LOG_ERROR("%s", status.message().c_str());
781 }
782 if (!status.ok())
783 {
784 SWSS_LOG_ERROR("Processing failed for extension APP_DB entry with key %s: %s",
785 QuotedVar(kfvKey(key_op_fvs_tuple)).c_str(), status.message().c_str());
786 }
787 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
788 status, /*replace=*/true);
789 }
790
791 it_m->second.clear();
792 }
793 }
794
795 // Now report error for all remaining un-processed entries
796 for (auto it_m = m_entriesTables.begin(); it_m != m_entriesTables.end(); it_m++)
797 {
798 for (const auto &key_op_fvs_tuple : it_m->second)
799 {
800 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
801 StatusCode::SWSS_RC_INVALID_PARAM, /*replace=*/true);
802 }
803
804 it_m->second.clear();
805 }
806}

809void ExtTablesManager::doExtCounterStatsTask()
810{
811 SWSS_LOG_ENTER();
812
813 if (!gP4Orch->tablesinfo)
814 {
815 return;
816 }
817
818 sai_stat_id_t stat_ids[] = { SAI_COUNTER_STAT_PACKETS, SAI_COUNTER_STAT_BYTES };
819 uint64_t stats[2];
820 std::vector<swss::FieldValueTuple> counter_stats_values;
821
822 for (auto table_it = gP4Orch->tablesinfo->m_tableInfoMap.begin();
823 table_it != gP4Orch->tablesinfo->m_tableInfoMap.end(); ++table_it)
824 {
825 if (!table_it->second.counter_bytes_enabled && !table_it->second.counter_packets_enabled)
826 {
827 continue;
828 }
829
830 auto table_name = table_it->second.name;
831 auto ext_table_it = m_extTables.find(table_name);
832 if (ext_table_it == m_extTables.end())
833 {
834 continue;
835 }
836
837 for (auto ext_table_entry_it = ext_table_it->second.begin();
838 ext_table_entry_it != ext_table_it->second.end(); ++ext_table_entry_it)
839 {
840 auto *ext_table_entry = &ext_table_entry_it->second;
841 if (ext_table_entry->sai_counter_oid == SAI_NULL_OBJECT_ID)
842 {
843 continue;
844 }
845
846 sai_status_t sai_status =
847 sai_counter_api->get_counter_stats(ext_table_entry->sai_counter_oid, 2, stat_ids, stats);
848 if (sai_status != SAI_STATUS_SUCCESS)
849 {
850 SWSS_LOG_WARN("Failed to set counters stats for extension entry %s:%s in COUNTERS_DB: ",
851 table_name.c_str(), ext_table_entry->table_key.c_str());
852 continue;
853 }
854
855 counter_stats_values.push_back(
856 swss::FieldValueTuple{P4_COUNTER_STATS_PACKETS, std::to_string(stats[0])});
857 counter_stats_values.push_back(
858 swss::FieldValueTuple{P4_COUNTER_STATS_BYTES, std::to_string(stats[1])});
859
860 // Set field value tuples for counters stats in COUNTERS_DB
861 m_countersTable->set(ext_table_entry->db_key, counter_stats_values);
862 }
863 }
