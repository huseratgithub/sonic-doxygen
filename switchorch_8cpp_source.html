351void SwitchOrch::doAppSwitchTableTask(Consumer &consumer)
352{
353 SWSS_LOG_ENTER();
354
355 auto it = consumer.m_toSync.begin();
356 while (it != consumer.m_toSync.end())
357 {
358 auto t = it->second;
359 auto op = kfvOp(t);
360 bool retry = false;
361
362 if (op == SET_COMMAND)
363 {
364 for (auto i : kfvFieldsValues(t))
365 {
366 auto attribute = fvField(i);
367
368 if (switch_non_sai_attribute_set.find(attribute) != switch_non_sai_attribute_set.end())
369 {
370 setSwitchNonSaiAttributes(i);
371 continue;
372 }
373 else if (switch_attribute_map.find(attribute) == switch_attribute_map.end())
374 {
375 // Check additionally 'switch_tunnel_attribute_map' for Switch Tunnel
376 if (switch_tunnel_attribute_map.find(attribute) == switch_tunnel_attribute_map.end())
377 {
378 SWSS_LOG_ERROR("Unsupported switch attribute %s", attribute.c_str());
379 break;
380 }
381
382 auto status = setSwitchTunnelVxlanParams(i);
383 if ((status != SAI_STATUS_SUCCESS) && (handleSaiSetStatus(SAI_API_SWITCH, status) == task_need_retry))
384 {
385 retry = true;
386 break;
387 }
388
389 continue;
390 }
391
392 auto value = fvValue(i);
393
394 sai_attribute_t attr;
395 attr.id = switch_attribute_map.at(attribute);
396
397 MacAddress mac_addr;
398 bool invalid_attr = false;
399 switch (attr.id)
400 {
401 case SAI_SWITCH_ATTR_FDB_UNICAST_MISS_PACKET_ACTION:
402 case SAI_SWITCH_ATTR_FDB_BROADCAST_MISS_PACKET_ACTION:
403 case SAI_SWITCH_ATTR_FDB_MULTICAST_MISS_PACKET_ACTION:
404 if (packet_action_map.find(value) == packet_action_map.end())
405 {
406 SWSS_LOG_ERROR("Unsupported packet action %s", value.c_str());
407 invalid_attr = true;
408 break;
409 }
410 attr.value.s32 = packet_action_map.at(value);
411 break;
412
413 case SAI_SWITCH_ATTR_ECMP_DEFAULT_HASH_SEED:
414 case SAI_SWITCH_ATTR_LAG_DEFAULT_HASH_SEED:
415 attr.value.u32 = to_uint<uint32_t>(value);
416 break;
417
418 case SAI_SWITCH_ATTR_FDB_AGING_TIME:
419 attr.value.u32 = to_uint<uint32_t>(value);
420 break;
421
422 case SAI_SWITCH_ATTR_SWITCH_SHELL_ENABLE:
423 attr.value.booldata = to_uint<bool>(value);
424 break;
425
426 case SAI_SWITCH_ATTR_VXLAN_DEFAULT_PORT:
427 attr.value.u16 = to_uint<uint16_t>(value);
428 break;
429
430 case SAI_SWITCH_ATTR_VXLAN_DEFAULT_ROUTER_MAC:
431 mac_addr = value;
432 gVxlanMacAddress = mac_addr;
433 memcpy(attr.value.mac, mac_addr.getMac(), sizeof(sai_mac_t));
434 break;
435
436 default:
437 invalid_attr = true;
438 break;
439 }
440 if (invalid_attr)
441 {
442 /* break from kfvFieldsValues for loop */
443 break;
444 }
445
446 sai_status_t status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
447 if (status != SAI_STATUS_SUCCESS)
448 {
449 SWSS_LOG_ERROR("Failed to set switch attribute %s to %s, rv:%d",
450 attribute.c_str(), value.c_str(), status);
451 retry = (handleSaiSetStatus(SAI_API_SWITCH, status) == task_need_retry);
452 break;
453 }
454
455 SWSS_LOG_NOTICE("Set switch attribute %s to %s", attribute.c_str(), value.c_str());
456 }
457 if (retry == true)
458 {
459 it++;
460 }
461 else
462 {
463 it = consumer.m_toSync.erase(it);
464 }
465 }
466 else
467 {
468 SWSS_LOG_WARN("Unsupported operation");
469 it = consumer.m_toSync.erase(it);
470 }
471 }
472}

659{

662 sai_attribute_t attr;
663 sai_status_t status;
664 std::vector<FieldValueTuple> values;

691 std::vector<int32_t> temp_list(m_numTempSensors);

709 SWSS_LOG_ERROR("ASIC sensors : failed to get SAI_SWITCH_ATTR_TEMP_LIST: %d", status);
710 }
711 }
712
713 if (m_sensorsMaxTempSupported)
714 {
715 const std::string &fieldName = "maximum_temperature";
716 values.emplace_back(fieldName, std::to_string(0));
717 m_asicSensorsTable->set("",values);
718 }
719
720 if (m_sensorsAvgTempSupported)
721 {
722 const std::string &fieldName = "average_temperature";
723 values.emplace_back(fieldName, std::to_string(0));
