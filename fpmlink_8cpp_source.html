37bool FpmLink::isRawProcessing(struct nlmsghdr *h)
38{
39 int len;
40 short encap_type = 0;
41 struct rtmsg *rtm;
42 struct rtattr *tb[RTA_MAX + 1] = {0};
43
44 rtm = (struct rtmsg *)NLMSG_DATA(h);
45
46 if (h->nlmsg_type != RTM_NEWROUTE && h->nlmsg_type != RTM_DELROUTE)
47 {
48 return false;
49 }
50
51 len = (int)(h->nlmsg_len - NLMSG_LENGTH(sizeof(struct rtmsg)));
52 if (len < 0)
53 {
54 return false;
55 }
56
57 netlink_parse_rtattr(tb, RTA_MAX, RTM_RTA(rtm), len);
58
59 if (!tb[RTA_MULTIPATH])
60 {
61 if (tb[RTA_ENCAP_TYPE])
62 {
63 encap_type = *(short *)RTA_DATA(tb[RTA_ENCAP_TYPE]);
64 }
65 }
66 else
67 {
68 /* This is a multipath route */
69 int len;
70 struct rtnexthop *rtnh = (struct rtnexthop *)RTA_DATA(tb[RTA_MULTIPATH]);
71 len = (int)RTA_PAYLOAD(tb[RTA_MULTIPATH]);
72 struct rtattr *subtb[RTA_MAX + 1];
73
74 for (;;)
75 {
76 if (len < (int)sizeof(*rtnh) || rtnh->rtnh_len > len)
77 {
78 break;
79 }
80
81 if (rtnh->rtnh_len > sizeof(*rtnh))
82 {
83 memset(subtb, 0, sizeof(subtb));
84 netlink_parse_rtattr(subtb, RTA_MAX, RTNH_DATA(rtnh),
85 (int)(rtnh->rtnh_len - sizeof(*rtnh)));
86 if (subtb[RTA_ENCAP_TYPE])
87 {
88 encap_type = *(uint16_t *)RTA_DATA(subtb[RTA_ENCAP_TYPE]);
89 break;
90 }
91 }
92
93 if (rtnh->rtnh_len == 0)
94 {
95 break;
96 }
97
98 len -= NLMSG_ALIGN(rtnh->rtnh_len);
99 rtnh = RTNH_NEXT(rtnh);
100 }
101 }
102
103 SWSS_LOG_INFO("Rx MsgType:%d Encap:%d", h->nlmsg_type, encap_type);
104
105 if (encap_type > 0)
106 {
107 return true;
108 }
109
110 return false;
111}

239void FpmLink::processFpmMessage(fpm_msg_hdr_t* hdr)
240{
241 size_t msg_len = fpm_msg_len(hdr);
242
243 if (hdr->msg_type != FPM_MSG_TYPE_NETLINK)
244 {
245 return;
246 }
247 nlmsghdr *nl_hdr = (nlmsghdr *)fpm_msg_data(hdr);
248
249 /* Read all netlink messages inside FPM message */
250 for (; NLMSG_OK (nl_hdr, msg_len); nl_hdr = NLMSG_NEXT(nl_hdr, msg_len))
251 {
252 /*
253 * EVPN Type5 Add Routes need to be process in Raw mode as they contain
254 * RMAC, VLAN and L3VNI information.
255 * Where as all other route will be using rtnl api to extract information
256 * from the netlink msg.
257 */
258 bool isRaw = isRawProcessing(nl_hdr);
259
260 nl_msg *msg = nlmsg_convert(nl_hdr);
261 if (msg == NULL)
262 {
263 throw system_error(make_error_code(errc::bad_message), "Unable to convert nlmsg");
264 }
265
266 nlmsg_set_proto(msg, NETLINK_ROUTE);
267
268 if (isRaw)
269 {
270 /* EVPN Type5 Add route processing */
271 processRawMsg(nl_hdr);
272 }
273 else
274 {
275 NetDispatcher::getInstance().onNetlinkMessage(msg);
276 }
277 nlmsg_free(msg);
278 }
