93// doTask processes updates from the consumer and modifies the state of the
94// following components:
95// 1) The ASIC, by creating, modifying, and deleting debug counters
96// 2) syncd, by creating, modifying, and deleting flex counters to
97// keep track of the debug counters
98//
99// Updates can fail due to the following:
100// 1) Malformed requests: if the update contains an unknown or unsupported
101// counter type or drop reason then the update will fail
102// 2) SAI failures: if the SAI returns an error for any reason then the
103// update will fail
104// It is guaranteed that failed updates will not modify the state of the
105// system.
106//
107// In addition, updates are idempotent - repeating the same request any number
108// of times will always result in the same external behavior.
109void DebugCounterOrch::doTask(Consumer& consumer)
110{
111 SWSS_LOG_ENTER();
112
113 // Currently we depend on 1) the switch and 2) the ports being ready
114 // before we can set up the counters. If debug counters for other
115 // object types are added we may need to update this dependency.
116 if (!gPortsOrch->allPortsReady())
117 {
118 return;
119 }
120
121 auto it = consumer.m_toSync.begin();
122 while (it != consumer.m_toSync.end())
123 {
124 KeyOpFieldsValuesTuple t = it->second;
125 string key = kfvKey(t);
126 string op = kfvOp(t);
127 std::vector<FieldValueTuple> values = kfvFieldsValues(t);
128
129 auto table_name = consumer.getTableName();
130 task_process_status task_status = task_process_status::task_ignore;
131 if (table_name == CFG_DEBUG_COUNTER_TABLE_NAME)
132 {
133 if (op == SET_COMMAND)
134 {
135 try
136 {
137 task_status = installDebugCounter(key, values);
138 }
139 catch (const std::runtime_error& e)
140 {
141 SWSS_LOG_ERROR("Failed to create debug counter '%s'", key.c_str());
142 task_status = task_process_status::task_failed;
143 }
144 }
145 else if (op == DEL_COMMAND)
146 {
147 try
148 {
149 task_status = uninstallDebugCounter(key);
150 }
151 catch (const std::runtime_error& e)
152 {
153 SWSS_LOG_ERROR("Failed to delete debug counter '%s'", key.c_str());
154 task_status = task_process_status::task_failed;
155 }
156 }
157 else
158 {
159 SWSS_LOG_ERROR("Unknown operation type %s\n", op.c_str());
160 }
161 }
162 else if (table_name == CFG_DEBUG_COUNTER_DROP_REASON_TABLE_NAME)
163 {
164 string counter_name, drop_reason;
165 parseDropReasonUpdate(key, '|', &counter_name, &drop_reason);
166
167 if (op == SET_COMMAND)
168 {
169 try
170 {
171 task_status = addDropReason(counter_name, drop_reason);
172 }
173 catch (const std::runtime_error& e)
174 {
175 SWSS_LOG_ERROR("Failed to add drop reason '%s' to counter '%s'", drop_reason.c_str(), counter_name.c_str());
176 task_status = task_process_status::task_failed;
177 }
178 }
179 else if (op == DEL_COMMAND)
180 {
181 try
182 {
183 task_status = removeDropReason(counter_name, drop_reason);
184 }
185 catch (const std::runtime_error& e)
186 {
187 SWSS_LOG_ERROR("Failed to remove drop reason '%s' from counter '%s'", drop_reason.c_str(), counter_name.c_str());
188 task_status = task_process_status::task_failed;
189 }
190 }
191 else
192 {
193 SWSS_LOG_ERROR("Unknown operation type %s\n", op.c_str());
194 }
195 }
196 else
197 {
198 SWSS_LOG_ERROR("Received update from unknown table '%s'", table_name.c_str());
199 }
200
201 switch (task_status)
202 {
203 case task_process_status::task_success:
204 consumer.m_toSync.erase(it++);
205 break;
206 case task_process_status::task_ignore:
207 SWSS_LOG_WARN("Debug counters '%s' task ignored", op.c_str());
208 consumer.m_toSync.erase(it++);
209 break;
210 case task_process_status::task_need_retry:
211 SWSS_LOG_NOTICE("Failed to process debug counters '%s' task, retrying", op.c_str());
212 ++it;
213 break;
214 case task_process_status::task_failed:
215 SWSS_LOG_ERROR("Failed to process debug counters '%s' task, error(s) occurred during execution", op.c_str());
216 consumer.m_toSync.erase(it++);
217 break;
218 default:
219 SWSS_LOG_ERROR("Invalid task status %d", task_status);
220 consumer.m_toSync.erase(it++);
221 break;
222 }
223 }
224}

228// publishDropCounterCapabilities queries the SAI for available drop counter
229// capabilities on this device and publishes the information to the
230// DROP_COUNTER_CAPABILITIES table in STATE_DB.
231void DebugCounterOrch::publishDropCounterCapabilities()
232{
233 supported_ingress_drop_reasons = DropCounter::getSupportedDropReasons(SAI_DEBUG_COUNTER_ATTR_IN_DROP_REASON_LIST);
234 supported_egress_drop_reasons = DropCounter::getSupportedDropReasons(SAI_DEBUG_COUNTER_ATTR_OUT_DROP_REASON_LIST);
235 supported_counter_types = DropCounter::getSupportedCounterTypes();
236
237 string ingress_drop_reason_str = DropCounter::serializeSupportedDropReasons(supported_ingress_drop_reasons);
238 string egress_drop_reason_str = DropCounter::serializeSupportedDropReasons(supported_egress_drop_reasons);
239
240 for (auto const &counter_type : DebugCounter::getDebugCounterTypeLookup())
241 {
242 string drop_reasons;
243
244 if (!supported_counter_types.count(counter_type.first))
245 {
246 continue;
247 }
248
249 if (counter_type.first == PORT_INGRESS_DROPS || counter_type.first == SWITCH_INGRESS_DROPS)
250 {
251 drop_reasons = ingress_drop_reason_str;
252 }
253 else
254 {
255 drop_reasons = egress_drop_reason_str;
256 }
257
258 // Don't bother publishing counters that have no drop reasons
259 if (drop_reasons.empty())
260 {
261 continue;
262 }
263
264 string num_counters = std::to_string(DropCounter::getSupportedDebugCounterAmounts(counter_type.second));
265
266 // Don't bother publishing counters that aren't available.
267 if (num_counters == "0")
268 {
269 continue;
270 }
271
272 vector<FieldValueTuple> fieldValues;
273 fieldValues.push_back(FieldValueTuple("count", num_counters));
274 fieldValues.push_back(FieldValueTuple("reasons", drop_reasons));
275
276 SWSS_LOG_DEBUG("Setting '%s' capabilities to count='%s', reasons='%s'", counter_type.first.c_str(), num_counters.c_str(), drop_reasons.c_str());
277 m_debugCapabilitiesTable->set(counter_type.first, fieldValues);
278 }
279}

283// Note that this function cannot be used to re-initialize a counter. To create a counter
284// with the same name but different attributes (e.g. type, drop reasons, etc.) you will need
285// to delete the original counter first or use a function like addDropReason, if available.
286task_process_status DebugCounterOrch::installDebugCounter(const string& counter_name, const vector<FieldValueTuple>& attributes)
287{
288 SWSS_LOG_ENTER();
289
290 if (debug_counters.find(counter_name) != debug_counters.end())
291 {
292 SWSS_LOG_DEBUG("Debug counter '%s' already exists", counter_name.c_str());
293 return task_process_status::task_success;
294 }
295
296 // NOTE: this method currently assumes that all counters are drop counters.
297 // If you are adding support for a non-drop counter than it may make sense
298 // to either: a) dispatch to different handlers in doTask or b) dispatch to
299 // different helper methods in this method.
300
301 string counter_type = getDebugCounterType(attributes);
302
303 if (supported_counter_types.find(counter_type) == supported_counter_types.end())
304 {
305 SWSS_LOG_ERROR("Specified counter type '%s' is not supported.", counter_type.c_str());
306 return task_process_status::task_failed;
307 }
308
309 addFreeCounter(counter_name, counter_type);
310 reconcileFreeDropCounters(counter_name);
311
312 SWSS_LOG_NOTICE("Successfully created drop counter %s", counter_name.c_str());
313 return task_process_status::task_success;

355task_process_status DebugCounterOrch::addDropReason(const string& counter_name, const string& drop_reason)
356{
357 SWSS_LOG_ENTER();
358
359 if (!isDropReasonValid(drop_reason))
360 {
361 SWSS_LOG_ERROR("Specified drop reason '%s' is invalid.", drop_reason.c_str());
362 return task_process_status::task_failed;
363 }
364
365 if (supported_ingress_drop_reasons.find(drop_reason) == supported_ingress_drop_reasons.end() &&
366 supported_egress_drop_reasons.find(drop_reason) == supported_egress_drop_reasons.end())
367 {
368 SWSS_LOG_ERROR("Specified drop reason '%s' is not supported.", drop_reason.c_str());
369 return task_process_status::task_failed;
370 }
371
372 auto it = debug_counters.find(counter_name);
373 if (it == debug_counters.end())
374 {
375 // In order to gracefully handle the case where the drop reason updates
376 // are received before the create counter update, we keep track of reasons
377 // we've seen in the free_drop_reasons table.
378 addFreeDropReason(counter_name, drop_reason);
379 SWSS_LOG_NOTICE("Added drop reason %s to drop counter %s", drop_reason.c_str(), counter_name.c_str());
380
381 reconcileFreeDropCounters(counter_name);
382 return task_process_status::task_success;
383 }
384
385 DropCounter *counter = dynamic_cast<DropCounter*>(it->second.get());
386 counter->addDropReason(drop_reason);
387
388 SWSS_LOG_NOTICE("Added drop reason %s to drop counter %s", drop_reason.c_str(), counter_name.c_str());
389 return task_process_status::task_success;
390}

392// A drop counter must always contain at least one drop reason, so this function
393// will do nothing if you attempt to remove the last drop reason.
394task_process_status DebugCounterOrch::removeDropReason(const string& counter_name, const string& drop_reason)
395{
396 SWSS_LOG_ENTER();
397
398 if (!isDropReasonValid(drop_reason))
399 {
400 return task_failed;
401 }
402
403 auto it = debug_counters.find(counter_name);
404 if (it == debug_counters.end())
405 {
406 deleteFreeDropReason(counter_name, drop_reason);
407 return task_success;
408 }
409
410 DropCounter *counter = dynamic_cast<DropCounter*>(it->second.get());
411 const unordered_set<string>& drop_reasons = counter->getDropReasons();
412
413 if (drop_reasons.size() <= 1)
414 {
415 SWSS_LOG_WARN("Attempted to remove all drop reasons from counter '%s'", counter_name.c_str());
416 return task_ignore;
417 }
418
419 counter->removeDropReason(drop_reason);
420
421 SWSS_LOG_NOTICE("Removed drop reason %s from drop counter %s", drop_reason.c_str(), counter_name.c_str());
422 return task_success;
423}

496{

601// NOTE: At this point COUNTER_TYPE is the only field from CONFIG_DB that we care about. In the future
602// if other fields become relevant it may make sense to extend this method and return a struct with the
603// relevant fields.
604std::string DebugCounterOrch::getDebugCounterType(const vector<FieldValueTuple>& values) const
605{
606 SWSS_LOG_ENTER();
607
608 std::string counter_type;
609 for (auto attr : values)
610 {
611 std::string attr_name = fvField(attr);
612 auto supported_debug_counter_attributes = DebugCounter::getSupportedDebugCounterAttributes();
613 auto attr_name_it = supported_debug_counter_attributes.find(attr_name);
614 if (attr_name_it == supported_debug_counter_attributes.end())
615 {
616 SWSS_LOG_ERROR("Unknown debug counter attribute '%s'", attr_name.c_str());
617 continue;
618 }
619
620 std::string attr_value = fvValue(attr);
621 if (attr_name == "type")
622 {
623 auto debug_counter_type_lookup = DebugCounter::getDebugCounterTypeLookup();
624 auto counter_type_it = debug_counter_type_lookup.find(attr_value);
625 if (counter_type_it == debug_counter_type_lookup.end())
626 {
627 SWSS_LOG_ERROR("Debug counter type '%s' does not exist", attr_value.c_str());
628 throw std::runtime_error("Failed to initialize debug counter");
629 }
630
631 counter_type = counter_type_it->first;
632 }
633 }
634
635 return counter_type;
636}

673bool DebugCounterOrch::isDropReasonValid(const string& drop_reason) const
674{
675 SWSS_LOG_ENTER();
676
677 if (!DropCounter::isIngressDropReasonValid(drop_reason) &&
