26/*
27 * Purpose: Perform the operations requested by APPL_DB users.
28 *
29 * Description: Iterate over the untreated operations list and resolve them.
30 * The operations supported are SET and DEL. If an operation
31 * could not be resolved, it will either remain in the list, or be
32 * removed, depending on the case.
33 *
34 * Params: IN consumer - The cosumer object.
35 *
36 * Returns: Nothing.
37 */
38void CbfNhgOrch::doTask(Consumer& consumer)
39{
40 SWSS_LOG_ENTER();
41
42 if (!gPortsOrch->allPortsReady())
43 {
44 return;
45 }
46
47 auto it = consumer.m_toSync.begin();
48
49 while (it != consumer.m_toSync.end())
50 {
51 swss::KeyOpFieldsValuesTuple t = it->second;
52
53 string index = kfvKey(t);
54 string op = kfvOp(t);
55
56 bool success;
57 const auto &cbf_nhg_it = m_syncdNextHopGroups.find(index);
58
59 if (op == SET_COMMAND)
60 {
61 string members;
62 string selection_map;
63
64 /*
65 * Get CBF group's members and selection map.
66 */
67 for (const auto &i : kfvFieldsValues(t))
68 {
69 if (fvField(i) == "members")
70 {
71 members = fvValue(i);
72 }
73 else if (fvField(i) == "selection_map")
74 {
75 selection_map = fvValue(i);
76 }
77 }
78
79 /*
80 * Validate the data.
81 */
82 auto p = getMembers(members);
83
84 if (!p.first)
85 {
86 SWSS_LOG_ERROR("CBF next hop group %s data is invalid.",
87 index.c_str());
88 it = consumer.m_toSync.erase(it);
89 continue;
90 }
91
92 /*
93 * If the CBF group does not exist, create it.
94 */
95 if (cbf_nhg_it == m_syncdNextHopGroups.end())
96 {
97 /*
98 * If we reached the NHG limit, postpone the creation.
99 */
100 if (gRouteOrch->getNhgCount() + NhgBase::getSyncedCount() >= gRouteOrch->getMaxNhgCount())
101 {
102 SWSS_LOG_WARN("Reached next hop group limit. Postponing creation.");
103 success = false;
104 }
105 else
106 {
107 auto cbf_nhg = std::make_unique<CbfNhg>(index, p.second, selection_map);
108 success = cbf_nhg->sync();
109
110 if (success)
111 {
112 /*
113 * If the CBF NHG contains temporary NHGs as members,
114 * we have to keep checking for updates.
115 */
116 if (cbf_nhg->hasTemps())
117 {
118 success = false;
119 }
120
121 m_syncdNextHopGroups.emplace(index, NhgEntry<CbfNhg>(move(cbf_nhg)));
122 }
123 }
124 }
125 /*
126 * If the CBF group exists, update it.
127 */
128 else
129 {
130 success = cbf_nhg_it->second.nhg->update(p.second, selection_map);
131
132 /*
133 * If the CBF NHG has temporary NHGs synced, we need to keep
134 * checking this group in case they are promoted.
135 */
136 if (cbf_nhg_it->second.nhg->hasTemps())
137 {
138 success = false;
139 }
140 }
141 }
142 else if (op == DEL_COMMAND)
143 {
144 /*
145 * If there is a pending SET after this DEL operation, skip the
146 * DEL operation to perform the update instead. Otherwise, in the
147 * scenario where the DEL operation may be blocked by the ref
148 * counter, we'd end up deleting the object after the SET operation
149 * is performed, which would not reflect the desired state of the
150 * object.
151 */
152 if (consumer.m_toSync.count(it->first) > 1)
153 {
154 success = true;
155 }
156 /* If the group doesn't exist, do nothing. */
157 else if (cbf_nhg_it == m_syncdNextHopGroups.end())
158 {
159 SWSS_LOG_WARN("Deleting inexistent CBF NHG %s", index.c_str());
160 /*
161 * Mark it as a success to remove the task from the consumer.
162 */
163 success = true;
164 }
165 /* If the group does exist but is still referenced, skip.*/
166 else if (cbf_nhg_it->second.ref_count > 0)
167 {
168 SWSS_LOG_WARN("Skipping removal of CBF next hop group %s which"
169 " is still referenced", index.c_str());
170 success = false;
171 }
172 /* Otherwise, delete it. */
173 else
174 {
175 success = cbf_nhg_it->second.nhg->remove();
176
177 if (success)
178 {
179 m_syncdNextHopGroups.erase(cbf_nhg_it);
180 }
181 }
182 }
183 else
184 {
185 SWSS_LOG_WARN("Unknown operation type %s", op.c_str());
186 /* Mark the operation as successful to consume it. */
187 success = true;
188 }
189
190 /* Depending on the operation success, consume it or skip it. */
191 if (success)
192 {
193 it = consumer.m_toSync.erase(it);
194 }
195 else
196 {
197 ++it;
198 }
199 }
200}

202/*
203 * Purpose: Validate the CBF members.
204 *
205 * Params: IN members - The members string to validate.
206 *
207 * Returns: Pair where:
208 * - the first element is a bool, representing if the members are
209 * valid or not
210 * - the second element is a vector of members
211 */
212pair<bool, vector<string>> CbfNhgOrch::getMembers(const string &members)
213{
214 SWSS_LOG_ENTER();
215
216 auto members_vec = swss::tokenize(members, ',');
217 set<string> members_set(members_vec.begin(), members_vec.end());
218 bool success = true;
219
220 /*
221 * Verify that the members list is not empty
222 */
223 if (members_set.empty())
224 {
225 SWSS_LOG_ERROR("CBF next hop group members list is empty.");
226 success = false;
227 }
228 /*
229 * Verify that the members are unique.
230 */
231 else if (members_set.size() != members_vec.size())
232 {
233 SWSS_LOG_ERROR("CBF next hop group members are not unique.");
234 success = false;
235 }
236
237 return {success, members_vec};
238}

279/*
280 * Purpose: Sync the CBF NHG over SAI, getting a SAI ID.
281 *
282 * Params: None.
283 *
284 * Returns: true, if the operation was successful,
285 * false, otherwise.
286 */
287bool CbfNhg::sync()
288{
289 SWSS_LOG_ENTER();
290
291 /* If the group is already synced, exit. */
292 if (isSynced())
293 {
294 return true;
295 }
296
297 /* Create the CBF next hop group over SAI. */
298 sai_attribute_t nhg_attr;
299 vector<sai_attribute_t> nhg_attrs;
300
301 nhg_attr.id = SAI_NEXT_HOP_GROUP_ATTR_TYPE;
302 nhg_attr.value.u32 = SAI_NEXT_HOP_GROUP_TYPE_CLASS_BASED;
303 nhg_attrs.push_back(move(nhg_attr));
304
305 /* Add the number of members. */
306 nhg_attr.id = SAI_NEXT_HOP_GROUP_ATTR_CONFIGURED_SIZE;
307 assert(m_members.size() <= UINT32_MAX);
308 nhg_attr.value.u32 = static_cast<sai_uint32_t>(m_members.size());
309 nhg_attrs.push_back(move(nhg_attr));
310
311 if (nhg_attr.value.u32 > gNhgMapOrch->getMaxNumFcs())
312 {
313 /* If there are more members than FCs then this may be an error, as some members won't be used. */
314 SWSS_LOG_WARN("More CBF NHG members configured than supported Forwarding Classes");
315 }
316
317 /* Add the selection map to the attributes. */
318 nhg_attr.id = SAI_NEXT_HOP_GROUP_ATTR_SELECTION_MAP;
319 nhg_attr.value.oid = gNhgMapOrch->getMapId(m_selection_map);
320
321 if (nhg_attr.value.oid == SAI_NULL_OBJECT_ID)
322 {
323 SWSS_LOG_ERROR("FC to NHG map index %s does not exist", m_selection_map.c_str());
324 return false;
325 }
326
327 if ((unsigned int)gNhgMapOrch->getLargestNhIndex(m_selection_map) >= m_members.size())
328 {
329 SWSS_LOG_ERROR("FC to NHG map references more NHG members than exist in group %s", m_key.c_str());
330 return false;
331 }
332
333 nhg_attrs.push_back(move(nhg_attr));
334
335 auto status = sai_next_hop_group_api->create_next_hop_group(
336 &m_id,
337 gSwitchId,
338 static_cast<uint32_t>(nhg_attrs.size()),
339 nhg_attrs.data());
340
341 if (status != SAI_STATUS_SUCCESS)
342 {
343 SWSS_LOG_ERROR("Failed to create CBF next hop group %s, rv %d",
344 m_key.c_str(),
345 status);
346 task_process_status handle_status = gCbfNhgOrch->handleSaiCreateStatus(SAI_API_NEXT_HOP_GROUP, status);
347 if (handle_status != task_success)
348 {
349 return gCbfNhgOrch->parseHandleSaiStatusFailure(handle_status);
350 }
351 }
352
353 /* Increase the reference counter for the selection map. */
354 gNhgMapOrch->incRefCount(m_selection_map);
355
356 /*
357 * Increment the amount of programmed next hop groups. */
358 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
359 incSyncedCount();
360
361 /* Sync the group members. */
362 set<string> members;
363
364 for (const auto &member : m_members)
365 {
366 members.insert(member.first);
367 }
368
369 if (!syncMembers(members))
370 {
371 SWSS_LOG_ERROR("Failed to sync CBF next hop group %s", m_key.c_str());
372 return false;
373 }
374
375 return true;
376}

444/*
445 * Purpose: Update a CBF next hop group.
446 *
447 * Params: IN members - The new members.
448 * IN selection_map - The new selection map.
449 *
450 * Returns: true, if the update was successful,
451 * false, otherwise.
452 */
453bool CbfNhg::update(const vector<string> &members, const string &selection_map)
454{
455 SWSS_LOG_ENTER();
456
457 /*
458 * Check if we're just checking if the temporary NHG members were updated.
459 * This would happen only if the given members are the same with the
460 * existing ones and in the same order. In this scenario, we'll update
461 * just the NEXT_HOP attribute of the temporary members if necessary.
462 */
463 if (!m_temp_nhgs.empty() && hasSameMembers(members))
464 {
465 /* Iterate over the temporary NHGs and check if they were updated. */
466 for (auto member = m_temp_nhgs.begin(); member != m_temp_nhgs.end();)
467 {
468 const auto &nhg = gNhgOrch->getNhg(member->first);
469
470 /*
471 * If the NHG ID has changed since storing the first occurence,
472 * we have to update the CBF NHG member's attribute.
473 */
474 if (nhg.getId() != member->second)
475 {
476 if (!m_members.at(member->first).updateNhAttr())
477 {
478 SWSS_LOG_ERROR("Failed to update temporary next hop group "
479 "member %s of CBF next hop group %s",
480 member->first.c_str(), m_key.c_str());
481 return false;
482 }
483
484 /* If the NHG was promoted, remove it from the temporary NHG map. */
485 if (!nhg.isTemp())
486 {
487 member = m_temp_nhgs.erase(member);
488 }
489 /*
490 * If the NHG was just updated, update its current NHG ID value
491 * in the map.
492 */
493 else
494 {
495 member->second = nhg.getId();
496 ++member;
497 }
498 }
499 else
500 {
501 ++member;
502 }
503 }
504 }
505 /* If the members are different, update the whole members list. */
506 else
507 {
508 /*
509 * Because the INDEX attribute is CREATE_ONLY, we have to remove all
510 * the existing members and sync the new ones back as the order of the
511 * members can change due to some of them being removed, which would
512 * invalidate all the other members following them, or inserting a new
513 * one somewhere in the front of the existing members which would also
514 * invalidate them.
515 */
516 set<string> members_set;
517
518 for (const auto &member : m_members)
519 {
520 members_set.insert(member.first);
521 }
522
523 /* Remove the existing members. */
524 if (!removeMembers(members_set))
525 {
526 SWSS_LOG_ERROR("Failed to remove members of CBF next hop group %s",
527 m_key.c_str());
528 return false;
529 }
530 m_members.clear();
531 m_temp_nhgs.clear();
532
533 /* Add the new members. */
534 uint8_t index = 0;
535 for (const auto &member : members)
536 {
537 m_members.emplace(member, CbfNhgMember(member, index++));
538 }
539
540 if ((unsigned int)gNhgMapOrch->getLargestNhIndex(m_selection_map) >= m_members.size())
541 {
542 SWSS_LOG_ERROR("FC to NHG map references more NHG members than exist in group %s",
543 m_key.c_str());
544 return false;
545 }
546
547 /* Sync the new members. */
548 if (!syncMembers({members.begin(), members.end()}))
549 {
550 SWSS_LOG_ERROR("Failed to sync members of CBF next hop group %s",
551 m_key.c_str());
552 return false;
553 }
554 }
555
556 /* Update the group map. */
557 if (m_selection_map != selection_map)
558 {
559 sai_attribute_t nhg_attr;
560 nhg_attr.id = SAI_NEXT_HOP_GROUP_ATTR_SELECTION_MAP;
561 nhg_attr.value.oid = gNhgMapOrch->getMapId(selection_map);
562
563 if (nhg_attr.value.oid == SAI_NULL_OBJECT_ID)
564 {
565 SWSS_LOG_ERROR("NHG map %s does not exist", selection_map.c_str());
566 return false;
567 }
568
569 if ((unsigned int)gNhgMapOrch->getLargestNhIndex(selection_map) >= m_members.size())
570 {
571 SWSS_LOG_ERROR("FC to NHG map references more NHG members than exist in group %s",
572 m_key.c_str());
573 return false;
574 }
575
576 auto status = sai_next_hop_group_api->set_next_hop_group_attribute(m_id, &nhg_attr);
577
578 if (status != SAI_STATUS_SUCCESS)
579 {
580 SWSS_LOG_ERROR("Failed to update CBF next hop group %s, rv %d",
581 m_key.c_str(),
582 status);
583 return false;
584 }
585
586 /* Update the selection map and update the previous and new map ref count. */
587 gNhgMapOrch->decRefCount(m_selection_map);
588 m_selection_map = selection_map;
589 gNhgMapOrch->incRefCount(m_selection_map);
590 }
591
592 return true;
593}

595/*
596 * Purpose: Sync the given CBF group members.
597 *
598 * Params: IN members - The members to sync.
599 *
600 * Returns: true, if the operation was successful,
601 * false, otherwise.
602 */
603bool CbfNhg::syncMembers(const set<string> &members)
604{
605 SWSS_LOG_ENTER();
606
607 /* The group should be synced at this point. */
608 if (!isSynced())
609 {
610 SWSS_LOG_ERROR("Trying to sync members of CBF next hop group %s which is not synced",
611 m_key.c_str());
612 throw logic_error("Syncing members of unsynced CBF next hop group");
613 }
614
615 /*
616 * Sync all the given members. If a NHG does not exist, is not yet synced
617 * or is temporary, stop immediately.
618 */
619 ObjectBulker<sai_next_hop_group_api_t> bulker(sai_next_hop_group_api,
620 gSwitchId,
621 gMaxBulkSize);
622 unordered_map<string, sai_object_id_t> nhgm_ids;
623
624 for (const auto &key : members)
625 {
626 auto &nhgm = m_members.at(key);
627
628 /* A next hop group member can't be already synced if it was passed into the syncMembers() method. */
629 if (nhgm.isSynced())
630 {
631 SWSS_LOG_ERROR("Trying to sync already synced CBF NHG member %s in group %s",
632 nhgm.to_string().c_str(), to_string().c_str());
633 throw std::logic_error("Syncing already synced NHG member");
634 }
635 else if (nhgm.getNhgId() == SAI_NULL_OBJECT_ID)
636 {
637 SWSS_LOG_WARN("CBF NHG member %s is not yet synced", nhgm.to_string().c_str());
638 return false;
639 }
640
641 /*
642 * Check if the group exists in NhgOrch.
643 */
644 if (!gNhgOrch->hasNhg(key))
645 {
646 SWSS_LOG_ERROR("Next hop group %s in CBF next hop group %s does "
647 "not exist", key.c_str(), m_key.c_str());
648 return false;
649 }
650
651 const auto &nhg = gNhgOrch->getNhg(key);
652
653 /*
654 * Check if the group is synced.
655 */
656 if (!nhg.isSynced())
657 {
658 SWSS_LOG_ERROR("Next hop group %s in CBF next hop group %s is not"
659 " synced",
660 key.c_str(), m_key.c_str());
661 return false;
662 }
663
664 /* Create the SAI attributes for syncing the NHG as a member. */
665 auto attrs = createNhgmAttrs(nhgm);
666
667 bulker.create_entry(&nhgm_ids[key],
668 (uint32_t)attrs.size(),
669 attrs.data());
670 }
671
672 /* Flush the bulker to perform the sync. */
673 bulker.flush();
674
675 /* Iterate over the synced members and set their SAI ID. */
676 bool success = true;
677
678 for (const auto &member : nhgm_ids)
679 {
680 if (member.second == SAI_NULL_OBJECT_ID)
681 {
682 SWSS_LOG_ERROR("Failed to create CBF next hop group %s member %s",
683 m_key.c_str(), member.first.c_str());
684 success = false;
685 }
686 else
687 {
688 m_members.at(member.first).sync(member.second);
689
690 const auto &nhg = gNhgOrch->getNhg(member.first);
691 /*
692 * If the member is temporary, store it in order to check if it is
693 * promoted at some point.
694 */
695 if (nhg.isTemp())
696 {
697 m_temp_nhgs.emplace(member.first, nhg.getId());
698 }
699 }
700 }
701
702 return success;
703}
