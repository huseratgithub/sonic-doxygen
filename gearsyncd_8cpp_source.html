111bool handleGearboxConfigFromConfigDB(ProducerStateTable &p, DBConnector &cfgDb, bool warm)
112{
113 cout << "Get gearbox configuration from ConfigDB..." << endl;
114
115 Table table(&cfgDb, CFG_GEARBOX_TABLE_NAME);
116 std::vector<FieldValueTuple> ovalues;
117 std::vector<string> keys;
118 table.getKeys(keys);
119
120 if (keys.empty())
121 {
122 cout << "No gearbox configuration in ConfigDB" << endl;
123 return false;
124 }
125
126 for ( auto &k : keys )
127 {
128 table.get(k, ovalues);
129 vector<FieldValueTuple> attrs;
130 for ( auto &v : ovalues )
131 {
132 FieldValueTuple attr(v.first, v.second);
133 attrs.push_back(attr);
134 }
135 if (!warm)
136 {
137 p.set(k, attrs);
138 }
139 }
140 if (!warm)
141 {
142 notifyGearboxConfigDone(p, true);
143 }
144
145 return true;
146}
