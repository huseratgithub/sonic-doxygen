148sai_status_t SwitchBCM81724::refresh_port_list(
149 _In_ const sai_attr_metadata_t *meta)
150{
151 SWSS_LOG_ENTER();
152
153 // since now port can be added or removed, we need to update port list
154 // dynamically
155
156 sai_attribute_t attr;
157
158 m_port_list.clear();
159
160 // iterate via ASIC state to find all the ports
161
162 for (const auto& it: m_objectHash.at(SAI_OBJECT_TYPE_PORT))
163 {
164 sai_object_id_t port_id;
165 sai_deserialize_object_id(it.first, port_id);
166
167 m_port_list.push_back(port_id);
168 }
169
170 uint32_t port_count = (uint32_t)m_port_list.size();
171
172 attr.id = SAI_SWITCH_ATTR_PORT_LIST;
173 attr.value.objlist.count = port_count;
174 attr.value.objlist.list = m_port_list.data();
175
176 CHECK_STATUS(set(SAI_OBJECT_TYPE_SWITCH, m_switch_id, &attr));
177
178 attr.id = SAI_SWITCH_ATTR_PORT_NUMBER;
179 attr.value.u32 = port_count;
180
181 CHECK_STATUS(set(SAI_OBJECT_TYPE_SWITCH, m_switch_id, &attr));
182
183 SWSS_LOG_NOTICE("refreshed port list, current port number: %zu, not counting cpu port", m_port_list.size());
184
185 return SAI_STATUS_SUCCESS;
186}
