96ReturnCodeOr<std::vector<sai_attribute_t>> NextHopManager::getSaiAttrs(const P4NextHopEntry &next_hop_entry)
97{
98 std::vector<sai_attribute_t> next_hop_attrs;
99 sai_attribute_t next_hop_attr;
100
101 if (!next_hop_entry.gre_tunnel_id.empty())
102 {
103 // From centralized mapper and, get gre tunnel that next hop depends on. Get
104 // underlay router interface from gre tunnel manager,
105 sai_object_id_t tunnel_oid;
106 if (!m_p4OidMapper->getOID(SAI_OBJECT_TYPE_TUNNEL,
107 KeyGenerator::generateTunnelKey(next_hop_entry.gre_tunnel_id), &tunnel_oid))
108 {
109 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
110 << "GRE Tunnel " << QuotedVar(next_hop_entry.gre_tunnel_id) << " does not exist");
111 }
112
113 next_hop_attr.id = SAI_NEXT_HOP_ATTR_TYPE;
114 next_hop_attr.value.s32 = SAI_NEXT_HOP_TYPE_TUNNEL_ENCAP;
115 next_hop_attrs.push_back(next_hop_attr);
116
117 next_hop_attr.id = SAI_NEXT_HOP_ATTR_TUNNEL_ID;
118 next_hop_attr.value.oid = tunnel_oid;
119 next_hop_attrs.push_back(next_hop_attr);
120 }
121 else
122 {
123 // From centralized mapper, get OID of router interface that next hop
124 // depends on.
125 sai_object_id_t rif_oid;
126 if (!m_p4OidMapper->getOID(SAI_OBJECT_TYPE_ROUTER_INTERFACE,
127 KeyGenerator::generateRouterInterfaceKey(next_hop_entry.router_interface_id),
128 &rif_oid))
129 {
130 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
131 << "Router intf " << QuotedVar(next_hop_entry.router_interface_id)
132 << " does not exist");
133 }
134 next_hop_attr.id = SAI_NEXT_HOP_ATTR_TYPE;
135 next_hop_attr.value.s32 = SAI_NEXT_HOP_TYPE_IP;
136 next_hop_attrs.push_back(next_hop_attr);
137
138 next_hop_attr.id = SAI_NEXT_HOP_ATTR_ROUTER_INTERFACE_ID;
139 next_hop_attr.value.oid = rif_oid;
140 next_hop_attrs.push_back(next_hop_attr);
141 }
142
143 next_hop_attr.id = SAI_NEXT_HOP_ATTR_IP;
144 swss::copy(next_hop_attr.value.ipaddr, next_hop_entry.neighbor_id);
145 next_hop_attrs.push_back(next_hop_attr);
146
147 return next_hop_attrs;

182void NextHopManager::drain()
183{
184 SWSS_LOG_ENTER();
185
186 for (const auto &key_op_fvs_tuple : m_entries)
187 {
188 std::string table_name;
189 std::string key;
190 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &key);
191 const std::vector<swss::FieldValueTuple> &attributes = kfvFieldsValues(key_op_fvs_tuple);
192
193 ReturnCode status;
194 auto app_db_entry_or = deserializeP4NextHopAppDbEntry(key, attributes);
195 if (!app_db_entry_or.ok())
196 {
197 status = app_db_entry_or.status();
198 SWSS_LOG_ERROR("Unable to deserialize APP DB entry with key %s: %s",
199 QuotedVar(table_name + ":" + key).c_str(), status.message().c_str());
200 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
201 status,
202 /*replace=*/true);
203 continue;
204 }
205 auto &app_db_entry = *app_db_entry_or;
206
207 const std::string next_hop_key = KeyGenerator::generateNextHopKey(app_db_entry.next_hop_id);
208
209 // Fulfill the operation.
210 const std::string &operation = kfvOp(key_op_fvs_tuple);
211 if (operation == SET_COMMAND)
212 {
213 status = validateAppDbEntry(app_db_entry);
214 if (!status.ok())
215 {
216 SWSS_LOG_ERROR("Validation failed for Nexthop APP DB entry with key %s: %s",
217 QuotedVar(kfvKey(key_op_fvs_tuple)).c_str(), status.message().c_str());
218 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
219 status,
220 /*replace=*/true);
221 continue;
222 }
223 auto *next_hop_entry = getNextHopEntry(next_hop_key);
224 if (next_hop_entry == nullptr)
225 {
226 // Create new next hop.
227 status = processAddRequest(app_db_entry);
228 }
229 else
230 {
231 // Modify existing next hop.
232 status = processUpdateRequest(app_db_entry, next_hop_entry);
233 }
234 }
235 else if (operation == DEL_COMMAND)
236 {
237 // Delete next hop.
238 status = processDeleteRequest(next_hop_key);
239 }
240 else
241 {
242 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Unknown operation type " << QuotedVar(operation);
243 SWSS_LOG_ERROR("%s", status.message().c_str());
244 }
245 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple), status,
246 /*replace=*/true);
247 }
248 m_entries.clear();
249}

267ReturnCodeOr<P4NextHopAppDbEntry> NextHopManager::deserializeP4NextHopAppDbEntry(
268 const std::string &key, const std::vector<swss::FieldValueTuple> &attributes)
269{
270 SWSS_LOG_ENTER();
271
272 P4NextHopAppDbEntry app_db_entry = {};
273 app_db_entry.neighbor_id = swss::IpAddress("0.0.0.0");
274
275 try
276 {
277 nlohmann::json j = nlohmann::json::parse(key);
278 app_db_entry.next_hop_id = j[prependMatchField(p4orch::kNexthopId)];
279 }
280 catch (std::exception &ex)
281 {
282 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to deserialize next hop id";
283 }
284
285 for (const auto &it : attributes)
286 {
287 const auto &field = fvField(it);
288 const auto &value = fvValue(it);
289 if (field == prependParamField(p4orch::kRouterInterfaceId))
290 {
291 app_db_entry.router_interface_id = value;
292 }
293 else if (field == prependParamField(p4orch::kNeighborId))
294 {
295 try
296 {
297 app_db_entry.neighbor_id = swss::IpAddress(value);
298 }
299 catch (std::exception &ex)
300 {
301 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
302 << "Invalid IP address " << QuotedVar(value) << " of field " << QuotedVar(field);
303 }
304 }
305 else if (field == prependParamField(p4orch::kTunnelId))
306 {
307 app_db_entry.gre_tunnel_id = value;
308 }
309 else if (field == p4orch::kAction)
310 {
311 app_db_entry.action_str = value;
312 }
313 else if (field != p4orch::kControllerMetadata)
314 {
315 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
316 << "Unexpected field " << QuotedVar(field) << " in table entry";
317 }
318 }
319
320 return app_db_entry;
321}

337ReturnCode NextHopManager::createNextHop(P4NextHopEntry &next_hop_entry)
338{
339 SWSS_LOG_ENTER();
340
341 // Check the existence of the next hop in next hop manager and centralized
342 // mapper.
343 if (getNextHopEntry(next_hop_entry.next_hop_key) != nullptr)
344 {
345 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_EXISTS)
346 << "Next hop with key " << QuotedVar(next_hop_entry.next_hop_key)
347 << " already exists in next hop manager");
348 }
349 if (m_p4OidMapper->existsOID(SAI_OBJECT_TYPE_NEXT_HOP, next_hop_entry.next_hop_key))
350 {
351 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Next hop with key " << QuotedVar(next_hop_entry.next_hop_key)
352 << " already exists in centralized mapper");
353 }
354
355 if (!next_hop_entry.gre_tunnel_id.empty())
356 {
357 auto gre_tunnel_or = gP4Orch->getGreTunnelManager()->getConstGreTunnelEntry(
358 KeyGenerator::generateTunnelKey(next_hop_entry.gre_tunnel_id));
359 if (!gre_tunnel_or.ok())
360 {
361 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
362 << "GRE Tunnel " << QuotedVar(next_hop_entry.gre_tunnel_id)
363 << " does not exist in GRE Tunnel Manager");
364 }
365 next_hop_entry.router_interface_id = (*gre_tunnel_or).router_interface_id;
366 // BRCM requires neighbor object to be created before GRE tunnel, referring
367 // to the one in GRE tunnel object when creating next_hop_entry_with
368 // setTunnelAction
369 next_hop_entry.neighbor_id = (*gre_tunnel_or).neighbor_id;
370 }
371
372 // Neighbor doesn't have OID and the IP addr needed in next hop creation is
373 // neighbor_id, so only check neighbor existence in centralized mapper.
374 const auto neighbor_key =
375 KeyGenerator::generateNeighborKey(next_hop_entry.router_interface_id, next_hop_entry.neighbor_id);
376 if (!m_p4OidMapper->existsOID(SAI_OBJECT_TYPE_NEIGHBOR_ENTRY, neighbor_key))
377 {
378 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
379 << "Neighbor with key " << QuotedVar(neighbor_key)
380 << " does not exist in centralized mapper");
381 }
382
383 ASSIGN_OR_RETURN(std::vector<sai_attribute_t> attrs, getSaiAttrs(next_hop_entry));
384
385 // Call SAI API.
386 CHECK_ERROR_AND_LOG_AND_RETURN(sai_next_hop_api->create_next_hop(&next_hop_entry.next_hop_oid, gSwitchId,
387 (uint32_t)attrs.size(), attrs.data()),
388 "Failed to create next hop " << QuotedVar(next_hop_entry.next_hop_key));
389
390 if (!next_hop_entry.gre_tunnel_id.empty())
391 {
392 // On successful creation, increment ref count for tunnel object
393 m_p4OidMapper->increaseRefCount(SAI_OBJECT_TYPE_TUNNEL,
394 KeyGenerator::generateTunnelKey(next_hop_entry.gre_tunnel_id));
395 }
396 else
397 {
398 // On successful creation, increment ref count for router intf object
399 m_p4OidMapper->increaseRefCount(SAI_OBJECT_TYPE_ROUTER_INTERFACE,
400 KeyGenerator::generateRouterInterfaceKey(next_hop_entry.router_interface_id));
401 }
402
403 m_p4OidMapper->increaseRefCount(SAI_OBJECT_TYPE_NEIGHBOR_ENTRY, neighbor_key);
404 if (next_hop_entry.neighbor_id.isV4())
405 {
406 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEXTHOP);
407 }
408 else
409 {
410 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEXTHOP);
411 }
412
413 // Add created entry to internal table.
414 m_nextHopTable.emplace(next_hop_entry.next_hop_key, next_hop_entry);
415
416 // Add the key to OID map to centralized mapper.
417 m_p4OidMapper->setOID(SAI_OBJECT_TYPE_NEXT_HOP, next_hop_entry.next_hop_key, next_hop_entry.next_hop_oid);
418
419 return ReturnCode();
420}

446ReturnCode NextHopManager::removeNextHop(const std::string &next_hop_key)
447{
448 SWSS_LOG_ENTER();
449
450 // Check the existence of the next hop in next hop manager and centralized
451 // mapper.
452 auto *next_hop_entry = getNextHopEntry(next_hop_key);
453 if (next_hop_entry == nullptr)
454 {
455 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
456 << "Next hop with key " << QuotedVar(next_hop_key)
457 << " does not exist in next hop manager");
458 }
459
460 // Check if there is anything referring to the next hop before deletion.
461 uint32_t ref_count;
462 if (!m_p4OidMapper->getRefCount(SAI_OBJECT_TYPE_NEXT_HOP, next_hop_key, &ref_count))
463 {
464 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Failed to get reference count for next hop "
465 << QuotedVar(next_hop_key));
466 }
467 if (ref_count > 0)
468 {
469 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
470 << "Next hop " << QuotedVar(next_hop_entry->next_hop_key)
471 << " referenced by other objects (ref_count = " << ref_count);
472 }
473
474 // Call SAI API.
475 CHECK_ERROR_AND_LOG_AND_RETURN(sai_next_hop_api->remove_next_hop(next_hop_entry->next_hop_oid),
476 "Failed to remove next hop " << QuotedVar(next_hop_entry->next_hop_key));
477
478 if (!next_hop_entry->gre_tunnel_id.empty())
479 {
480 // On successful deletion, decrement ref count for tunnel object
481 m_p4OidMapper->decreaseRefCount(SAI_OBJECT_TYPE_TUNNEL,
482 KeyGenerator::generateTunnelKey(next_hop_entry->gre_tunnel_id));
483 }
484 else
485 {
486 // On successful deletion, decrement ref count for router intf object
487 m_p4OidMapper->decreaseRefCount(SAI_OBJECT_TYPE_ROUTER_INTERFACE,
488 KeyGenerator::generateRouterInterfaceKey(next_hop_entry->router_interface_id));
489 }
490
491 std::string router_interface_id = next_hop_entry->router_interface_id;
492 if (!next_hop_entry->gre_tunnel_id.empty())
493 {
494 auto gre_tunnel_or = gP4Orch->getGreTunnelManager()->getConstGreTunnelEntry(
495 KeyGenerator::generateTunnelKey(next_hop_entry->gre_tunnel_id));
496 if (!gre_tunnel_or.ok())
497 {
498 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
499 << "GRE Tunnel " << QuotedVar(next_hop_entry->gre_tunnel_id)
500 << " does not exist in GRE Tunnel Manager");
501 }
502 router_interface_id = (*gre_tunnel_or).router_interface_id;
503 }
504 m_p4OidMapper->decreaseRefCount(
505 SAI_OBJECT_TYPE_NEIGHBOR_ENTRY,
506 KeyGenerator::generateNeighborKey(router_interface_id, next_hop_entry->neighbor_id));
507 if (next_hop_entry->neighbor_id.isV4())
508 {
509 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEXTHOP);
510 }
511 else
512 {
513 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEXTHOP);
514 }
515
516 // Remove the key to OID map to centralized mapper.
517 m_p4OidMapper->eraseOID(SAI_OBJECT_TYPE_NEXT_HOP, next_hop_key);
518
519 // Remove the entry from internal table.
520 m_nextHopTable.erase(next_hop_key);
521
522 return ReturnCode();
523}

525std::string NextHopManager::verifyState(const std::string &key, const std::vector<swss::FieldValueTuple> &tuple)
526{
527 SWSS_LOG_ENTER();
528
529 auto pos = key.find_first_of(kTableKeyDelimiter);
530 if (pos == std::string::npos)
531 {
532 return std::string("Invalid key: ") + key;
533 }
534 std::string p4rt_table = key.substr(0, pos);
535 std::string p4rt_key = key.substr(pos + 1);
536 if (p4rt_table != APP_P4RT_TABLE_NAME)
537 {
538 return std::string("Invalid key: ") + key;
539 }
540 std::string table_name;
541 std::string key_content;
542 parseP4RTKey(p4rt_key, &table_name, &key_content);
543 if (table_name != APP_P4RT_NEXTHOP_TABLE_NAME)
544 {
545 return std::string("Invalid key: ") + key;
546 }
547
548 ReturnCode status;
549 auto app_db_entry_or = deserializeP4NextHopAppDbEntry(key_content, tuple);
550 if (!app_db_entry_or.ok())
551 {
552 status = app_db_entry_or.status();
553 std::stringstream msg;
554 msg << "Unable to deserialize key " << QuotedVar(key) << ": " << status.message();
555 return msg.str();
556 }
557 auto &app_db_entry = *app_db_entry_or;
558 const std::string next_hop_key = KeyGenerator::generateNextHopKey(app_db_entry.next_hop_id);
559 auto *next_hop_entry = getNextHopEntry(next_hop_key);
560 if (next_hop_entry == nullptr)
561 {
562 std::stringstream msg;
563 msg << "No entry found with key " << QuotedVar(key);
564 return msg.str();
565 }
566
567 std::string cache_result = verifyStateCache(app_db_entry, next_hop_entry);
568 std::string asic_db_result = verifyStateAsicDb(next_hop_entry);
569 if (cache_result.empty())
570 {
571 return asic_db_result;
572 }
573 if (asic_db_result.empty())
574 {
575 return cache_result;
576 }
577 return cache_result + "; " + asic_db_result;
578}

580std::string NextHopManager::verifyStateCache(const P4NextHopAppDbEntry &app_db_entry,
581 const P4NextHopEntry *next_hop_entry)
582{
583 const std::string next_hop_key = KeyGenerator::generateNextHopKey(app_db_entry.next_hop_id);
584 if (next_hop_entry->next_hop_key != next_hop_key)
585 {
586 std::stringstream msg;
587 msg << "Nexthop with key " << QuotedVar(next_hop_key) << " does not match internal cache "
588 << QuotedVar(next_hop_entry->next_hop_key) << " in nexthop manager.";
