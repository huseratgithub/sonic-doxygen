151static void lexical_convert(const std::string &buffer, MACsecSAK &sak)
152{
153 SWSS_LOG_ENTER();
154
155 bool convert_done = false;
156 memset(&sak, 0, sizeof(sak));
157 // One hex indicates 4 bits
158 size_t bit_count = buffer.length() * 4;
159 // 128 bits SAK
160 if (bit_count == 128)
161 {
162 // 128-bit SAK uses only Bytes 16..31.
163 sak.m_sak_256_enable = false;
164 convert_done = swss::hex_to_binary(
165 buffer,
166 &sak.m_sak[16],
167 16);
168 }
169 // 256 bits SAK
170 else if (bit_count == 256)
171 {
172 sak.m_sak_256_enable = true;
173 convert_done = swss::hex_to_binary(
174 buffer,
175 sak.m_sak,
176 32);
177 }
178
179 if (!convert_done)
180 {
181 SWSS_LOG_THROW("Invalid SAK %s", buffer.c_str());
182 }
183}

691 auto task = TaskMap.find(std::make_tuple(table_name, op));
692 if (task != TaskMap.end())
693 {
694 task_done = (this->*task->second)(
695 kfvKey(message),
696 kfvFieldsValues(message));
697 }
698 else
699 {
700 SWSS_LOG_ERROR(
701 "Unknown task : %s - %s",
702 table_name.c_str(),
703 op.c_str());
704 }
705
706 if (task_done == task_need_retry)
707 {
708 SWSS_LOG_DEBUG(
709 "Task %s - %s need retry",
710 table_name.c_str(),
711 op.c_str());
712 ++itr;
713 }
714 else
715 {
716 if (task_done != task_success)
717 {
718 SWSS_LOG_WARN("Task %s - %s fail",
719 table_name.c_str(),
720 op.c_str());
721 }
722 else
723 {
724 SWSS_LOG_DEBUG(
725 "Task %s - %s success",
726 table_name.c_str(),
727 op.c_str());
728 }
729
730 itr = consumer.m_toSync.erase(itr);
731 }
732 }

1012 SWSS_LOG_ENTER();
1013
1014 RecoverStack recover;
1015
1016 auto macsec_obj = m_macsec_objs.emplace(switch_id, MACsecObject());
1017 if (!macsec_obj.second)

1019 SWSS_LOG_INFO("The MACsec has been initialized at the switch 0x%" PRIx64, switch_id);
1020 return true;
1021 }
1022 recover.add_action([&]() { m_macsec_objs.erase(macsec_obj.first); });
1023
1024 sai_attribute_t attr;
1025 std::vector<sai_attribute_t> attrs;
1026
1027 attr.id = SAI_MACSEC_ATTR_DIRECTION;
1028 attr.value.s32 = SAI_MACSEC_DIRECTION_EGRESS;
1029 attrs.push_back(attr);
1030
1031 attr.id = SAI_MACSEC_ATTR_PHYSICAL_BYPASS_ENABLE;
1032 attr.value.booldata = true;
1033 attrs.push_back(attr);
1034
1035 sai_status_t status = sai_macsec_api->create_macsec(
1036 &macsec_obj.first->second.m_egress_id,
1037 switch_id,
1038 static_cast<uint32_t>(attrs.size()),
1039 attrs.data());
1040 if (status != SAI_STATUS_SUCCESS)
1041 {
1042 SWSS_LOG_WARN("Cannot initialize MACsec egress object at the switch 0x%" PRIx64, switch_id);
1043 task_process_status handle_status = handleSaiCreateStatus(SAI_API_MACSEC, status);
1044 if (handle_status != task_success)
1045 {
1046 return parseHandleSaiStatusFailure(handle_status);
1047 }
1048 }
1049 recover.add_action([&]() { sai_macsec_api->remove_macsec(macsec_obj.first->second.m_egress_id); });
1050
1051 attrs.clear();
1052 attr.id = SAI_MACSEC_ATTR_DIRECTION;
1053 attr.value.s32 = SAI_MACSEC_DIRECTION_INGRESS;
1054 attrs.push_back(attr);
1055
1056 attr.id = SAI_MACSEC_ATTR_PHYSICAL_BYPASS_ENABLE;
1057 attr.value.booldata = true;
1058 attrs.push_back(attr);
1059
1060 status = sai_macsec_api->create_macsec(
1061 &macsec_obj.first->second.m_ingress_id,
1062 switch_id,
1063 static_cast<uint32_t>(attrs.size()),
1064 attrs.data());
1065 if (status != SAI_STATUS_SUCCESS)
1066 {
1067 SWSS_LOG_WARN("Cannot initialize MACsec ingress object at the switch 0x%" PRIx64, switch_id);
1068 task_process_status handle_status = handleSaiCreateStatus(SAI_API_MACSEC, status);
1069 if (handle_status != task_success)
1070 {
1071 return parseHandleSaiStatusFailure(handle_status);
1072 }
1073 }
1074 recover.add_action([&]() { sai_macsec_api->remove_macsec(macsec_obj.first->second.m_ingress_id); });
1075
1076 attrs.clear();
1077 attr.id = SAI_MACSEC_ATTR_SCI_IN_INGRESS_MACSEC_ACL;
1078 attrs.push_back(attr);
1079 status = sai_macsec_api->get_macsec_attribute(
1080 macsec_obj.first->second.m_ingress_id,
1081 static_cast<uint32_t>(attrs.size()),
1082 attrs.data());
1083 if (status != SAI_STATUS_SUCCESS)
1084 {
1085 SWSS_LOG_WARN(
1086 "Cannot get MACsec attribution SAI_MACSEC_ATTR_SCI_IN_INGRESS_MACSEC_ACL at the switch 0x%" PRIx64,
1087 switch_id);
1088 task_process_status handle_status = handleSaiGetStatus(SAI_API_MACSEC, status);
1089 if (handle_status != task_process_status::task_success)
1090 {
1091 return false;
1092 }
1093 }
1094 macsec_obj.first->second.m_sci_in_ingress_macsec_acl = attrs.front().value.booldata;
1095
1096 attrs.clear();
1097 attr.id = SAI_MACSEC_ATTR_MAX_SECURE_ASSOCIATIONS_PER_SC;
1098 attrs.push_back(attr);
1099 status = sai_macsec_api->get_macsec_attribute(
1100 macsec_obj.first->second.m_ingress_id,
1101 static_cast<uint32_t>(attrs.size()),
1102 attrs.data());
1103 if (status != SAI_STATUS_SUCCESS)
1104 {
1105 // Default to 4 if SAI_MACSEC_ATTR_MAX_SECURE_ASSOCIATION_PER_SC isn't supported
1106 macsec_obj.first->second.m_max_sa_per_sc = 4;
1107 } else {
1108 switch (attrs.front().value.s32)
1109 {
1110 case SAI_MACSEC_MAX_SECURE_ASSOCIATIONS_PER_SC_TWO:
1111 macsec_obj.first->second.m_max_sa_per_sc = 2;
1112 break;
1113 case SAI_MACSEC_MAX_SECURE_ASSOCIATIONS_PER_SC_FOUR:
1114 macsec_obj.first->second.m_max_sa_per_sc = 4;
1115 break;
1116 default:
1117 SWSS_LOG_WARN( "Unsupported value returned from SAI_MACSEC_ATTR_MAX_SECURE_ASSOCIATION_PER_SC" );
1118 return false;
1119 }
1120 }
1121
1122 recover.clear();
1123 return true;
1124}

1708bool MACsecOrch::createMACsecSC(
1709 MACsecPort &macsec_port,
1710 const std::string &port_name,
1711 const TaskArgs &sc_attr,
1712 const MACsecObject &macsec_obj,
1713 sai_uint64_t sci,
1714 sai_object_id_t switch_id,
1715 sai_macsec_direction_t direction)
1716{
1717 SWSS_LOG_ENTER();
1718
1719 RecoverStack recover;
1720
1721 const std::string port_sci = swss::join(':', port_name, MACsecSCI(sci));
1722
1723 auto scs =
1724 (direction == SAI_MACSEC_DIRECTION_EGRESS)
1725 ? &macsec_port.m_egress_scs
1726 : &macsec_port.m_ingress_scs;
1727 auto sc_itr = scs->emplace(
1728 std::piecewise_construct,
1729 std::forward_as_tuple(sci),
1730 std::forward_as_tuple());
1731 if (!sc_itr.second)
1732 {
1733 SWSS_LOG_ERROR("The SC %s has been created.", port_sci.c_str());
1734 return false;
1735 }
1736 recover.add_action([scs, sc_itr]() {
1737 scs->erase(sc_itr.first->first);
1738 });
1739 auto sc = &sc_itr.first->second;
1740
1741 if (direction == SAI_MACSEC_DIRECTION_EGRESS)
1742 {
1743 get_value(sc_attr, "encoding_an", sc->m_encoding_an);
1744 }
1745
1746 sc->m_flow_id = SAI_NULL_OBJECT_ID;
1747 // If SCI can only be used as ACL field
1748 // Which means one flow can be associated with only one ACL entry and one SC.
1749 if (macsec_obj.m_sci_in_ingress_macsec_acl)
1750 {
1751 if (!createMACsecFlow(sc->m_flow_id, switch_id, direction))
1752 {
1753 SWSS_LOG_WARN("Cannot create MACsec Flow");
1754 return false;
1755 }
1756 recover.add_action([this, sc]() { this->deleteMACsecFlow(sc->m_flow_id); });
1757 }
1758 else
1759 {
1760 sc->m_flow_id =
1761 (direction == SAI_MACSEC_DIRECTION_EGRESS)
1762 ? macsec_port.m_egress_flow_id
1763 : macsec_port.m_ingress_flow_id;
1764 }
1765
1766 if (!createMACsecSC(
1767 sc->m_sc_id,
1768 switch_id,
1769 direction,
1770 sc->m_flow_id,
1771 sci,
1772 macsec_port.m_enable_encrypt,
1773 macsec_port.m_sci_in_sectag,
1774 macsec_port.m_cipher_suite))
1775 {
1776 SWSS_LOG_WARN("Create MACsec SC %s fail.", port_sci.c_str());
1777 return false;
1778 }
1779 recover.add_action([this, sc]() { this->deleteMACsecSC(sc->m_sc_id); });
1780
1781 auto table =
1782 (direction == SAI_MACSEC_DIRECTION_EGRESS)
1783 ? &macsec_port.m_egress_acl_table
1784 : &macsec_port.m_ingress_acl_table;
1785 if (table->m_available_acl_priorities.empty())
1786 {
1787 SWSS_LOG_WARN("Available ACL priorities have been exhausted.");
1788 return false;
1789 }
1790 sc->m_acl_priority = *(table->m_available_acl_priorities.begin());
1791 table->m_available_acl_priorities.erase(table->m_available_acl_priorities.begin());
1792 if (!createMACsecACLDataEntry(
1793 sc->m_entry_id,
1794 table->m_table_id,
1795 switch_id,
1796 macsec_port.m_sci_in_sectag,
1797 sci,
1798 sc->m_acl_priority))
1799 {
1800 SWSS_LOG_WARN("Cannot create ACL Data entry");
1801 return false;
1802 }

1804 deleteMACsecACLEntry(sc->m_entry_id);
1805 table->m_available_acl_priorities.insert(sc->m_acl_priority);
1806 });
1807
1808 SWSS_LOG_NOTICE("MACsec SC %s is created.", port_sci.c_str());
1809
1810 std::vector<FieldValueTuple> fvVector;
1811 fvVector.emplace_back("state", "ok");

1813 {
1814 m_state_macsec_egress_sc.set(swss::join('|', port_name, MACsecSCI(sci)), fvVector);
1815 }
1816 else
1817 {
1818 m_state_macsec_ingress_sc.set(swss::join('|', port_name, MACsecSCI(sci)), fvVector);
1819 }
1820
1821 recover.clear();
1822 return true;
1823}
