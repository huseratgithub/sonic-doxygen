52sai_status_t Sai::initialize(
53 _In_ uint64_t flags,
54 _In_ const sai_service_method_table_t *service_method_table)
55{
56 MUTEX();
57 SWSS_LOG_ENTER();
58
59 if (m_apiInitialized)
60 {
61 SWSS_LOG_ERROR("%s: api already initialized", __PRETTY_FUNCTION__);
62
63 return SAI_STATUS_FAILURE;
64 }
65
66 if (flags != 0)
67 {
68 SWSS_LOG_ERROR("invalid flags passed to SAI API initialize");
69
70 return SAI_STATUS_INVALID_PARAMETER;
71 }
72
73 if ((service_method_table == NULL) ||
74 (service_method_table->profile_get_next_value == NULL) ||
75 (service_method_table->profile_get_value == NULL))
76 {
77 SWSS_LOG_ERROR("invalid service_method_table handle passed to SAI API initialize");
78
79 return SAI_STATUS_INVALID_PARAMETER;
80 }
81
82 memcpy(&m_service_method_table, service_method_table, sizeof(m_service_method_table));
83
84 m_recorder = std::make_shared<Recorder>();
85
86 const char* contextConfig = service_method_table->profile_get_value(0, SAI_REDIS_KEY_CONTEXT_CONFIG);
87
88 auto ccc = ContextConfigContainer::loadFromFile(contextConfig);
89
90 for (auto&cc: ccc->getAllContextConfigs())
91 {
92 auto context = std::make_shared<Context>(cc, m_recorder, std::bind(&Sai::handle_notification, this, _1, _2));
93
94 m_contextMap[cc->m_guid] = context;
95 }
96
97 m_apiInitialized = true;
98
99 return SAI_STATUS_SUCCESS;
100}

122sai_status_t Sai::create(
123 _In_ sai_object_type_t objectType,
124 _Out_ sai_object_id_t* objectId,
125 _In_ sai_object_id_t switchId,
126 _In_ uint32_t attr_count,
127 _In_ const sai_attribute_t *attr_list)
128{
129 MUTEX();
130 SWSS_LOG_ENTER();
131 REDIS_CHECK_API_INITIALIZED();
132
133 REDIS_CHECK_CONTEXT(switchId);
134
135 if (objectType == SAI_OBJECT_TYPE_SWITCH && attr_count > 0 && attr_list)
136 {
137 uint32_t globalContext = 0; // default
138
139 if (attr_list[attr_count - 1].id == SAI_REDIS_SWITCH_ATTR_CONTEXT)
140 {
141 globalContext = attr_list[--attr_count].value.u32;
142 }
143
144 SWSS_LOG_NOTICE("request switch create with context %u", globalContext);
145
146 context = getContext(globalContext);
147
148 if (context == nullptr)
149 {
150 SWSS_LOG_ERROR("no global context defined at index %u", globalContext);
151
152 return SAI_STATUS_FAILURE;
153 }
154 }
155
156 auto status = context->m_meta->create(
157 objectType,
158 objectId,
159 switchId,
160 attr_count,
161 attr_list);
162
163 if (objectType == SAI_OBJECT_TYPE_SWITCH && status == SAI_STATUS_SUCCESS)
164 {
165 auto *attr = sai_metadata_get_attr_by_id(
166 SAI_SWITCH_ATTR_INIT_SWITCH,
167 attr_count,
168 attr_list);
169
170 if (attr && attr->value.booldata == false)
171 {
172 // request to connect to existing switch
173
174 SWSS_LOG_NOTICE("Sai::create call context populateMetadata");
175
176 context->populateMetadata(*objectId);
177 }
178 }
179
180 return status;
181}

195sai_status_t Sai::set(
196 _In_ sai_object_type_t objectType,
197 _In_ sai_object_id_t objectId,
198 _In_ const sai_attribute_t *attr)
199{
200 MUTEX();
201 SWSS_LOG_ENTER();
202 REDIS_CHECK_API_INITIALIZED();
203
204 if (RedisRemoteSaiInterface::isRedisAttribute(objectType, attr))
205 {
206 if (attr->id == SAI_REDIS_SWITCH_ATTR_REDIS_COMMUNICATION_MODE)
207 {
208 // Since communication mode destroys current channel and creates
209 // new one, it may happen, that during this SET api execution when
210 // api mutex is acquired, channel destructor will be blocking on
211 // thread->join() and channel thread will start processing
212 // incoming notification. That notification will be synchronized
213 // with api mutex and will cause deadlock, so to mitigate this
214 // scenario we will unlock api mutex here.
215 //
216 // This is not the perfect, but assuming that communication mode is
217 // changed in single thread and before switch create then we should
218 // not hit race condition.
219
220 SWSS_LOG_NOTICE("unlocking api mutex for communication mode");
221
222 MUTEX_UNLOCK();
223 }
224
225 // skip metadata if attribute is redis extension attribute
226
227 // TODO this is setting on all contexts, but maybe we want one specific?
228 // and do set on all if objectId == NULL
229
230 bool success = true;
231
232 for (auto& kvp: m_contextMap)
233 {
234 sai_status_t status = kvp.second->m_redisSai->set(objectType, objectId, attr);
235
236 success &= (status == SAI_STATUS_SUCCESS);
237
238 SWSS_LOG_INFO("setting attribute 0x%x status: %s",
239 attr->id,
240 sai_serialize_status(status).c_str());
241 }
242
243 return success ? SAI_STATUS_SUCCESS : SAI_STATUS_FAILURE;
244 }
245
246 REDIS_CHECK_CONTEXT(objectId);
247
248 return context->m_meta->set(objectType, objectId, attr);
249}

703/*
704 * NOTE: Notifications during switch create and switch remove.
705 *
706 * It is possible that when we create switch we will immediately start getting
707 * notifications from it, and it may happen that this switch will not be yet
708 * put to switch container and notification won't find it. But before
709 * notification will be processed it will first try to acquire mutex, so create
710 * switch function will end and switch will be put inside container.
711 *
712 * Similar it can happen that we receive notification when we are removing
713 * switch, then switch will be removed from switch container and notification
714 * will not find existing switch, but that's ok since switch was removed, and
715 * notification can be ignored.
716 */
717
718sai_switch_notifications_t Sai::handle_notification(
719 _In_ std::shared_ptr<Notification> notification,
720 _In_ Context* context)
721{
722 MUTEX();
723 SWSS_LOG_ENTER();
724
725 if (!m_apiInitialized)
726 {
727 SWSS_LOG_ERROR("%s: api not initialized", __PRETTY_FUNCTION__);
728
729 return { };
730 }
731
732 return context->m_redisSai->syncProcessNotification(notification);
733}
