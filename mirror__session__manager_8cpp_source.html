57void MirrorSessionManager::drain()
58{
59 SWSS_LOG_ENTER();
60
61 for (const auto &key_op_fvs_tuple : m_entries)
62 {
63 std::string table_name;
64 std::string key;
65 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &key);
66 const std::vector<swss::FieldValueTuple> &attributes = kfvFieldsValues(key_op_fvs_tuple);
67
68 ReturnCode status;
69 auto app_db_entry_or = deserializeP4MirrorSessionAppDbEntry(key, attributes);
70 if (!app_db_entry_or.ok())
71 {
72 status = app_db_entry_or.status();
73 SWSS_LOG_ERROR("Unable to deserialize APP DB entry with key %s: %s",
74 QuotedVar(table_name + ":" + key).c_str(), status.message().c_str());
75 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
76 status,
77 /*replace=*/true);
78 continue;
79 }
80 auto &app_db_entry = *app_db_entry_or;
81
82 const std::string mirror_session_key = KeyGenerator::generateMirrorSessionKey(app_db_entry.mirror_session_id);
83
84 // Fulfill the operation.
85 const std::string &operation = kfvOp(key_op_fvs_tuple);
86 if (operation == SET_COMMAND)
87 {
88 auto *mirror_session_entry = getMirrorSessionEntry(mirror_session_key);
89 if (mirror_session_entry == nullptr)
90 {
91 // Create new mirror session.
92 status = processAddRequest(app_db_entry);
93 }
94 else
95 {
96 // Modify existing mirror session.
97 status = processUpdateRequest(app_db_entry, mirror_session_entry);
98 }
99 }
100 else if (operation == DEL_COMMAND)
101 {
102 // Delete mirror session.
103 status = processDeleteRequest(mirror_session_key);
104 }
105 else
106 {
107 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Unknown operation type " << QuotedVar(operation);
108 SWSS_LOG_ERROR("%s", status.message().c_str());
109 }
110 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple), status,
111 /*replace=*/true);
112 }
113 m_entries.clear();
114}

182ReturnCodeOr<P4MirrorSessionAppDbEntry> MirrorSessionManager::deserializeP4MirrorSessionAppDbEntry(
183 const std::string &key, const std::vector<swss::FieldValueTuple> &attributes)
184{
185 SWSS_LOG_ENTER();
186
187 P4MirrorSessionAppDbEntry app_db_entry = {};
188
189 try
190 {
191 nlohmann::json j = nlohmann::json::parse(key);
192 app_db_entry.mirror_session_id = j[prependMatchField(p4orch::kMirrorSessionId)];
193 }
194 catch (std::exception &ex)
195 {
196 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to deserialize mirror session id";
197 }
198
199 for (const auto &it : attributes)
200 {
201 const auto &field = fvField(it);
202 const auto &value = fvValue(it);
203 if (field == prependParamField(p4orch::kPort))
204 {
205 swss::Port port;
206 if (!gPortsOrch->getPort(value, port))
207 {
208 return ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
209 << "Failed to get port info for port " << QuotedVar(value);
210 }
211 if (port.m_type != Port::Type::PHY)
212 {
213 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
214 << "Port " << QuotedVar(value) << "'s type " << port.m_type
215 << " is not physical and is invalid as destination port for "
216 "mirror packet.";
217 }
218 app_db_entry.port = value;
219 app_db_entry.has_port = true;
220 }
221 else if (field == prependParamField(p4orch::kSrcIp))
222 {
223 try
224 {
225 app_db_entry.src_ip = swss::IpAddress(value);
226 app_db_entry.has_src_ip = true;
227 }
228 catch (std::exception &ex)
229 {
230 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
231 << "Invalid IP address " << QuotedVar(value) << " of field " << QuotedVar(field);
232 }
233 }
234 else if (field == prependParamField(p4orch::kDstIp))
235 {
236 try
237 {
238 app_db_entry.dst_ip = swss::IpAddress(value);
239 app_db_entry.has_dst_ip = true;
240 }
241 catch (std::exception &ex)
242 {
243 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
244 << "Invalid IP address " << QuotedVar(value) << " of field " << QuotedVar(field);
245 }
246 }
247 else if (field == prependParamField(p4orch::kSrcMac))
248 {
249 try
250 {
251 app_db_entry.src_mac = swss::MacAddress(value);
252 app_db_entry.has_src_mac = true;
253 }
254 catch (std::exception &ex)
255 {
256 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
257 << "Invalid MAC address " << QuotedVar(value) << " of field " << QuotedVar(field);
258 }
259 }
260 else if (field == prependParamField(p4orch::kDstMac))
261 {
262 try
263 {
264 app_db_entry.dst_mac = swss::MacAddress(value);
265 app_db_entry.has_dst_mac = true;
266 }
267 catch (std::exception &ex)
268 {
269 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
270 << "Invalid MAC address " << QuotedVar(value) << " of field " << QuotedVar(field);
271 }
272 }
273 else if (field == prependParamField(p4orch::kTtl))
274 {
275 try
276 {
277 app_db_entry.ttl = static_cast<uint8_t>(std::stoul(value, 0, /*base=*/16));
278 app_db_entry.has_ttl = true;
279 }
280 catch (std::exception &ex)
281 {
282 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
283 << "Invalid TTL " << QuotedVar(value) << " of field " << QuotedVar(field);
284 }
285 }
286 else if (field == prependParamField(p4orch::kTos))
287 {
288 try
289 {
290 app_db_entry.tos = static_cast<uint8_t>(std::stoul(value, 0, /*base=*/16));
291 app_db_entry.has_tos = true;
292 }
293 catch (std::exception &ex)
294 {
295 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
296 << "Invalid TOS " << QuotedVar(value) << " of field " << QuotedVar(field);
297 }
298 }
299 else if (field == p4orch::kAction)
300 {
301 if (value != p4orch::kMirrorAsIpv4Erspan)
302 {
303 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
304 << "Action value " << QuotedVar(value) << " is not mirror_as_ipv4_erspan.";
305 }
306 }
307 else if (field != p4orch::kControllerMetadata)
308 {
309 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
310 << "Unexpected field " << QuotedVar(field) << " in table entry";
311 }
312 }
313
314 return app_db_entry;

331ReturnCode MirrorSessionManager::processAddRequest(const P4MirrorSessionAppDbEntry &app_db_entry)
332{
333 SWSS_LOG_ENTER();
334
335 ReturnCode status;
336 // Check if all required fields for add operation are given in APP DB entry.
337 if (app_db_entry.has_port && app_db_entry.has_src_ip && app_db_entry.has_dst_ip && app_db_entry.has_src_mac &&

340 P4MirrorSessionEntry mirror_session_entry(

344 status = createMirrorSession(std::move(mirror_session_entry));
345 }

347 {
348 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
349 << "Mirror session entry with mirror_session_id " << QuotedVar(app_db_entry.mirror_session_id)
350 << " doesn't specify all required fields for ADD operation.";
351 SWSS_LOG_ERROR("%s", status.message().c_str());
352 }

357ReturnCode MirrorSessionManager::createMirrorSession(P4MirrorSessionEntry mirror_session_entry)
358{
359 SWSS_LOG_ENTER();
360
361 // Check the existence of the mirror session in centralized mapper.
362 if (m_p4OidMapper->existsOID(SAI_OBJECT_TYPE_MIRROR_SESSION, mirror_session_entry.mirror_session_key))
363 {
364 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Mirror session with key "
365 << QuotedVar(mirror_session_entry.mirror_session_key)
366 << " already exists in centralized mapper");
367 }
368 // Prepare attributes for the SAI creation call.
369 ASSIGN_OR_RETURN(std::vector<sai_attribute_t> attrs, getSaiAttrs(mirror_session_entry));
370
371 // Call SAI API.
372 CHECK_ERROR_AND_LOG_AND_RETURN(
373 sai_mirror_api->create_mirror_session(&mirror_session_entry.mirror_session_oid, gSwitchId,
374 (uint32_t)attrs.size(), attrs.data()),
375 "Failed to create mirror session " << QuotedVar(mirror_session_entry.mirror_session_key));
376
377 // On successful creation, increment ref count.
378 gPortsOrch->increasePortRefCount(mirror_session_entry.port);
379
380 // Add the key to OID map to centralized mapper.
381 m_p4OidMapper->setOID(SAI_OBJECT_TYPE_MIRROR_SESSION, mirror_session_entry.mirror_session_key,
382 mirror_session_entry.mirror_session_oid);
383
384 // Add created entry to internal table.
385 m_mirrorSessionTable.emplace(mirror_session_entry.mirror_session_key, mirror_session_entry);
386
387 return ReturnCode();
388}

390ReturnCode MirrorSessionManager::processUpdateRequest(const P4MirrorSessionAppDbEntry &app_db_entry,
391 P4MirrorSessionEntry *existing_mirror_session_entry)
392{
393 SWSS_LOG_ENTER();
394
395 // Check the existence of the mirror session in mirror manager and centralized
396 // mapper.
397 if (existing_mirror_session_entry == nullptr)
398 {
399 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("existing_mirror_session_entry is nullptr");
400 }
401 if (!m_p4OidMapper->existsOID(SAI_OBJECT_TYPE_MIRROR_SESSION, existing_mirror_session_entry->mirror_session_key))
402 {
403 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Mirror session with key "
404 << QuotedVar(existing_mirror_session_entry->mirror_session_key)
405 << " doesn't exist in centralized mapper");
406 }
407
408 P4MirrorSessionEntry mirror_session_entry_before_update(*existing_mirror_session_entry);
409
410 // Because SAI mirror set API sets attr one at a time, it is possible attr
411 // updates fail in the middle. Up on failure, all successful operations need
412 // to be undone.
413 ReturnCode ret;
414 bool update_fail_in_middle = false;
415 if (!update_fail_in_middle && app_db_entry.has_port)
416 {
417 ret = setPort(app_db_entry.port, existing_mirror_session_entry);
418 if (!ret.ok())
419 update_fail_in_middle = true;
420 }
421 if (!update_fail_in_middle && app_db_entry.has_src_ip)
422 {
423 ret = setSrcIp(app_db_entry.src_ip, existing_mirror_session_entry);
424 if (!ret.ok())
425 update_fail_in_middle = true;
426 }
427 if (!update_fail_in_middle && app_db_entry.has_dst_ip)
428 {
429 ret = setDstIp(app_db_entry.dst_ip, existing_mirror_session_entry);
430 if (!ret.ok())
431 update_fail_in_middle = true;
432 }
433 if (!update_fail_in_middle && app_db_entry.has_src_mac)
434 {
435 ret = setSrcMac(app_db_entry.src_mac, existing_mirror_session_entry);
436 if (!ret.ok())
437 update_fail_in_middle = true;
438 }
439 if (!update_fail_in_middle && app_db_entry.has_dst_mac)
440 {
441 ret = setDstMac(app_db_entry.dst_mac, existing_mirror_session_entry);
442 if (!ret.ok())
443 update_fail_in_middle = true;
444 }
445 if (!update_fail_in_middle && app_db_entry.has_ttl)
446 {
447 ret = setTtl(app_db_entry.ttl, existing_mirror_session_entry);
448 if (!ret.ok())
449 update_fail_in_middle = true;
450 }
451 if (!update_fail_in_middle && app_db_entry.has_tos)
452 {
453 ret = setTos(app_db_entry.tos, existing_mirror_session_entry);
454 if (!ret.ok())
455 update_fail_in_middle = true;
456 }
457
458 if (update_fail_in_middle)
459 {
460 ReturnCode status = setMirrorSessionEntry(mirror_session_entry_before_update, existing_mirror_session_entry);
461 if (!status.ok())
462 {
463 ret << "Failed to recover mirror session entry to the state before "
464 "update operation.";
465 SWSS_RAISE_CRITICAL_STATE("Failed to recover mirror session entry to the state before update "
466 "operation.");
467 }
468 }
469
470 return ret;
471}

473ReturnCode MirrorSessionManager::setPort(const std::string &new_port_name,
474 P4MirrorSessionEntry *existing_mirror_session_entry)
475{
476 SWSS_LOG_ENTER();
477
478 if (new_port_name == existing_mirror_session_entry->port)
479 {
480 return ReturnCode();
481 }
482
483 swss::Port new_port;
484 if (!gPortsOrch->getPort(new_port_name, new_port))
485 {
486 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
487 << "Failed to get port info for port " << QuotedVar(new_port_name));
488 }
489 if (new_port.m_type != Port::Type::PHY)
490 {
491 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
492 << "Port " << QuotedVar(new_port.m_alias) << "'s type " << new_port.m_type
493 << " is not physical and is invalid as destination "
494 "port for mirror packet.");
495 }
496
497 sai_attribute_t attr;
498 attr.id = SAI_MIRROR_SESSION_ATTR_MONITOR_PORT;
499 attr.value.oid = new_port.m_port_id;
500
501 // Call SAI API.
502 CHECK_ERROR_AND_LOG_AND_RETURN(
503 sai_mirror_api->set_mirror_session_attribute(existing_mirror_session_entry->mirror_session_oid, &attr),
504 "Failed to set new port " << QuotedVar(new_port.m_alias) << " for mirror session "
505 << QuotedVar(existing_mirror_session_entry->mirror_session_key));
506
507 // Update ref count.
508 gPortsOrch->decreasePortRefCount(existing_mirror_session_entry->port);
509 gPortsOrch->increasePortRefCount(new_port.m_alias);
510
511 // Update the entry in table
512 existing_mirror_session_entry->port = new_port_name;
513
514 return ReturnCode();

724ReturnCode MirrorSessionManager::processDeleteRequest(const std::string &mirror_session_key)
725{
726 SWSS_LOG_ENTER();
727
728 const P4MirrorSessionEntry *mirror_session_entry = getMirrorSessionEntry(mirror_session_key);
729 if (mirror_session_entry == nullptr)
730 {
731 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
732 << "Mirror session with key " << QuotedVar(mirror_session_key)
733 << " does not exist in mirror session manager");
734 }
735
736 // Check if there is anything referring to the mirror session before deletion.
737 uint32_t ref_count;
738 if (!m_p4OidMapper->getRefCount(SAI_OBJECT_TYPE_MIRROR_SESSION, mirror_session_key, &ref_count))
739 {
740 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Failed to get reference count for mirror session "
741 << QuotedVar(mirror_session_key));
742 }
743 if (ref_count > 0)
744 {
745 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_IN_USE)
746 << "Mirror session " << QuotedVar(mirror_session_entry->mirror_session_key)
747 << " referenced by other objects (ref_count = " << ref_count);
748 }
749
750 // Call SAI API.
751 CHECK_ERROR_AND_LOG_AND_RETURN(sai_mirror_api->remove_mirror_session(mirror_session_entry->mirror_session_oid),
752 "Failed to remove mirror session "
753 << QuotedVar(mirror_session_entry->mirror_session_key));
754
755 // On successful deletion, decrement ref count.
756 gPortsOrch->decreasePortRefCount(mirror_session_entry->port);
757
758 // Delete the key to OID map from centralized mapper.
759 m_p4OidMapper->eraseOID(SAI_OBJECT_TYPE_MIRROR_SESSION, mirror_session_entry->mirror_session_key);
760
761 // Delete entry from internal table.
762 m_mirrorSessionTable.erase(mirror_session_entry->mirror_session_key);
763
764 return ReturnCode();
765}

767std::string MirrorSessionManager::verifyState(const std::string &key, const std::vector<swss::FieldValueTuple> &tuple)
768{
769 SWSS_LOG_ENTER();
770
771 auto pos = key.find_first_of(kTableKeyDelimiter);
772 if (pos == std::string::npos)
773 {
774 return std::string("Invalid key: ") + key;
775 }
776 std::string p4rt_table = key.substr(0, pos);
777 std::string p4rt_key = key.substr(pos + 1);
778 if (p4rt_table != APP_P4RT_TABLE_NAME)
779 {
780 return std::string("Invalid key: ") + key;
781 }
782 std::string table_name;
783 std::string key_content;
784 parseP4RTKey(p4rt_key, &table_name, &key_content);
785 if (table_name != APP_P4RT_MIRROR_SESSION_TABLE_NAME)
786 {
787 return std::string("Invalid key: ") + key;
788 }
789
790 ReturnCode status;
791 auto app_db_entry_or = deserializeP4MirrorSessionAppDbEntry(key_content, tuple);
792 if (!app_db_entry_or.ok())
793 {
794 status = app_db_entry_or.status();
795 std::stringstream msg;
796 msg << "Unable to deserialize key " << QuotedVar(key) << ": " << status.message();
797 return msg.str();
798 }
799 auto &app_db_entry = *app_db_entry_or;
800 const std::string mirror_session_key = KeyGenerator::generateMirrorSessionKey(app_db_entry.mirror_session_id);
801 auto *mirror_session_entry = getMirrorSessionEntry(mirror_session_key);
802 if (mirror_session_entry == nullptr)
803 {
804 std::stringstream msg;
805 msg << "No entry found with key " << QuotedVar(key);
806 return msg.str();
807 }
808
809 std::string cache_result = verifyStateCache(app_db_entry, mirror_session_entry);
810 std::string asic_db_result = verifyStateAsicDb(mirror_session_entry);
811 if (cache_result.empty())
812 {
813 return asic_db_result;
814 }
815 if (asic_db_result.empty())
816 {
817 return cache_result;
818 }
819 return cache_result + "; " + asic_db_result;
820}
