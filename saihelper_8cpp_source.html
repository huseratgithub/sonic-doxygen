247void initSaiRedis(const string &record_location, const std::string &record_filename)
248{
255 sai_attribute_t attr;
256 sai_status_t status;
257
258 /* set recording dir before enable recording */
259
260 if (gSairedisRecord)
261 {
262 attr.id = SAI_REDIS_SWITCH_ATTR_RECORDING_OUTPUT_DIR;
263 attr.value.s8list.count = (uint32_t)record_location.size();
264 attr.value.s8list.list = (int8_t*)const_cast<char *>(record_location.c_str());
265
266 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
267 if (status != SAI_STATUS_SUCCESS)
268 {
269 SWSS_LOG_ERROR("Failed to set SAI Redis recording output folder to %s, rv:%d",
270 record_location.c_str(), status);
271 exit(EXIT_FAILURE);
272 }
273
274 attr.id = SAI_REDIS_SWITCH_ATTR_RECORDING_FILENAME;
275 attr.value.s8list.count = (uint32_t)record_filename.size();
276 attr.value.s8list.list = (int8_t*)const_cast<char *>(record_filename.c_str());
277
278 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
279 if (status != SAI_STATUS_SUCCESS)
280 {
281 SWSS_LOG_ERROR("Failed to set SAI Redis recording logfile to %s, rv:%d",
282 record_filename.c_str(), status);
283 exit(EXIT_FAILURE);
284 }
285
286 }
287
288 /* Disable/enable SAI Redis recording */
289
290 attr.id = SAI_REDIS_SWITCH_ATTR_RECORD;
291 attr.value.booldata = gSairedisRecord;
292
293 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
294 if (status != SAI_STATUS_SUCCESS)
295 {
296 SWSS_LOG_ERROR("Failed to %s SAI Redis recording, rv:%d",
297 gSairedisRecord ? "enable" : "disable", status);
298 exit(EXIT_FAILURE);
299 }
300
301 attr.id = SAI_REDIS_SWITCH_ATTR_USE_PIPELINE;
302 attr.value.booldata = true;
303
304 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
305 if (status != SAI_STATUS_SUCCESS)
306 {
307 SWSS_LOG_ERROR("Failed to enable redis pipeline, rv:%d", status);
308 exit(EXIT_FAILURE);
309 }
310 SWSS_LOG_NOTICE("Enable redis pipeline");
311
312 char *platform = getenv("platform");
313 if (platform && (strstr(platform, MLNX_PLATFORM_SUBSTRING) || strstr(platform, XS_PLATFORM_SUBSTRING)))
314 {
315 /* We set this long timeout in order for Orchagent to wait enough time for
316 * response from syncd. It is needed since in init, systemd syncd startup
317 * script first calls FW upgrade script (that might take up to 7 minutes
318 * in systems with Gearbox) and only then launches syncd container */
319 attr.id = SAI_REDIS_SWITCH_ATTR_SYNC_OPERATION_RESPONSE_TIMEOUT;
320 attr.value.u64 = SAI_REDIS_SYNC_OPERATION_RESPONSE_TIMEOUT;
321 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
322
323 if (status != SAI_STATUS_SUCCESS)
324 {
325 SWSS_LOG_ERROR("Failed to set SAI REDIS response timeout");
326 exit(EXIT_FAILURE);
327 }
328
329 SWSS_LOG_NOTICE("SAI REDIS response timeout set successfully to %" PRIu64 " ", attr.value.u64);
330 }
331
332 attr.id = SAI_REDIS_SWITCH_ATTR_NOTIFY_SYNCD;
333 attr.value.s32 = SAI_REDIS_NOTIFY_SYNCD_INIT_VIEW;
334 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
335
336 if (status != SAI_STATUS_SUCCESS)
337 {
338 SWSS_LOG_ERROR("Failed to notify syncd INIT_VIEW, rv:%d gSwitchId %" PRIx64, status, gSwitchId);
339 exit(EXIT_FAILURE);
340 }
341 SWSS_LOG_NOTICE("Notify syncd INIT_VIEW");
342
343 if (platform && (strstr(platform, MLNX_PLATFORM_SUBSTRING) || strstr(platform, XS_PLATFORM_SUBSTRING)))
344 {
345 /* Set timeout back to the default value */
346 attr.id = SAI_REDIS_SWITCH_ATTR_SYNC_OPERATION_RESPONSE_TIMEOUT;
347 attr.value.u64 = SAI_REDIS_DEFAULT_SYNC_OPERATION_RESPONSE_TIMEOUT;
348 status = sai_switch_api->set_switch_attribute(gSwitchId, &attr);
349
350 if (status != SAI_STATUS_SUCCESS)
351 {
352 SWSS_LOG_ERROR("Failed to set SAI REDIS response timeout");
353 exit(EXIT_FAILURE);
354 }
355
356 SWSS_LOG_NOTICE("SAI REDIS response timeout set successfully to %" PRIu64 " ", attr.value.u64);
357 }
358}

360sai_status_t initSaiPhyApi(swss::gearbox_phy_t *phy)
361{
362 sai_object_id_t phyOid;
363 sai_attribute_t attr;
364 vector<sai_attribute_t> attrs;
365 sai_status_t status;
366 char fwPath[PATH_MAX];
367 char hwinfo[HWINFO_MAX_SIZE + 1];
368
369 SWSS_LOG_ENTER();
370
371 attr.id = SAI_SWITCH_ATTR_INIT_SWITCH;
372 attr.value.booldata = true;
373 attrs.push_back(attr);
374
375 attr.id = SAI_SWITCH_ATTR_TYPE;
376 attr.value.u32 = SAI_SWITCH_TYPE_PHY;
377 attrs.push_back(attr);
378
379 attr.id = SAI_SWITCH_ATTR_SWITCH_PROFILE_ID;
380 attr.value.u32 = 0;
381 attrs.push_back(attr);
382
383 if( phy->hwinfo.length() > HWINFO_MAX_SIZE ) {
384 SWSS_LOG_ERROR( "hwinfo string attribute is too long." );
385 return SAI_STATUS_FAILURE;
386 }
387 strncpy(hwinfo, phy->hwinfo.c_str(), phy->hwinfo.length());
388
389 attr.id = SAI_SWITCH_ATTR_SWITCH_HARDWARE_INFO;
390 attr.value.s8list.count = (uint32_t) phy->hwinfo.length();
391 attr.value.s8list.list = (int8_t *) hwinfo;
392 attrs.push_back(attr);
393
394 if (phy->firmware.length() == 0)
395 {
396 attr.id = SAI_SWITCH_ATTR_FIRMWARE_LOAD_METHOD;
397 attr.value.u32 = SAI_SWITCH_FIRMWARE_LOAD_METHOD_NONE;
398 attrs.push_back(attr);
399 }
400 else
401 {
402 attr.id = SAI_SWITCH_ATTR_FIRMWARE_LOAD_METHOD;
403 attr.value.u32 = SAI_SWITCH_FIRMWARE_LOAD_METHOD_INTERNAL;
404 attrs.push_back(attr);
405
406 strncpy(fwPath, phy->firmware.c_str(), PATH_MAX - 1);
407
408 attr.id = SAI_SWITCH_ATTR_FIRMWARE_PATH_NAME;
409 attr.value.s8list.list = (int8_t *) fwPath;
410 attr.value.s8list.count = (uint32_t) strlen(fwPath) + 1;
411 attrs.push_back(attr);
412
413 attr.id = SAI_SWITCH_ATTR_FIRMWARE_LOAD_TYPE;
414 attr.value.u32 = SAI_SWITCH_FIRMWARE_LOAD_TYPE_AUTO;
415 attrs.push_back(attr);
416 }
417
418 attr.id = SAI_SWITCH_ATTR_REGISTER_READ;
419 attr.value.ptr = (void *) mdio_read;
420 attrs.push_back(attr);
421
422 attr.id = SAI_SWITCH_ATTR_REGISTER_WRITE;
423 attr.value.ptr = (void *) mdio_write;
424 attrs.push_back(attr);
425
426 attr.id = SAI_SWITCH_ATTR_HARDWARE_ACCESS_BUS;
427 attr.value.u32 = hardware_access_map[phy->access];
428 attrs.push_back(attr);
429
430 attr.id = SAI_SWITCH_ATTR_PLATFROM_CONTEXT;
431 attr.value.u64 = phy->address;
432 attrs.push_back(attr);
433
434 /* Must be last Attribute */
435 attr.id = SAI_REDIS_SWITCH_ATTR_CONTEXT;
436 attr.value.u64 = phy->context_id;
437 attrs.push_back(attr);
438
439 status = sai_switch_api->create_switch(&phyOid, (uint32_t)attrs.size(), attrs.data());
440
441 if (status != SAI_STATUS_SUCCESS)
442 {
443 SWSS_LOG_ERROR("BOX: Failed to create PHY:%d rtn:%d", phy->phy_id, status);
444 return status;
445 }
446 SWSS_LOG_NOTICE("BOX: Created PHY:%d Oid:0x%" PRIx64, phy->phy_id, phyOid);
447
448 phy->phy_oid = sai_serialize_object_id(phyOid);
449
450 if (phy->firmware.length() != 0)
451 {
452 attr.id = SAI_SWITCH_ATTR_FIRMWARE_MAJOR_VERSION;
453 status = sai_switch_api->get_switch_attribute(phyOid, 1, &attr);
454 if (status != SAI_STATUS_SUCCESS)
455 {
456 SWSS_LOG_ERROR("BOX: Failed to get firmware major version for hwinfo:%s, phy:%d, rtn:%d",
457 phy->hwinfo.c_str(), phy->phy_id, status);
458 return status;
459 }
460 else
461 {
462 phy->firmware_major_version = string(attr.value.chardata);
463 }
464 }
465 return status;
466}
