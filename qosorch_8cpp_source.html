115task_process_status QosMapHandler::processWorkItem(Consumer& consumer, KeyOpFieldsValuesTuple &tuple)
116{
117 SWSS_LOG_ENTER();
118
119 sai_object_id_t sai_object = SAI_NULL_OBJECT_ID;
120 string qos_object_name = kfvKey(tuple);
121 string qos_map_type_name = consumer.getTableName();
122 string op = kfvOp(tuple);
123
124 if (QosOrch::getTypeMap()[qos_map_type_name]->find(qos_object_name) != QosOrch::getTypeMap()[qos_map_type_name]->end())
125 {
126 sai_object = (*(QosOrch::getTypeMap()[qos_map_type_name]))[qos_object_name].m_saiObjectId;
127 if ((*(QosOrch::getTypeMap()[qos_map_type_name]))[qos_object_name].m_pendingRemove && op == SET_COMMAND)
128 {
129 SWSS_LOG_NOTICE("Entry %s %s is pending remove, need retry", qos_map_type_name.c_str(), qos_object_name.c_str());
130 return task_process_status::task_need_retry;
131 }
132 }
133 if (op == SET_COMMAND)
134 {
135 vector<sai_attribute_t> attributes;
136 if (!convertFieldValuesToAttributes(tuple, attributes))
137 {
138 return task_process_status::task_invalid_entry;
139 }
140 if (SAI_NULL_OBJECT_ID != sai_object)
141 {
142 if (!modifyQosItem(sai_object, attributes))
143 {
144 SWSS_LOG_ERROR("Failed to set [%s:%s]", qos_map_type_name.c_str(), qos_object_name.c_str());
145 freeAttribResources(attributes);
146 return task_process_status::task_failed;
147 }
148 SWSS_LOG_NOTICE("Set [%s:%s]", qos_map_type_name.c_str(), qos_object_name.c_str());
149 }
150 else
151 {
152 sai_object = addQosItem(attributes);
153 if (sai_object == SAI_NULL_OBJECT_ID)
154 {
155 SWSS_LOG_ERROR("Failed to create [%s:%s]", qos_map_type_name.c_str(), qos_object_name.c_str());
156 freeAttribResources(attributes);
157 return task_process_status::task_failed;
158 }
159 (*(QosOrch::getTypeMap()[qos_map_type_name]))[qos_object_name].m_saiObjectId = sai_object;
160 (*(QosOrch::getTypeMap()[qos_map_type_name]))[qos_object_name].m_pendingRemove = false;
161 SWSS_LOG_NOTICE("Created [%s:%s]", qos_map_type_name.c_str(), qos_object_name.c_str());
162 }
163 freeAttribResources(attributes);
164 }
165 else if (op == DEL_COMMAND)
166 {
167 if (SAI_NULL_OBJECT_ID == sai_object)
168 {
169 SWSS_LOG_ERROR("Object with name:%s not found.", qos_object_name.c_str());
170 return task_process_status::task_invalid_entry;
171 }
172 if (gQosOrch->isObjectBeingReferenced(QosOrch::getTypeMap(), qos_map_type_name, qos_object_name))
173 {
174 auto hint = gQosOrch->objectReferenceInfo(QosOrch::getTypeMap(), qos_map_type_name, qos_object_name);
175 SWSS_LOG_NOTICE("Can't remove object %s due to being referenced (%s)", qos_object_name.c_str(), hint.c_str());
176 (*(QosOrch::getTypeMap()[qos_map_type_name]))[qos_object_name].m_pendingRemove = true;
177 return task_process_status::task_need_retry;
178 }
179 if (!removeQosItem(sai_object))
180 {
181 SWSS_LOG_ERROR("Failed to remove dscp_to_tc map. db name:%s sai object:%" PRIx64, qos_object_name.c_str(), sai_object);
182 return task_process_status::task_failed;
183 }
184 auto it_to_delete = (QosOrch::getTypeMap()[qos_map_type_name])->find(qos_object_name);
185 (QosOrch::getTypeMap()[qos_map_type_name])->erase(it_to_delete);
186 }
187 else
188 {
189 SWSS_LOG_ERROR("Unknown operation type %s", op.c_str());
190 return task_process_status::task_invalid_entry;
191 }
192 return task_process_status::task_success;
193}

359 for (const auto &fv : kfvFieldsValues(tuple))

362 {
363 dot1p_map_list.list[i].key.dot1p = static_cast<sai_uint8_t>(stoi(fvField(fv)));

373 SWSS_LOG_ERROR("Out of range dot1p to tc argument %s:%s to %s()", fvField(fv).c_str(), fvValue(fv).c_str(), e.what());

375 }
376

378 }
379 dot1p_map_list.count = static_cast<uint32_t>(i);

381 sai_attribute_t attr;

383 attr.value.qosmap.count = dot1p_map_list.count;

522bool WredMapHandler::convertFieldValuesToAttributes(KeyOpFieldsValuesTuple &tuple, vector<sai_attribute_t> &attribs)
523{
524 SWSS_LOG_ENTER();
525 sai_attribute_t attr;
526 vector<sai_attribute_t> deferred_attributes;
527 auto &key = kfvKey(tuple);
528 auto &storedProfile = WredMapHandler::m_wredProfiles[key];
529 qos_wred_thresholds_t currentProfile = storedProfile;
530 sai_uint32_t threshold;
531
532 /*
533 * Setting WRED profile can fail in case
534 * - the current min threshold is greater than the new max threshold
535 * - or the current max threshold is less than the new min threshold
536 * for any color at any time, on some vendor's platforms.
537 *
538 * The root cause
539 * There can be only one attribute in each SAI SET operation, which means
540 * the vendor SAI do not have a big picture regarding what attributes are being set
541 * and can only perform the sanity check against each SET operation.
542 * In the above case, the sanity check will fail.
543 *
544 * The fix
545 * The thresholds that have been applied to SAI will be stored in orchagent.
546 *
547 * The original logic is to handle each attribute to be set and append it to an attribute list.
548 * To resolve the issue, a 2nd half attribute list is introduced and
549 * will be appended to the original attribute list after all the attributes have been handled.
550 *
551 * In the new logic, each threshold to be set will be checked against the stored data.
552 * In case it violates the condition, the violating attribute will be deferred, done via putting it into the 2nd half attributes list.
553 *
554 * For any color, there can be only 1 threshold violating the condition.
555 * Otherwise, it means both new min > old max and new max > old min, which means either old max < old min or new max < new min,
556 * which means either old or new data is illegal.
557 * This can not happen because illegal data can not be applied and stored.
558 *
559 * By doing so, the other threshold will be applied first, which extends the threshold range and breaks the violating condition.
560 * A logic is also introduced to guarantee the min threshold is always less than the max threshold in the new profile to be set.
561 *
562 * For example:
563 * Current min=1M, max=2M, new min=3M, new max=4M
564 * The min is set first, so current max (2M) < new min (3M), which violates the condition
565 * By the new logic, min threshold will be deferred so the new max will be applied first and then the new min is applied and no violating.
566 * min = 1M, max = 2M
567 * => min = 1M, max = 4M
568 * => min = 3M, max = 4M
569 */
570
571 for (auto i = kfvFieldsValues(tuple).begin(); i != kfvFieldsValues(tuple).end(); i++)
572 {
573 if (fvField(*i) == yellow_max_threshold_field_name)
574 {
575 threshold = stoi(fvValue(*i));
576 appendThresholdToAttributeList(SAI_WRED_ATTR_YELLOW_MAX_THRESHOLD,
577 threshold,
578 (storedProfile.yellow_min_threshold > threshold),
579 attribs,
580 deferred_attributes,
581 currentProfile.yellow_max_threshold);
582 }
583 else if (fvField(*i) == yellow_min_threshold_field_name)
584 {
585 threshold = stoi(fvValue(*i));
586 appendThresholdToAttributeList(SAI_WRED_ATTR_YELLOW_MIN_THRESHOLD,
587 threshold,
588 (storedProfile.yellow_max_threshold < threshold),
589 attribs,
590 deferred_attributes,
591 currentProfile.yellow_min_threshold);
592 }
593 else if (fvField(*i) == green_max_threshold_field_name)
594 {
595 threshold = stoi(fvValue(*i));
596 appendThresholdToAttributeList(SAI_WRED_ATTR_GREEN_MAX_THRESHOLD,
597 threshold,
598 (storedProfile.green_min_threshold > threshold),
599 attribs,
600 deferred_attributes,
601 currentProfile.green_max_threshold);
602 }
603 else if (fvField(*i) == green_min_threshold_field_name)
604 {
605 threshold = stoi(fvValue(*i));
606 appendThresholdToAttributeList(SAI_WRED_ATTR_GREEN_MIN_THRESHOLD,
607 threshold,
608 (storedProfile.green_max_threshold < threshold),
609 attribs,
610 deferred_attributes,
611 currentProfile.green_min_threshold);
612 }
613 else if (fvField(*i) == red_max_threshold_field_name)
614 {
615 threshold = stoi(fvValue(*i));
616 appendThresholdToAttributeList(SAI_WRED_ATTR_RED_MAX_THRESHOLD,
617 threshold,
618 (storedProfile.red_min_threshold > threshold),
619 attribs,
620 deferred_attributes,
621 currentProfile.red_max_threshold);
622 }
623 else if (fvField(*i) == red_min_threshold_field_name)
624 {
625 threshold = stoi(fvValue(*i));
626 appendThresholdToAttributeList(SAI_WRED_ATTR_RED_MIN_THRESHOLD,
627 threshold,
628 (storedProfile.red_max_threshold < threshold),
629 attribs,
630 deferred_attributes,
631 currentProfile.red_min_threshold);
632 }
633 else if (fvField(*i) == green_drop_probability_field_name)
634 {
635 attr.id = SAI_WRED_ATTR_GREEN_DROP_PROBABILITY;
636 attr.value.s32 = stoi(fvValue(*i));
637 attribs.push_back(attr);
638 }
639 else if (fvField(*i) == yellow_drop_probability_field_name)
640 {
641 attr.id = SAI_WRED_ATTR_YELLOW_DROP_PROBABILITY;
642 attr.value.s32 = stoi(fvValue(*i));
643 attribs.push_back(attr);
644 }
645 else if (fvField(*i) == red_drop_probability_field_name)
646 {
647 attr.id = SAI_WRED_ATTR_RED_DROP_PROBABILITY;
648 attr.value.s32 = stoi(fvValue(*i));
649 attribs.push_back(attr);
650 }
651 else if (fvField(*i) == wred_green_enable_field_name)
652 {
653 attr.id = SAI_WRED_ATTR_GREEN_ENABLE;
654 if(!convertBool(fvValue(*i),attr.value.booldata))
655 {
656 return false;
657 }
658 attribs.push_back(attr);
659 }
660 else if (fvField(*i) == wred_yellow_enable_field_name)
661 {
662 attr.id = SAI_WRED_ATTR_YELLOW_ENABLE;
663 if(!convertBool(fvValue(*i),attr.value.booldata))
664 {
665 return false;
666 }
667 attribs.push_back(attr);
668 }
669 else if (fvField(*i) == wred_red_enable_field_name)
670 {
671 attr.id = SAI_WRED_ATTR_RED_ENABLE;
672 if(!convertBool(fvValue(*i),attr.value.booldata))
673 {
674 return false;
675 }
676 attribs.push_back(attr);
677 }
678 else if (fvField(*i) == ecn_field_name)
679 {
680 attr.id = SAI_WRED_ATTR_ECN_MARK_MODE;
681 sai_ecn_mark_mode_t ecn = ecn_map.at(fvValue(*i));
682 attr.value.s32 = ecn;
683 attribs.push_back(attr);
684 }
685 else {
686 SWSS_LOG_ERROR("Unknown wred profile field:%s", fvField(*i).c_str());
687 return false;
688 }
689 }
690
691 if ((currentProfile.green_min_threshold > currentProfile.green_max_threshold)
692 || (currentProfile.yellow_min_threshold > currentProfile.yellow_max_threshold)
693 || (currentProfile.red_min_threshold > currentProfile.red_max_threshold))
694 {
695 SWSS_LOG_ERROR("Wrong wred profile: min threshold is greater than max threshold");
696 return false;
697 }
698
699 attribs.insert(attribs.end(), deferred_attributes.begin(), deferred_attributes.end());
700 storedProfile = currentProfile;
701
702 return true;
703}

725 sai_object_id_t sai_object;
726 sai_attribute_t attr;
727 vector<sai_attribute_t> attrs;
728 uint8_t drop_prob_set = 0;
729 uint8_t wred_enable_set = 0;
730
731 attr.id = SAI_WRED_ATTR_WEIGHT;
732 attr.value.s32 = 0;
733 attrs.push_back(attr);
734
735 for(auto attrib : attribs)
736 {
737 attrs.push_back(attrib);
738
739 switch (attrib.id)
740 {
741 case SAI_WRED_ATTR_GREEN_ENABLE:
742 if (attrib.value.booldata)
743 {
744 wred_enable_set |= GREEN_WRED_ENABLED;
745 }
746 break;
747 case SAI_WRED_ATTR_YELLOW_ENABLE:
748 if (attrib.value.booldata)
749 {

763 drop_prob_set |= YELLOW_DROP_PROBABILITY_SET;
764 break;
765 case SAI_WRED_ATTR_RED_DROP_PROBABILITY:

767 break;
768 default:
769 break;
770 }
771 }

788 attr.value.s32 = 100;
789 attrs.push_back(attr);
790 }
791
792 sai_status = sai_wred_api->create_wred(&sai_object, gSwitchId, (uint32_t)attrs.size(), attrs.data());
793 if (sai_status != SAI_STATUS_SUCCESS)

976bool DscpToFcMapHandler::convertFieldValuesToAttributes(KeyOpFieldsValuesTuple &tuple, vector<sai_attribute_t> &attributes)
977{
978 SWSS_LOG_ENTER();
979

981

1019 list_attr.value.qosmap.list[ind].value.fc);

1283task_process_status QosOrch::handleSchedulerTable(Consumer& consumer, KeyOpFieldsValuesTuple &tuple)
1284{
1285 SWSS_LOG_ENTER();
1286
1287 sai_status_t sai_status;
1288 sai_object_id_t sai_object = SAI_NULL_OBJECT_ID;
1289
1290 string qos_map_type_name = CFG_SCHEDULER_TABLE_NAME;
1291 string qos_object_name = kfvKey(tuple);
1292 string op = kfvOp(tuple);
1293
1294 if (m_qos_maps[qos_map_type_name]->find(qos_object_name) != m_qos_maps[qos_map_type_name]->end())
1295 {
1296 sai_object = (*(m_qos_maps[qos_map_type_name]))[qos_object_name].m_saiObjectId;
1297 if (sai_object == SAI_NULL_OBJECT_ID)
1298 {
1299 SWSS_LOG_ERROR("Error sai_object must exist for key %s", qos_object_name.c_str());
1300 return task_process_status::task_invalid_entry;
1301 }
1302 if ((*(m_qos_maps[qos_map_type_name]))[qos_object_name].m_pendingRemove && op == SET_COMMAND)
1303 {
1304 SWSS_LOG_NOTICE("Entry %s %s is pending remove, need retry", qos_map_type_name.c_str(), qos_object_name.c_str());
1305 return task_process_status::task_need_retry;
1306 }
1307 }
1308 if (op == SET_COMMAND)
1309 {
1310 vector<sai_attribute_t> sai_attr_list;
1311 sai_attribute_t attr;
1312 for (auto i = kfvFieldsValues(tuple).begin(); i != kfvFieldsValues(tuple).end(); i++)
1313 {
1314 if (fvField(*i) == scheduler_algo_type_field_name)
1315 {
1316 attr.id = SAI_SCHEDULER_ATTR_SCHEDULING_TYPE;
1317 if (fvValue(*i) == scheduler_algo_DWRR)
1318 {
1319 attr.value.s32 = SAI_SCHEDULING_TYPE_DWRR;
1320 }
1321 else if (fvValue(*i) == scheduler_algo_WRR)
1322 {
1323 attr.value.s32 = SAI_SCHEDULING_TYPE_WRR;
1324 }
1325 else if (fvValue(*i) == scheduler_algo_STRICT)
1326 {
1327 attr.value.s32 = SAI_SCHEDULING_TYPE_STRICT;
1328 }
1329 else {
1330 SWSS_LOG_ERROR("Unknown scheduler type value:%s", fvField(*i).c_str());
1331 return task_process_status::task_invalid_entry;
1332 }
1333 sai_attr_list.push_back(attr);
1334 }
1335 else if (fvField(*i) == scheduler_weight_field_name)
1336 {
1337 attr.id = SAI_SCHEDULER_ATTR_SCHEDULING_WEIGHT;
1338 attr.value.u8 = (uint8_t)stoi(fvValue(*i));
1339 sai_attr_list.push_back(attr);
1340 }
1341 else if (fvField(*i) == scheduler_priority_field_name)
1342 {
1343 // TODO: The meaning is to be able to adjust priority of the given scheduler group.
1344 // However currently SAI model does not provide such ability.
1345 }
1346 else if (fvField(*i) == scheduler_meter_type_field_name)
1347 {
1348 sai_meter_type_t meter_value = scheduler_meter_map.at(fvValue(*i));
1349 attr.id = SAI_SCHEDULER_ATTR_METER_TYPE;
1350 attr.value.s32 = meter_value;
1351 sai_attr_list.push_back(attr);
1352 }
1353 else if (fvField(*i) == scheduler_min_bandwidth_rate_field_name)
1354 {
1355 attr.id = SAI_SCHEDULER_ATTR_MIN_BANDWIDTH_RATE;
1356 attr.value.u64 = stoull(fvValue(*i));
1357 sai_attr_list.push_back(attr);
1358 }
1359 else if (fvField(*i) == scheduler_min_bandwidth_burst_rate_field_name)
1360 {
1361 attr.id = SAI_SCHEDULER_ATTR_MIN_BANDWIDTH_BURST_RATE;
1362 attr.value.u64 = stoull(fvValue(*i));
1363 sai_attr_list.push_back(attr);
1364 }
1365 else if (fvField(*i) == scheduler_max_bandwidth_rate_field_name)
1366 {
1367 attr.id = SAI_SCHEDULER_ATTR_MAX_BANDWIDTH_RATE;
1368 attr.value.u64 = stoull(fvValue(*i));
1369 sai_attr_list.push_back(attr);
1370 }
1371 else if (fvField(*i) == scheduler_max_bandwidth_burst_rate_field_name)
1372 {
1373 attr.id = SAI_SCHEDULER_ATTR_MAX_BANDWIDTH_BURST_RATE;
1374 attr.value.u64 = stoull(fvValue(*i));
1375 sai_attr_list.push_back(attr);
1376 }
1377 else {
1378 SWSS_LOG_ERROR("Unknown field:%s", fvField(*i).c_str());
1379 return task_process_status::task_invalid_entry;
1380 }
1381 }
1382
1383 if (SAI_NULL_OBJECT_ID != sai_object)
1384 {
1385 for (auto attr : sai_attr_list)
1386 {
1387 sai_status = sai_scheduler_api->set_scheduler_attribute(sai_object, &attr);
1388 if (sai_status != SAI_STATUS_SUCCESS)
1389 {
1390 SWSS_LOG_ERROR("fail to set scheduler attribute, id:%d", attr.id);
1391 task_process_status handle_status = handleSaiSetStatus(SAI_API_SCHEDULER, sai_status);
1392 if (handle_status != task_process_status::task_success)
1393 {
1394 return handle_status;
1395 }
1396 }
1397 }
1398 }
1399 else
1400 {
1401 sai_status = sai_scheduler_api->create_scheduler(&sai_object, gSwitchId, (uint32_t)sai_attr_list.size(), sai_attr_list.data());
1402 if (sai_status != SAI_STATUS_SUCCESS)
1403 {
1404 SWSS_LOG_ERROR("Failed to create scheduler profile [%s:%s], rv:%d",
1405 qos_map_type_name.c_str(), qos_object_name.c_str(), sai_status);
1406 task_process_status handle_status = handleSaiCreateStatus(SAI_API_SCHEDULER, sai_status);
1407 if (handle_status != task_process_status::task_success)
1408 {
1409 return handle_status;
1410 }
1411 }
1412 SWSS_LOG_NOTICE("Created [%s:%s]", qos_map_type_name.c_str(), qos_object_name.c_str());
1413 (*(m_qos_maps[qos_map_type_name]))[qos_object_name].m_saiObjectId = sai_object;
1414 (*(m_qos_maps[qos_map_type_name]))[qos_object_name].m_pendingRemove = false;
1415 }
1416 }
1417 else if (op == DEL_COMMAND)
1418 {
1419 if (SAI_NULL_OBJECT_ID == sai_object)
1420 {
1421 SWSS_LOG_ERROR("Object with name:%s not found.", qos_object_name.c_str());
1422 return task_process_status::task_invalid_entry;
1423 }
1424 if (gQosOrch->isObjectBeingReferenced(QosOrch::getTypeMap(), qos_map_type_name, qos_object_name))
1425 {
1426 auto hint = gQosOrch->objectReferenceInfo(QosOrch::getTypeMap(), qos_map_type_name, qos_object_name);
1427 SWSS_LOG_NOTICE("Can't remove object %s due to being referenced (%s)", qos_object_name.c_str(), hint.c_str());
1428 (*(m_qos_maps[qos_map_type_name]))[qos_object_name].m_pendingRemove = true;
1429 return task_process_status::task_need_retry;
1430 }
1431 sai_status = sai_scheduler_api->remove_scheduler(sai_object);
1432 if (SAI_STATUS_SUCCESS != sai_status)
1433 {
1434 SWSS_LOG_ERROR("Failed to remove scheduler profile. db name:%s sai object:%" PRIx64, qos_object_name.c_str(), sai_object);
1435 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_SCHEDULER, sai_status);
1436 if (handle_status != task_process_status::task_success)
1437 {
1438 return handle_status;
1439 }
1440 }
1441 auto it_to_delete = (m_qos_maps[qos_map_type_name])->find(qos_object_name);
1442 (m_qos_maps[qos_map_type_name])->erase(it_to_delete);
1443 }
1444 else
1445 {
1446 SWSS_LOG_ERROR("Unknown operation type %s", op.c_str());
1447 return task_process_status::task_invalid_entry;
1448 }
1449 return task_process_status::task_success;
1450}

1452sai_object_id_t QosOrch::getSchedulerGroup(const Port &port, const sai_object_id_t queue_id)
1453{
1454 SWSS_LOG_ENTER();
1455
1456 sai_attribute_t attr;
1457 sai_status_t sai_status;
1458
1459 const auto it = m_scheduler_group_port_info.find(port.m_port_id);
1460 if (it == m_scheduler_group_port_info.end())
1461 {
1462 /* Get max sched groups count */
1463 attr.id = SAI_PORT_ATTR_QOS_NUMBER_OF_SCHEDULER_GROUPS;
1464 sai_status = sai_port_api->get_port_attribute(port.m_port_id, 1, &attr);
1465 if (SAI_STATUS_SUCCESS != sai_status)
1466 {
1467 SWSS_LOG_ERROR("Failed to get number of scheduler groups for port:%s", port.m_alias.c_str());
1468 task_process_status handle_status = handleSaiGetStatus(SAI_API_PORT, sai_status);
1469 if (handle_status != task_process_status::task_success)
1470 {
1471 return SAI_NULL_OBJECT_ID;
1472 }
1473 }
1474
1475 /* Get total groups list on the port */
1476 uint32_t groups_count = attr.value.u32;
1477 std::vector<sai_object_id_t> groups(groups_count);
1478
1479 attr.id = SAI_PORT_ATTR_QOS_SCHEDULER_GROUP_LIST;
1480 attr.value.objlist.list = groups.data();
1481 attr.value.objlist.count = groups_count;
1482 sai_status = sai_port_api->get_port_attribute(port.m_port_id, 1, &attr);
1483 if (SAI_STATUS_SUCCESS != sai_status)
1484 {
1485 SWSS_LOG_ERROR("Failed to get scheduler group list for port:%s", port.m_alias.c_str());
1486 task_process_status handle_status = handleSaiGetStatus(SAI_API_PORT, sai_status);
1487 if (handle_status != task_process_status::task_success)
1488 {
1489 return SAI_NULL_OBJECT_ID;
1490 }
1491 }
1492
1493 m_scheduler_group_port_info[port.m_port_id] = {
1494 .groups = std::move(groups),
1495 .child_groups = std::vector<std::vector<sai_object_id_t>>(groups_count),
1496 .group_has_been_initialized = std::vector<bool>(groups_count)
1497 };
1498
1499 SWSS_LOG_INFO("Port %s has been initialized with %u group(s)", port.m_alias.c_str(), groups_count);
1500 }
1501
1502 /* Lookup groups to which queue belongs */
1503 auto& scheduler_group_port_info = m_scheduler_group_port_info[port.m_port_id];
1504 const auto& groups = scheduler_group_port_info.groups;
1505 for (uint32_t ii = 0; ii < groups.size() ; ii++)
1506 {
1507 const auto& group_id = groups[ii];
1508 const auto& child_groups_per_group = scheduler_group_port_info.child_groups[ii];
1509 if (child_groups_per_group.empty())
1510 {
1511 if (scheduler_group_port_info.group_has_been_initialized[ii])
1512 {
1513 // skip this iteration if it has been initialized which means there're no children in this group
1514 SWSS_LOG_INFO("No child group for port %s group 0x%" PRIx64 ", skip", port.m_alias.c_str(), group_id);
1515 continue;
1516 }
1517
1518 attr.id = SAI_SCHEDULER_GROUP_ATTR_CHILD_COUNT;//Number of queues/groups childs added to scheduler group
1519 sai_status = sai_scheduler_group_api->get_scheduler_group_attribute(group_id, 1, &attr);
1520 if (SAI_STATUS_SUCCESS != sai_status)
1521 {
1522 SWSS_LOG_ERROR("Failed to get child count for scheduler group:0x%" PRIx64 " of port:%s", group_id, port.m_alias.c_str());
1523 task_process_status handle_status = handleSaiGetStatus(SAI_API_SCHEDULER_GROUP, sai_status);
1524 if (handle_status != task_process_status::task_success)
1525 {
1526 return SAI_NULL_OBJECT_ID;
1527 }
1528 }
1529
1530 uint32_t child_count = attr.value.u32;
1531
1532 SWSS_LOG_INFO("Port %s group 0x%" PRIx64 " has been initialized with %u child group(s)", port.m_alias.c_str(), group_id, child_count);
1533 scheduler_group_port_info.group_has_been_initialized[ii] = true;
1534
1535 // skip this iteration if there're no children in this group
1536 if (child_count == 0)
1537 {
1538 continue;
1539 }
1540
1541 vector<sai_object_id_t> child_groups(child_count);
1542 attr.id = SAI_SCHEDULER_GROUP_ATTR_CHILD_LIST;
1543 attr.value.objlist.list = child_groups.data();
1544 attr.value.objlist.count = child_count;
1545 sai_status = sai_scheduler_group_api->get_scheduler_group_attribute(group_id, 1, &attr);
1546 if (SAI_STATUS_SUCCESS != sai_status)
1547 {
1548 SWSS_LOG_ERROR("Failed to get child list for scheduler group:0x%" PRIx64 " of port:%s", group_id, port.m_alias.c_str());
1549 task_process_status handle_status = handleSaiGetStatus(SAI_API_SCHEDULER_GROUP, sai_status);
1550 if (handle_status != task_process_status::task_success)
1551 {
1552 return SAI_NULL_OBJECT_ID;
1553 }
1554 }
1555
1556 scheduler_group_port_info.child_groups[ii] = std::move(child_groups);
1557 }
1558
1559 for (const auto& child_group_id: child_groups_per_group)
1560 {
1561 if (child_group_id == queue_id)
1562 {
1563 return group_id;
1564 }
1565 }
1566 }
1567
1568 return SAI_NULL_OBJECT_ID;

1642task_process_status QosOrch::handleQueueTable(Consumer& consumer, KeyOpFieldsValuesTuple &tuple)
1643{
1644 SWSS_LOG_ENTER();
1645 Port port;
1646 bool result;
1647 string key = kfvKey(tuple);
1648 string op = kfvOp(tuple);
1649 size_t queue_ind = 0;
1650 vector<string> tokens;
1651
1652 sai_uint32_t range_low, range_high;
1653 vector<string> port_names;
1654
1655 ref_resolve_status resolve_result;
1656 // sample "QUEUE: {Ethernet4|0-1}"
1657 tokens = tokenize(key, config_db_key_delimiter);
1658 if (tokens.size() != 2)
1659 {
1660 SWSS_LOG_ERROR("malformed key:%s. Must contain 2 tokens", key.c_str());
1661 return task_process_status::task_invalid_entry;
1662 }
1663 port_names = tokenize(tokens[0], list_item_delimiter);
1664 if (!parseIndexRange(tokens[1], range_low, range_high))
1665 {
1666 SWSS_LOG_ERROR("Failed to parse range:%s", tokens[1].c_str());
1667 return task_process_status::task_invalid_entry;
1668 }
1669
1670 bool donotChangeScheduler = false;
1671 bool donotChangeWredProfile = false;
1672 sai_object_id_t sai_scheduler_profile;
1673 sai_object_id_t sai_wred_profile;
1674
1675 if (op == SET_COMMAND)
1676 {
1677 string scheduler_profile_name;
1678 resolve_result = resolveFieldRefValue(m_qos_maps, scheduler_field_name,
1679 qos_to_ref_table_map.at(scheduler_field_name), tuple,
1680 sai_scheduler_profile, scheduler_profile_name);
1681 if (ref_resolve_status::success != resolve_result)
1682 {
1683 if (resolve_result != ref_resolve_status::field_not_found)
1684 {
1685 if(ref_resolve_status::not_resolved == resolve_result)
1686 {
1687 SWSS_LOG_INFO("Missing or invalid scheduler reference");
1688 return task_process_status::task_need_retry;
1689 }
1690 SWSS_LOG_ERROR("Resolving scheduler reference failed");
1691 return task_process_status::task_failed;
1692 }
1693
1694 if (doesObjectExist(m_qos_maps, CFG_QUEUE_TABLE_NAME, key, scheduler_field_name, scheduler_profile_name))
1695 {
1696 SWSS_LOG_NOTICE("QUEUE|%s %s was configured but is not any more. Remove it", key.c_str(), scheduler_field_name.c_str());
1697 removeMeFromObjsReferencedByMe(m_qos_maps, CFG_QUEUE_TABLE_NAME, key, scheduler_field_name, scheduler_profile_name);
1698 sai_scheduler_profile = SAI_NULL_OBJECT_ID;
1699 }
1700 else
1701 {
1702 // Did not exist and do not exist. No action
1703 donotChangeScheduler = true;
1704 }
1705 }
1706 else
1707 {
1708 setObjectReference(m_qos_maps, CFG_QUEUE_TABLE_NAME, key, scheduler_field_name, scheduler_profile_name);
1709 SWSS_LOG_INFO("QUEUE %s Field %s %s has been resolved to %" PRIx64 , key.c_str(), scheduler_field_name.c_str(), scheduler_profile_name.c_str(), sai_scheduler_profile);
1710 }
1711
1712 string wred_profile_name;
1713 resolve_result = resolveFieldRefValue(m_qos_maps, wred_profile_field_name,
1714 qos_to_ref_table_map.at(wred_profile_field_name), tuple,
1715 sai_wred_profile, wred_profile_name);
1716 if (ref_resolve_status::success != resolve_result)
1717 {
1718 if (resolve_result != ref_resolve_status::field_not_found)
1719 {
1720 if(ref_resolve_status::not_resolved == resolve_result)
1721 {
1722 SWSS_LOG_INFO("Missing or invalid wred profile reference");
1723 return task_process_status::task_need_retry;
1724 }
1725 SWSS_LOG_ERROR("Resolving wred profile reference failed");
1726 return task_process_status::task_failed;
1727 }
1728
1729 if (doesObjectExist(m_qos_maps, CFG_QUEUE_TABLE_NAME, key, wred_profile_field_name, wred_profile_name))
1730 {
1731 SWSS_LOG_NOTICE("QUEUE|%s %s was configured but is not any more. Remove it", key.c_str(), wred_profile_field_name.c_str());
1732 removeMeFromObjsReferencedByMe(m_qos_maps, CFG_QUEUE_TABLE_NAME, key, wred_profile_field_name, wred_profile_name);
1733 sai_wred_profile = SAI_NULL_OBJECT_ID;
1734 }
1735 else
1736 {
1737 donotChangeWredProfile = true;
1738 }
1739 }
1740 else
1741 {
1742 setObjectReference(m_qos_maps, CFG_QUEUE_TABLE_NAME, key, wred_profile_field_name, wred_profile_name);
1743 }
1744 }
1745 else if (op == DEL_COMMAND)
1746 {
1747 removeObject(QosOrch::getTypeMap(), CFG_QUEUE_TABLE_NAME, key);
1748 sai_scheduler_profile = SAI_NULL_OBJECT_ID;
1749 sai_wred_profile = SAI_NULL_OBJECT_ID;
1750 }
1751 else
1752 {
1753 SWSS_LOG_ERROR("Unknown operation type %s", op.c_str());
1754 return task_process_status::task_invalid_entry;
1755 }
1756
1757 for (string port_name : port_names)
1758 {
1759 Port port;
1760 SWSS_LOG_DEBUG("processing port:%s", port_name.c_str());
1761 if (!gPortsOrch->getPort(port_name, port))
1762 {
1763 SWSS_LOG_ERROR("Port with alias:%s not found", port_name.c_str());
1764 return task_process_status::task_invalid_entry;
1765 }
1766 SWSS_LOG_DEBUG("processing range:%d-%d", range_low, range_high);
1767 for (size_t ind = range_low; ind <= range_high; ind++)
1768 {
1769 queue_ind = ind;
1770 SWSS_LOG_DEBUG("processing queue:%zd", queue_ind);
1771
1772 if (!donotChangeScheduler)
1773 {
1774 result = applySchedulerToQueueSchedulerGroup(port, queue_ind, sai_scheduler_profile);
1775
1776 if (!result)
1777 {
1778 SWSS_LOG_ERROR("Failed setting field:%s to port:%s, queue:%zd, line:%d", scheduler_field_name.c_str(), port.m_alias.c_str(), queue_ind, __LINE__);
1779 return task_process_status::task_failed;
1780 }
1781 SWSS_LOG_DEBUG("Applied scheduler to port:%s", port_name.c_str());
1782 }
1783
1784 if (!donotChangeWredProfile)
1785 {
1786 result = applyWredProfileToQueue(port, queue_ind, sai_wred_profile);
1787
1788 if (!result)
1789 {
1790 SWSS_LOG_ERROR("Failed setting field:%s to port:%s, queue:%zd, line:%d", wred_profile_field_name.c_str(), port.m_alias.c_str(), queue_ind, __LINE__);
1791 return task_process_status::task_failed;
1792 }
1793 SWSS_LOG_DEBUG("Applied wred profile to port:%s", port_name.c_str());
1794 }
1795 }
1796 }
1797 SWSS_LOG_DEBUG("finished");
1798 return task_process_status::task_success;
1799}

1895task_process_status QosOrch::handlePortQosMapTable(Consumer& consumer, KeyOpFieldsValuesTuple &tuple)
1896{
1897 SWSS_LOG_ENTER();
1898
1899 string key = kfvKey(tuple);
1900 string op = kfvOp(tuple);
1901
1902 if (key == PORT_NAME_GLOBAL)
1903 {
1904 return handleGlobalQosMap(op, tuple);
1905 }
1906
1907 vector<string> port_names = tokenize(key, list_item_delimiter);
1908
1909 if (op == DEL_COMMAND)
1910 {
1911 /* Handle DEL command. Just set all the maps to oid:0x0 */
1912 for (string port_name : port_names)
1913 {
1914 Port port;
1915
1916 /* Skip port which is not found */
1917 if (!gPortsOrch->getPort(port_name, port))
1918 {
1919 SWSS_LOG_ERROR("Failed to apply QoS maps to port %s. Port is not found.", port_name.c_str());
1920 continue;
1921 }
1922
1923 for (auto &mapRef : qos_to_attr_map)
1924 {
1925 string referenced_obj;
1926 if (!doesObjectExist(m_qos_maps, CFG_PORT_QOS_MAP_TABLE_NAME, key, mapRef.first, referenced_obj))
1927 {
1928 continue;
1929 }
1930
1931 sai_attribute_t attr;
1932 attr.id = mapRef.second;
1933 attr.value.oid = SAI_NULL_OBJECT_ID;
1934
1935 sai_status_t status = sai_port_api->set_port_attribute(port.m_port_id, &attr);
1936 if (status != SAI_STATUS_SUCCESS)
1937 {
1938 SWSS_LOG_ERROR("Failed to remove %s on port %s, rv:%d",
1939 mapRef.first.c_str(), port_name.c_str(), status);
1940 task_process_status handle_status = handleSaiSetStatus(SAI_API_PORT, status);
1941 if (handle_status != task_process_status::task_success)
1942 {
1943 return task_process_status::task_invalid_entry;
1944 }
1945 }
1946 SWSS_LOG_INFO("Removed %s on port %s", mapRef.first.c_str(), port_name.c_str());
1947 }
1948
1949 if (!gPortsOrch->setPortPfc(port.m_port_id, 0))
1950 {
1951 SWSS_LOG_ERROR("Failed to disable PFC on port %s", port_name.c_str());
1952 }
1953
1954 SWSS_LOG_INFO("Disabled PFC on port %s", port_name.c_str());
1955 }
1956
1957 removeObject(m_qos_maps, CFG_PORT_QOS_MAP_TABLE_NAME, key);
1958
1959 return task_process_status::task_success;
1960 }
1961
1962 sai_uint8_t pfc_enable = 0;
1963 sai_uint8_t pfcwd_sw_enable = 0;
1964 map<sai_port_attr_t, pair<string, sai_object_id_t>> update_list;
1965 for (auto it = kfvFieldsValues(tuple).begin(); it != kfvFieldsValues(tuple).end(); it++)
1966 {
1967 /* Check all map instances are created before applying to ports */
1968 if (qos_to_attr_map.find(fvField(*it)) != qos_to_attr_map.end())
1969 {
1970 sai_object_id_t id;
1971 string object_name;
1972 string &map_type_name = fvField(*it), &map_name = fvValue(*it);
1973 ref_resolve_status status = resolveFieldRefValue(m_qos_maps, map_type_name, qos_to_ref_table_map.at(map_type_name), tuple, id, object_name);
1974
1975 if (status != ref_resolve_status::success)
1976 {
1977 SWSS_LOG_INFO("Port QoS map %s is not yet created", map_name.c_str());
1978 return task_process_status::task_need_retry;
1979 }
1980
1981 update_list[qos_to_attr_map[map_type_name]] = make_pair(map_name, id);
1982 setObjectReference(m_qos_maps, CFG_PORT_QOS_MAP_TABLE_NAME, key, map_type_name, object_name);
1983 }
1984
1985 else if (fvField(*it) == pfc_enable_name || fvField(*it) == pfcwd_sw_enable_name)
1986 {
1987 sai_uint8_t bitmask = 0;
1988 vector<string> queue_indexes;
1989 queue_indexes = tokenize(fvValue(*it), list_item_delimiter);
1990 for(string q_ind : queue_indexes)
1991 {
1992 sai_uint8_t q_val = (uint8_t)stoi(q_ind);
1993 bitmask |= (uint8_t)(1 << q_val);
1994 }
1995
1996 if (fvField(*it) == pfc_enable_name)
1997 {
1998 pfc_enable = bitmask;
1999 }
2000 else
2001 {
2002 pfcwd_sw_enable = bitmask;
2003 }
2004 }
2005 }
2006
2007 /* Remove any map that was configured but isn't there any longer. */
2008 for (auto &mapRef : qos_to_attr_map)
2009 {
2010 auto &sai_attribute = mapRef.second;
2011 if (update_list.find(sai_attribute) == update_list.end())
2012 {
2013 string referenced_obj;
2014 if (!doesObjectExist(m_qos_maps, CFG_PORT_QOS_MAP_TABLE_NAME, key, mapRef.first, referenced_obj))
2015 {
2016 continue;
2017 }
2018 SWSS_LOG_NOTICE("PORT_QOS_MAP|%s %s was configured but is not any more. Remove it", key.c_str(), mapRef.first.c_str());
2019 removeMeFromObjsReferencedByMe(m_qos_maps, CFG_PORT_QOS_MAP_TABLE_NAME, key, mapRef.first, referenced_obj);
2020 update_list[mapRef.second] = make_pair("NULL", SAI_NULL_OBJECT_ID);
2021 }
2022 }
2023
2024 for (string port_name : port_names)
2025 {
2026 Port port;
2027
2028 /* Skip port which is not found */
2029 if (!gPortsOrch->getPort(port_name, port))
2030 {
2031 SWSS_LOG_ERROR("Failed to apply QoS maps to port %s. Port is not found.", port_name.c_str());
2032 continue;
2033 }
2034
2035 /* Apply a list of attributes to be applied */
2036 for (auto it = update_list.begin(); it != update_list.end(); it++)
2037 {
2038 sai_attribute_t attr;
2039 attr.id = it->first;
2040 attr.value.oid = it->second.second;
2041
2042 sai_status_t status = sai_port_api->set_port_attribute(port.m_port_id, &attr);
2043 if (status != SAI_STATUS_SUCCESS)
2044 {
2045 SWSS_LOG_ERROR("Failed to apply %s to port %s, rv:%d",
2046 it->second.first.c_str(), port_name.c_str(), status);
2047 task_process_status handle_status = handleSaiSetStatus(SAI_API_PORT, status);
2048 if (handle_status != task_process_status::task_success)
2049 {
2050 return task_process_status::task_invalid_entry;
2051 }
2052 }
2053 SWSS_LOG_INFO("Applied %s to port %s", it->second.first.c_str(), port_name.c_str());
2054 }
2055
2056 sai_uint8_t old_pfc_enable = 0;
2057 if (!gPortsOrch->getPortPfc(port.m_port_id, &old_pfc_enable))
2058 {
2059 SWSS_LOG_ERROR("Failed to retrieve PFC bits on port %s", port_name.c_str());
2060 }
2061
2062 if (pfc_enable || old_pfc_enable)
2063 {
2064 if (!gPortsOrch->setPortPfc(port.m_port_id, pfc_enable))
2065 {
2066 SWSS_LOG_ERROR("Failed to apply PFC bits 0x%x to port %s", pfc_enable, port_name.c_str());
2067 }
2068
2069 SWSS_LOG_INFO("Applied PFC bits 0x%x to port %s", pfc_enable, port_name.c_str());
2070 }
2071
2072 // Save pfd_wd bitmask unconditionally
2073 gPortsOrch->setPortPfcWatchdogStatus(port.m_port_id, pfcwd_sw_enable);
2074 }
2075
2076 SWSS_LOG_NOTICE("Applied QoS maps to ports");
2077 return task_process_status::task_success;
2078}

2084 auto *port_qos_map_cfg_exec = getExecutor(CFG_PORT_QOS_MAP_TABLE_NAME);
