61std::unordered_map<sai_attr_id_t, sai_attribute_value_t> CreateAttributeListForL3AdmitObject(
62 const P4L3AdmitAppDbEntry &app_entry, const sai_object_id_t &port_oid)
63{
64 std::unordered_map<sai_attr_id_t, sai_attribute_value_t> my_mac_attrs;
65 sai_attribute_t my_mac_attr;
66
67 my_mac_attr.id = SAI_MY_MAC_ATTR_PRIORITY;
68 my_mac_attr.value.u32 = app_entry.priority;
69 my_mac_attrs.insert({my_mac_attr.id, my_mac_attr.value});

73 my_mac_attrs.insert({my_mac_attr.id, my_mac_attr.value});
74
75 my_mac_attr.id = SAI_MY_MAC_ATTR_MAC_ADDRESS_MASK;
76 memcpy(my_mac_attr.value.mac, app_entry.mac_address_mask.getMac(), sizeof(sai_mac_t));
77 my_mac_attrs.insert({my_mac_attr.id, my_mac_attr.value});
78
79 if (port_oid != SAI_NULL_OBJECT_ID)
80 {
81 my_mac_attr.id = SAI_MY_MAC_ATTR_PORT_ID;
82 my_mac_attr.value.oid = port_oid;
83 my_mac_attrs.insert({my_mac_attr.id, my_mac_attr.value});
84 }
85
86 return my_mac_attrs;

89// Verifies whether the attribute list is the same as expected.
90// Returns true if they match; otherwise, false.
91bool MatchCreateL3AdmitArgAttrList(const sai_attribute_t *attr_list,
92 const std::unordered_map<sai_attr_id_t, sai_attribute_value_t> &expected_attr_list)
93{
94 if (attr_list == nullptr)
95 {
96 return false;
97 }
98
99 // Sanity check for expected_attr_list.
100 const auto end = expected_attr_list.end();
101 if (expected_attr_list.size() < 3 || expected_attr_list.find(SAI_MY_MAC_ATTR_PRIORITY) == end ||
102 expected_attr_list.find(SAI_MY_MAC_ATTR_MAC_ADDRESS) == end ||
103 expected_attr_list.find(SAI_MY_MAC_ATTR_MAC_ADDRESS_MASK) == end)
104 {
105 return false;
106 }
107
108 size_t valid_attrs_num = 0;
109 for (size_t i = 0; i < expected_attr_list.size(); ++i)
110 {
111 switch (attr_list[i].id)
112 {
113 case SAI_MY_MAC_ATTR_PRIORITY: {
114 if (attr_list[i].value.u32 != expected_attr_list.at(SAI_MY_MAC_ATTR_PRIORITY).u32)
115 {
116 return false;
117 }
118 valid_attrs_num++;
119 break;
120 }
121 case SAI_MY_MAC_ATTR_MAC_ADDRESS: {
122 auto macaddr = swss::MacAddress(attr_list[i].value.mac);
123 auto expected_macaddr = swss::MacAddress(expected_attr_list.at(SAI_MY_MAC_ATTR_MAC_ADDRESS).mac);
124 if (macaddr != expected_macaddr)
125 {
126 return false;
127 }
128 valid_attrs_num++;
129 break;
130 }
131 case SAI_MY_MAC_ATTR_MAC_ADDRESS_MASK: {
132 auto macaddr = swss::MacAddress(attr_list[i].value.mac);
133 auto expected_macaddr = swss::MacAddress(expected_attr_list.at(SAI_MY_MAC_ATTR_MAC_ADDRESS_MASK).mac);
134 if (macaddr != expected_macaddr)
135 {
136 return false;
137 }
138 valid_attrs_num++;
139 break;
140 }
141 case SAI_MY_MAC_ATTR_PORT_ID: {
142 if (expected_attr_list.find(SAI_MY_MAC_ATTR_PORT_ID) == end ||
143 expected_attr_list.at(SAI_MY_MAC_ATTR_PORT_ID).oid != attr_list[i].value.oid)
144 {
145 return false;
146 }
147 valid_attrs_num++;
148 break;
149 }
150 default:
151 return false;
152 }
153 }
154
155 if (expected_attr_list.size() != valid_attrs_num)
156 {
157 return false;
158 }
159
160 return true;
161}
