49void AsicView::fromDump(
50 _In_ const swss::TableDump &dump)
51{
52 SWSS_LOG_ENTER();
53
54 /*
55 * Input should be also existing objects, so they could be created
56 * here right away but we would need VIDs as well.
57 */
58
59 int switchesCount = 0;
60
61 for (const auto &key: dump)
62 {
63 auto start = key.first.find_first_of(":");
64
65 if (start == std::string::npos)
66 {
67 SWSS_LOG_THROW("failed to find colon in %s", key.first.c_str());
68 }
69
70 std::shared_ptr<SaiObj> o = std::make_shared<SaiObj>();
71
72 // TODO we could use sai deserialize object meta key
73
74 o->m_str_object_type = key.first.substr(0, start);
75 o->m_str_object_id = key.first.substr(start + 1);
76
77 sai_deserialize_object_type(o->m_str_object_type, o->m_meta_key.objecttype);
78
79 o->m_info = sai_metadata_get_object_type_info(o->m_meta_key.objecttype);
80
81 /*
82 * Since neighbor/route/fdb structs objects contains OIDs, we
83 * need to increase vid reference. With new metadata for SAI
84 * 1.0 this can be done in generic way for all non object ids.
85 */
86
87 switch (o->m_meta_key.objecttype)
88 {
89 case SAI_OBJECT_TYPE_FDB_ENTRY:
90 sai_deserialize_fdb_entry(o->m_str_object_id, o->m_meta_key.objectkey.key.fdb_entry);
91 m_soFdbs[o->m_str_object_id] = o;
92 break;
93
94 case SAI_OBJECT_TYPE_NEIGHBOR_ENTRY:
95 sai_deserialize_neighbor_entry(o->m_str_object_id, o->m_meta_key.objectkey.key.neighbor_entry);
96 m_soNeighbors[o->m_str_object_id] = o;
97
98 m_neighborsByIp[sai_serialize_ip_address(o->m_meta_key.objectkey.key.neighbor_entry.ip_address)].push_back(o->m_str_object_id);
99
100 break;
101
102 case SAI_OBJECT_TYPE_ROUTE_ENTRY:
103 sai_deserialize_route_entry(o->m_str_object_id, o->m_meta_key.objectkey.key.route_entry);
104 m_soRoutes[o->m_str_object_id] = o;
105
106 m_routesByPrefix[sai_serialize_ip_prefix(o->m_meta_key.objectkey.key.route_entry.destination)].push_back(o->m_str_object_id);
107
108 break;
109
110 case SAI_OBJECT_TYPE_NAT_ENTRY:
111 sai_deserialize_nat_entry(o->m_str_object_id, o->m_meta_key.objectkey.key.nat_entry);
112 m_soNatEntries[o->m_str_object_id] = o;
113 break;
114
115 case SAI_OBJECT_TYPE_INSEG_ENTRY:
116 sai_deserialize_inseg_entry(o->m_str_object_id, o->m_meta_key.objectkey.key.inseg_entry);
117 m_soInsegs[o->m_str_object_id] = o;
118 break;
119
120 default:
121
122 if (o->m_info->isnonobjectid)
123 {
124 SWSS_LOG_THROW("object %s is non object id, not handled, FIXME", key.first.c_str());
125 }
126
127 sai_deserialize_object_id(o->m_str_object_id, o->m_meta_key.objectkey.key.object_id);
128
129 m_soOids[o->m_str_object_id] = o;
130 m_oOids[o->m_meta_key.objectkey.key.object_id] = o;
131
132 if (o->m_meta_key.objecttype == SAI_OBJECT_TYPE_SWITCH)
133 {
134 switchesCount++;
135 }
136
137 break;
138 }
139
140 m_soAll[o->m_str_object_id] = o;
141 m_sotAll[o->m_meta_key.objecttype][o->m_str_object_id] = o;
142
143 if (o->m_info->isnonobjectid)
144 {
145 updateNonObjectIdVidReferenceCountByValue(o, 1);
146 }
147 else
148 {
149 /*
150 * Here is only object VID declaration, since we don't
151 * know what objects were processed previously but on
152 * some of previous object attributes this VID could be
153 * used, so value can be already greater than zero, but
154 * here we need to just mark that vid exists in
155 * vidReference.
156 */
157
158 m_vidReference[o->m_meta_key.objectkey.key.object_id] += 0;
159 }
160
161 populateAttributes(o, key.second);
162 }
163
164 if (switchesCount != 1)
165 {
166 // NOTE: In our solution multiple switches are not supported in single AsicView
167
168 SWSS_LOG_THROW("only one switch is expected in ASIC view, got: %d switches", switchesCount);
169 }
170}

226void AsicView::releaseVidReference(
227 _In_ sai_object_id_t vid)
228{
229 SWSS_LOG_ENTER();
230
231 if (vid == SAI_NULL_OBJECT_ID)
232 {
233 return;
234 }
235
236 auto it = m_vidReference.find(vid);
237
238 if (it == m_vidReference.end())
239 {
240 SWSS_LOG_THROW("vid %s doesn't exist in reference map",
241 sai_serialize_object_id(vid).c_str());
242 }
243
244 int referenceCount = --(it->second);
245
246 if (referenceCount < 0)
247 {
248 SWSS_LOG_THROW("vid %s decreased reference count too many times: %d, BUG",
249 sai_serialize_object_id(vid).c_str(),
250 referenceCount);
251 }
252
253 SWSS_LOG_INFO("decreased vid %s refrence to %d",
254 sai_serialize_object_id(vid).c_str(),
255 referenceCount);
256
257 if (referenceCount == 0)
258 {
259 m_vidToAsicOperationId[vid] = m_asicOperationId;
260 }

321{
322 SWSS_LOG_ENTER();
323
324 if (vid == SAI_NULL_OBJECT_ID)
325 {

328
329 /*

347

349

563void AsicView::asicSetAttribute(
564 _In_ const std::shared_ptr<SaiObj> &currentObj,
565 _In_ const std::shared_ptr<SaiAttr> &attr)
566{
567 SWSS_LOG_ENTER();
568
569 SWSS_LOG_INFO("%s: %s -> %s:%s", currentObj->m_str_object_type.c_str(), currentObj->m_str_object_id.c_str(),
570 attr->getStrAttrId().c_str(), attr->getStrAttrValue().c_str());
571
572 m_asicOperationId++;
573
574 /*
575 * Release previous references if attribute is object id and bind
576 * new reference in that place.
577 */
578
579 auto meta = attr->getAttrMetadata();
580
581 auto currentAttr = currentObj->tryGetSaiAttr(meta->attrid);
582
583 if (attr->isObjectIdAttr())
584 {
585 if (currentObj->hasAttr(meta->attrid))
586 {
587 /*
588 * Since previous attribute exists, lets release previous links if
589 * they are not NULL.
590 */
591
592 releaseExisgingLinks(currentObj->getSaiAttr(meta->attrid));
593 }
594
595 currentObj->setAttr(attr);
596
597 bindNewLinks(currentObj->getSaiAttr(meta->attrid));
598 }
599 else
600 {
601 /*
602 * This SET don't contain any OIDs so no extra operations are required,
603 * we don't need to break any references and decrease any
604 * reference count.
605 *
606 * Also this attribute may exist already on this object or it will be
607 * just set now, so just in case lets make copy of it.
608 *
609 * Making copy here is not necessary since default attribute will be
610 * created dynamically anyway, and temporary attributes will not change
611 * also.
612 */
613
614 currentObj->setAttr(attr);
615 }
616
617 auto entry = SaiAttributeList::serialize_attr_list(
618 currentObj->getObjectType(),
619 1,
620 attr->getSaiAttr(),
621 false);
622
623 std::string key = currentObj->m_str_object_type + ":" + currentObj->m_str_object_id;
624
625 auto kco = std::make_shared<swss::KeyOpFieldsValuesTuple>(key, "set", entry);
626
627 sai_object_id_t vid = (currentObj->isOidObject()) ? currentObj->getVid() : SAI_NULL_OBJECT_ID;
628
629 m_asicOperations.push_back(AsicOperation(m_asicOperationId, vid, false, kco));
630
631 if (currentAttr)
632 {
633 // if current attribute exists, save it value for log purpose
634 m_asicOperations.rbegin()->m_currentValue = currentAttr->getStrAttrValue();
635 }
636
637 dumpRef("set");
638}

653void AsicView::asicCreateObject(
654 _In_ const std::shared_ptr<SaiObj> &currentObj)
655{
656 SWSS_LOG_ENTER();
657
658 SWSS_LOG_INFO("%s: %s", currentObj->m_str_object_type.c_str(), currentObj->m_str_object_id.c_str());
659
660 m_asicOperationId++;
661
662 if (currentObj->isOidObject())
663 {
664 m_soOids[currentObj->m_str_object_id] = currentObj;
665 m_oOids[currentObj->m_meta_key.objectkey.key.object_id] = currentObj;
666
667 m_soAll[currentObj->m_str_object_id] = currentObj;
668 m_sotAll[currentObj->m_meta_key.objecttype][currentObj->m_str_object_id] = currentObj;
669
670 /*
671 * Since we are creating object, we just need to mark that
672 * reference is there. But at this point object is not used
673 * anywhere.
674 */
675
676 m_vidReference[currentObj->m_meta_key.objectkey.key.object_id] += 0;
677 }
678 else
679 {
680 /*
681 * Since neighbor/route/fdb structs objects contains OIDs, we
682 * need to increase vid reference. With new metadata for SAI
683 * 1.0 this can be done in generic way for all non object ids.
684 */
685
686 // TODO why we are doing deserialize here? meta key should be populated already
687
688 switch (currentObj->getObjectType())
689 {
690 case SAI_OBJECT_TYPE_FDB_ENTRY:
691 //sai_deserialize_fdb_entry(currentObj->m_str_object_id, currentObj->m_meta_key.objectkey.key.fdb_entry);
692 m_soFdbs[currentObj->m_str_object_id] = currentObj;
693 break;
694
695 case SAI_OBJECT_TYPE_NEIGHBOR_ENTRY:
696 //sai_deserialize_neighbor_entry(currentObj->m_str_object_id, currentObj->m_meta_key.objectkey.key.neighbor_entry);
697 m_soNeighbors[currentObj->m_str_object_id] = currentObj;
698 break;
699
700 case SAI_OBJECT_TYPE_ROUTE_ENTRY:
701 //sai_deserialize_route_entry(currentObj->m_str_object_id, currentObj->m_meta_key.objectkey.key.route_entry);
702 m_soRoutes[currentObj->m_str_object_id] = currentObj;
703 break;
704
705 case SAI_OBJECT_TYPE_NAT_ENTRY:
706 m_soNatEntries[currentObj->m_str_object_id] = currentObj;
707 break;
708
709 case SAI_OBJECT_TYPE_INSEG_ENTRY:
710 m_soInsegs[currentObj->m_str_object_id] = currentObj;
711 break;
712
713 default:
714
715 SWSS_LOG_THROW("unsupported object type: %s",
716 sai_serialize_object_type(currentObj->getObjectType()).c_str());
717 }
718
719 m_soAll[currentObj->m_str_object_id] = currentObj;
720 m_sotAll[currentObj->m_meta_key.objecttype][currentObj->m_str_object_id] = currentObj;
721
722 updateNonObjectIdVidReferenceCountByValue(currentObj, 1);
723 }
724
725 bindNewLinks(currentObj); // handle attribute references
726
727 /*
728 * Generate asic commands.
729 */
730
731 std::vector<swss::FieldValueTuple> entry;
732
733 for (auto const &pair: currentObj->getAllAttributes())
734 {
735 const auto &attr = pair.second;
736
737 swss::FieldValueTuple fvt(attr->getStrAttrId(), attr->getStrAttrValue());
738
739 entry.push_back(fvt);
740 }
741
742 if (entry.size() == 0)
743 {
744 /*
745 * Make sure that we put object into db even if there are no
746 * attributes set.
747 */
748
749 swss::FieldValueTuple null("NULL", "NULL");
750
751 entry.push_back(null);
752 }
753
754 std::string key = currentObj->m_str_object_type + ":" + currentObj->m_str_object_id;
755
756 auto kco = std::make_shared<swss::KeyOpFieldsValuesTuple>(key, "create", entry);
757
758 sai_object_id_t vid = (currentObj->isOidObject()) ? currentObj->getVid() : SAI_NULL_OBJECT_ID;
759
760 m_asicOperations.push_back(AsicOperation(m_asicOperationId, vid, false, kco));
761
762 dumpRef("create");
763}

770void AsicView::asicRemoveObject(
771 _In_ const std::shared_ptr<SaiObj> &currentObj)
772{
773 SWSS_LOG_ENTER();
774
775 SWSS_LOG_INFO("%s: %s", currentObj->m_str_object_type.c_str(), currentObj->m_str_object_id.c_str());
776
777 if (currentObj->getObjectStatus() != SAI_OBJECT_STATUS_NOT_PROCESSED)
778 {
779 SWSS_LOG_THROW("FATAL: removing object with status: %d, logic error", currentObj->getObjectStatus());
780 }
781
782 m_asicOperationId++;
783
784 if (currentObj->isOidObject())
785 {
786 /*
787 * Reference count is already check externally, but we can move
788 * that check here also as sanity check.
789 */
790
791 int count = getVidReferenceCount(currentObj->getVid());
792
793 if (count != 0)
794 {
795 SWSS_LOG_THROW("can't remove existing object %s:%s since reference count is %d, FIXME",
796 currentObj->m_str_object_type.c_str(),
797 currentObj->m_str_object_id.c_str(),
798 count);
799 }
800
801 m_soOids.erase(currentObj->m_str_object_id);
802 m_oOids.erase(currentObj->m_meta_key.objectkey.key.object_id);
803
804 m_soAll.erase(currentObj->m_str_object_id);
805 m_sotAll.at(currentObj->m_meta_key.objecttype).erase(currentObj->m_str_object_id);
806
807 m_vidReference[currentObj->m_meta_key.objectkey.key.object_id] -= 1;
808
809 /*
810 * Clear object also from rid/vid maps.
811 */
812
813 sai_object_id_t vid = currentObj->getVid();
814 sai_object_id_t rid = m_vidToRid.at(vid);
815
816 /*
817 * This will have impact on translate_vid_to_rid, we need to put
818 * this in other view.
819 */
820
821 m_ridToVid.erase(rid);
822 m_vidToRid.erase(vid);
823
824 /*
825 * We could remove this VID also from m_vidReference, but it's not
826 * required.
827 */
828
829 m_removedVidToRid[vid] = rid;
830 }
831 else
832 {
833 /*
834 * Since neighbor/route/fdb structs objects contains OIDs, we
835 * need to decrease VID reference. With new metadata for SAI
836 * 1.0 this can be done in generic way for all non object ids.
837 */
838
839 switch (currentObj->getObjectType())
840 {
841 case SAI_OBJECT_TYPE_FDB_ENTRY:
842 m_soFdbs.erase(currentObj->m_str_object_id);
843 break;
844
845 case SAI_OBJECT_TYPE_NEIGHBOR_ENTRY:
846 m_soNeighbors.erase(currentObj->m_str_object_id);
847 break;
848
849 case SAI_OBJECT_TYPE_ROUTE_ENTRY:
850 m_soRoutes.erase(currentObj->m_str_object_id);
851 break;
852
853 case SAI_OBJECT_TYPE_NAT_ENTRY:
854 m_soNatEntries.erase(currentObj->m_str_object_id);
855 break;
856
857 case SAI_OBJECT_TYPE_INSEG_ENTRY:
858 m_soInsegs.erase(currentObj->m_str_object_id);
859 break;
860
861 default:
862
863 SWSS_LOG_THROW("unsupported object type: %s",
864 sai_serialize_object_type(currentObj->getObjectType()).c_str());
865 }
866
867 m_soAll.erase(currentObj->m_str_object_id);
868 m_sotAll.at(currentObj->m_meta_key.objecttype).erase(currentObj->m_str_object_id);
869
870 updateNonObjectIdVidReferenceCountByValue(currentObj, -1);
871 }
872
873 releaseExisgingLinks(currentObj); // handle attribute references
874
875 /*
876 * Generate asic commands.
877 */
878
879 std::vector<swss::FieldValueTuple> entry;
880
881 std::string key = currentObj->m_str_object_type + ":" + currentObj->m_str_object_id;
882
883 auto kco = std::make_shared<swss::KeyOpFieldsValuesTuple>(key, "remove", entry);
884
885 sai_object_id_t vid = (currentObj->isOidObject()) ? currentObj->getVid() : SAI_NULL_OBJECT_ID;
886
887 if (currentObj->isOidObject())
888 {
889 m_asicOperations.push_back(AsicOperation(m_asicOperationId, vid, true, kco));
890 }
891 else
892 {
893 /*
894 * When doing remove of non object id, let's put it on front,
895 * since when removing next hop group from group last member,
896 * and group is in use by some route, then remove fails since
897 * group cant be empty.
898 *
899 * Of course this doesn't guarantee that remove all routes will
900 * be at the beginning, also it may happen that default route
901 * will be removed first which maybe not allowed.
902 */
903
904 m_asicRemoveOperationsNonObjectId.push_back(AsicOperation(m_asicOperationId, vid, true, kco));
905 }
906
907 dumpRef("remove");
908}

910std::vector<AsicOperation> AsicView::asicGetWithOptimizedRemoveOperations() const
911{
912 SWSS_LOG_ENTER();
913
914 SWSS_LOG_TIMER("optimizing asic remove operations");
915
916 std::vector<AsicOperation> v;
917
918 /*
919 * First push remove operations on non object id at the beginning of list.
920 */
921
922 for (const auto &n: m_asicRemoveOperationsNonObjectId)
923 {
924 v.push_back(n);
925 }
926
927 size_t index = v.size();
928
929 size_t moved = 0;
930
931 for (auto opit = m_asicOperations.begin(); opit != m_asicOperations.end(); ++opit)
932 {
933 const auto &op = *opit;
934
935 if (!op.m_isRemove)
936 {
937 /*
938 * This is create or set operation, put it at the list end.
939 */
940
941 v.push_back(op);
942
943 continue;
944 }
945
946 /*
947 * NOTE: When operation is SET on OID object it can release
948 * references on that OID, and if that OID is later to be
949 * removed a wrong order of operations will happen:
950 *
951 * not optimized scenario:
952 * - create object A
953 * - set object B attr (releases reference on C)
954 * - remove object D (release reference on C)
955 * - remove object C
956 *
957 * this logic could result with wrong order:
958 * - remove D
959 * - remove C (will break reference count on C, since used in B)
960 * since D is considered last OP releasing C reference
961 * - create A
962 * - set object B with A
963 *
964 * correct optimized logic:
965 * - remove D
966 * - create A
967 * - set object B with A
968 * - remove C
969 *
970 * This is fixed by checking if last operation to release reference was REMOVE
971 */
972
973 if (op.m_vid == SAI_NULL_OBJECT_ID)
974 {
975 SWSS_LOG_THROW("non object id remove not expected here");
976 }
977
978 auto mit = m_vidToAsicOperationId.find(op.m_vid);
979
980 if (mit == m_vidToAsicOperationId.end())
981 {
982 /*
983 * This vid is not present in map, so we can move remove
984 * operation all the way to the top since there is no
985 * operation that decreased this vid reference.
986 *
987 * This can be NHG member, vlan member etc.
988 */
989
990 v.insert(v.begin() + index, op);
991
992 auto ot = VidManager::objectTypeQuery(op.m_vid);
993
994 SWSS_LOG_INFO("move %s all way up (not in map): %s to index: %zu",
995 sai_serialize_object_id(op.m_vid).c_str(),
996 sai_serialize_object_type(ot).c_str(),
997 index);
998
999 index++;
1000
1001 moved++;
1002
1003 continue;
1004 }
1005
1006 /*
1007 * This last operation id that decreased VID reference to zero
1008 * can be before or after current iterator, so it may be not
1009 * found on list from current iterator to list end. This will
1010 * mean that we can insert this remove at current iterator
1011 * position.
1012 *
1013 * If operation is found then we need to insert this op after
1014 * current iterator and forward iterator.
1015 *
1016 * But there is a catch here, if that last operation was REMOVE,
1017 * then it was forwarded all the way to the UP, but in the
1018 * middle we could have some "SET" operations that would
1019 * release this reference also we in this case, we can't just
1020 * move this remove just after previous remove.
1021 */
1022
1023 int lastOpIdDecRef = mit->second;
1024
1025 auto itr = find_if(v.begin(), v.end(), [lastOpIdDecRef] (const AsicOperation& ao) { return ao.m_opId == lastOpIdDecRef; } );
1026
1027 if (itr == v.end())
1028 {
1029 SWSS_LOG_THROW("something wrong, vid %s in map, but not found on list!",
1030 sai_serialize_object_id(op.m_vid).c_str());
1031 }
1032
1033 if (itr->m_isRemove)
1034 {
1035 SWSS_LOG_INFO("previous operation to release reference %s was also REMOVE, we push current REMOVE op at the list end",
1036 sai_serialize_object_id(mit->first).c_str());
1037
1038 /*
1039 * If previous operation was REMOVE (it could be pushed to
1040 * the top) so push current operation to the list here,
1041 * instead of possible break reference count on SET
1042 * operations.
1043 */
1044
1045 v.push_back(op);
1046
1047 continue;
1048 }
1049
1050 /*
1051 * We add +1 since we need to insert current object AFTER the one that we found
1052 */
1053
1054 size_t lastOpIdDecRefIndex = itr - v.begin() + 1;
1055
1056 if (lastOpIdDecRefIndex > index)
1057 {
1058 SWSS_LOG_INFO("index update from %zu to %zu", index, lastOpIdDecRefIndex);
1059
1060 index = lastOpIdDecRefIndex;
1061 }
1062
1063 v.insert(v.begin() + index, op);
1064
1065 auto ot = VidManager::objectTypeQuery(op.m_vid);
1066
1067 SWSS_LOG_INFO("move 0x%" PRIx64 " in the middle up: %s (last: %zu curr: %zu)",
1068 op.m_vid,
1069 sai_serialize_object_type(ot).c_str(),
1070 lastOpIdDecRefIndex,
1071 index);
1072
1073 index++;
1074
1075 moved++;
1076 }
1077
1078 SWSS_LOG_NOTICE("moved %zu REMOVE operations upper in stack from total %zu operations", moved, v.size());
1079
1080 return v;
1081}

1178 _In_ const swss::TableMap &map)
1179{
1180 SWSS_LOG_ENTER();
1181
1182 for (const auto& field: map)

1184 std::shared_ptr<SaiAttr> attr = std::make_shared<SaiAttr>(field.first, field.second);

1199
1200 continue;
1201
1202 default:

1204 }
