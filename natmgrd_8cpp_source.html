82void cleanup()
83{
84 int ret = 0;
85 std::string res;
86 const std::string conntrackFlush = "conntrack -F";
87
88 SWSS_LOG_ENTER();
89
90 /*If there are any conntrack entries, clean them */
91 ret = swss::exec(conntrackFlush, res);
92 if (ret)
93 {
94 SWSS_LOG_ERROR("Command '%s' failed with rc %d", conntrackFlush.c_str(), ret);
95 }
96
97 /* Send notification to Orchagent to clean up the REDIS and ASIC database */
98 if (cleanupNotifier != NULL)
99 {
100 SWSS_LOG_NOTICE("Sending notification to orchagent to cleanup NAT entries in REDIS/ASIC");
101
102 std::vector<swss::FieldValueTuple> entry;
103
104 cleanupNotifier->send("nat_cleanup", "all", entry);
105 }
106
107 if (natmgr)

109 natmgr->removeStaticNatIptables();
110 natmgr->removeStaticNaptIptables();
111 natmgr->removeDynamicNatRules();

113 natmgr->cleanupMangleIpTables();
114 natmgr->cleanupPoolIpTable();

123 SWSS_LOG_NOTICE("--- Starting natmgrd ---");

125 try
126 {
127 vector<string> cfg_tables = {
128 CFG_STATIC_NAT_TABLE_NAME,
129 CFG_STATIC_NAPT_TABLE_NAME,
130 CFG_NAT_POOL_TABLE_NAME,

140
141 DBConnector cfgDb("CONFIG_DB", 0);
142 DBConnector appDb("APPL_DB", 0);
143 DBConnector stateDb("STATE_DB", 0);
144
145 cleanupNotifier = std::make_shared<swss::NotificationProducer>(&appDb, "NAT_DB_CLEANUP_NOTIFICATION");
146
147 struct sigaction sigact = {};
148 sigact.sa_handler = sigterm_handler;
149 if (sigaction(SIGTERM, &sigact, &old_sigaction))
150 {
151 SWSS_LOG_ERROR("failed to setup SIGTERM action handler");
152 exit(EXIT_FAILURE);
153 }
154
155 natmgr = new NatMgr(&cfgDb, &appDb, &stateDb, cfg_tables);
156
157 natmgr->isPortInitDone(&appDb);
158
159 std::vector<Orch *> cfgOrchList = {natmgr};
160
161 swss::Select s;
162 for (Orch *o : cfgOrchList)
163 {
164 s.addSelectables(o->getSelectables());
165 }
166
167 timeoutNotificationsConsumer = new NotificationConsumer(&appDb, "SETTIMEOUTNAT");
168 s.addSelectable(timeoutNotificationsConsumer);
169
170 flushNotificationsConsumer = new NotificationConsumer(&appDb, "FLUSHNATENTRIES");
171 s.addSelectable(flushNotificationsConsumer);
172
173 SWSS_LOG_NOTICE("starting main loop");
174 while (!gExit)
175 {
176 Selectable *sel;
177 int ret;
178
179 ret = s.select(&sel, SELECT_TIMEOUT);
180 if (ret == Select::ERROR)
181 {
182 SWSS_LOG_NOTICE("Error: %s!", strerror(errno));
183 continue;
184 }
185
186 if (sel == timeoutNotificationsConsumer)
187 {
188 std::string op;
189 std::string data;
190 std::vector<swss::FieldValueTuple> values;
191
192 timeoutNotificationsConsumer->pop(op, data, values);
193 natmgr->timeoutNotifications(op, data);
194 continue;
195 }
196
197 if (sel == flushNotificationsConsumer)
198 {
199 std::string op;
200 std::string data;
201 std::vector<swss::FieldValueTuple> values;
202
203 flushNotificationsConsumer->pop(op, data, values);
204 natmgr->flushNotifications(op, data);
205 continue;
206 }
207
208 if (ret == Select::TIMEOUT)
209 {
210 natmgr->doTask();
211 continue;
212 }
213
214 auto *c = (Executor *)sel;
215 c->execute();
216 }
217
218 cleanup();
219 }
220 catch(const std::exception &e)
221 {
222 SWSS_LOG_ERROR("Runtime error: %s", e.what());
223 return EXIT_FAILURE;
224 }
225
226 return 0;
