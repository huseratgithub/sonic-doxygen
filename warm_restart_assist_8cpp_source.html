131void AppRestartAssist::readTablesToMap()
132{
133 vector<string> keys;
134
135 for (auto it = m_appTables.begin(); it != m_appTables.end(); it++)

167/*
168 * Check and insert to CacheMap Logic:
169 * if delete_key:
170 * mark the entry as "DELETE";
171 * else:
172 * if key exist {
173 * if it has different value: update with "NEW" flag.
174 * if same value: mark it as "SAME";
175 * } else {
176 * insert with "NEW" flag.
177 * }
178 */
179void AppRestartAssist::insertToMap(string tableName, string key, vector<FieldValueTuple> fvVector, bool delete_key)
180{
181 SWSS_LOG_INFO("Received message %s, key: %s, "
182 "%s, delete = %d", tableName.c_str(), key.c_str(), joinVectorString(fvVector).c_str(), delete_key);
183
184 auto found = appTableCacheMap[tableName].find(key);
185
186 if (delete_key)
187 {
188 SWSS_LOG_NOTICE("%s, delete key: %s, ", tableName.c_str(), key.c_str());
189 /* mark it as DELETE if exist, otherwise, no-op */
190 if (found != appTableCacheMap[tableName].end())
191 {
192 setCacheEntryState(found->second, DELETE);
193 }
194 }
195 else if (found != appTableCacheMap[tableName].end())
196 {
197 // check only the original vector range (exclude cache-state field/value)
198 if(! contains(found->second, fvVector))
199 {
200 SWSS_LOG_NOTICE("%s, found key: %s, new value ", tableName.c_str(), key.c_str());
201
202 FieldValueTuple state(CACHE_STATE_FIELD, "");
203 fvVector.push_back(state);
204
205 // mark as NEW flag
206 setCacheEntryState(fvVector, NEW);
207 appTableCacheMap[tableName][key] = fvVector;
208 }
209 else
210 {
211 SWSS_LOG_INFO("%s, found key: %s, same value", tableName.c_str(), key.c_str());
212
213 // mark as SAME flag
214 setCacheEntryState(found->second, SAME);
215 }
216 }
217 else
218 {
219 // not found, mark the entry as NEW and insert to map
220 SWSS_LOG_NOTICE("%s, not found key: %s, new", tableName.c_str(), key.c_str());
221 FieldValueTuple state(CACHE_STATE_FIELD, "");
222 fvVector.push_back(state);
223 setCacheEntryState(fvVector, NEW);
224 appTableCacheMap[tableName][key] = fvVector;
225 }
226 return;
227}

229/*
230 * Reconcile logic:
231 * iterate through the cache map
232 * if the entry has "SAME" flag, do nothing
233 * if has "STALE/DELETE" flag, delete it from appDB.
234 * else if "NEW" flag, add it to appDB
235 * else, throw (should never happen)
236 */
237void AppRestartAssist::reconcile()
238{
239 std::string tableName;
240
241 SWSS_LOG_ENTER();
242 for (auto tableIter = appTableCacheMap.begin(); tableIter != appTableCacheMap.end(); ++tableIter)
243 {
244 tableName = tableIter->first;
245 for (auto it = (tableIter->second).begin(); it != (tableIter->second).end(); ++it)
246 {
247 string s = joinVectorString(it->second);
248 auto state = getCacheEntryState(it->second);
249
250 if (state == SAME)
251 {
252 SWSS_LOG_INFO("%s SAME, key: %s, %s",
253 tableName.c_str(), it->first.c_str(), s.c_str());
254 continue;
255 }
256 else if (state == STALE || state == DELETE)
257 {
258 SWSS_LOG_NOTICE("%s STALE/DELETE, key: %s, %s",
259 tableName.c_str(), it->first.c_str(), s.c_str());
260
261 //delete from appDB
262 m_psTables[tableName]->del(it->first);
263 }
264 else if (state == NEW)
265 {
266 SWSS_LOG_NOTICE("%s NEW, key: %s, %s",
267 tableName.c_str(), it->first.c_str(), s.c_str());
268
269 //add to appDB, exclude the state
270 it->second.pop_back();
271 m_psTables[tableName]->set(it->first, it->second);
272 }
273 else
274 {
275 throw std::logic_error("cache entry state is invalid");
276 }
277 }
278 // reconcile finished, clear the map, mark the warmstart state
279 appTableCacheMap[tableName].clear();
280 }
281 appTableCacheMap.clear();
282 WarmStart::setWarmStartState(m_appName, WarmStart::RECONCILED);
283 m_warmStartInProgress = false;
284 return;
285}
