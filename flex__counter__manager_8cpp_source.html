52FlexCounterManager *FlexManagerDirectory::createFlexCounterManager(const string& group_name,
53 const StatsMode stats_mode,
54 const uint polling_interval,
55 const bool enabled,
56 FieldValueTuple fv_plugin)
57{
58 if (m_managers.find(group_name) != m_managers.end())
59 {
60 if (stats_mode != m_managers[group_name]->getStatsMode())
61 {
62 SWSS_LOG_ERROR("Stats mode mismatch with already created flex counter manager %s",
63 group_name.c_str());
64 return NULL;
65 }
66 if (polling_interval != m_managers[group_name]->getPollingInterval())
67 {
68 SWSS_LOG_ERROR("Polling interval mismatch with already created flex counter manager %s",
69 group_name.c_str());
70 return NULL;
71 }
72 if (enabled != m_managers[group_name]->getEnabled())
73 {
74 SWSS_LOG_ERROR("Enabled field mismatch with already created flex counter manager %s",
75 group_name.c_str());
76 return NULL;
77 }
78 return m_managers[group_name];
79 }
80 FlexCounterManager *fc_manager = new FlexCounterManager(group_name, stats_mode, polling_interval,
81 enabled, fv_plugin);
82 m_managers[group_name] = fc_manager;
83 return fc_manager;
84}

238 flex_counter_table->set(getFlexCounterTableKey(group_name, object_id), field_values);
239 installed_counters.insert(object_id);

241 SWSS_LOG_DEBUG("Updated flex counter id list for object '%" PRIu64 "' in group '%s'.",
242 object_id,

250 SWSS_LOG_ENTER();

254 {
255 SWSS_LOG_WARN("No counters found on object '%" PRIu64 "' in group '%s'.",

258 return;
259 }
260
261 flex_counter_table->del(getFlexCounterTableKey(group_name, object_id));
