25// Describes the format of a value.
26enum Format
27{
28 // Hex string, e.g. 0x0a8b. All lowercase, and always of length
29 // ceil(num_bits/4)+2 (1 character for every 4 bits, zero-padded to be
30 // divisible by 4, and 2 characters for the '0x' prefix).
31 HEX_STRING = 0,
32 // MAC address, e.g. 00:11:ab:cd:ef:22. All lowercase, and always 17
33 // characters long.
34 MAC = 1,
35 // IPv4 address, e.g. 10.0.0.2.
36 IPV4 = 2,
37 // IPv6 address, e.g. fe80::21a:11ff:fe17:5f80. All lowercase, formatted
38 // according to RFC5952. This can be used for any bitwidth of 128 or less. If
39 // the bitwidth n is less than 128, then by convention only the upper n bits
40 // can be set.
41 IPV6 = 3,
42 // String format, only printable characters.
43 STRING = 4,
44};
