97/*
98 * Function Name: processFDBFlushUpdate
99 * Description:
100 * Goal of this function is to delete neighbor/ARP entries
101 * when a port belonging to a VLAN gets removed.
102 * This function is called whenever neighbor orchagent receives
103 * SUBJECT_TYPE_FDB_FLUSH_CHANGE notification. Currently we only care for
104 * deleted FDB entries. We flush neighbor entry that matches its
105 * in-coming interface and MAC with FDB entry's VLAN name and MAC
106 * respectively.
107 */
108void NeighOrch::processFDBFlushUpdate(const FdbFlushUpdate& update)
109{
110 SWSS_LOG_INFO("processFDBFlushUpdate port: %s",
111 update.port.m_alias.c_str());
112
113 for (auto entry : update.entries)
114 {
115 // Get Vlan object
116 Port vlan;
117 if (!m_portsOrch->getPort(entry.bv_id, vlan))
118 {
119 SWSS_LOG_NOTICE("FdbOrch notification: Failed to locate vlan port \
120 from bv_id 0x%" PRIx64 ".", entry.bv_id);
121 continue;
122 }
123 SWSS_LOG_INFO("Flushing ARP for port: %s, VLAN: %s",
124 vlan.m_alias.c_str(), update.port.m_alias.c_str());
125
126 // If the FDB entry MAC matches with neighbor/ARP entry MAC,
127 // and ARP entry incoming interface matches with VLAN name,
128 // flush neighbor/arp entry.
129 for (const auto &neighborEntry : m_syncdNeighbors)
130 {
131 if (neighborEntry.first.alias == vlan.m_alias &&
132 neighborEntry.second.mac == entry.mac)
133 {
134 resolveNeighborEntry(neighborEntry.first, neighborEntry.second.mac);
135 }
136 }
137 }
138 return;

192 nh.ip_address.to_string().c_str(), nh.alias.c_str());
193 return false;

200 nh.ip_address.to_string().c_str(), nh.alias.c_str());
201 return false;
202 }
203 }
204

207 {
208 //For remote system ports kernel nexthops are always on inband. Change the key
209 Port inbp;
210 gPortsOrch->getInbandPort(inbp);
211 assert(inbp.m_alias.length());
212
213 nexthop.alias = inbp.m_alias;
214 }
215
216 assert(!hasNextHop(nexthop));
217 sai_object_id_t rif_id = m_intfsOrch->getRouterIntfsId(nh.alias);
218
219 vector<sai_attribute_t> next_hop_attrs;
220
221 vector<Label> label_stack;
222 sai_attribute_t next_hop_attr;
223 if (nexthop.isMplsNextHop())
224 {
225 next_hop_attr.id = SAI_NEXT_HOP_ATTR_TYPE;
226 next_hop_attr.value.s32 = SAI_NEXT_HOP_TYPE_MPLS;
227 next_hop_attrs.push_back(next_hop_attr);
228
229 next_hop_attr.id = SAI_NEXT_HOP_ATTR_OUTSEG_TYPE;
230 next_hop_attr.value.s32 = nexthop.label_stack.m_outseg_type;
231 next_hop_attrs.push_back(next_hop_attr);
232
233 next_hop_attr.id = SAI_NEXT_HOP_ATTR_LABELSTACK;
234 label_stack = nexthop.label_stack.getLabelStack();
235 next_hop_attr.value.u32list.list = label_stack.data();
236 next_hop_attr.value.u32list.count = static_cast<uint32_t>(nexthop.label_stack.getSize());
237 next_hop_attrs.push_back(next_hop_attr);
238 }
239 else
240 {
241 next_hop_attr.id = SAI_NEXT_HOP_ATTR_TYPE;
242 next_hop_attr.value.s32 = SAI_NEXT_HOP_TYPE_IP;
243 next_hop_attrs.push_back(next_hop_attr);
244 }
245
246 next_hop_attr.id = SAI_NEXT_HOP_ATTR_IP;
247 copy(next_hop_attr.value.ipaddr, nexthop.ip_address);
248 next_hop_attrs.push_back(next_hop_attr);
249
250 next_hop_attr.id = SAI_NEXT_HOP_ATTR_ROUTER_INTERFACE_ID;
251 next_hop_attr.value.oid = rif_id;
252 next_hop_attrs.push_back(next_hop_attr);
253
254 sai_object_id_t next_hop_id;
255 sai_status_t status = sai_next_hop_api->create_next_hop(&next_hop_id, gSwitchId, (uint32_t)next_hop_attrs.size(), next_hop_attrs.data());
256 if (status != SAI_STATUS_SUCCESS)

259 nexthop.ip_address.to_string().c_str(), nexthop.alias.c_str(), status);
260 task_process_status handle_status = handleSaiCreateStatus(SAI_API_NEXT_HOP, status);
261 if (handle_status != task_success)
262 {
263 return parseHandleSaiStatusFailure(handle_status);
264 }
265 }
266
267 SWSS_LOG_NOTICE("Created next hop %s on %s",
268 nexthop.ip_address.to_string().c_str(), nexthop.alias.c_str());
269 if (m_neighborToResolve.find(nexthop) != m_neighborToResolve.end())
270 {
271 clearResolvedNeighborEntry(nexthop);
272 m_neighborToResolve.erase(nexthop);
273 SWSS_LOG_INFO("Resolved neighbor for %s", nexthop.to_string().c_str());
274 }
275
276 NextHopEntry next_hop_entry;
277 next_hop_entry.next_hop_id = next_hop_id;
278 next_hop_entry.ref_count = 0;
279 next_hop_entry.nh_flags = 0;
280 m_syncdNextHops[nexthop] = next_hop_entry;
281
282 m_intfsOrch->increaseRouterIntfsRefCount(nexthop.alias);
283
284 if (nexthop.isMplsNextHop())
285 {
286 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_MPLS_NEXTHOP);
287 }
288 else
289 {
290 if (nexthop.ip_address.isV4())
291 {
292 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEXTHOP);
293 }
294 else
295 {
296 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEXTHOP);
297 }
298 }
299
300 gFgNhgOrch->validNextHopInNextHopGroup(nexthop);
301
302 // For nexthop with incoming port which has down oper status, NHFLAGS_IFDOWN
303 // flag should be set on it.
304 // This scenario may happen under race condition where buffered neighbor event
305 // is processed after incoming port is down.
306 if (p.m_oper_status == SAI_PORT_OPER_STATUS_DOWN)
307 {
308 if (setNextHopFlag(nexthop, NHFLAGS_IFDOWN) == false)
309 {
310 SWSS_LOG_WARN("Failed to set NHFLAGS_IFDOWN on nexthop %s for interface %s",
311 nexthop.ip_address.to_string().c_str(), nexthop.alias.c_str());
312 }
313 }
314 return true;

649void NeighOrch::doTask(Consumer &consumer)
650{
651 SWSS_LOG_ENTER();
652
653 if (!gPortsOrch->allPortsReady())
654 {
655 return;
656 }
657
658 string table_name = consumer.getTableName();
659 if(table_name == CHASSIS_APP_SYSTEM_NEIGH_TABLE_NAME)
660 {
661 doVoqSystemNeighTask(consumer);
662 return;
663 }
664
665 auto it = consumer.m_toSync.begin();
666 while (it != consumer.m_toSync.end())
667 {
668 KeyOpFieldsValuesTuple t = it->second;
669
670 string key = kfvKey(t);
671 string op = kfvOp(t);
672
673 size_t found = key.find(':');
674 if (found == string::npos)
675 {
676 SWSS_LOG_ERROR("Failed to parse key %s", key.c_str());
677 it = consumer.m_toSync.erase(it);
678 continue;
679 }
680
681 string alias = key.substr(0, found);
682
683 if (alias == "eth0" || alias == "lo" || alias == "docker0"
684 || ((op == SET_COMMAND) && m_intfsOrch->isInbandIntfInMgmtVrf(alias)))
685 {
686 it = consumer.m_toSync.erase(it);
687 continue;
688 }
689
690 if(gPortsOrch->isInbandPort(alias))
691 {
692 Port ibport;
693 gPortsOrch->getInbandPort(ibport);
694 if(ibport.m_type != Port::VLAN)
695 {
696 //For "port" type Inband, the neighbors are only remote neighbors.
697 //Hence, this is the neigh learned due to the kernel entry added on
698 //Inband interface for the remote system port neighbors. Skip
699 it = consumer.m_toSync.erase(it);
700 continue;
701 }
702 //For "vlan" type inband, may identify the remote neighbors and skip
703 }
704
705 IpAddress ip_address(key.substr(found+1));
706
707 NeighborEntry neighbor_entry = { ip_address, alias };
708
709 if (op == SET_COMMAND)
710 {
711 Port p;
712 if (!gPortsOrch->getPort(alias, p))
713 {
714 SWSS_LOG_INFO("Port %s doesn't exist", alias.c_str());
715 it++;
716 continue;
717 }
718
719 if (!p.m_rif_id)
720 {
721 SWSS_LOG_INFO("Router interface doesn't exist on %s", alias.c_str());
722 it++;
723 continue;
724 }
725
726 MacAddress mac_address;
727 for (auto i = kfvFieldsValues(t).begin();
728 i != kfvFieldsValues(t).end(); i++)
729 {
730 if (fvField(*i) == "neigh")
731 mac_address = MacAddress(fvValue(*i));
732 }
733
734 if (m_syncdNeighbors.find(neighbor_entry) == m_syncdNeighbors.end()
735 || m_syncdNeighbors[neighbor_entry].mac != mac_address)
736 {
737 // only for unresolvable neighbors that are new
738 if (!mac_address)
739 {
740 if (m_syncdNeighbors.find(neighbor_entry) == m_syncdNeighbors.end())
741 {
742 addZeroMacTunnelRoute(neighbor_entry, mac_address);
743 }
744 it = consumer.m_toSync.erase(it);
745 }
746 else if (addNeighbor(neighbor_entry, mac_address))
747 {
748 it = consumer.m_toSync.erase(it);
749 }
750 else
751 {
752 it++;
753 continue;
754 }
755 }
756 else
757 {
758 /* Duplicate entry */
759 it = consumer.m_toSync.erase(it);
760 }
761
762 /* Remove remaining DEL operation in m_toSync for the same neighbor.
763 * Since DEL operation is supposed to be executed before SET for the same neighbor
764 * A remaining DEL after the SET operation means the DEL operation failed previously and should not be executed anymore
765 */
766 auto rit = make_reverse_iterator(it);
767 while (rit != consumer.m_toSync.rend() && rit->first == key && kfvOp(rit->second) == DEL_COMMAND)
768 {
769 consumer.m_toSync.erase(next(rit).base());
770 SWSS_LOG_NOTICE("Removed pending neighbor DEL operation for %s after SET operation", key.c_str());
771 }
772 }
773 else if (op == DEL_COMMAND)
774 {
775 if (m_syncdNeighbors.find(neighbor_entry) != m_syncdNeighbors.end())
776 {
777 if (removeNeighbor(neighbor_entry))
778 {
779 it = consumer.m_toSync.erase(it);
780 }
781 else
782 {
783 it++;
784 }
785 }
786 else
787 /* Cannot locate the neighbor */
788 it = consumer.m_toSync.erase(it);
789 }
790 else
791 {
792 SWSS_LOG_ERROR("Unknown operation type %s", op.c_str());
793 it = consumer.m_toSync.erase(it);
794 }
795 }

798bool NeighOrch::addNeighbor(const NeighborEntry &neighborEntry, const MacAddress &macAddress)
799{
800 SWSS_LOG_ENTER();
801
802 sai_status_t status;
803 IpAddress ip_address = neighborEntry.ip_address;
804 string alias = neighborEntry.alias;
805
806 sai_object_id_t rif_id = m_intfsOrch->getRouterIntfsId(alias);
807 if (rif_id == SAI_NULL_OBJECT_ID)
808 {
809 SWSS_LOG_INFO("Failed to get rif_id for %s", alias.c_str());
810 return false;
811 }
812
813 sai_neighbor_entry_t neighbor_entry;
814 neighbor_entry.rif_id = rif_id;
815 neighbor_entry.switch_id = gSwitchId;
816 copy(neighbor_entry.ip_address, ip_address);
817
818 vector<sai_attribute_t> neighbor_attrs;
819 sai_attribute_t neighbor_attr;
820
821 neighbor_attr.id = SAI_NEIGHBOR_ENTRY_ATTR_DST_MAC_ADDRESS;
822 memcpy(neighbor_attr.value.mac, macAddress.getMac(), 6);
823 neighbor_attrs.push_back(neighbor_attr);
824
825 if ((ip_address.getAddrScope() == IpAddress::LINK_SCOPE) && (ip_address.isV4()))
826 {
827 /* Check if this prefix is a configured ip, if not allow */
828 IpPrefix ipll_prefix(ip_address.getV4Addr(), 16);
829 if (!m_intfsOrch->isPrefixSubnet (ipll_prefix, alias))
830 {
831 neighbor_attr.id = SAI_NEIGHBOR_ENTRY_ATTR_NO_HOST_ROUTE;
832 neighbor_attr.value.booldata = 1;
833 neighbor_attrs.push_back(neighbor_attr);
834 }
835 }
836
837 MuxOrch* mux_orch = gDirectory.get<MuxOrch*>();
838 bool hw_config = isHwConfigured(neighborEntry);
839
840 if (gMySwitchType == "voq")
841 {
842 if (!addVoqEncapIndex(alias, ip_address, neighbor_attrs))
843 {
844 return false;
845 }
846 }
847
848 if (!hw_config && mux_orch->isNeighborActive(ip_address, macAddress, alias))
849 {
850 status = sai_neighbor_api->create_neighbor_entry(&neighbor_entry,
851 (uint32_t)neighbor_attrs.size(), neighbor_attrs.data());
852 if (status != SAI_STATUS_SUCCESS)
853 {
854 if (status == SAI_STATUS_ITEM_ALREADY_EXISTS)
855 {
856 SWSS_LOG_ERROR("Entry exists: neighbor %s on %s, rv:%d",
857 macAddress.to_string().c_str(), alias.c_str(), status);
858 /* Returning True so as to skip retry */
859 return true;
860 }
861 else
862 {
863 SWSS_LOG_ERROR("Failed to create neighbor %s on %s, rv:%d",
864 macAddress.to_string().c_str(), alias.c_str(), status);
865 task_process_status handle_status = handleSaiCreateStatus(SAI_API_NEIGHBOR, status);
866 if (handle_status != task_success)
867 {
868 return parseHandleSaiStatusFailure(handle_status);
869 }
870 }
871 }
872 SWSS_LOG_NOTICE("Created neighbor ip %s, %s on %s", ip_address.to_string().c_str(),
873 macAddress.to_string().c_str(), alias.c_str());
874 m_intfsOrch->increaseRouterIntfsRefCount(alias);
875
876 if (neighbor_entry.ip_address.addr_family == SAI_IP_ADDR_FAMILY_IPV4)
877 {
878 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEIGHBOR);
879 }
880 else
881 {
882 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEIGHBOR);
883 }
884
885 if (!addNextHop(NextHopKey(ip_address, alias)))
886 {
887 status = sai_neighbor_api->remove_neighbor_entry(&neighbor_entry);
888 if (status != SAI_STATUS_SUCCESS)
889 {
890 SWSS_LOG_ERROR("Failed to remove neighbor %s on %s, rv:%d",
891 macAddress.to_string().c_str(), alias.c_str(), status);
892 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_NEIGHBOR, status);
893 if (handle_status != task_success)
894 {
895 return parseHandleSaiStatusFailure(handle_status);
896 }
897 }
898 m_intfsOrch->decreaseRouterIntfsRefCount(alias);
899
900 if (neighbor_entry.ip_address.addr_family == SAI_IP_ADDR_FAMILY_IPV4)
901 {
902 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEIGHBOR);
903 }
904 else
905 {
906 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEIGHBOR);
907 }
908
909 return false;
910 }
911 hw_config = true;
912 }
913 else if (isHwConfigured(neighborEntry))
914 {
915 status = sai_neighbor_api->set_neighbor_entry_attribute(&neighbor_entry, &neighbor_attr);
916 if (status != SAI_STATUS_SUCCESS)
917 {
918 SWSS_LOG_ERROR("Failed to update neighbor %s on %s, rv:%d",
919 macAddress.to_string().c_str(), alias.c_str(), status);
920 task_process_status handle_status = handleSaiSetStatus(SAI_API_NEIGHBOR, status);
921 if (handle_status != task_success)
922 {
923 return parseHandleSaiStatusFailure(handle_status);
924 }
925 }
926 SWSS_LOG_NOTICE("Updated neighbor %s on %s", macAddress.to_string().c_str(), alias.c_str());
927 }
928
929 m_syncdNeighbors[neighborEntry] = { macAddress, hw_config };
930
931 NeighborUpdate update = { neighborEntry, macAddress, true };
932 notify(SUBJECT_TYPE_NEIGH_CHANGE, static_cast<void *>(&update));
933
934 if(gMySwitchType == "voq")
935 {
936 //Sync the neighbor to add to the CHASSIS_APP_DB
937 voqSyncAddNeigh(alias, ip_address, macAddress, neighbor_entry);
938 }
939
940 return true;
941}

943bool NeighOrch::removeNeighbor(const NeighborEntry &neighborEntry, bool disable)
944{
945 SWSS_LOG_ENTER();
946
947 sai_status_t status;
948 IpAddress ip_address = neighborEntry.ip_address;
949 string alias = neighborEntry.alias;
950
951 NextHopKey nexthop = { ip_address, alias };
952 if(m_intfsOrch->isRemoteSystemPortIntf(alias))
953 {
954 //For remote system ports kernel nexthops are always on inband. Change the key
955 Port inbp;
956 gPortsOrch->getInbandPort(inbp);
957 assert(inbp.m_alias.length());
958
959 nexthop.alias = inbp.m_alias;
960 }
961
962 if (m_syncdNeighbors.find(neighborEntry) == m_syncdNeighbors.end())
963 {
964 return true;
965 }
966
967 if (m_syncdNextHops.find(nexthop) != m_syncdNextHops.end() && m_syncdNextHops[nexthop].ref_count > 0)
968 {
969 SWSS_LOG_INFO("Failed to remove still referenced neighbor %s on %s",
970 m_syncdNeighbors[neighborEntry].mac.to_string().c_str(), alias.c_str());
971 return false;
972 }
973
974 if (isHwConfigured(neighborEntry))
975 {
976 sai_object_id_t rif_id = m_intfsOrch->getRouterIntfsId(alias);
977
978 sai_neighbor_entry_t neighbor_entry;
979 neighbor_entry.rif_id = rif_id;
980 neighbor_entry.switch_id = gSwitchId;
981 copy(neighbor_entry.ip_address, ip_address);
982
983 sai_object_id_t next_hop_id = m_syncdNextHops[nexthop].next_hop_id;
984 status = sai_next_hop_api->remove_next_hop(next_hop_id);
985 if (status != SAI_STATUS_SUCCESS)
986 {
987 /* When next hop is not found, we continue to remove neighbor entry. */
988 if (status == SAI_STATUS_ITEM_NOT_FOUND)
989 {
990 SWSS_LOG_ERROR("Failed to locate next hop %s on %s, rv:%d",
991 ip_address.to_string().c_str(), alias.c_str(), status);
992 }
993 else
994 {
995 SWSS_LOG_ERROR("Failed to remove next hop %s on %s, rv:%d",
996 ip_address.to_string().c_str(), alias.c_str(), status);
997 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_NEXT_HOP, status);
998 if (handle_status != task_success)
999 {
1000 return parseHandleSaiStatusFailure(handle_status);
1001 }
1002 }
1003 }
1004
1005 if (status != SAI_STATUS_ITEM_NOT_FOUND)
1006 {
1007 if (neighbor_entry.ip_address.addr_family == SAI_IP_ADDR_FAMILY_IPV4)
1008 {
1009 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEXTHOP);
1010 }
1011 else
1012 {
1013 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEXTHOP);
1014 }
1015 }
1016
1017 SWSS_LOG_NOTICE("Removed next hop %s on %s",
1018 ip_address.to_string().c_str(), alias.c_str());
1019
1020 status = sai_neighbor_api->remove_neighbor_entry(&neighbor_entry);
1021 if (status != SAI_STATUS_SUCCESS)
1022 {
1023 if (status == SAI_STATUS_ITEM_NOT_FOUND)
1024 {
1025 SWSS_LOG_ERROR("Failed to locate neighbor %s on %s, rv:%d",
1026 m_syncdNeighbors[neighborEntry].mac.to_string().c_str(), alias.c_str(), status);
1027 return true;
1028 }
1029 else
1030 {
1031 SWSS_LOG_ERROR("Failed to remove neighbor %s on %s, rv:%d",
1032 m_syncdNeighbors[neighborEntry].mac.to_string().c_str(), alias.c_str(), status);
1033 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_NEIGHBOR, status);
1034 if (handle_status != task_success)
1035 {
1036 return parseHandleSaiStatusFailure(handle_status);
1037 }
1038 }
1039 }
1040
1041 if (neighbor_entry.ip_address.addr_family == SAI_IP_ADDR_FAMILY_IPV4)
1042 {
1043 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV4_NEIGHBOR);
1044 }
1045 else
1046 {
1047 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_IPV6_NEIGHBOR);
1048 }
1049
1050 removeNextHop(ip_address, alias);
1051 m_intfsOrch->decreaseRouterIntfsRefCount(alias);
1052 }
1053
1054 SWSS_LOG_NOTICE("Removed neighbor %s on %s",
1055 m_syncdNeighbors[neighborEntry].mac.to_string().c_str(), alias.c_str());
1056
1057 /* Do not delete entry from cache if its disable request */
1058 if (disable)
1059 {
1060 m_syncdNeighbors[neighborEntry].hw_configured = false;
1061 return true;
1062 }
1063
1064 m_syncdNeighbors.erase(neighborEntry);
1065
1066 NeighborUpdate update = { neighborEntry, MacAddress(), false };
1067 notify(SUBJECT_TYPE_NEIGH_CHANGE, static_cast<void *>(&update));
1068
1069 if(gMySwitchType == "voq")
1070 {
1071 //Sync the neighbor to delete from the CHASSIS_APP_DB
1072 voqSyncDelNeigh(alias, ip_address);
1073 }
1074
1075 return true;
1076}

1194
1195void NeighOrch::doVoqSystemNeighTask(Consumer &consumer)
1196{
1197 SWSS_LOG_ENTER();
1198
1199 //Local inband port as the outgoing interface of the static neighbor and static route
1200 Port ibif;
1201 if(!gPortsOrch->getInbandPort(ibif))
1202 {
1203 //Inband port is not ready yet.
1204 return;
1205 }
1206
1207 // For "port" type inband interface, wait till the Inband interface is both admin up and oper up
1208 if (ibif.m_type != Port::VLAN)
1209 {
1210 if (ibif.m_admin_state_up != true || ibif.m_oper_status != SAI_PORT_OPER_STATUS_UP)
1211 {
1212 // Inband port is not operational yet
1213 return;
1214 }
1215 }
1216
1217 auto it = consumer.m_toSync.begin();
1218 while (it != consumer.m_toSync.end())
1219 {
1220 KeyOpFieldsValuesTuple t = it->second;
1221 string key = kfvKey(t);
1222 string op = kfvOp(t);
1223
1224 size_t found = key.find_last_of(consumer.getConsumerTable()->getTableNameSeparator().c_str());
1225 if (found == string::npos)
1226 {
1227 SWSS_LOG_ERROR("Failed to parse key %s", key.c_str());
1228 it = consumer.m_toSync.erase(it);
1229 continue;
1230 }
1231
1232 string alias = key.substr(0, found);
1233
1234 if(gIntfsOrch->isLocalSystemPortIntf(alias))
1235 {
1236 //Synced local neighbor. Skip
1237 it = consumer.m_toSync.erase(it);
1238 continue;
1239 }
1240
1241 IpAddress ip_address(key.substr(found+1));
1242
1243 NeighborEntry neighbor_entry = { ip_address, alias };
1244
1245 string state_key = alias + state_db_key_delimiter + ip_address.to_string();
1246
1247 if (op == SET_COMMAND)
1248 {
1249 Port p;
1250 if (!gPortsOrch->getPort(alias, p))
1251 {
1252 SWSS_LOG_INFO("Port %s doesn't exist", alias.c_str());
1253 it++;
1254 continue;
1255 }
1256
1257 if (!p.m_rif_id)
1258 {
1259 SWSS_LOG_INFO("Router interface doesn't exist on %s", alias.c_str());
1260 it++;
1261 continue;
1262 }
1263
1264 MacAddress mac_address;
1265 uint32_t encap_index = 0;
1266 for (auto i = kfvFieldsValues(t).begin();
1267 i != kfvFieldsValues(t).end(); i++)
1268 {
1269 if (fvField(*i) == "neigh")
1270 mac_address = MacAddress(fvValue(*i));
1271
1272 if(fvField(*i) == "encap_index")
1273 {
1274 encap_index = (uint32_t)stoul(fvValue(*i));
1275 }
1276 }
1277
1278 if(!encap_index)
1279 {
1280 //Encap index is not available yet. Since this is remote neighbor, we need to wait till
1281 //Encap index is made available either by dynamic syncing or by static config
1282 it++;
1283 continue;
1284 }
1285
1286 if (m_syncdNeighbors.find(neighbor_entry) == m_syncdNeighbors.end() ||
1287 m_syncdNeighbors[neighbor_entry].mac != mac_address ||
1288 m_syncdNeighbors[neighbor_entry].voq_encap_index != encap_index)
1289 {
1290
1291 if (m_syncdNeighbors.find(neighbor_entry) != m_syncdNeighbors.end() &&
1292 m_syncdNeighbors[neighbor_entry].voq_encap_index != encap_index)
1293 {
1294
1295 // Encap index changed. Set encap index attribute with new encap index
1296 if (!updateVoqNeighborEncapIndex(neighbor_entry, encap_index))
1297 {
1298 // Setting encap index failed. SAI does not support change of encap index for
1299 // existing neighbors. Remove the neighbor but do not errase from consumer sync
1300 // buffer. The next iteration will add the neighbor back with new encap index
1301
1302 SWSS_LOG_NOTICE("VOQ encap index set failed for neighbor %s. Removing and re-adding", kfvKey(t).c_str());
1303
1304 //Remove neigh from SAI
1305 if (removeNeighbor(neighbor_entry))
1306 {
1307 //neigh successfully deleted from SAI. Set STATE DB to signal to remove entries from kernel
1308 m_stateSystemNeighTable->del(state_key);
1309 }
1310 else
1311 {
1312 SWSS_LOG_ERROR("Failed to remove voq neighbor %s from SAI during encap index update", kfvKey(t).c_str());
1313 }
1314 it++;
1315 }
1316 else
1317 {
1318 SWSS_LOG_NOTICE("VOQ encap index updated for neighbor %s", kfvKey(t).c_str());
1319 it = consumer.m_toSync.erase(it);
1320 }
1321 continue;
1322 }
1323
1324 //Add neigh to SAI
1325 if (addNeighbor(neighbor_entry, mac_address))
1326 {
1327 //neigh successfully added to SAI. Set STATE DB to signal kernel programming by neighbor manager
1328
1329 //If the inband interface type is not VLAN, same MAC can be used for the inband interface for
1330 //kernel programming.
1331 if(ibif.m_type != Port::VLAN)
1332 {
1333 mac_address = gMacAddress;
1334
1335 // For VS platforms, the mac of the static neigh should not be same as asic's own mac.
1336 // This is because host originated packets will have same mac for both src and dst which
1337 // will result in host NOT sending packet out. To address this problem which is specific
1338 // to port type inband interfaces, set the mac to the neighbor's owner asic's mac. Since
1339 // the owner asic's mac is not readily avaiable here, the owner asic mac is derived from
1340 // the switch id and lower 5 bytes of asic mac which is assumed to be same for all asics
