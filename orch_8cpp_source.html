73void Consumer::addToSync(const KeyOpFieldsValuesTuple &entry)
74{
75 SWSS_LOG_ENTER();
76
77
78 string key = kfvKey(entry);
79 string op = kfvOp(entry);
80
81 /* Record incoming tasks */
82 if (gSwssRecord)
83 {
84 Orch::recordTuple(*this, entry);
85 }
86
87 /*
88 * m_toSync is a multimap which will allow one key with multiple values,
89 * Also, the order of the key-value pairs whose keys compare equivalent
90 * is the order of insertion and does not change. (since C++11)
91 */
92
93 /* If a new task comes we directly put it into getConsumerTable().m_toSync map */
94 if (m_toSync.find(key) == m_toSync.end())
95 {
96 m_toSync.emplace(key, entry);
97 }
98
99 /* if a DEL task comes, we overwrite the old key */
100 else if (op == DEL_COMMAND)
101 {
102 m_toSync.erase(key);
103 m_toSync.emplace(key, entry);
104 }
105 else
106 {
107 /*
108 * Now we are trying to add the key-value with SET.
109 * We maintain maximum two values per key.
110 * In case there is one key-value, it should be DEL or SET
111 * In case there are two key-value pairs, it should be DEL then SET
112 * The code logic is following:
113 * We iterate the values with the key, we skip the value with DEL and then
114 * check if that was the only one (I,E, the iter pointer now points to end or next key),
115 * in such case, we insert the key-value with SET.
116 * If there was a SET already (I,E, the pointer still points to the same key), we combine the kfv.
117 */
118 auto ret = m_toSync.equal_range(key);
119 auto iter = ret.first;
120 for (; iter != ret.second; ++iter)
121 {
122 auto old_op = kfvOp(iter->second);
123 if (old_op == SET_COMMAND)
124 break;
125 }
126 if (iter == ret.second)
127 {
128 m_toSync.emplace(key, entry);
129 }
130 else
131 {
132 KeyOpFieldsValuesTuple existing_data = iter->second;
133
134 auto new_values = kfvFieldsValues(entry);
135 auto existing_values = kfvFieldsValues(existing_data);
136
137
138 for (auto it : new_values)
139 {
140 string field = fvField(it);
141 string value = fvValue(it);
142
143 auto iu = existing_values.begin();
144 while (iu != existing_values.end())
145 {
146 string ofield = fvField(*iu);
147 if (field == ofield)
148 iu = existing_values.erase(iu);
149 else
150 iu++;
151 }
152 existing_values.push_back(FieldValueTuple(field, value));
153 }
154 iter->second = KeyOpFieldsValuesTuple(key, op, existing_values);
155 }
156 }

316- Validates reference has proper format which is object_name
317- validates table_name exists
318- validates object with object_name exists
319
320- Special case:
321- Deem reference format [] as valid, and return true. But in such a case,

323- indication to the caller of the special case

325bool Orch::parseReference(type_map &type_maps, string &ref_in, const string &type_name, string &object_name)
326{
327 SWSS_LOG_ENTER();
328
329 SWSS_LOG_DEBUG("input:%s", ref_in.c_str());
330
331 if (ref_in.size() == 0)
332 {
333 // value set by user is ""
334 // Deem it as a valid format
335 // clear both type_name and object_name
336 // as an indication to the caller that
337 // such a case has been encountered
338 object_name.clear();
339 return true;
340 }
341
342 if ((ref_in[0] == ref_start) || (ref_in[ref_in.size()-1] == ref_end))
343 {
344 SWSS_LOG_ERROR("malformed reference:%s. Must not be surrounded by [ ]\n", ref_in.c_str());
345 return false;
346 }
347 auto type_it = type_maps.find(type_name);
348 if (type_it == type_maps.end())
349 {
350 SWSS_LOG_ERROR("not recognized type:%s\n", type_name.c_str());
351 return false;
352 }
353 auto obj_map = type_maps[type_name];
354 auto obj_it = obj_map->find(ref_in);
355 if (obj_it == obj_map->end())
356 {
357 SWSS_LOG_INFO("map:%s does not contain object with name:%s\n", type_name.c_str(), ref_in.c_str());
358 return false;
359 }
360 if (obj_it->second.m_pendingRemove)
361 {
362 SWSS_LOG_NOTICE("map:%s contains a pending removed object %s, skip\n", type_name.c_str(), ref_in.c_str());
363 return false;
364 }
365 object_name = ref_in;
366 SWSS_LOG_DEBUG("parsed: type_name:%s, object_name:%s", type_name.c_str(), object_name.c_str());
367 return true;
368}

370ref_resolve_status Orch::resolveFieldRefValue(
371 type_map &type_maps,
372 const string &field_name,
373 const string &ref_type_name,
374 KeyOpFieldsValuesTuple &tuple,
375 sai_object_id_t &sai_object,
376 string &referenced_object_name)

378 SWSS_LOG_ENTER();
379
380 bool hit = false;

403 }
404 }
405 if (!hit)

407 return ref_resolve_status::field_not_found;

413void Orch::removeMeFromObjsReferencedByMe(
414 type_map &type_maps,
415 const string &table,
416 const string &obj_name,
417 const string &field,
418 const string &old_referenced_obj_name,
419 bool remove_field)
420{
421 vector<string> objects = tokenize(old_referenced_obj_name, list_item_delimiter);
422 for (auto &obj : objects)
423 {
424 // obj_name references token
425 auto tokens = tokenize(obj, delimiter);
426 auto &referenced_table = tokens[0];
427 auto &ref_obj_name = tokens[1];
428 auto &old_referenced_obj = (*type_maps[referenced_table])[ref_obj_name];
429 old_referenced_obj.m_objsDependingOnMe.erase(obj_name);
430 SWSS_LOG_INFO("Obj %s.%s Field %s: Remove reference to %s %s (now %s)",
431 table.c_str(), obj_name.c_str(), field.c_str(),
432 referenced_table.c_str(), ref_obj_name.c_str(),
433 to_string(old_referenced_obj.m_objsDependingOnMe.size()).c_str());
434 }
435
436 if (remove_field)
437 {
438 auto &referencing_object = (*type_maps[table])[obj_name];
439 referencing_object.m_objsReferencingByMe.erase(field);
440 }

443void Orch::setObjectReference(
444 type_map &type_maps,
445 const string &table,
446 const string &obj_name,
447 const string &field,
448 const string &referenced_obj)
449{
450 auto &obj = (*type_maps[table])[obj_name];
451 auto field_ref = obj.m_objsReferencingByMe.find(field);
452
453 if (field_ref != obj.m_objsReferencingByMe.end())
454 removeMeFromObjsReferencedByMe(type_maps, table, obj_name, field, field_ref->second, false);
455
456 obj.m_objsReferencingByMe[field] = referenced_obj;
457
458 // Add the reference to the new object being referenced
459 vector<string> objs = tokenize(referenced_obj, list_item_delimiter);
460 for (auto &obj : objs)
461 {
462 auto tokens = tokenize(obj, delimiter);
463 auto &referenced_table = tokens[0];
464 auto &referenced_obj_name = tokens[1];
465 auto &new_obj_being_referenced = (*type_maps[referenced_table])[referenced_obj_name];
466 new_obj_being_referenced.m_objsDependingOnMe.insert(obj_name);
467 SWSS_LOG_INFO("Obj %s.%s Field %s: Add reference to %s %s (now %s)",
468 table.c_str(), obj_name.c_str(), field.c_str(),
469 referenced_table.c_str(), referenced_obj_name.c_str(),
470 to_string(new_obj_being_referenced.m_objsDependingOnMe.size()).c_str());
471 }

498 const string &table,

604ref_resolve_status Orch::resolveFieldRefArray(
605 type_map &type_maps,
606 const string &field_name,
607 const string &ref_type_name,
608 KeyOpFieldsValuesTuple &tuple,
609 vector<sai_object_id_t> &sai_object_arr,
610 string &object_name_list)
611{
612 // example: e_port.profile0,e_port.profile1
613 SWSS_LOG_ENTER();
614 size_t count = 0;
615 sai_object_arr.clear();
616 for (auto i = kfvFieldsValues(tuple).begin(); i != kfvFieldsValues(tuple).end(); i++)
617 {
618 if (fvField(*i) == field_name)
619 {
620 if (count > 1)
621 {
622 SWSS_LOG_ERROR("Singleton field with name:%s must have only 1 instance, actual count:%zd\n", field_name.c_str(), count);
623 return ref_resolve_status::multiple_instances;
624 }
625 string object_name;
626 string list = fvValue(*i);
627 vector<string> list_items;
628 if (list.find(list_item_delimiter) != string::npos)
629 {
630 list_items = tokenize(list, list_item_delimiter);
631 }
632 else
633 {
634 list_items.push_back(list);
635 }
636 for (size_t ind = 0; ind < list_items.size(); ind++)
637 {
638 if (!parseReference(type_maps, list_items[ind], ref_type_name, object_name))
639 {
640 SWSS_LOG_NOTICE("Failed to parse profile reference:%s\n", list_items[ind].c_str());
641 return ref_resolve_status::not_resolved;
642 }
643 sai_object_id_t sai_obj = (*(type_maps[ref_type_name]))[object_name].m_saiObjectId;
644 SWSS_LOG_DEBUG("Resolved to sai_object:0x%" PRIx64 ", type:%s, name:%s", sai_obj, ref_type_name.c_str(), object_name.c_str());
645 sai_object_arr.push_back(sai_obj);
646 if (!object_name_list.empty())
647 object_name_list += list_item_delimiter;
648 object_name_list += ref_type_name + delimiter + object_name;
649 }
650 count++;
651 }
652 }
653 if (0 == count)
654 {
655 SWSS_LOG_NOTICE("field with name:%s not found\n", field_name.c_str());
656 return ref_resolve_status::field_not_found;
657 }
658 return ref_resolve_status::success;
659}

661bool Orch::parseIndexRange(const string &input, sai_uint32_t &range_low, sai_uint32_t &range_high)
662{
663 SWSS_LOG_ENTER();
664 SWSS_LOG_DEBUG("input:%s", input.c_str());
665 if (input.find(range_specifier) != string::npos)
666 {
667 vector<string> range_values;
668 range_values = tokenize(input, range_specifier);
669 if (range_values.size() != 2)
670 {
671 SWSS_LOG_ERROR("malformed index range in:%s. Must contain 2 tokens\n", input.c_str());
672 return false;
673 }

675 range_high = (uint32_t)stoul(range_values[1]);

681 }
682 else

684 range_low = range_high = (uint32_t)stoul(input);

686 SWSS_LOG_DEBUG("resulting range:%d-%d", range_low, range_high);

690/*
691 * generateBitMapFromIdsStr
692 *
693 * Generates the bit map representing the idsMap in string
694 * Args:
695 * idsStr: The string representing the IDs.
696 * Typically it's part of a key of BUFFER_QUEUE or BUFFER_PG, like "3-4" from "Ethernet0|3-4"
697 * Return:
698 * idsMap: The bitmap of IDs. The LSB stands for ID 0.
699 *
700 * Example:
701 * Input idsMap: 3-4
702 * Return: 00011000b
703 */
704unsigned long Orch::generateBitMapFromIdsStr(const string &idsStr)
705{
706 sai_uint32_t lowerBound, upperBound;
707 unsigned long idsMap = 0;
708
709 if (!parseIndexRange(idsStr, lowerBound, upperBound))
710 return 0;
711
712 for (sai_uint32_t id = lowerBound; id <= upperBound; id ++)
713 {
714 idsMap |= (1 << id);
715 }
716
717 return idsMap;
718}

720/*
721 * generateIdListFromMap
722 *
723 * Parse the idsMap and generate a vector which contains slices representing bits in idsMap
724 * Args:
725 * idsMap: The bitmap of IDs. The LSB stands for ID 0.
726 * maxId: The (exclusive) upperbound of ID range.
727 * Eg: if ID range is 0~7, maxId should be 8.
728 * Return:
729 * A vector which contains slices representing bits in idsMap
730 *
731 * Example:
732 * Input idsMap: 00100110b, maxId: 8
733 * Return vector: ["1-2", "5"]
734 */
735set<string> Orch::generateIdListFromMap(unsigned long idsMap, sai_uint32_t maxId)
736{
737 unsigned long currentIdMask = 1;
738 bool started = false, needGenerateMap = false;
739 sai_uint32_t lower, upper;
740 set<string> idStringList;
741 for (sai_uint32_t id = 0; id <= maxId; id ++)
742 {
743 // currentIdMask represents the bit mask corresponding to id: (1<<id)
744 if (idsMap & currentIdMask)
745 {
746 if (!started)
747 {
748 started = true;
749 lower = id;
750 }
751 }
752 else
753 {
754 if (started)
755 {
756 started = false;
757 upper = id - 1;
758 needGenerateMap = true;
759 }
760 }
761
762 if (needGenerateMap)
763 {
764 if (lower != upper)
765 {
766 idStringList.insert(to_string(lower) + "-" + to_string(upper));
767 }
768 else
769 {
770 idStringList.insert(to_string(lower));
771 }
772 needGenerateMap = false;
773 }

782/*
783 * isItemIdsMapContinuous
784 *
785 * Check whether the input idsMap is continuous.
786 * An idsMap is continuous means there is no "0"s between any two "1"s in the map.
787 * Args:
788 * idsMap: The bitmap of IDs. The LSB stands for ID 0.
789 * maxId: The maximum value of ID.
790 * Return:
791 * Whether the idsMap is continous
792 *
793 * Example:
794 * idsMaps like 00011100, 00001000, 00000000 are continuous
795 * while 00110010 is not because there are 2 "0"s in bit 2, 3 surrounded by "1"s
796 */
797bool Orch::isItemIdsMapContinuous(unsigned long idsMap, sai_uint32_t maxId)
798{
799 unsigned long currentIdMask = 1;
800 bool isCurrentBitValid = false, hasValidBits = false, hasZeroAfterValidBit = false;
801
802 for (sai_uint32_t id = 0; id < maxId; id ++)
803 {
804 isCurrentBitValid = ((idsMap & currentIdMask) != 0);
805 if (isCurrentBitValid)
806 {
807 if (!hasValidBits)
808 hasValidBits = true;
809 if (hasZeroAfterValidBit)
810 return false;
811 }
812 else
813 {
814 if (hasValidBits && !hasZeroAfterValidBit)
815 hasZeroAfterValidBit = true;
816 }
817 currentIdMask <<= 1;
818 }
819
820 return true;
821}
