471std::vector<int>
472findTablePrecedence (int tables, std::vector<std::pair<int, int>> preReq, TablesInfo *tables_info)
473{
474 std::unordered_map<int, std::unordered_set<int>> graph = createGraph(preReq);
475 std::unordered_map<int, int> degrees = computeIndegree(graph);
476 std::vector<int> visited;
477 std::vector<int> toposort;
478 std::queue<int> zeros;
479
480 // initialize queue with tables having no dependencies
481 for (auto table_it = tables_info->m_tableInfoMap.begin();
482 table_it != tables_info->m_tableInfoMap.end(); table_it++)
483 {
484 TableInfo table_info = table_it->second;
485 if (degrees.find(table_info.id) == degrees.end())
486 {
487 zeros.push(table_info.id);
488 visited.push_back(table_info.id);
489 }
490 }
491
492 for (int i = 0; i < tables; i++)
493 {
494 // Err input data like possible cyclic dependencies, could not build precedence order
495 if (zeros.empty())
496 {
497 SWSS_LOG_ERROR("Filed to build table precedence order");
498 return {};
499 }
500
501 // Run BFS
502 int zero = zeros.front();
503 zeros.pop();
504 toposort.push_back(zero);
505 auto g_it = graph.find(zero);
506 if (g_it != graph.end())
507 {
508 for (int neigh : g_it->second)
509 {
510 auto n_it = degrees.find(neigh);
511 if (n_it != degrees.end())
512 {
513 if (!n_it->second)
514 {
515 if (std::find(visited.begin(), visited.end(), neigh) == visited.end())
516 {
517 zeros.push(neigh);
518 visited.push_back(neigh);
519 }
520 }
521 else
522 {
523 n_it->second--;
524 }
525 }
526 }
527 }
528 }
529
530 return toposort;
531}

534void
535buildTablePrecedence (TablesInfo *tables_info)
536{
537 std::vector<std::pair<int, int>> preReq;
538 std::vector<int> orderedTables;
539 int tables = 0;
540
541 if (!tables_info) {
542 return;
543 }
544
545 // build dependencies
546 for (auto table_it = tables_info->m_tableInfoMap.begin();
547 table_it != tables_info->m_tableInfoMap.end(); table_it++)
548 {
549 TableInfo table_info = table_it->second;
550 tables++;
551
552 for (std::size_t i = 0; i < table_info.action_ref_tables.size(); i++)
553 {
559 if (FixedTablesMap.find(table_info.action_ref_tables[i]) != FixedTablesMap.end())
560 {
561 continue;
562 }
563
564 TableInfo ref_table_info = tables_info->m_tableInfoMap[table_info.action_ref_tables[i]];
565 if (std::find(preReq.begin(), preReq.end(),
566 std::make_pair(table_info.id, ref_table_info.id)) == preReq.end())
567 {
568 preReq.push_back(std::make_pair(table_info.id, ref_table_info.id));
569 }
570 }
571 }
572
573 // find precedence of tables based on dependencies
574 orderedTables = findTablePrecedence(tables, preReq, tables_info);
575
576 // update each table with calculated precedence value and build table precedence map
577 for (std::size_t i = 0; i < orderedTables.size(); i++)
578 {
579 auto table_id = orderedTables[i];
580 auto id_it = tables_info->m_tableIdNameMap.find(std::to_string(table_id));
581 if (id_it == tables_info->m_tableIdNameMap.end())
582 {
583 continue;
584 }
585
586 auto table_it = tables_info->m_tableInfoMap.find(id_it->second);
587 if (table_it == tables_info->m_tableInfoMap.end())
588 {
589 continue;
590 }
591
592 table_it->second.precedence = (int)i;
593 tables_info->m_tablePrecedenceMap[(int)i] = table_it->second.name;
594 }
595
596 return;
597}

605void TablesDefnManager::drain()
606{
607 SWSS_LOG_ENTER();
608
609 for (const auto &key_op_fvs_tuple : m_entries)
610 {
611 std::string table_name;
612 std::string key;
613 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &key);
614 const std::vector<swss::FieldValueTuple> &attributes = kfvFieldsValues(key_op_fvs_tuple);
615
616 ReturnCode status;
617 auto app_db_entry_or = deserializeTablesInfoEntry(key, attributes);
618 if (!app_db_entry_or.ok())
619 {
620 status = app_db_entry_or.status();
621 SWSS_LOG_ERROR("Unable to deserialize APP DB entry with key %s: %s",
622 QuotedVar(table_name + ":" + key).c_str(), status.message().c_str());
623 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
624 status, /*replace=*/true);
625 continue;
626 }
627 auto &app_db_entry = *app_db_entry_or;
628
629 status = validateTablesInfoAppDbEntry(app_db_entry);
630 if (!status.ok())
631 {
632 SWSS_LOG_ERROR("Validation failed for tables definition APP DB entry with key %s: %s",
633 QuotedVar(table_name + ":" + key).c_str(), status.message().c_str());
634 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
635 status, /*replace=*/true);
636 continue;
637 }
638
639 const std::string context_key = KeyGenerator::generateTablesInfoKey(app_db_entry.context);
640
641 const std::string &operation = kfvOp(key_op_fvs_tuple);
642 if (operation == SET_COMMAND)
643 {
644 auto *tablesinfo = getTablesInfoEntry(context_key);
645 if (tablesinfo == nullptr)
646 {
647 // Create TablesInfo
648 status = processAddRequest(app_db_entry, context_key);
649 }
650 else
651 {
652 // Modify existing TablesInfo
653 status = processUpdateRequest(app_db_entry, context_key);
654 }
655 }
656 else if (operation == DEL_COMMAND)
657 {
658 // Delete TablesInfo
659 status = processDeleteRequest(context_key);
660 }
661 else
662 {
663 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Unknown operation type " << QuotedVar(operation);
664 SWSS_LOG_ERROR("%s", status.message().c_str());
665 }
666 if (!status.ok())
667 {
668 SWSS_LOG_ERROR("Processing failed for tables definition APP DB entry with key %s: %s",
669 QuotedVar(table_name + ":" + key).c_str(), status.message().c_str());
670 }
671 else
672 {
673 buildTablePrecedence(gP4Orch->tablesinfo);
674 }
675 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
676 status, /*replace=*/true);
677 }
678 m_entries.clear();
679}
