147// If the drop reason is already present on this counter, this method has no
148// effect.
149//
150// If the update fails, this method throws a runtime error.
151void DropCounter::addDropReason(const std::string& drop_reason)
152{
153 SWSS_LOG_ENTER();
154
155 if (drop_reasons.find(drop_reason) != drop_reasons.end())
156 {
157 SWSS_LOG_DEBUG("Drop reason '%s' already present on '%s'", drop_reason.c_str(), name.c_str());
158 return;
159 }
160
161 try
162 {
163 drop_reasons.emplace(drop_reason);
164 updateDropReasonsInSAI();
165 }
166 catch (const std::runtime_error& e)
167 {
168 drop_reasons.erase(drop_reason);
169 throw;
170 }
171}

220// serializeDropReasons takes the list of drop reasons associated with this
221// counter and stores them in a SAI readable format in drop_reason_attribute.
222//
223// This method assumes that drop_reason_list points to a region in memory with
224// enough space for drop_reason_count drop reasons to be stored.
225//
226// If any of the provided drop reasons (or their serialization) is undefined,
227// then this method throws a runtime error.
228void DropCounter::serializeDropReasons(uint32_t drop_reason_count, int32_t *drop_reason_list, sai_attribute_t *drop_reason_attribute)
229{
230 SWSS_LOG_ENTER();
231
232 if (type == PORT_INGRESS_DROPS || type == SWITCH_INGRESS_DROPS)
233 {
234 drop_reason_attribute->id = SAI_DEBUG_COUNTER_ATTR_IN_DROP_REASON_LIST;
235 drop_reason_attribute->value.s32list.count = drop_reason_count;
236 drop_reason_attribute->value.s32list.list = drop_reason_list;
237
238 int index = 0;
239 for (auto drop_reason: drop_reasons)
240 {
241 auto reason_it = ingress_drop_reason_lookup.find(drop_reason);
242 if (reason_it == ingress_drop_reason_lookup.end())
243 {
244 SWSS_LOG_ERROR("Ingress drop reason '%s' not found", drop_reason.c_str());
245 throw runtime_error("Ingress drop reason not found");
246 }
247
248 drop_reason_list[index++] = static_cast<int32_t>(reason_it->second);
249 }
250 }
251 else if (type == PORT_EGRESS_DROPS || type == SWITCH_EGRESS_DROPS)
252 {
253 drop_reason_attribute->id = SAI_DEBUG_COUNTER_ATTR_OUT_DROP_REASON_LIST;
254 drop_reason_attribute->value.s32list.count = drop_reason_count;
255 drop_reason_attribute->value.s32list.list = drop_reason_list;
256
257 int index = 0;
258 for (auto drop_reason: drop_reasons)
259 {
260 auto reason_it = egress_drop_reason_lookup.find(drop_reason);
261 if (reason_it == egress_drop_reason_lookup.end())
262 {
263 SWSS_LOG_ERROR("Egress drop reason '%s' not found", drop_reason.c_str());
264 throw runtime_error("Egress drop reason not found");
265 }
266
267 drop_reason_list[index++] = static_cast<int32_t>(reason_it->second);
268 }
269 }
270 else
271 {
272 SWSS_LOG_ERROR("Serialization undefined for drop counter type '%s'", type.c_str());
273 throw runtime_error("Failed to serialize drop counter attributes");
274 }
275}

292// Multiple calls to this function are guaranteed to return the same reasons
293// (assuming the device has not been rebooted between calls). The order of
294// the list of reasons is not guaranteed to be the same between calls.
295//
296// If the device does not support querying drop reasons, this method will
297// return an empty list.
298unordered_set<string> DropCounter::getSupportedDropReasons(sai_debug_counter_attr_t drop_reason_type)
299{
300 sai_s32_list_t drop_reason_list;
301 int32_t supported_reasons[maxDropReasons];
302 drop_reason_list.count = maxDropReasons;
303 drop_reason_list.list = supported_reasons;
304
305 if (sai_query_attribute_enum_values_capability(gSwitchId,
306 SAI_OBJECT_TYPE_DEBUG_COUNTER,
307 drop_reason_type,
308 &drop_reason_list) != SAI_STATUS_SUCCESS)
309 {
310 SWSS_LOG_NOTICE("This device does not support querying drop reasons");
311 return {};
312 }
313
314 unordered_set<string> supported_drop_reasons;
315 for (uint32_t i = 0; i < drop_reason_list.count; i++)
316 {
317 string drop_reason;
318 if (drop_reason_type == SAI_DEBUG_COUNTER_ATTR_IN_DROP_REASON_LIST)
319 {
320 drop_reason = sai_serialize_ingress_drop_reason(static_cast<sai_in_drop_reason_t>(drop_reason_list.list[i]));
321 // in case of unsupported counter, enum value is returned as a str
322 if (drop_reason.length() < INGRESS_DROP_REASON_PREFIX_LENGTH)
323 {
324 continue;
325 }
326 drop_reason = drop_reason.substr(INGRESS_DROP_REASON_PREFIX_LENGTH);
327 }
328 else
329 {
330 drop_reason = sai_serialize_egress_drop_reason(static_cast<sai_out_drop_reason_t>(drop_reason_list.list[i]));
331 // in case of unsupported counter, enum value is returned as a str
332 if (drop_reason.length() < EGRESS_DROP_REASON_PREFIX_LENGTH)
333 {
334 continue;
335 }
336 drop_reason = drop_reason.substr(EGRESS_DROP_REASON_PREFIX_LENGTH);
337 }
338
339 supported_drop_reasons.emplace(drop_reason);
340 }
341
342 return supported_drop_reasons;
343}

345// Returns a set of supported counter types.
346unordered_set<string> DropCounter::getSupportedCounterTypes()
347{
348 sai_status_t status = SAI_STATUS_FAILURE;
349
350 const auto& countersTypeLookup = getDebugCounterTypeLookup();
351 unordered_set<string> supportedCounterTypes;
352
353 sai_s32_list_t enumValuesCapabilities;
354 vector<int32_t> saiCounterTypes;
355
356 const auto* meta = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_DEBUG_COUNTER,
357 SAI_DEBUG_COUNTER_ATTR_TYPE);
358 if (!meta)
359 {
360 SWSS_LOG_ERROR("SAI BUG: metadata null pointer returned by "
361 "sai_metadata_get_attr_metadata for SAI_DEBUG_COUNTER_ATTR_TYPE");
362 return {};
363 }
364
365 if (!meta->isenum || !meta->enummetadata)
366 {
367 SWSS_LOG_ERROR("SAI BUG: SAI_DEBUG_COUNTER_ATTR_TYPE value type is not an enum");
368 return {};
369 }
370
371 saiCounterTypes.assign(meta->enummetadata->valuescount, 0);
372
373 enumValuesCapabilities.count = static_cast<uint32_t>(saiCounterTypes.size());
374 enumValuesCapabilities.list = saiCounterTypes.data();
375
376 status = sai_query_attribute_enum_values_capability(gSwitchId,
377 SAI_OBJECT_TYPE_DEBUG_COUNTER,
378 SAI_DEBUG_COUNTER_ATTR_TYPE,
379 &enumValuesCapabilities);
380 if (status != SAI_STATUS_SUCCESS)
381 {
382 SWSS_LOG_NOTICE("This device does not support querying drop counters");
383 return {};
384 }
385
386 for (uint32_t i = 0; i < enumValuesCapabilities.count; i++)
387 {
388 auto enumValue = static_cast<sai_debug_counter_type_t>(enumValuesCapabilities.list[i]);
389 for (const auto& it: countersTypeLookup)
390 {
391 if (it.second == enumValue)
392 {
393 supportedCounterTypes.emplace(it.first);
394 break;
395 }
396 }
397 }
398
399 return supportedCounterTypes;
400}

402// serializeSupportedDropReasons takes a list of drop reasons and returns that
403// list as a string.
404//
405// e.g. { "SMAC_EQUALS_DMAC", "INGRESS_VLAN_FILTER" } -> "["SMAC_EQUALS_DMAC","INGRESS_VLAN_FILTER"]"
406// e.g. { } -> "[]"
407string DropCounter::serializeSupportedDropReasons(unordered_set<string> drop_reasons)
408{
409 if (drop_reasons.size() == 0)
410 {
411 return "[]";
412 }
413
414 string supported_drop_reasons;
415 for (auto const& drop_reason : drop_reasons)
416 {
417 supported_drop_reasons += ',';
418 supported_drop_reasons += drop_reason;
419 }
420
421 supported_drop_reasons[0] = '[';
422 return supported_drop_reasons + ']';
423}
