20// RETURN_IF_ERROR evaluates an expression that returns a ReturnCode. If the
21// result is not ok, returns the result. Otherwise, continues.
22//
23// Example:
24// ReturnCode Foo() {...}
25// ReturnCode Bar() {
26// RETURN_IF_ERROR(Foo());
27// return ReturnCode();
28// }
29#define RETURN_IF_ERROR(expr) \
30 do \
31 { \
32 ReturnCode RETURN_IF_ERROR_RC_ = expr; \
33 if (!RETURN_IF_ERROR_RC_.ok()) \
34 return RETURN_IF_ERROR_RC_; \
35 } while (0)
