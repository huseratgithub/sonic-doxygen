52{
53 SWSS_LOG_ENTER();
54
55 for (auto& i : m_mcCountersMap)
56 {
57 auto oid = i.first;
58 auto mcCounters = i.second;
59 uint8_t pfcMask = 0;
60
61 Port port;
62 if (!gPortsOrch->getPort(oid, port))
63 {
64 SWSS_LOG_ERROR("Invalid port oid 0x%" PRIx64, oid);
65 continue;
66 }
67
68 auto newMcCounters = getQueueMcCounters(port);
69
70 if (!gPortsOrch->getPortPfc(port.m_port_id, &pfcMask))
71 {
72 SWSS_LOG_ERROR("Failed to get PFC mask on port %s", port.m_alias.c_str());
73 continue;
74 }
75
76 for (size_t prio = 0; prio != mcCounters.size(); prio++)
77 {
78 bool isLossy = ((1 << prio) & pfcMask) == 0;
79 if (newMcCounters[prio] == numeric_limits<uint64_t>::max())
80 {
81 SWSS_LOG_WARN("Could not retreive MC counters on queue %zu port %s",
82 prio,
83 port.m_alias.c_str());
84 }
85 else if (!isLossy && mcCounters[prio] < newMcCounters[prio])
86 {
87 SWSS_LOG_WARN("Got Multicast %" PRIu64 " frame(s) on lossless queue %zu port %s",
88 newMcCounters[prio] - mcCounters[prio],
89 prio,
90 port.m_alias.c_str());
91 }
