90ReturnCode WcmpManager::validateWcmpGroupEntry(const P4WcmpGroupEntry &app_db_entry)
91{
92 for (auto &wcmp_group_member : app_db_entry.wcmp_group_members)
93 {
94 if (wcmp_group_member->weight <= 0)

121ReturnCodeOr<P4WcmpGroupEntry> WcmpManager::deserializeP4WcmpGroupAppDbEntry(
122 const std::string &key, const std::vector<swss::FieldValueTuple> &attributes)
123{
124 P4WcmpGroupEntry app_db_entry = {};
125 try
126 {
127 nlohmann::json j = nlohmann::json::parse(key);
128 if (!j.is_object())
129 {
130 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Invalid WCMP group key: should be a JSON object.";
131 }
132 app_db_entry.wcmp_group_id = j[prependMatchField(kWcmpGroupId)];
133 }
134 catch (std::exception &ex)
135 {
136 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to deserialize WCMP group key";
137 }
138
139 for (const auto &it : attributes)
140 {
141 const auto &field = fvField(it);
142 const auto &value = fvValue(it);
143 if (field == kActions)
144 {
145 try
146 {
147 nlohmann::json j = nlohmann::json::parse(value);
148 if (!j.is_array())
149 {
150 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
151 << "Invalid WCMP group actions " << QuotedVar(value) << ", expecting an array.";
152 }
153 for (auto &action_item : j)
154 {
155 std::shared_ptr<P4WcmpGroupMemberEntry> wcmp_group_member =
156 std::make_shared<P4WcmpGroupMemberEntry>();
157 std::string action = action_item[kAction];
158 if (action != kSetNexthopId)
159 {
160 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
161 << "Unexpected action " << QuotedVar(action) << " in WCMP group entry";
162 }
163 if (action_item[prependParamField(kNexthopId)].empty())
164 {
165 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
166 << "Next hop id was not found in entry member for WCMP "
167 "group "
168 << QuotedVar(app_db_entry.wcmp_group_id);
169 }
170 wcmp_group_member->next_hop_id = action_item[prependParamField(kNexthopId)];
171 if (!action_item[kWeight].empty())
172 {
173 wcmp_group_member->weight = action_item[kWeight];
174 }
175 if (!action_item[kWatchPort].empty())
176 {
177 wcmp_group_member->watch_port = action_item[kWatchPort];
178 }
179 wcmp_group_member->wcmp_group_id = app_db_entry.wcmp_group_id;
180 wcmp_group_member->pruned = false;
181 app_db_entry.wcmp_group_members.push_back(wcmp_group_member);
182 }
183 }
184 catch (std::exception &ex)
185 {
186 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
187 << "Failed to deserialize WCMP group actions fields: " << QuotedVar(value);
188 }
189 }
190 else if (field != kControllerMetadata)
191 {
192 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
193 << "Unexpected field " << QuotedVar(field) << " in table entry";
194 }
195 }
196
197 return app_db_entry;
198}

282ReturnCode WcmpManager::processWcmpGroupMembersAddition(
283 const std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> &members, const std::string &wcmp_group_key,
284 sai_object_id_t wcmp_group_oid, std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> &created_wcmp_group_members)
285{
286 SWSS_LOG_ENTER();
287 ReturnCode status;
288 vector<sai_object_id_t> nhgm_ids(members.size(), SAI_NULL_OBJECT_ID);
289 for (size_t i = 0; i < members.size(); ++i)
290 {
291 bool insert_member = true;
292 auto &member = members[i];
293 if (!member->watch_port.empty())
294 {
295 // Create member in SAI only for operationally up ports
296 sai_port_oper_status_t oper_status = SAI_PORT_OPER_STATUS_DOWN;
297 status = fetchPortOperStatus(member->watch_port, &oper_status);
298 if (!status.ok())
299 {
300 break;
301 }
302
303 if (oper_status != SAI_PORT_OPER_STATUS_UP)
304 {
305 insert_member = false;
306 member->pruned = true;
307 SWSS_LOG_NOTICE("Member %s in group %s not created in asic as the associated "
308 "watchport "
309 "(%s) is not operationally up",
310 member->next_hop_id.c_str(), member->wcmp_group_id.c_str(), member->watch_port.c_str());
311 }
312 }
313 if (insert_member)
314 {
315 auto attrs = getSaiMemberAttrs(*(member.get()), wcmp_group_oid);
316 gNextHopGroupMemberBulker.create_entry(&nhgm_ids[i], (uint32_t)attrs.size(), attrs.data());
317 }
318 }
319 if (status.ok())
320 {
321 gNextHopGroupMemberBulker.flush();
322 for (size_t i = 0; i < members.size(); ++i)
323 {
324 auto &member = members[i];
325 if (!member->pruned)
326 {
327 if (nhgm_ids[i] == SAI_NULL_OBJECT_ID)
328 {
329 if (status.ok())
330 {
331 status = ReturnCode(StatusCode::SWSS_RC_UNKNOWN)
332 << "Fail to create wcmp group member: " << QuotedVar(member->next_hop_id);
333 }
334 else
335 {
336 status << "; Fail to create wcmp group member: " << QuotedVar(member->next_hop_id);
337 }
338 continue;
339 }
340 member->member_oid = nhgm_ids[i];
341 m_p4OidMapper->setOID(SAI_OBJECT_TYPE_NEXT_HOP_GROUP_MEMBER,
342 getWcmpGroupMemberKey(wcmp_group_key, member->member_oid), member->member_oid);
343 m_p4OidMapper->increaseRefCount(SAI_OBJECT_TYPE_NEXT_HOP_GROUP, wcmp_group_key);
344 }
345 if (!member->watch_port.empty())
346 {
347 // Add member to port_name_to_wcmp_group_member_map
348 insertMemberInPortNameToWcmpGroupMemberMap(member);
349 }
350 const std::string &next_hop_key = KeyGenerator::generateNextHopKey(member->next_hop_id);
351 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP_MEMBER);
352 m_p4OidMapper->increaseRefCount(SAI_OBJECT_TYPE_NEXT_HOP, next_hop_key);
353 created_wcmp_group_members.push_back(member);
354 }
355 }
356 return status;
357}

359ReturnCode WcmpManager::createWcmpGroup(P4WcmpGroupEntry *wcmp_group)
360{
361 SWSS_LOG_ENTER();
362
363 auto attrs = getSaiGroupAttrs(*wcmp_group);
364 CHECK_ERROR_AND_LOG_AND_RETURN(sai_next_hop_group_api->create_next_hop_group(&wcmp_group->wcmp_group_oid, gSwitchId,
365 (uint32_t)attrs.size(), attrs.data()),
366 "Failed to create next hop group " << QuotedVar(wcmp_group->wcmp_group_id));
367 // Update reference count
368 const auto &wcmp_group_key = KeyGenerator::generateWcmpGroupKey(wcmp_group->wcmp_group_id);
369 gCrmOrch->incCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
370 m_p4OidMapper->setOID(SAI_OBJECT_TYPE_NEXT_HOP_GROUP, wcmp_group_key, wcmp_group->wcmp_group_oid);
371
372 // Create next hop group members
373 std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> created_wcmp_group_members;
374 ReturnCode status = processWcmpGroupMembersAddition(wcmp_group->wcmp_group_members, wcmp_group_key,
375 wcmp_group->wcmp_group_oid, created_wcmp_group_members);
376 if (!status.ok())
377 {
378 // Clean up created group members and the group
379 recoverGroupMembers(wcmp_group, wcmp_group_key, created_wcmp_group_members, {});
380 auto sai_status = sai_next_hop_group_api->remove_next_hop_group(wcmp_group->wcmp_group_oid);
381 if (sai_status != SAI_STATUS_SUCCESS)
382 {
383 std::stringstream ss;
384 ss << "Failed to delete WCMP group with id " << QuotedVar(wcmp_group->wcmp_group_id);
385 SWSS_LOG_ERROR("%s SAI_STATUS: %s", ss.str().c_str(), sai_serialize_status(sai_status).c_str());
386 SWSS_RAISE_CRITICAL_STATE(ss.str());
387 }
388 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
389 m_p4OidMapper->eraseOID(SAI_OBJECT_TYPE_NEXT_HOP_GROUP, wcmp_group_key);
390 return status;
391 }
392
393 m_wcmpGroupTable[wcmp_group->wcmp_group_id] = *wcmp_group;
394 return ReturnCode();
395}

397void WcmpManager::recoverGroupMembers(
398 p4orch::P4WcmpGroupEntry *wcmp_group_entry, const std::string &wcmp_group_key,
399 const std::vector<std::shared_ptr<p4orch::P4WcmpGroupMemberEntry>> &created_wcmp_group_members,
400 const std::vector<std::shared_ptr<p4orch::P4WcmpGroupMemberEntry>> &removed_wcmp_group_members)
401{
402 SWSS_LOG_ENTER();
403 std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> members;
404 ReturnCode recovery_status;
405 // Clean up created group members - remove created new members
406 if (created_wcmp_group_members.size() != 0)
407 {
408 recovery_status = processWcmpGroupMembersRemoval(created_wcmp_group_members, wcmp_group_key, members)
409 .prepend("Error during recovery: ");
410 }
411
412 // Clean up removed group members - create removed old members
413 if (recovery_status.ok() && removed_wcmp_group_members.size() != 0)
414 {
415 recovery_status = processWcmpGroupMembersAddition(removed_wcmp_group_members, wcmp_group_key,
416 wcmp_group_entry->wcmp_group_oid, members)
417 .prepend("Error during recovery: ");
418 }
419
420 if (!recovery_status.ok())
421 {
422 SWSS_RAISE_CRITICAL_STATE(recovery_status.message());
423 }

426ReturnCode WcmpManager::processUpdateRequest(P4WcmpGroupEntry *wcmp_group_entry)
427{
428 SWSS_LOG_ENTER();
429 auto *old_wcmp = getWcmpGroupEntry(wcmp_group_entry->wcmp_group_id);
430 wcmp_group_entry->wcmp_group_oid = old_wcmp->wcmp_group_oid;
431 const auto &wcmp_group_key = KeyGenerator::generateWcmpGroupKey(wcmp_group_entry->wcmp_group_id);
432 // Keep record of created next hop group members
433 std::vector<std::shared_ptr<p4orch::P4WcmpGroupMemberEntry>> created_wcmp_group_members;
434 // Keep record of removed next hop group members
435 std::vector<std::shared_ptr<p4orch::P4WcmpGroupMemberEntry>> removed_wcmp_group_members;
436
437 // Update group members steps:
438 // 1. Find the old member in the list with the smallest weight
439 // 2. Find the new member in the list with the smallest weight
440 // 3. Make SAI calls to remove old members except the reserved member with the
441 // smallest weight
442 // 4. Make SAI call to create the new member with the smallest weight
443 // 5. Make SAI call to remove the reserved old member
444 // 6. Make SAI calls to create remaining new members
445 ReturnCode update_request_status;
446 auto find_smallest_index = [&](p4orch::P4WcmpGroupEntry *wcmp,
447 std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> &other_members) -> int {
448 other_members.clear();
449 if (wcmp->wcmp_group_members.empty())
450 {
451 return -1;
452 }
453 int reserved_idx = 0;
454 for (int i = 1; i < (int)wcmp->wcmp_group_members.size(); i++)
455 {
456 if (wcmp->wcmp_group_members[i]->weight < wcmp->wcmp_group_members[reserved_idx]->weight)
457 {
458 other_members.push_back(wcmp->wcmp_group_members[reserved_idx]);
459 reserved_idx = i;
460 }
461 else
462 {
463 other_members.push_back(wcmp->wcmp_group_members[i]);
464 }
465 }
466 return reserved_idx;
467 };
468 // Find the old member who has the smallest weight, -1 if the member list is
469 // empty
470 std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> other_old_members;
471 int reserved_old_member_index = find_smallest_index(old_wcmp, other_old_members);
472 // Find the new member who has the smallest weight, -1 if the member list is
473 // empty
474 std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> other_new_members;
475 int reserved_new_member_index = find_smallest_index(wcmp_group_entry, other_new_members);
476
477 // Remove stale group members except the member with the smallest weight
478 if (other_old_members.size() != 0)
479 {
480 update_request_status =
481 processWcmpGroupMembersRemoval(other_old_members, wcmp_group_key, removed_wcmp_group_members);
482 if (!update_request_status.ok())
483 {
484 recoverGroupMembers(wcmp_group_entry, wcmp_group_key, created_wcmp_group_members,
485 removed_wcmp_group_members);
486 return update_request_status;
487 }
488 }
489
490 // Create the new member with the smallest weight if member list is nonempty
491 if (reserved_new_member_index != -1)
492 {
493 update_request_status = processWcmpGroupMembersAddition(
494 {wcmp_group_entry->wcmp_group_members[reserved_new_member_index]}, wcmp_group_key,
495 wcmp_group_entry->wcmp_group_oid, created_wcmp_group_members);
496 if (!update_request_status.ok())
497 {
498 recoverGroupMembers(wcmp_group_entry, wcmp_group_key, created_wcmp_group_members,
499 removed_wcmp_group_members);
500 return update_request_status;
501 }
502 }
503
504 // Remove the old member with the smallest weight if member list is nonempty
505 if (reserved_old_member_index != -1)
506 {
507 update_request_status = processWcmpGroupMembersRemoval(
508 {old_wcmp->wcmp_group_members[reserved_old_member_index]}, wcmp_group_key, removed_wcmp_group_members);
509 if (!update_request_status.ok())
510 {
511 recoverGroupMembers(wcmp_group_entry, wcmp_group_key, created_wcmp_group_members,
512 removed_wcmp_group_members);
513 return update_request_status;
514 }
515 }
516
517 // Create new group members
518 if (other_new_members.size() != 0)
519 {
520 update_request_status = processWcmpGroupMembersAddition(
521 other_new_members, wcmp_group_key, wcmp_group_entry->wcmp_group_oid, created_wcmp_group_members);
522 if (!update_request_status.ok())
523 {
524 recoverGroupMembers(wcmp_group_entry, wcmp_group_key, created_wcmp_group_members,
525 removed_wcmp_group_members);
526 return update_request_status;
527 }
528 }
529
530 m_wcmpGroupTable[wcmp_group_entry->wcmp_group_id] = *wcmp_group_entry;
531 return update_request_status;
532}

548ReturnCode WcmpManager::processWcmpGroupMembersRemoval(
549 const std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> &members, const std::string &wcmp_group_key,
550 std::vector<std::shared_ptr<P4WcmpGroupMemberEntry>> &removed_wcmp_group_members)
551{
552 SWSS_LOG_ENTER();
553 ReturnCode status;
554 std::vector<sai_status_t> statuses(members.size(), SAI_STATUS_FAILURE);
555 for (size_t i = 0; i < members.size(); ++i)
556 {
557 auto &member = members[i];
558 if (!member->pruned)
559 {
560 gNextHopGroupMemberBulker.remove_entry(&statuses[i], member->member_oid);
561 }
562 }
563 gNextHopGroupMemberBulker.flush();
564 for (size_t i = 0; i < members.size(); ++i)
565 {
566 auto &member = members[i];
567 if (member->pruned)
568 {
569 SWSS_LOG_NOTICE("Removed pruned member %s from group %s", member->next_hop_id.c_str(),
570 member->wcmp_group_id.c_str());
571 member->pruned = false;
572 }
573 else
574 {
575 if (statuses[i] != SAI_STATUS_SUCCESS)
576 {
577 if (status.ok())
578 {

604ReturnCode WcmpManager::removeWcmpGroup(const std::string &wcmp_group_id)
605{
606 SWSS_LOG_ENTER();
607 auto *wcmp_group = getWcmpGroupEntry(wcmp_group_id);
608 if (wcmp_group == nullptr)
609 {
610 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
611 << "WCMP group with id " << QuotedVar(wcmp_group_id) << " was not found.");
612 }
613 // Check refcount before deleting group members
614 uint32_t expected_refcount = (uint32_t)wcmp_group->wcmp_group_members.size();
615 uint32_t wcmp_group_refcount = 0;
616 const auto &wcmp_group_key = KeyGenerator::generateWcmpGroupKey(wcmp_group_id);
617 m_p4OidMapper->getRefCount(SAI_OBJECT_TYPE_NEXT_HOP_GROUP, wcmp_group_key, &wcmp_group_refcount);
618 if (wcmp_group_refcount > expected_refcount)
619 {
620 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_IN_USE)
621 << "Failed to remove WCMP group with id " << QuotedVar(wcmp_group_id) << ", as it has "
622 << wcmp_group_refcount - expected_refcount << " more objects than its group members (size="
623 << expected_refcount << ") referencing it.");
624 }
625
626 // Delete group members
627 std::vector<std::shared_ptr<p4orch::P4WcmpGroupMemberEntry>> removed_wcmp_group_members;
628 ReturnCode status =
629 processWcmpGroupMembersRemoval(wcmp_group->wcmp_group_members, wcmp_group_key, removed_wcmp_group_members);
630
631 // Delete group
632 if (status.ok())
633 {
634 auto sai_status = sai_next_hop_group_api->remove_next_hop_group(wcmp_group->wcmp_group_oid);
635 if (sai_status == SAI_STATUS_SUCCESS)
636 {
637 m_p4OidMapper->eraseOID(SAI_OBJECT_TYPE_NEXT_HOP_GROUP, wcmp_group_key);
638 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
639 m_wcmpGroupTable.erase(wcmp_group->wcmp_group_id);
640 return ReturnCode();
641 }
642 status = ReturnCode(sai_status) << "Failed to delete WCMP group with id "
643 << QuotedVar(wcmp_group->wcmp_group_id);
644 SWSS_LOG_ERROR("%s SAI_STATUS: %s", status.message().c_str(), sai_serialize_status(sai_status).c_str());
645 }
646 // Recover group members.
647 recoverGroupMembers(wcmp_group, wcmp_group_key, {}, removed_wcmp_group_members);
648 return status;
649}

769void WcmpManager::drain()
770{
771 SWSS_LOG_ENTER();
772
773 for (const auto &key_op_fvs_tuple : m_entries)
774 {
775 std::string table_name;
776 std::string db_key;
777 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &db_key);
778 const std::vector<swss::FieldValueTuple> &attributes = kfvFieldsValues(key_op_fvs_tuple);
779
780 ReturnCode status;
781 auto app_db_entry_or = deserializeP4WcmpGroupAppDbEntry(db_key, attributes);
782 if (!app_db_entry_or.ok())
783 {
784 status = app_db_entry_or.status();
785 SWSS_LOG_ERROR("Unable to deserialize APP DB WCMP group entry with key %s: %s",
786 QuotedVar(table_name + ":" + db_key).c_str(), status.message().c_str());
787 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
788 status,
789 /*replace=*/true);
790 continue;
791 }
792 auto &app_db_entry = *app_db_entry_or;
793
794 const std::string &operation = kfvOp(key_op_fvs_tuple);
795 if (operation == SET_COMMAND)
796 {
797 status = validateWcmpGroupEntry(app_db_entry);
798 if (!status.ok())
799 {
800 SWSS_LOG_ERROR("Invalid WCMP group with id %s: %s", QuotedVar(app_db_entry.wcmp_group_id).c_str(),
801 status.message().c_str());
802 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
803 status,
804 /*replace=*/true);
805 continue;
806 }
807 auto *wcmp_group_entry = getWcmpGroupEntry(app_db_entry.wcmp_group_id);
808 if (wcmp_group_entry == nullptr)
809 {
810 // Create WCMP group
811 status = processAddRequest(&app_db_entry);
812 }
813 else
814 {
815 // Modify existing WCMP group
816 status = processUpdateRequest(&app_db_entry);
817 }
818 }
819 else if (operation == DEL_COMMAND)
820 {
821 // Delete WCMP group
822 status = removeWcmpGroup(app_db_entry.wcmp_group_id);
823 }
824 else
825 {
826 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
827 << "Unknown operation type: " << QuotedVar(operation) << " for WCMP group entry with key "
828 << QuotedVar(table_name) << ":" << QuotedVar(db_key)
829 << "; only SET and DEL operations are allowed.";
830 SWSS_LOG_ERROR("Unknown operation type %s\n", QuotedVar(operation).c_str());
831 }
832 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple), status,
833 /*replace=*/true);
834 }
835 m_entries.clear();
836}

838std::string WcmpManager::verifyState(const std::string &key, const std::vector<swss::FieldValueTuple> &tuple)
839{
840 SWSS_LOG_ENTER();
841
842 auto pos = key.find_first_of(kTableKeyDelimiter);
843 if (pos == std::string::npos)
844 {
845 return std::string("Invalid key: ") + key;
846 }
847 std::string p4rt_table = key.substr(0, pos);
848 std::string p4rt_key = key.substr(pos + 1);
849 if (p4rt_table != APP_P4RT_TABLE_NAME)
850 {
851 return std::string("Invalid key: ") + key;
852 }
853 std::string table_name;
854 std::string key_content;
855 parseP4RTKey(p4rt_key, &table_name, &key_content);
856 if (table_name != APP_P4RT_WCMP_GROUP_TABLE_NAME)
857 {
858 return std::string("Invalid key: ") + key;
859 }
860
861 ReturnCode status;
862 auto app_db_entry_or = deserializeP4WcmpGroupAppDbEntry(key_content, tuple);
863 if (!app_db_entry_or.ok())
864 {
865 status = app_db_entry_or.status();
866 std::stringstream msg;
867 msg << "Unable to deserialize key " << QuotedVar(key) << ": " << status.message();
868 return msg.str();
869 }
870 auto &app_db_entry = *app_db_entry_or;
871
872 auto *wcmp_group_entry = getWcmpGroupEntry(app_db_entry.wcmp_group_id);
873 if (wcmp_group_entry == nullptr)
874 {
875 std::stringstream msg;
876 msg << "No entry found with key " << QuotedVar(key);
877 return msg.str();
878 }
879
880 std::string cache_result = verifyStateCache(app_db_entry, wcmp_group_entry);
881 std::string asic_db_result = verifyStateAsicDb(wcmp_group_entry);
882 if (cache_result.empty())
883 {
884 return asic_db_result;
885 }
886 if (asic_db_result.empty())
887 {
888 return cache_result;
889 }
890 return cache_result + "; " + asic_db_result;
891}
