300std::vector<std::string> ValuesStore::update_storage(const HashOfRecords & storage)
301{
302 std::vector<std::string> to_update;
303
304 for (const auto & entry_pair: storage)
305 {
306 const auto & entry_key = entry_pair.first;
307 const auto & entry_values = entry_pair.second;
308 if (m_storage.find(entry_key) == m_storage.end())
309 {
310 m_storage.emplace(entry_pair);
311 to_update.emplace_back(entry_key);
312 }
313 else
314 {
315 bool is_changed = false;
316 for (const auto & row_pair: entry_values)
317 {
318 const auto & row_key = row_pair.first;
319 const auto & row_value = row_pair.second;
320 if (m_storage[entry_key][row_key] != row_value)
321 {
322 is_changed = true;
323 break;
324 }
325 }
326
327 if (is_changed)
328 {
329 m_storage.erase(entry_key);
330 m_storage.emplace(entry_pair);
331 to_update.emplace_back(entry_key);
332 }
333 }
334 }
335
336 return to_update;
337}
