59void Sai::channelOpSetReadOnlyAttribute(
60 _In_ const std::string &key,
61 _In_ const std::vector<swss::FieldValueTuple> &values)
62{
63 SWSS_LOG_ENTER();
64
65 for (const auto &v: values)
66 {
67 SWSS_LOG_DEBUG("attr: %s: %s", fvField(v).c_str(), fvValue(v).c_str());
68 }
69
70 if (values.size() != 1)
71 {
72 SWSS_LOG_ERROR("expected 1 value only, but given: %zu", values.size());
73 return;
74 }
75
76 const std::string &str_object_type = key.substr(0, key.find(":"));
77 const std::string &str_object_id = key.substr(key.find(":") + 1);
78
79 sai_object_type_t object_type;
80 sai_deserialize_object_type(str_object_type, object_type);
81
82 if (object_type == SAI_OBJECT_TYPE_NULL || object_type >= SAI_OBJECT_TYPE_EXTENSIONS_MAX)
83 {
84 SWSS_LOG_ERROR("invalid object type: %d", object_type);
85 return;
86 }
87
88 auto info = sai_metadata_get_object_type_info(object_type);
89
90 if (info->isnonobjectid)
91 {
92 SWSS_LOG_ERROR("non object id %s is not supported yet", str_object_type.c_str());
93 return;
94 }
95
96 sai_object_id_t object_id;
97
98 sai_deserialize_object_id(str_object_id, object_id);
99
100 sai_object_type_t ot = objectTypeQuery(object_id);
101
102 if (ot != object_type)
103 {
104 SWSS_LOG_ERROR("object type is different than provided %s, but oid is %s",
105 str_object_type.c_str(), sai_serialize_object_type(ot).c_str());
106 return;
107 }
108
109 sai_object_id_t switch_id = switchIdQuery(object_id);
110
111 if (switch_id == SAI_NULL_OBJECT_ID)
112 {
113 SWSS_LOG_ERROR("failed to find switch id for oid %s", str_object_id.c_str());
114 return;
115 }
116
117 // oid is validated and we got switch id
118
119 const std::string &str_attr_id = fvField(values.at(0));
120 const std::string &str_attr_value = fvValue(values.at(0));
121
122 auto meta = sai_metadata_get_attr_metadata_by_attr_id_name(str_attr_id.c_str());
123
124 if (meta == NULL)
125 {
126 SWSS_LOG_ERROR("failed to find attr %s", str_attr_id.c_str());
127 return;
128 }
129
130 if (meta->objecttype != ot)
131 {
132 SWSS_LOG_ERROR("attr %s belongs to different object type than oid: %s",
133 str_attr_id.c_str(), sai_serialize_object_type(ot).c_str());
134 return;
135 }
136
137 // we got attr metadata
138
139 sai_attribute_t attr;
140
141 attr.id = meta->attrid;
142
143 sai_deserialize_attr_value(str_attr_value, *meta, attr);
144
145 SWSS_LOG_NOTICE("switch id is %s", sai_serialize_object_id(switch_id).c_str());
146
147 sai_status_t status = m_meta->meta_unittests_allow_readonly_set_once(meta->objecttype, meta->attrid);
148
149 if (status != SAI_STATUS_SUCCESS)
150 {
151 SWSS_LOG_ERROR("failed to enable SET readonly attribute once: %s", sai_serialize_status(status).c_str());
152 return;
153 }
154
155 sai_object_meta_key_t meta_key = { .objecttype = ot, .objectkey = { .key = { .object_id = object_id } } };
156
157 status = m_meta->set(meta_key, &attr); // name hidden in base class (but same name overloaded in derived class)
158
159 if (status != SAI_STATUS_SUCCESS)
160 {
161 SWSS_LOG_ERROR("failed to set %s to %s on %s",
162 str_attr_id.c_str(), str_attr_value.c_str(), str_object_id.c_str());
163 }
164 else
165 {
166 SWSS_LOG_NOTICE("SUCCESS to set %s to %s on %s",
167 str_attr_id.c_str(), str_attr_value.c_str(), str_object_id.c_str());
168 }
169
170 sai_deserialize_free_attribute_value(meta->attrvaluetype, attr);
171}

173void Sai::channelOpSetStats(
174 _In_ const std::string &key,
175 _In_ const std::vector<swss::FieldValueTuple> &values)
176{
177 SWSS_LOG_ENTER();
178
179 // NOTE: add support for non object id stats
180
181 sai_object_id_t oid;
182
183 sai_deserialize_object_id(key, oid);
184
185 sai_object_type_t ot = objectTypeQuery(oid);
186
187 if (ot == SAI_OBJECT_TYPE_NULL)
188 {
189 SWSS_LOG_ERROR("invalid object id: %s", key.c_str());
190 return;
191 }
192
193 sai_object_id_t switch_id = switchIdQuery(oid);
194
195 if (switch_id == SAI_NULL_OBJECT_ID)
196 {
197 SWSS_LOG_ERROR("unable to get switch_id from oid: %s", key.c_str());
198 return;
199 }
200
201 auto statenum = sai_metadata_get_object_type_info(ot)->statenum;
202
203 if (statenum == NULL)
204 {
205 SWSS_LOG_ERROR("object %s does not support statistics",
206 sai_serialize_object_type(ot).c_str());
207 return;
208 }
209
210 /*
211 * Check if object for statistics have statistic map created, if not
212 * create empty map.
213 */
214
215 std::map<sai_stat_id_t, uint64_t> stats;
216
217 for (auto v: values)
218 {
219 // value format: stat_enum_name:uint64
220
221 auto name = fvField(v);
222 auto val = fvValue(v);
223
224 uint64_t value;
225
226 if (sscanf(val.c_str(), "%" PRIu64, &value) != 1)
227 {
228 SWSS_LOG_ERROR("failed to deserialize %s as counter value uint64_t", val.c_str());
229 continue;
230 }
231
232 // linear search
233
234 int enumvalue = -1;
235
236 for (size_t i = 0; i < statenum->valuescount; ++i)
237 {
238 if (statenum->valuesnames[i] == name)
239 {
240 enumvalue = statenum->values[i];
241 break;
242 }
243 }
244
245 if (enumvalue == -1)
246 {
247 SWSS_LOG_ERROR("failed to find enum value: %s", name.c_str());
248 continue;
249 }
250
251 SWSS_LOG_INFO("writing %s = %lu on %s", name.c_str(), value, key.c_str());
252
253 stats[enumvalue] = value;
254 }
255
256 m_vsSai->debugSetStats(oid, stats);
257}

259void Sai::handleUnittestChannelOp(
260 _In_ const std::string &op,
261 _In_ const std::string &key,
262 _In_ const std::vector<swss::FieldValueTuple> &values)
263{
264 MUTEX();
265
266 SWSS_LOG_ENTER();
267
268 // at this point api are initialized since unit test thread is started if
269 // initialization will be successful
270
271 /*
272 * Since we will access and modify DB we need to be under mutex.
273 *
274 * NOTE: since this unittest channel is handled in thread, then that means
275 * there is a DELAY from producer and consumer thread in VS, so if user
276 * will set value on the specific READ_ONLY value he should wait for some
277 * time until that value will be propagated to virtual switch.
278 */
279
280 SWSS_LOG_NOTICE("op = %s, key = %s", op.c_str(), key.c_str());
281
282 for (const auto &v: values)
283 {
284 SWSS_LOG_NOTICE("attr: %s: %s", fvField(v).c_str(), fvValue(v).c_str());
285 }
286
287 if (op == SAI_VS_UNITTEST_ENABLE_UNITTESTS)
288 {
289 channelOpEnableUnittest(key, values);
290 }
291 else if (op == SAI_VS_UNITTEST_SET_RO_OP)
292 {
293 channelOpSetReadOnlyAttribute(key, values);
294 }
295 else if (op == SAI_VS_UNITTEST_SET_STATS_OP)
296 {
297 channelOpSetStats(key, values);
298 }
299 else
300 {
301 SWSS_LOG_THROW("unknown unittest operation: %s", op.c_str());
302 }
303}

305void Sai::unittestChannelThreadProc()
306{
307 SWSS_LOG_ENTER();
308
309 SWSS_LOG_NOTICE("enter VS unittest channel thread");
310
311 swss::Select s;
312
313 s.addSelectable(m_unittestChannelNotificationConsumer.get());
314 s.addSelectable(m_unittestChannelThreadEvent.get());
315
316 while (m_unittestChannelRun)
317 {
318 swss::Selectable *sel = nullptr;
319
320 int result = s.select(&sel);
321
322 if (sel == m_unittestChannelThreadEvent.get())
323 {
324 // user requested shutdown_switch
325 break;
326 }
327
328 if (result == swss::Select::OBJECT)
329 {
330 swss::KeyOpFieldsValuesTuple kco;
331
332 std::string op;
333 std::string data;
334 std::vector<swss::FieldValueTuple> values;
335
336 m_unittestChannelNotificationConsumer->pop(op, data, values);
337
338 SWSS_LOG_DEBUG("notification: op = %s, data = %s", op.c_str(), data.c_str());
339
340 try
341 {
342 handleUnittestChannelOp(op, data, values);
343 }
344 catch (const std::exception &e)
345 {
346 SWSS_LOG_ERROR("Exception: op = %s, data = %s, %s", op.c_str(), data.c_str(), e.what());
347 }
348 }
349 }
350
351 SWSS_LOG_NOTICE("exit VS unittest channel thread");
352}
