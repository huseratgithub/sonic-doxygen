187void ComparisonLogic::matchOids(
188 _In_ AsicView &currentView,
189 _In_ AsicView &temporaryView)
190{
191 SWSS_LOG_ENTER();
192
193 auto coldBootDiscoveredVids = m_switch->getColdBootDiscoveredVids();
194
195 for (const auto &temporaryIt: temporaryView.m_oOids)
196 {
197 sai_object_id_t temporaryVid = temporaryIt.first;
198
199 const auto &currentIt = currentView.m_oOids.find(temporaryVid);
200
201 if (currentIt == currentView.m_oOids.end())
202 {
203 continue;
204 }
205
206 sai_object_id_t vid = temporaryVid;
207 sai_object_id_t rid = currentView.m_vidToRid.at(vid);
208
209 // save VID and RID in temporary view
210
211 temporaryView.m_ridToVid[rid] = vid;
212 temporaryView.m_vidToRid[vid] = rid;
213
214 // set both objects status as matched
215
216 temporaryIt.second->setObjectStatus(SAI_OBJECT_STATUS_MATCHED);
217 currentIt->second->setObjectStatus(SAI_OBJECT_STATUS_MATCHED);
218
219 SWSS_LOG_INFO("matched %s RID %s VID %s",
220 currentIt->second->m_str_object_type.c_str(),
221 sai_serialize_object_id(rid).c_str(),
222 sai_serialize_object_id(vid).c_str());
223
224 if (coldBootDiscoveredVids.find(vid) == coldBootDiscoveredVids.end())
225 {
226 auto ot = currentIt->second->getObjectType();
227
228 switch (ot)
229 {
230 case SAI_OBJECT_TYPE_INGRESS_PRIORITY_GROUP:
231 case SAI_OBJECT_TYPE_SCHEDULER_GROUP:
232 case SAI_OBJECT_TYPE_QUEUE:
233 case SAI_OBJECT_TYPE_PORT:
234 break;
235
236 default:
237
238 // Should we also add check if also not in removed?
239 // This is for case of only GET:
240 // 1. cold boot:
241 // - OA assigns buffer profile to Queue
242 // 2. warm boot:
243 // - OA do GET on Queue and got previous buffer profile
244 // - OA assigns new buffer profile on Queue
245 //
246 // Question: what should happen to old buffer profile? should it be removed?
247 // No, since OA still hold's the reference to that VID and may use it later.
248 SWSS_LOG_INFO("matched %s VID %s was not in cold boot, possible only GET?",
249 sai_serialize_object_id(vid).c_str(),
250 currentIt->second->m_str_object_type.c_str());
251 break;
252 }
253 }
254 }
255
256 SWSS_LOG_NOTICE("matched oids");

373void ComparisonLogic::procesObjectAttributesForViewTransition(
374 _In_ AsicView &currentView,
375 _In_ AsicView &temporaryView,
376 _In_ const std::shared_ptr<SaiObj> &temporaryObj)
377{
378 SWSS_LOG_ENTER();
379
380 SWSS_LOG_INFO("%s %s", temporaryObj->m_str_object_type.c_str(), temporaryObj->m_str_object_id.c_str());
381
382 /*
383 * First we need to make sure if all attributes of this temporary object
384 * are in FINAL or MATCHED state, then we can process this object and find
385 * best match in current asic view.
386 */
387
388 for (auto &at: temporaryObj->getAllAttributes())
389 {
390 auto &attribute = at.second;
391
392 SWSS_LOG_INFO("attr %s", attribute->getStrAttrId().c_str());
393
394 /*
395 * For each object id in attributes go recursively to match those
396 * objects.
397 */
398
399 for (auto vid: attribute->getOidListFromAttribute())
400 {
401 if (vid == SAI_NULL_OBJECT_ID)
402 {
403 continue;
404 }
405
406 SWSS_LOG_INFO("- processing attr VID %s", sai_serialize_object_id(vid).c_str());
407
408 auto tempParent = temporaryView.m_oOids.at(vid);
409
410 processObjectForViewTransition(currentView, temporaryView, tempParent); // recursion
411
412 /*
413 * Temporary object here is never changed, even if we do recursion
414 * here all that could been removed are objects in current view
415 * tree so we don't need to worry about any temporary object
416 * removal.
417 */
418 }
419 }
420
421 if (temporaryObj->isOidObject())
422 {
423 return;
424 }
425
426 /*
427 * For non object id types like NEIGHBOR or ROUTE they have object id
428 * inside object struct entry, they also need to be processed, in SAI 1.0
429 * this can be automated since we have metadata for those structs.
430 */
431
432 for (size_t j = 0; j < temporaryObj->m_info->structmemberscount; ++j)
433 {
434 const sai_struct_member_info_t *m = temporaryObj->m_info->structmembers[j];
435
436 if (m->membervaluetype != SAI_ATTR_VALUE_TYPE_OBJECT_ID)
437 {
438 continue;
439 }
440
441 sai_object_id_t vid = m->getoid(&temporaryObj->m_meta_key);
442
443 SWSS_LOG_INFO("- processing %s (%s) VID %s",
444 temporaryObj->m_str_object_type.c_str(),
445 m->membername,
446 sai_serialize_object_id(vid).c_str());
447
448 auto structObject = temporaryView.m_oOids.at(vid);
449
450 processObjectForViewTransition(currentView, temporaryView, structObject); // recursion
451 }

541void ComparisonLogic::removeExistingObjectFromCurrentView(
542 _In_ AsicView &currentView,
543 _In_ const AsicView &temporaryView,
544 _In_ const std::shared_ptr<SaiObj> &currentObj)
545{
546 SWSS_LOG_ENTER();
547
548 if (currentObj->getObjectStatus() != SAI_OBJECT_STATUS_NOT_PROCESSED)
549 {
550 SWSS_LOG_THROW("FATAL: removing object with status: %d, logic error", currentObj->getObjectStatus());
551 }
552
553 /*
554 * This decreasing VID reference will be hard when actual reference will
555 * be one of default objects like CPU or default trap group when we "bring
556 * default value" or maybe in oid case we will not bring default values and
557 * just remove and recreate.
558 */
559
560 if (currentObj->isOidObject())
561 {
562 int count = currentView.getVidReferenceCount(currentObj->getVid());
563
564 if (count != 0)
565 {
566 /*
567 * If references count is not zero, we need to remove child
568 * first for that we need dependency tree, not supported yet.
569 *
570 * NOTE: Object can be non removable and can have reference on it.
571 */
572
573 SWSS_LOG_THROW("can't remove existing object %s:%s since reference count is %d, FIXME",
574 currentObj->m_str_object_type.c_str(),
575 currentObj->m_str_object_id.c_str(),
576 count);
577 }
578 }
579
580 /*
581 * If some object can't be removed from current view and it's missing from
582 * temp, then it needs to be transferred to temp as well and bring to
583 * default values.
584 *
585 * First we need to check if object can be removed, like port's cant be
586 * removed, vlan 1, queues, ingress pg etc.
587 *
588 * Remove for those objects is not supported now lets bring back
589 * object to default state.
590 *
591 * For oid values we can figure out if we would have list of
592 * defaults which can be removed and which can't and use list same
593 * for queues, schedulers etc.
594 *
595 * We could use switch here and non created objects to make this simpler.
596 */
597
598
599 if (currentObj->getObjectType() == SAI_OBJECT_TYPE_SWITCH)
600 {
601 /*
602 * Remove switch is much simpler, we just need check if switch doesn't
603 * exist in temporary view, and we can just remove it. In here when
604 * attributes don't match it's much complicated.
605 */
606
607 SWSS_LOG_THROW("switch can't be removed from current view, not supported yet");
608 }
609
610 if (isNonRemovableObject(currentView, temporaryView, currentObj))
611 {
612 bringNonRemovableObjectToDefaultState(currentView, currentObj);
613 }
614 else
615 {
616 /*
617 * Asic remove object is decreasing reference count on non object ID if
618 * current object is non object id, release existing links only looks
619 * into attributes.
620 */
621
622 currentView.asicRemoveObject(currentObj);
623
624 currentObj->setObjectStatus(SAI_OBJECT_STATUS_REMOVED);
625 }

628sai_object_id_t ComparisonLogic::translateTemporaryVidToCurrentVid(
629 _In_ const AsicView &currentView,
630 _In_ const AsicView &temporaryView,
631 _In_ sai_object_id_t tvid)
632{
633 SWSS_LOG_ENTER();
634
635 /*
636 * This method is used to translate temporary VID to current VID using RID
637 * which should be present in both views. If RID doesn't exist, then we
638 * check whether object was created if so, then we return temporary VID
639 * instead of creating new current VID and we don't need to track mapping
640 * of those vids not having actual RID. This function should be used only
641 * when we creating new objects in current view.
642 */
643
644 auto temporaryIt = temporaryView.m_vidToRid.find(tvid);
645
646 if (temporaryIt == temporaryView.m_vidToRid.end())
647 {
648 /*
649 * This can also happen when this VID object was created dynamically,
650 * and don't have RID assigned yet, in that case we should use exact
651 * same VID. We need to get temporary object for that VID and see if
652 * flag created is set, and same VID must then exists in current view.
653 */
654
655 /*
656 * We need to find object that has this vid
657 */
658
659 auto tempIt = temporaryView.m_oOids.find(tvid);
660
661 if (tempIt == temporaryView.m_oOids.end())
662 {
663 SWSS_LOG_THROW("temporary VID %s not found in temporary view",
664 sai_serialize_object_id(tvid).c_str());
665 }
666
667 const auto &tempObj = tempIt->second;
668
669 if (tempObj->m_createdObject)
670 {
671 SWSS_LOG_DEBUG("translated temp VID %s to current, since object was created",
672 sai_serialize_object_id(tvid).c_str());
673
674 return tvid;
675 }
676
677 SWSS_LOG_THROW("VID %s was not found in temporary view, was object created? FIXME",
678 sai_serialize_object_id(tvid).c_str());
679 }
680
681 sai_object_id_t rid = temporaryIt->second;
682
683 auto currentIt = currentView.m_ridToVid.find(rid);
684
685 if (currentIt == currentView.m_ridToVid.end())
686 {
687 SWSS_LOG_THROW("RID %s was not found in current view",
688 sai_serialize_object_id(rid).c_str());
689 }
690
691 sai_object_id_t cvid = currentIt->second;
692
693 SWSS_LOG_DEBUG("translated temp VID %s using RID %s to current VID %s",
694 sai_serialize_object_id(tvid).c_str(),
695 sai_serialize_object_id(rid).c_str(),
696 sai_serialize_object_id(cvid).c_str());
697
698 return cvid;
699}

701std::shared_ptr<SaiAttr> ComparisonLogic::translateTemporaryVidsToCurrentVids(
702 _In_ const AsicView &currentView,
703 _In_ const AsicView &temporaryView,
704 _In_ const std::shared_ptr<SaiObj> &currentObj,
705 _In_ const std::shared_ptr<SaiAttr> &inattr)
706{
707 SWSS_LOG_ENTER();
708
709 /*
710 * We are creating copy here since we will modify contents of that
711 * attribute.
712 */
713
714 auto attr = std::make_shared<SaiAttr>(inattr->getStrAttrId(), inattr->getStrAttrValue());
715
716 if (!attr->isObjectIdAttr())
717 {
718 return attr;
719 }
720
721 /*
722 * We need temporary VID translation to current view RID.
723 *
724 * We also need simpler version that will translate simple VID for
725 * oids present inside non object id structs.
726 */
727
728 uint32_t count = 0;
729
730 sai_object_id_t *objectIdList = NULL;
731
732 auto &at = *attr->getRWSaiAttr();
733
734 switch (attr->getAttrMetadata()->attrvaluetype)
735 {
736 case SAI_ATTR_VALUE_TYPE_OBJECT_ID:
737 count = 1;
738 objectIdList = &at.value.oid;
739 break;
740
741 case SAI_ATTR_VALUE_TYPE_OBJECT_LIST:
742 count = at.value.objlist.count;
743 objectIdList = at.value.objlist.list;
744 break;
745
746 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_ID:
747
748 if (at.value.aclfield.enable)
749 {
750 count = 1;
751 objectIdList = &at.value.aclfield.data.oid;
752 }
753
754 break;
755
756 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_LIST:
757
758 if (at.value.aclfield.enable)
759 {
760 count = at.value.aclfield.data.objlist.count;
761 objectIdList = at.value.aclfield.data.objlist.list;
762 }
763
764 break;
765
766 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_ID:
767
768 if (at.value.aclaction.enable)
769 {
770 count = 1;
771 objectIdList = &at.value.aclaction.parameter.oid;
772 }
773
774 break;
775
776 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_LIST:
777
778 if (at.value.aclaction.enable)
779 {
780 count = at.value.aclaction.parameter.objlist.count;
781 objectIdList = at.value.aclaction.parameter.objlist.list;
782 }
783
784 break;
785
786 default:
787
788 // TODO check is is oid object and throw as not translated
789
790 break;
791 }
792
793 for (uint32_t i = 0; i < count; ++i)
794 {
795 sai_object_id_t tvid = objectIdList[i];
796
797 if (tvid == SAI_NULL_OBJECT_ID)
798 {
799 continue;
800 }
801
802 /*
803 * Do actual translation.
804 */
805
806 objectIdList[i] = translateTemporaryVidToCurrentVid(currentView, temporaryView, tvid);
807 }
808
809 /*
810 * Since we probably performed updates on this attribute, since VID was
811 * exchanged, we need to update value string of that attribute.
812 */
813
814 attr->updateValue();
815
816 return attr;
817}

834 _In_ const AsicView &temporaryView,
835 _In_ const std::shared_ptr<SaiObj> &currentObj,
836 _In_ const std::shared_ptr<SaiAttr> &inattr)
837{
838 SWSS_LOG_ENTER();
839
840 /*
841 * At the beginning just small assert to check if object is create and set.
842 */
843
844 const auto meta = inattr->getAttrMetadata();
845
846 if (!SAI_HAS_FLAG_CREATE_AND_SET(meta->flags))
847 {
848 SWSS_LOG_THROW("can't set attribute %s on current object %s:%s since it's not CREATE_AND_SET",
849 meta->attridname,
850 currentObj->m_str_object_type.c_str(),
851 currentObj->m_str_object_id.c_str());
852 }
853
854 /*
855 * NOTE: Since attribute we SET can be OID attribute, and that can be a VID
856 * or list of VIDs from temporary view. So after this "set" operation we
857 * will end up with MIXED VIDs on current view, some object will contain
858 * VIDs from temporary view, and this can lead to fail in
859 * translate_vid_to_rid since we will need to look inside two views to find
860 * right RID.
861 *
862 * We need a flag to say that VID is for object that was created
863 * and we can mix only those VID's for objects that was created.
864 *
865 * In here we need to translate attribute VIDs to current VIDs to keep
866 * track in dependency tree, since if we use temporary VID that will point
867 * to the same RID then we will lost track in dependency tree of that VID
868 * reference count.
869 *
870 * This should also done for object id's inside non object ids.
871 */
872
873 std::shared_ptr<SaiAttr> attr = translateTemporaryVidsToCurrentVids(currentView, temporaryView, currentObj, inattr);
874
875 currentView.asicSetAttribute(currentObj, attr);
876}
877
888void ComparisonLogic::createNewObjectFromTemporaryObject(
889 _In_ AsicView &currentView,
890 _In_ const AsicView &temporaryView,
891 _In_ const std::shared_ptr<SaiObj> &temporaryObj)
892{
893 SWSS_LOG_ENTER();
894
895 SWSS_LOG_INFO("creating object %s:%s",
896 temporaryObj->m_str_object_type.c_str(),
897 temporaryObj->m_str_object_id.c_str());
898
899 /*
900 * This trap can be default trap group, so let's check if it's, then we
901 * can't create it, we need to set new attributes if that possible. But
902 * this case should not happen since on syncd start we are putting default
903 * trap group to asic view, so if user will query default one, they will be
904 * matched by RID.
905 *
906 * Default trap group is transferred to view on init if it doesn't exist.
907 *
908 * There should be no such action here, since this scenario would mean that
909 * there is default switch object in temporary view, but not in current
910 * view. All default objects are put to ASIC state table when switch is
911 * created. So what can happen is opposite scenario.
912 */
913
914 /*
915 * NOTE: Since attributes we pass to create can be OID attributes, and that
916 * can be a VID or list of VIDs from temporary view. So after this create
917 * operation we will end up with MIXED VIDs on current view, some object
918 * will contain VIDs from temporary view, and this can lead to fail in
919 * translate_vid_to_rid since we will need to look inside two views to find
920 * right RID.
921 *
922 * We need a flag to say that VID is for object that was created and we can
923 * mix only those VID's for objects that was created.
924 *
925 * In here we need to translate attribute VIDs to current VIDs to keep
926 * track in dependency tree, since if we use temporary VID that will point
927 * to the same RID then we will lost track in dependency tree of that VID
928 * reference count.
929 *
930 * This should also be done for object id's inside non object ids.
931 */
932
933 /*
934 * We need to loop through all attributes and create copy of this object
935 * and translate all attributes for getting correct VIDs like in set
936 * operation, we can't use VID's from temporary view to not mix them with
937 * current view, except created object ID's, since we won't be creating any
938 * new VID on the way (but we could) maybe we can do that later - but for
939 * that we will need some kind of RID which would later need to be removed.
940 */
941
942 std::shared_ptr<SaiObj> currentObj = std::make_shared<SaiObj>();
943
944 /*
945 * TODO Find out better way to do this, copy operator ?
946 */
947
948 currentObj->m_str_object_type = temporaryObj->m_str_object_type;
949 currentObj->m_str_object_id = temporaryObj->m_str_object_id; // temporary VID / non object id
950 currentObj->m_meta_key = temporaryObj->m_meta_key; // temporary VID / non object id
951 currentObj->m_info = temporaryObj->m_info;
952
953 if (!temporaryObj->isOidObject())
954 {
955 /*
956 * Since object contains OID inside struct, this OID may already exist
957 * in current view, so we need to do translation here.
958 */
959
960 for (size_t j = 0; j < currentObj->m_info->structmemberscount; ++j)
961 {
962 const sai_struct_member_info_t *m = currentObj->m_info->structmembers[j];
963
964 if (m->membervaluetype != SAI_ATTR_VALUE_TYPE_OBJECT_ID)
965 {
966 continue;
967 }
968
969 sai_object_id_t vid = m->getoid(&currentObj->m_meta_key);
970
971 vid = translateTemporaryVidToCurrentVid(currentView, temporaryView, vid);
972
973 m->setoid(&currentObj->m_meta_key, vid);
974
975 /*
976 * Bind new vid reference is already done inside asicCreateObject.
977 */
978 }
979
980 /*
981 * Since it's possible that object id may had been changed, we need to
982 * update string as well.
983 */
984
985 switch (temporaryObj->getObjectType())
986 {
987 case SAI_OBJECT_TYPE_ROUTE_ENTRY:
988
989 currentObj->m_str_object_id = sai_serialize_route_entry(currentObj->m_meta_key.objectkey.key.route_entry);
990 break;
991
992 case SAI_OBJECT_TYPE_NEIGHBOR_ENTRY:
993
994 currentObj->m_str_object_id = sai_serialize_neighbor_entry(currentObj->m_meta_key.objectkey.key.neighbor_entry);
995 break;
996
997 case SAI_OBJECT_TYPE_FDB_ENTRY:
998 currentObj->m_str_object_id = sai_serialize_fdb_entry(currentObj->m_meta_key.objectkey.key.fdb_entry);
999 break;
1000
1001 case SAI_OBJECT_TYPE_NAT_ENTRY:
1002 currentObj->m_str_object_id = sai_serialize_nat_entry(currentObj->m_meta_key.objectkey.key.nat_entry);
1003 break;
1004
1005 case SAI_OBJECT_TYPE_INSEG_ENTRY:
1006 currentObj->m_str_object_id = sai_serialize_inseg_entry(currentObj->m_meta_key.objectkey.key.inseg_entry);
1007 break;
1008
1009 default:
1010
1011 SWSS_LOG_THROW("unexpected non object id type: %s",
1012 sai_serialize_object_type(temporaryObj->getObjectType()).c_str());
1013 }
1014 }
1015
1016 /*
1017 * CreateObject flag is set to true, so when we we will be looking to
1018 * translate VID between temporary view and current view (since status will
1019 * be FINAL) then we will know that there is no actual RID yet, and both
1020 * VIDs are the same. So this is how we cam match both objects between
1021 * views.
1022 */
1023
1024 currentObj->m_createdObject = true;
1025 temporaryObj->m_createdObject = true;
1026
1027 for (const auto &pair: temporaryObj->getAllAttributes())
1028 {
1029 const auto &tmpattr = pair.second;
1030
1031 std::shared_ptr<SaiAttr> attr = translateTemporaryVidsToCurrentVids(currentView, temporaryView, currentObj, tmpattr);
1032
1033 if (attr->isObjectIdAttr())
1034 {
1035 /*
1036 * This is new attribute so we don't need to release any new links.
1037 * But we need to bind new links on current object.
1038 */
1039
1040 currentObj->setAttr(attr);
1041 }
1042 else
1043 {
1044 /*
1045 * This create attribute don't contain any OIDs so no extra
1046 * operations are required, we don't need to break any references
1047 * and decrease any reference count.
1048 */
1049
1050 currentObj->setAttr(attr);
1051 }
1052 }
1053
1054 /*
1055 * Asic create object inserts new reference to track inside asic view.
1056 */
1057
1058 currentView.asicCreateObject(currentObj);
1059
1060 /*
1061 * Move both object status to FINAL since both objects were processed
1062 * successfully and object was created.
1063 */
1064
1065 currentObj->setObjectStatus(SAI_OBJECT_STATUS_FINAL);
1066 temporaryObj->setObjectStatus(SAI_OBJECT_STATUS_FINAL);
1067}
1068
1069void ComparisonLogic::updateObjectStatus(
1070 _In_ AsicView &currentView,
1071 _In_ AsicView &temporaryView,
1072 _In_ const std::shared_ptr<SaiObj> &currentBestMatch,
1073 _In_ const std::shared_ptr<SaiObj> &temporaryObj)
1074{
1075 SWSS_LOG_ENTER();
1076
1077 bool notprocessed = ((temporaryObj->getObjectStatus() == SAI_OBJECT_STATUS_NOT_PROCESSED) &&
1078 (currentBestMatch->getObjectStatus() == SAI_OBJECT_STATUS_NOT_PROCESSED));
1079
1080 bool matched = ((temporaryObj->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED) &&
1081 (currentBestMatch->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED));
1082
1083 if (notprocessed || matched)
1084 {
1085 /*
1086 * Since everything was processed, then move those object status to
1087 * FINAL.
1088 */
1089
1090 temporaryObj->setObjectStatus(SAI_OBJECT_STATUS_FINAL);
1091 currentBestMatch->setObjectStatus(SAI_OBJECT_STATUS_FINAL);
1092
1093 if (temporaryObj->isOidObject())
1094 {
1095 /*
1096 * When objects are object id type, we need to update map in
1097 * temporary view for correct VID/RID.
1098 *
1099 * In here RID should always exists, since when objects are MATCHED
1100 * then RID exists and both VID's are the same, and when both
1101 * objects are not processed, then RID also exists since current
1102 * object was selected as current best match. Other options are
1103 * object was removed, but the it could not be selected, or object
1104 * was created, but then is in FINAL state so also could not be
1105 * selected here.
1106 */
1107
1108 sai_object_id_t tvid = temporaryObj->getVid();
1109 sai_object_id_t cvid = currentBestMatch->getVid();
1110 sai_object_id_t rid = currentView.m_vidToRid.at(cvid);
1111
1112 SWSS_LOG_INFO("remapped current VID %s to temp VID %s using RID %s",
1113 sai_serialize_object_id(cvid).c_str(),
1114 sai_serialize_object_id(tvid).c_str(),
1115 sai_serialize_object_id(rid).c_str());
1116
1117 temporaryView.m_ridToVid[rid] = tvid;
1118 temporaryView.m_vidToRid[tvid] = rid;
1119
1120 /*
1121 * TODO: Set new VID if it doesn't exist in current view with NULL RID
1122 * that will mean we created new object, this VID will be later
1123 * used to count references and as a sanity check if we are
1124 * increasing valid reference.
1125 *
1126 * Revisit. We have create flag now.
1127 */
1128 }
1129 else
1130 {
1131 /*
1132 * For non object id we will create map by object type and map
1133 * current to temporary and vice versa. For some objects like TRAP
1134 * or VLAN values will be the same but for ROUTES or NEIGHBORS can
1135 * be different since they contain VID which in temporary view can
1136 * be different and because of that we will need a map.
1137 */
1138
1139 sai_object_type_t objectType = temporaryObj->getObjectType();
1140
1141 SWSS_LOG_INFO("remapped %s current %s to temp %s",
1142 sai_serialize_object_type(temporaryObj->getObjectType()).c_str(),
1143 currentBestMatch->m_str_object_id.c_str(),
1144 temporaryObj->m_str_object_id.c_str());
1145
1146 temporaryView.m_nonObjectIdMap[objectType][temporaryObj->m_str_object_id] = currentBestMatch->m_str_object_id;
1147 currentView.m_nonObjectIdMap[objectType][currentBestMatch->m_str_object_id] = temporaryObj->m_str_object_id;
1148 }
1149 }
1150 else
1151 {
1152 /*
1153 * This method should be used only for objects that have status MATCHED
1154 * or NON_PROCESSED, other combinations are not supported since other
1155 * actions will be either create new object from temporary object or
1156 * remove existing current object.
1157 */
1158
1159 SWSS_LOG_THROW("unexpected status combination: current %d, temporary %d",
1160 currentBestMatch->getObjectStatus(),
1161 temporaryObj->getObjectStatus());
1162 }
1163}
1164
1165bool ComparisonLogic::performObjectSetTransition(
1166 _In_ AsicView &currentView,
1167 _In_ AsicView &temporaryView,
1168 _In_ const std::shared_ptr<SaiObj> currentBestMatch,
1169 _In_ std::shared_ptr<SaiObj> temporaryObj,
1170 _In_ bool performTransition)
1171{
1172 SWSS_LOG_ENTER();
1173
1174 /*
1175 * All parents (if any) are in final state here, we could now search for
1176 * best match in current view that some of the objects in temp final state
1177 * could been created so they should exist in current view but without
1178 * actual RID since all creation and RID mapping is done after all
1179 * matching, so it may cause problems for finding RID for compare.
1180 */
1181
1182 /*
1183 * When we have best match we need to determine whether current object can
1184 * be updated to "this temporary" object or whether current needs to be
1185 * destroyed and recreated according to temporary.
1186 */
1187
1188 std::set<sai_attr_id_t> processedAttributes;
1189
1190 /*
1191 * Matched objects can have different attributes so we need to mark in
1192 * processed attributes which one were processed so if current object has
1193 * more attributes then we need to bring them back to default values if
1194 * possible.
1195 */
1196
1197 /*
1198 * Depending on performTransition flag this method used in first pass will
1199 * determine whether current object can be updated to temporary one. If
1200 * first pass was successful second pass when flag is set to true, actual
1201 * SET operations will be generated and current view will be modified. No
1202 * actual ASIC operations will be performed, all ASIC changes will be done
1203 * after all object will be moved to final state.
1204 */
1205
1206 /*
1207 * XXX If objects are matched (same vid/rid on object id) then this
1208 * function must return true, just skip all create only attributes.
1209 */
1210
1211 for (auto &at: temporaryObj->getAllAttributes())
1212 {
1213 auto &temporaryAttr = at.second;
1214
1215 SWSS_LOG_INFO("first pass (temp): attr %s", temporaryAttr->getStrAttrId().c_str());
1216
1217 const auto meta = temporaryAttr->getAttrMetadata();
1218
1219 const sai_attribute_t &attr = *temporaryAttr->getSaiAttr();
1220
1221 processedAttributes.insert(attr.id); // mark attr id as processed
1222
1223 if (currentBestMatch->hasAttr(attr.id))
1224 {
1225 /*
1226 * Same attribute exists on current and temp view, check if it's
1227 * the same. Previously we used hasEqualAttribute method to find
1228 * best match but now we are looking for different attribute
1229 * values.
1230 */
1231
1232 auto currentAttr = currentBestMatch->getSaiAttr(attr.id);
1233
1234 SWSS_LOG_INFO("compare attr value curr %s vs temp %s",
1235 currentBestMatch->getSaiAttr(attr.id)->getStrAttrValue().c_str(),
1236 temporaryObj->getSaiAttr(attr.id)->getStrAttrValue().c_str());
1237
1238 if (BestCandidateFinder::hasEqualAttribute(currentView, temporaryView, currentBestMatch, temporaryObj, attr.id))
1239 {
1240 /*
1241 * Attributes are equal so go for next attribute
1242 */
1243
1244 continue;
1245 }
1246
1247 /*
1248 * Now we know that attribute values are different.
1249 */
1250
1251 /*
1252 * Here we don't need to check if attribute is mandatory on create
1253 * or conditional since attribute is present on both objects. If
1254 * attribute is CREATE_AND_SET that means we can update attribute
1255 * value on current best match object.
1256 */
1257
1258 if (SAI_HAS_FLAG_CREATE_AND_SET(meta->flags))
1259 {
1260 SWSS_LOG_DEBUG("Attr %s can be updated from %s to %s",
1261 meta->attridname,
1262 currentAttr->getStrAttrValue().c_str(),
1263 temporaryAttr->getStrAttrValue().c_str());
1264
1265 /*
1266 * Generate action and update current view in second pass
1267 * and continue for next attribute.
1268 */
1269
1270 if (performTransition)
1271 {
1272 setAttributeOnCurrentObject(currentView, temporaryView, currentBestMatch, temporaryAttr);
1273 }
1274
1275 continue;
1276 }
1277
1278 /*
1279 * In this place we know attribute is CREATE_ONLY and it's value is
1280 * different on both objects. Object must be destroyed and new
1281 * object must be created. In this case does not matter whether
1282 * attribute is mandatory on create or conditional since attribute
1283 * is present on both objects.
1284 */
1285
1286 if (currentBestMatch->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED)
1287 {
1288 /*
1289 * This should not happen, since this mean, that attribute is
1290 * create only, object is matched, and attribute value is
1291 * different! DB is broken?
1292 */
1293
1294 SWSS_LOG_THROW("Attr %s CAN'T be updated from %s to %s on VID %s when MATCHED and CREATE_ONLY, FATAL",
1295 meta->attridname,
1296 currentAttr->getStrAttrValue().c_str(),
1297 temporaryAttr->getStrAttrValue().c_str(),
1298 temporaryObj->m_str_object_id.c_str());
1299 }
1300
1301 SWSS_LOG_WARN("Attr %s CAN'T be updated from %s to %s since it's CREATE_ONLY",
1302 meta->attridname,
1303 currentAttr->getStrAttrValue().c_str(),
1304 temporaryAttr->getStrAttrValue().c_str());
1305
1306 /*
1307 * We return false since object can't be updated. Object creation
1308 * is in different place when current best match is not found.
1309 */
1310
1311 return false;
1312 }
1313
1314 /*
1315 * In this case attribute exists only on temporary object. Because of
1316 * different flags, and conditions this maybe not easy task to
1317 * determine what should happen.
1318 *
1319 * Depends on attribute order processing, we may process mandatory on
1320 * create conditional attribute first, before finding out that
1321 * condition attribute is different, but if condition would be the same
1322 * then this conditional attribute would be also present on current
1323 * best match.
1324 *
1325 * There are also default values here that come into play. We can
1326 * expand this logic in the future.
1327 */
1328
1329 bool conditional = meta->isconditional;
1330
1331 /*
1332 * If attribute is CREATE_AND_SET and not conditional then it's
1333 * safe to make SET operation.
1334 *
1335 * XXX previously we had (meta->flags == SAI_ATTR_FLAGS_CREATE_AND_SET)
1336 * If it's not conditional current SAI_HAS_FLAG should not matter. But it
1337 * can also be mandatory on create but this does not matter since if
1338 * it's mandatory on create then current object already exists co we can
1339 * still perform update on this attribute because it was passed during
1340 * creation.
1341 */
1342
1343 if (SAI_HAS_FLAG_CREATE_AND_SET(meta->flags) && !conditional)
1344 {
1345 SWSS_LOG_INFO("Missing current attr %s can be set to %s",
1346 meta->attridname,
1347 temporaryAttr->getStrAttrValue().c_str());
1348
1349 /*
1350 * There is another case here, if this attribute exists only in
1351 * temporary view and it has default value, and SET value is the
1352 * same as default, then there is no need for ASIC operation.
1353 *
1354 * NOTE: This can lead to not put attributes with default VALUE to
1355 * redis database and could be confusing when debugging.
1356 */
1357
1358 const auto defaultValueAttr = BestCandidateFinder::getSaiAttrFromDefaultValue(currentView, m_switch, *meta);
1359
1360 if (defaultValueAttr != nullptr)
1361 {
1362 std::string defStr = sai_serialize_attr_value(*meta, *defaultValueAttr->getSaiAttr());
1363
1364 if (defStr == temporaryAttr->getStrAttrValue())
1365 {
1366 SWSS_LOG_NOTICE("explicit %s:%s is the same as default, no need for ASIC SET action",
1367 meta->attridname, defStr.c_str());
1368
1369 continue;
1370 }
1371 }
1372
1373 /*
1374 * Generate action and update current view in second pass
1375 * and continue for next attribute.
1376 */
1377
1378 if (performTransition)
1379 {
1380 setAttributeOnCurrentObject(currentView, temporaryView, currentBestMatch, temporaryAttr);
1381 }
1382
1383 continue;
1384 }
1385
1386 if (currentBestMatch->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED)
1387 {
1388 if (SAI_HAS_FLAG_CREATE_ONLY(meta->flags))
1389 {
1390 /*
1391 * Attribute is create only attribute on matched object. This
1392 * can happen when we are have create only attributes in asic
1393 * view, those attributes were put by snoop logic. Since we
1394 * skipping only read-only attributes then we snoop create-only
1395 * also, but on "existing" objects this will cause problem and
1396 * during apply logic we need to skip this attribute since we
1397 * won't be able to SET it anyway on matched object, and value
1398 * is the same as current object.
1399 */
1400
1401 SWSS_LOG_INFO("Skipping create only attr on matched object: %s:%s",
1402 meta->attridname,
1403 temporaryAttr->getStrAttrValue().c_str());
1404
1405 continue;
1406 }
1407 }
1408
1409 /*
1410 * This is the most interesting case, we currently leave it here and we
1411 * will support it later. Some other cases here also can be considered
1412 * as success, for example if default value is the same as current
1413 * value.
1414 */
1415
1416 SWSS_LOG_WARN("Missing current attr %s (conditional: %d) CAN'T be set to %s, flags: 0x%x, FIXME",
1417 meta->attridname,
1418 conditional,
1419 temporaryAttr->getStrAttrValue().c_str(),
1420 meta->flags);
1421
1422 /*
1423 * We can't continue with update in that case, so return false.
1424 */
1425
1426 return false;
1427 }
1428
1429 const bool beginTempSizeZero = temporaryObj->getAllAttributes().size() == 0;
1430
1431 /*
1432 * Current best match can have more attributes than temporary object.
1433 * let see if we can bring them to default value if possible.
1434 */
1435
1436 for (auto &ac: currentBestMatch->getAllAttributes())
1437 {
1438 auto &currentAttr = ac.second;
1439
1440 const auto &meta = currentAttr->getAttrMetadata();
1441
1442 const sai_attribute_t &attr = *currentAttr->getSaiAttr();
1443
1444 if (processedAttributes.find(attr.id) != processedAttributes.end())
1445 {
1446 /*
1447 * This attribute was processed in previous temporary attributes processing so skip it here.
1448 */
1449
1450 continue;
1451 }
1452
1453 SWSS_LOG_INFO("first pass (curr): attr %s", currentAttr->getStrAttrId().c_str());
1454
1455 /*
1456 * Check if we are bringing one of the default created objects to
1457 * default value, since for that we will need dependency TREE. Most of
1458 * the time user should just query configuration and just set new
1459 * values based on get results, so this will not be needed.
1460 *
1461 * And even if we will have dependency tree, those values may not be
1462 * synced because of remove etc, so we will need to check if default
1463 * values actually exists.
1464 */
1465
1466 if (currentBestMatch->isOidObject())
1467 {
1468 sai_object_id_t vid = currentBestMatch->getVid();
1469
1470 /*
1471 * Current best match may be created, check if vid/rid exist.
1472 */
1473
1474 if (currentView.hasVid(vid))
1475 {
1476 sai_object_id_t rid = currentView.m_vidToRid.at(vid);
1477
1478 if (m_switch->isDiscoveredRid(rid))
1479 {
1480 SWSS_LOG_INFO("performing default on existing object VID %s: %s: %s, we need default dependency TREE, FIXME",
1481 sai_serialize_object_id(vid).c_str(),
1482 meta->attridname,
1483 currentAttr->getStrAttrValue().c_str());
1484 }
1485 }
1486 }
1487
1488 /*
1489 * We should not have MANDATORY_ON_CREATE attributes here since all
1490 * mandatory on create (even conditional) should be present in in
1491 * previous loop and they are matching, so we should get here
1492 * CREATE_ONLY or CREATE_AND_SET attributes only. So we should not get
1493 * conditional attributes here also, but lets take extra care about that
1494 * just as sanity check.
1495 */
1496
1497 bool conditional = meta->isconditional;
1498
1499 if (conditional || SAI_HAS_FLAG_MANDATORY_ON_CREATE(meta->flags))
1500 {
1501 if (currentBestMatch->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED &&
1502 SAI_HAS_FLAG_CREATE_AND_SET(meta->flags))
1503 {
1504 if (meta->objecttype == SAI_OBJECT_TYPE_PORT &&
1505 meta->attrid == SAI_PORT_ATTR_SPEED)
1506 {
1507 /*
1508 * NOTE: for SPEED we could query each port at start and
1509 * save it's speed to recover here, or even we could query
1510 * each attribute on existing object during discovery
1511 * process.
1512 */
1513
1514 SWSS_LOG_WARN("No previous value specified on %s (VID), can't bring to default, leaving attr unchanged: %s:%s",
1515 sai_serialize_object_id(currentBestMatch->getVid()).c_str(),
1516 meta->attridname,
1517 currentAttr->getStrAttrValue().c_str());
1518

1520 }

1523 meta->attrid == SAI_SCHEDULER_GROUP_ATTR_SCHEDULER_PROFILE_ID)

1527 * objects which maybe required to be destroyed, that's why
1528 * we need to bring real value. What if real value were
1529 * removed?
1530 */

1533

1542

1544 {

1546 * We found default value
1547 */
1548
1549 SWSS_LOG_DEBUG("found default rid %s, vid %s for %s",
1550 sai_serialize_object_id(rid).c_str(),
1551 sai_serialize_object_id(vid).c_str(),
1552 meta->attridname);
1553
1554 def = currentView.m_ridToVid.at(rid);
1555 }
1556 }
1557

1559
1560 defattr.id = meta->attrid;
1561 defattr.value.oid = def;
1562
1563 std::string str_attr_value = sai_serialize_attr_value(*meta, defattr, false);
1564
1565 auto defaultValueAttr = std::make_shared<SaiAttr>(meta->attridname, str_attr_value);
1566
1567 if (performTransition)
1568 {
1569 setAttributeOnCurrentObject(currentView, temporaryView, currentBestMatch, defaultValueAttr);
1570 }
1571
1572 continue;
1573 }
1574
1575 // current best match is MATCHED
1576
1577 //auto vid = currentBestMatch->getVid();
1578
1579 // TODO don't transfer oid attributes, we don't know how to handle this yet
1580 // If OA did GET on any attributes, snoop in syncd should catch that and write
1581 // to database so we would have some attributes here.
1582
1583 if (beginTempSizeZero && !meta->isoidattribute)
1584 {
1585 SWSS_LOG_WARN("current attr is MoC|CaS and object is MATCHED: %s transferring %s:%s to temp object (was empty)",
1586 currentBestMatch->m_str_object_id.c_str(),
1587 meta->attridname,
1588 currentAttr->getStrAttrValue().c_str());
1589
1590 std::shared_ptr<SaiAttr> transferedAttr = std::make_shared<SaiAttr>(
1591 currentAttr->getStrAttrId(),
1592 currentAttr->getStrAttrValue());
1593
1594 temporaryObj->setAttr(transferedAttr);
1595
1596 continue;
1597 }

1599 // SAI_QUEUE_ATTR_PARENT_SCHEDULER_NODE
1600 // SAI_SCHEDULER_GROUP_ATTR_SCHEDULER_PROFILE_ID*
1601 // SAI_SCHEDULER_GROUP_ATTR_PARENT_NODE
1602 // SAI_BRIDGE_PORT_ATTR_BRIDGE_ID
1603 //
1604 // TODO matched by ID (MATCHED state) should always be updatable
1605 // except those 4 above (at least for those above since they can have
1606 // default value present after switch creation
1607
1608 // TODO SAI_SCHEDULER_GROUP_ATTR_SCHEDULER_PROFILE_ID is mandatory on create but also SET
1609 // if attribute is set we and object is in MATCHED state then that means we are able to
1610 // bring this attribute to default state not for all attributes!
1611 // *SAI_SCHEDULER_GROUP_ATTR_SCHEDULER_PROFILE_ID - is not any more mandatory on create, so default should be NULL
1612
1613 SWSS_LOG_ERROR("current attribute is mandatory on create, crate and set, and object MATCHED, FIXME %s %s:%s",
1614 currentBestMatch->m_str_object_id.c_str(),
1615 meta->attridname,
1616 currentAttr->getStrAttrValue().c_str());
1617
1618 return false;
1619 }
1620
1621 if (currentBestMatch->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED)
1622 {
1623 if (SAI_HAS_FLAG_CREATE_ONLY(meta->flags))
1624 {

1626 * Attribute is create only attribute on matched object. This
1627 * can happen when we are have create only attributes in asic
1628 * view, those attributes were put by snoop logic. Since we
1629 * skipping only read-only attributes then we snoop create-only
1630 * also, but on "existing" objects this will cause problem and
1631 * during apply logic we need to skip this attribute since we
1632 * won't be able to SET it anyway on matched object, and value
1633 * is the same as current object.
1634 */

1637 meta->attridname,
1638 currentAttr->getStrAttrValue().c_str());

1640 // don't produce too much noise for queues

1642 {
1643 SWSS_LOG_WARN("current attr is CREATE_ONLY and object is MATCHED: %s transferring %s:%s to temp object",
1644 currentBestMatch->m_str_object_id.c_str(),
1645 meta->attridname,
1646 currentAttr->getStrAttrValue().c_str());
1647 }
1648
1649 std::shared_ptr<SaiAttr> transferedAttr = std::make_shared<SaiAttr>(
1650 currentAttr->getStrAttrId(),
1651 currentAttr->getStrAttrValue());
1652
1653 temporaryObj->setAttr(transferedAttr);

1655 continue;
1656 }

1660 meta->attridname,
1661 currentAttr->getStrAttrValue().c_str());

1664 * We don't expect conditional or mandatory on create attributes,
1665 * since in previous loop all attributes were matching they also
1666 * had to match. If we hit this case we need to take a closer look
1667 * since it will mean we have a bug somewhere.
1668 */
1669
1670 return false;
1671 }

1689
1690 const auto defaultValueAttr = BestCandidateFinder::getSaiAttrFromDefaultValue(currentView, m_switch, *meta);
1691
1692 if (defaultValueAttr == nullptr)
1693 {
1694 SWSS_LOG_WARN("Can't get default value for present current attr %s:%s, FIXME",
1695 meta->attridname,
1696 currentAttr->getStrAttrValue().c_str());
1697
1698 /*
1699 * If we can't get default value then we can't do set, because
1700 * we don't know with what, so we need to destroy current
1701 * object and recreate new one from temporary.
1702 */
1703
1704 return false;
1705 }
1706
1707 if (currentAttr->getStrAttrValue() == defaultValueAttr->getStrAttrValue())
1708 {
1709 SWSS_LOG_INFO("Present current attr %s value %s is the same as default value, no action needed",
1710 meta->attridname,
1711 currentAttr->getStrAttrValue().c_str());
1712
1713 continue;
1714 }
1715
1716 if (meta->flags == SAI_ATTR_FLAGS_CREATE_ONLY)
1717 {
1718 SWSS_LOG_WARN("Present current attr %s:%s has default that CAN'T be set to %s since it's CREATE_ONLY",

1720 currentAttr->getStrAttrValue().c_str(),

1722

1727 meta->attridname,
1728 currentAttr->getStrAttrValue().c_str(),
1729 defaultValueAttr->getStrAttrValue().c_str());
1730
1731 /*
1732 * Generate action and update current view in second pass
1733 * and continue for next attribute.
1734 */
1735

1737 {

1739 }
1740
1741 continue;
1742 }
1743
1744 SWSS_LOG_THROW("we should not get here, we have a bug, current present attribute %s:%s has some wrong flags 0x%x",
1745 meta->attridname,
1746 currentAttr->getStrAttrValue().c_str(),
1747 meta->flags);
1748 }
1749
1750 /*
1751 * All attributes were processed, and ether no changes are required or all
1752 * changes can be performed or some missing attributes has exact value as
1753 * default value.
1754 */
1755
1756 return true;

1790void ComparisonLogic::processObjectForViewTransition(
1791 _In_ AsicView &currentView,
1792 _In_ AsicView &temporaryView,
1793 _Inout_ std::shared_ptr<SaiObj> temporaryObj)
1794{
1795 SWSS_LOG_ENTER();
1796
1797 if (temporaryObj->getObjectStatus() == SAI_OBJECT_STATUS_FINAL)
1798 {
1799 /*
1800 * If object is in final state, no need to process it again, nothing
1801 * about this object will change.
1802 */
1803
1804 return;
1805 }
1806
1807 SWSS_LOG_INFO("processing: %s:%s", temporaryObj->m_str_object_type.c_str(), temporaryObj->m_str_object_id.c_str());
1808
1809 procesObjectAttributesForViewTransition(currentView, temporaryView, temporaryObj);
1810
1811 /*
1812 * Now since all object ids (VIDs) were processed for temporary object we
1813 * can try to find current best match.
1814 */
1815
1816 auto bcf = std::make_shared<BestCandidateFinder>(currentView, temporaryView, m_switch);
1817
1818 std::shared_ptr<SaiObj> currentBestMatch = bcf->findCurrentBestMatch(temporaryObj);
1819
1820 /*
1821 * So there will be interesting problem, when we don't find best matching
1822 * object, but actual object will exist, and it will have the same KEY
1823 * attribute like, and some other attribute will be create_only and it will
1824 * be different then we can't just create temporary object because of key
1825 * will match and we have conflict, so we need first in this case remove
1826 * previous object with key
1827 *
1828 * so there are 2 options here:
1829 *
1830 * - either our finding best match function will always include matching
1831 * keys (if more keys, they all should (or any?) match should be right
1832 * away taken) and this should be only object returned on list possible
1833 * candidates as best match then below logic should figure out that we
1834 * cant do "set" and we need to destroy object and destroy it now, and
1835 * create temporary later this means that finding best logic can return
1836 * objects that are not upgradable "cant do set"
1837 *
1838 * - or second solution, in this case, finding best match will not include
1839 * this object on list, since other create only attribute will be
1840 * different which also means we need to destroy and recreate, but in
1841 * this case finding best match will figure this out and will destroy
1842 * object on the way since it will know that set transition is not
1843 * possible so here when we find best match (other object) in that case
1844 * we always will be able to do "SET", or when no object will be returned
1845 * then we can just right away create this object since all keys objects
1846 * were removed by finding best match
1847 *
1848 * In both cases logic is the same, someone needs to figure out whether
1849 * updating object and set is possible or whether we leave object alone, or
1850 * we delete it before creating new one object.
1851 *
1852 * Preferably this logic could be in both but that duplicates compare
1853 * actions i would rather want this in finding best match, but that would
1854 * modify current view which i don't like much and here it seems like to
1855 * much hassle since after finding best match here we would have simple
1856 * operations.
1857 *
1858 * Currently KEY are in: vlan, queue, port, hostif_trap
1859 * - port - hw lane, we don't need to worry since we match them all
1860 * - queue - all other attributes can be set
1861 * - vlan - all other attributes can be set
1862 * - hostif_trap - all other attributes can be set
1863 *
1864 * We could add a check for that in sanity.
1865 */
1866
1867 if (currentBestMatch == nullptr)
1868 {
1869 /*
1870 * Since we didn't found current best match, just create this object as
1871 * a new object, and put it also in current view in final state.
1872 */
1873
1874 SWSS_LOG_INFO("failed to find best match %s %s in current view, will create new object",
1875 temporaryObj->m_str_object_type.c_str(),
1876 temporaryObj->m_str_object_id.c_str());
1877
1878 /*
1879 * We can hit this scenario when, we will not have current best match,
1880 * and we still want to create new object, but resources are limited,
1881 * so it can happen that in this place we also would need to remove
1882 * some objects first, before we create new one.
1883 */
1884 breakBeforeMake(currentView, temporaryView, currentBestMatch, temporaryObj);
1885
1886 createNewObjectFromTemporaryObject(currentView, temporaryView, temporaryObj);
1887 return;
1888 }
1889
1890 SWSS_LOG_INFO("found best match %s: current: %s temporary: %s",
1891 currentBestMatch->m_str_object_type.c_str(),
1892 currentBestMatch->m_str_object_id.c_str(),
1893 temporaryObj->m_str_object_id.c_str());
1894
1895 /*
1896 * We need to two passes, for not matched parameters since if first
1897 * attribute will modify current object like SET operation, but second
1898 * attribute will not be able to do update because of CREATE_ONLY etc, then
1899 * we will end up with half modified current object and VIEW. So we have
1900 * two choices here when update is not possible:
1901 *
1902 * - remove current object and it's childs and create temporary one, or
1903 * - leave current object as unprocessed (dry run) for maybe future best
1904 * match, and just create temporary object
1905 *
1906 * We will choose approach witch leaving object untouched if something will
1907 * go wrong, we can always switch back to destroy current best match if
1908 * that would be better approach. It could be actually param of comparison
1909 * logic.
1910 *
1911 * NOTE: this function is called twice if first time will be successful
1912 * then logs will be doubled in syslog.
1913 */
1914
1915 bool passed = performObjectSetTransition(currentView, temporaryView, currentBestMatch, temporaryObj, false);
1916
1917 if (!passed)
1918 {
1919 if (temporaryObj->getObjectStatus() == SAI_OBJECT_STATUS_MATCHED)
1920 {
1921 /*
1922 * If object status is MATCHED, then we have the same object in
1923 * current view and temporary view, so it must be possible to make
1924 * transition to temporary view, hence for MATCHED objects
1925 * performObjectSetTransition must pass. There can be some corner
1926 * cases like existing objects after switch create and their attributes
1927 * that are OID set to existing objects (like QUEUES).
1928 */
1929
1930 SWSS_LOG_THROW("performObjectSetTransition on MATCHED object (%s) FAILED! bug?",
1931 temporaryObj->m_str_object_id.c_str());
1932 }
1933
1934 /*
1935 * First pass was a failure, so we can't update existing object with
1936 * temporary one, probably because of CRATE_ONLY attributes.
1937 *
1938 * Our strategy here will be to leave untouched current object (second
1939 * strategy will be to remove it and it's children if link can't be
1940 * broken, Objects with non id like ROUTE FDB NEIGHBOR they need to be
1941 * removed first anyway, since we got best match, and for those best
1942 * match is via struct elements so we can't create next route with the
1943 * same elements in the struct.
1944 *
1945 * Then we create new object from temporary one and set it state to
1946 * final.
1947 */
1948
1949 if (temporaryObj->getObjectType() == SAI_OBJECT_TYPE_SWITCH)
1950 {
1951 SWSS_LOG_THROW("we should not get switch object here, bug?");
1952 }
1953
1954 if (!temporaryObj->isOidObject())
1955 {
1956 removeExistingObjectFromCurrentView(currentView, temporaryView, currentBestMatch);
1957 }
1958 else
1959 {
1960 /*
1961 * Later on if we decide we want to remove objects before creating
1962 * new one's we need to put here this action, or just remove this
1963 * entire switch and call remove.
1964 */
1965
1966 breakBeforeMake(currentView, temporaryView, currentBestMatch, temporaryObj);
1967 }
1968
1969 // No need to store VID since at this point we don't have RID yet, it will be
1970 // created on execute asic and RID will be saved to maps in both views.
1971
1972 SWSS_LOG_INFO("found best match, but set failed: %s: current: %s temporary: %s",
1973 currentBestMatch->m_str_object_type.c_str(),
1974 currentBestMatch->m_str_object_id.c_str(),
1975 temporaryObj->m_str_object_id.c_str());
1976
1977 createNewObjectFromTemporaryObject(currentView, temporaryView, temporaryObj);
1978
1979 return;
1980 }
1981
1982 /*
1983 * First pass was successful, so we can do update on current object, lets do that now!
1984 */
1985
1986 if (temporaryObj->isOidObject() && (temporaryObj->getObjectStatus() != SAI_OBJECT_STATUS_MATCHED))
1987 {
1988 /*
1989 * We track all oid object references, and since this object was
1990 * matched we need to insert it's vid to reference map since when we
1991 * will update attributes in current view from temporary view
1992 * attributes, then this object must be in reference map. This is
1993 * sanity check. Object can't be in matched state since matched objects
1994 * have the same RID and VID in both views, so VID already exists in
1995 * reference map.
1996 *
1997 * This object VID will be remapped inside updateObjectStatus to
1998 * rid/vid map since if we are here then we matched one of current
1999 * objects to this temporary object and update can be performed.
2000 */
2001
2002 /*
2003 * TODO: This is temporary VID, it's not available in current view so
2004 * we need to remove this code here, since we will translate this VID
2005 * in temporary view to current view.
2006 */
2007
2008 sai_object_id_t vid = temporaryObj->getVid();
2009
2010 currentView.insertNewVidReference(vid);
2011
2012 // save temporary vid to rid after match
2013 // can't be here since vim to rid map will not match
2014 // currentView.m_vidToRid[vid] = currentView.m_vidToRid.at(currentBestMatch->getVid());
2015 }
2016
2017 performObjectSetTransition(currentView, temporaryView, currentBestMatch, temporaryObj, true);
2018
2019 /*
2020 * Since we got here, that means matching or updating current best match
2021 * object was successful, and we can now set their state to final.
2022 */
2023
2024 updateObjectStatus(currentView, temporaryView, currentBestMatch, temporaryObj);
2025}

2034 if (!currentObj->isOidObject())
2035 {

2037
2038 removeExistingObjectFromCurrentView(currentView, temporaryView, currentObj);
2039
2040 return;

2043 // check reference count and remove other objects if necessary
2044
2045 const int count = currentView.getVidReferenceCount(currentObj->getVid());
2046

2052

2054 // need to lookout on loops on some objects

2073
2074 for (auto& obj: objs)
2075 {

2077 // they can have the same value, but they can have different
2078 // attributes (like set/create_only)

2080 auto status = obj->getObjectStatus();
2081

2083 status != SAI_OBJECT_STATUS_MATCHED)
2084 {
2085 continue;
2086 }

2088 auto attr = obj->tryGetSaiAttr(revgraph->attrmetadata->attrid);
2089
2090 if (attr == nullptr)
2091 {
2092 // no such attribute
2093 continue;
2094 }

2097 revgraph->attrmetadata->attrvaluetype != SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_ID)

2100 SWSS_LOG_THROW("attr value type %d, not supported yet, FIXME",
2101 revgraph->attrmetadata->attrvaluetype);
2102 }
2103
2104 if (attr->getOid() != currentObj->getVid())
2105 {
2106 // VID is not matching, skip this attribute
2107 continue;
2108 }
2109
2110 // we found object and attribute that is using current VID,
2111 // it's possible this VID is used on multiple attributes on the
2112 // same object
2113
2114 SWSS_LOG_INFO("found reference object: %s", obj->m_str_object_id.c_str());
2115
2116 if (revgraph->attrmetadata->iscreateonly && status == SAI_OBJECT_STATUS_NOT_PROCESSED)
2117 {
2118 // attribute is create only, and object was not processed
2119 // yet this means that we need to remove object, since we
2120 // can't break the link

2123 }
2124 else if (revgraph->attrmetadata->iscreateandset && status == SAI_OBJECT_STATUS_NOT_PROCESSED)

2128 // we can also remove entire object here too
2129

2131 }
2132 else
2133 {
2134 // attribute is create and set, but not allow null object id
2135 // so probably attribute is mandatory on create, we can't break
2136 // the link, but we can remove entire object
2137
2138 removeCurrentObjectDependencyTree(currentView, temporaryView, obj); // recursion
2139 }
2140 }
2141 else if (revgraph->attrmetadata->iscreateandset && status == SAI_OBJECT_STATUS_MATCHED)
2142 {
2143 SWSS_LOG_THROW("matched break the link is not implemented yet, FIXME");
2144 }
2145 else
2146 {
2147 SWSS_LOG_THROW("remove on %s, obj status: %d, not supported, FIXME", revgraph->attrmetadata->attridname, status);
2148 }
2149 }
2150 }
2151 }

2153 removeExistingObjectFromCurrentView(currentView, temporaryView, currentObj);

2156void ComparisonLogic::breakBeforeMake(
2157 _In_ AsicView &currentView,
2158 _In_ AsicView &temporaryView,
2159 _In_ const std::shared_ptr<SaiObj>& currentBestMatch, // can be nullptr
2160 _In_ const std::shared_ptr<SaiObj>& temporaryObj)
2161{
2162 SWSS_LOG_ENTER();
2163
2164 /*
2165 * Break Before Make rule.
2166 *
2167 * We can have 2 paths here:
2168 *
2169 * - current best match object was not found (nullptr), in case of limited
2170 * resources, this mean that we want to remove some existing object
2171 * before creating new, so number of existing objects will not exceed
2172 * initial value, for example we remove acl table, then we create acl
2173 * table. Of course this will not hold true, if temporary number of
2174 * objects of given type is greater than current existing objects.
2175 *
2176 * - current best match was found, but perform Object Set Transition
2177 * operation failed, so we can't move current object attributes to
2178 * temporary object attributes, because for example some are create only.
2179 */
2180
2181 if (!m_breakConfig->shouldBreakBeforeMake(temporaryObj->getObjectType()))
2182 {
2183 // skip object if not in break config
2184
2185 return;
2186 }
2187
2188 if (currentBestMatch == nullptr)
2189 {
2190 // it can happen that current best match is not found for temporary
2191 // object then we need to find best object for removal from existing
2192 // ones the most suitable should be the one with most same attributes,
2193 // since maybe only one read only attribute has been changed, and this
2194 // will automatically result in null best match
2195
2196 auto bcf = std::make_shared<BestCandidateFinder>(currentView, temporaryView, m_switch);
2197
2198 std::shared_ptr<SaiObj> similarBestMatch = bcf->findSimilarBestMatch(temporaryObj);
2199
2200 if (similarBestMatch == nullptr)
2201 {
2202 SWSS_LOG_WARN("similar best match is null, not removing");
2203 return;

2311

2313
2314 auto rids = m_switch->getDiscoveredRids();
2315
2316 /*
2317 * We should transfer existing objects from current view to temporary view.
2318 * But not all objects, since user could remove some default removable
2319 * objects like vlan member, bridge port etc. We collected those removed
2320 * objects to initViewRemovedVidSet and we will use this as a reference to
2321 * transfer objects to temporary view.
2322 *
2323 * TODO: still sairedis metadata database have no idea about existing
2324 * objects so user can create object which already exists like vlan 1 if he
2325 * didn't queried it yet. We need to transfer all dependency tree to
2326 * sairedis after switch create.
2327 */
2328

2377 {
2378 /*
2379 * If this vid was removed during init view mode, it still exits in
2380 * current view, but don't put it to existing objects, then
2381 * comparison logic will take care of that and, it will remove it
2382 * from current view as well.
2383 */

2386 sai_serialize_object_type(m_vendorSai->objectTypeQuery(rid)).c_str(),

2394 * In case of warm boot, it may happen that user set some created
2395 * objects on default existing objects, like for example buffer profile
2396 * on ingress priority group. In this case buffer profile should not
2397 * be considered as matched object and copied to temporary view, since

2400 * possible new buffer profile created in temporary view. This may
2401 * happen if OA will not care what was set previously on ingress
2402 * priority group and just create new buffer profile and assign it. In
2403 * this case we don't want any asic operations to happen. Also if we
2404 * would pass this buffer profile as existing to temporary view, it

2408 *

2410 * new user created buffer profile. That's why we need default existing

2412 * created objects.
2413 *
2414 * For default existing objects, we don't need to copy attributes, since

2416 *

2418 * if this object contains any OID attributes, discovery logic queried

2442 case SAI_OBJECT_TYPE_INGRESS_PRIORITY_GROUP:
2443 case SAI_OBJECT_TYPE_SCHEDULER_GROUP:

2447 // warm boot, it may happen that current logic will be
2448 // confused which of those objects are from previous warm
2449 // boot or second one, need better way to mark changes to
2450 // those objects in redis DB between warm boots
2451
2452 performColdCheck = false;
2453

2456 break;
2457 }
2458 }
2459

2461 {
2462 // We have some new discovered VIDs after warm boot, we need to
2463 // create temporary objects from them, so comparison logic will not
2464 // get confused and will not remove them.
2465 //
2466 // TODO: there could be potential issue here, when user will remove
2467 // one of the new discovered object in init view phase, then we
2468 // can't put that object to DB, and it should be removed.
2469
2470 performColdCheck = false;
2471
2472 sai_object_type_t ot = VidManager::objectTypeQuery(vid);
2473
2474 SWSS_LOG_NOTICE("creating temporary object for new discovered VID %s:%s",
2475 sai_serialize_object_type(ot).c_str(),
2476 sai_serialize_object_id(vid).c_str());
2477 }
2478
2479 if (performColdCheck && coldBootDiscoveredVids.find(vid) == coldBootDiscoveredVids.end())
2480 {
2481 SWSS_LOG_INFO("object is not on default existing list: %s RID %s VID %s",
2482 sai_serialize_object_type(m_vendorSai->objectTypeQuery(rid)).c_str(),
2483 sai_serialize_object_id(rid).c_str(),
2484 sai_serialize_object_id(vid).c_str());
2485
2486 continue;
2487 }
2488
2489 temporaryView.createDummyExistingObject(rid, vid);
2490
2491 SWSS_LOG_INFO("populate existing %s RID %s VID %s",
2492 sai_serialize_object_type(m_vendorSai->objectTypeQuery(rid)).c_str(),
2493 sai_serialize_object_id(rid).c_str(),
2494 sai_serialize_object_id(vid).c_str());
2495
2496 /*
2497 * Move both objects to matched state since match oids was already
2498 * called, and here we created some new objects that should be matched.
2499 */
2500
2501 currentView.m_oOids.at(vid)->setObjectStatus(SAI_OBJECT_STATUS_MATCHED);
2502 temporaryView.m_oOids.at(vid)->setObjectStatus(SAI_OBJECT_STATUS_MATCHED);
2503 }

2697 return;

2700
2701 if (cObj->getObjectType() != tObj->getObjectType())
2702 return;

2809void ComparisonLogic::createPreMatchMap(
2810 _In_ const AsicView& cur,
2811 _Inout_ AsicView& tmp)
2812{
2813 SWSS_LOG_ENTER();
2814
2815 /*
2816 * When comparison logic is running on object A it runs recursively on each
2817 * object that was fount in object A attributes, because we need to make
2818 * sure all objects are matched before actually processing object A. For
2819 * example before processing ROUTE_ENTRY, we process first NEXT_HOP and
2820 * before that ROUTER_INTERFACE and before that PORT. Object processing
2821 * could be described going from down to top. But figuring out for top
2822 * object ex. WRED could be hard since we would need to check not matched
2823 * yet buffer profile and buffer pool before we use QUEUE or IPG as an
2824 * anchor object. We can actually leverage that and when processing graph
2825 * from top to bottom, we can create helper map which will contain
2826 * predictions which object will be suitable for current processing
2827 * objects. We can have N candidates objects and instead of choosing 1 at
2828 * random, to reduce number of ASIC operations we will use a pre match map
2829 * which is created in this method.
2830 *
2831 * This map should be almost exact match in warn boot case, but it will be
2832 * treated only as hint, not as actual mapping.
2833 *
2834 * We will create map for all matched objects and for route_entry,
2835 * fdb_entry and neighbor_entry.
2836 */
2837
2838 std::set<std::string> processed;
2839
2840 SWSS_LOG_TIMER("create preMatch map");
2841
2842 for (auto& ok: tmp.m_soAll)
2843 {
2844 auto& tObj = ok.second;
2845
2846 if (tObj->getObjectStatus() != SAI_OBJECT_STATUS_MATCHED)
2847 continue;
2848
2849 sai_object_id_t cObjVid = cur.m_ridToVid.at(tmp.m_vidToRid.at(tObj->getVid()));
2850
2851 auto& cObj = cur.m_oOids.at(cObjVid);
