20// P4GreTunnelEntry holds GreTunnelManager's internal cache of P4 GRE tunnel
21// entry. Example: P4RT_TABLE:FIXED_TUNNEL_TABLE:{"match/tunnel_id":"tunnel-1"}
22// "action" = "mark_for_tunnel_encap",
23// "param/router_interface_id" = "intf-eth-1/2/3",
24// "param/encap_src_ip" = "2607:f8b0:8096:3110::1",
25// "param/encap_dst_ip" = "2607:f8b0:8096:311a::2",
26// "controller_metadata" = "..."
27struct P4GreTunnelEntry
28{
29 // Key of this entry, built from tunnel_id.
30 std::string tunnel_key;
31
32 // Fields from P4 table.
33 // Match
34 std::string tunnel_id;
35 // Action
36 std::string router_interface_id;
37 swss::IpAddress encap_src_ip;
38 swss::IpAddress encap_dst_ip;
39 // neighbor_id is required to be equal to encap_dst_ip by BRCM. And the
40 // neighbor entry needs to be created before GRE tunnel object
41 swss::IpAddress neighbor_id;
42
43 // SAI OID associated with this entry.
44 sai_object_id_t tunnel_oid = SAI_NULL_OBJECT_ID;
45 // SAI OID of a loopback rif for SAI_TUNNEL_ATTR_OVERLAY_INTERFACE
46 sai_object_id_t overlay_if_oid = SAI_NULL_OBJECT_ID;
47 // SAI OID of the router_interface_id for SAI_TUNNEL_ATTR_UNDERLAY_INTERFACE
48 sai_object_id_t underlay_if_oid = SAI_NULL_OBJECT_ID;
49
50 P4GreTunnelEntry(const std::string &tunnel_id, const std::string &router_interface_id,
51 const swss::IpAddress &encap_src_ip, const swss::IpAddress &encap_dst_ip,
52 const swss::IpAddress &neighbor_id);
53};
