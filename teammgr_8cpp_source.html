169void TeamMgr::cleanTeamProcesses()
170{
171 SWSS_LOG_ENTER();
172 SWSS_LOG_NOTICE("Cleaning up LAGs during shutdown...");
173
174 std::unordered_map<std::string, pid_t> aliasPidMap;
175
176 for (const auto& alias: m_lagList)
177 {
178 std::string res;
179 pid_t pid;
180
181 try
182 {
183 std::stringstream cmd;
184 cmd << "cat " << shellquote("/var/run/teamd/" + alias + ".pid");
185 EXEC_WITH_ERROR_THROW(cmd.str(), res);
186 }
187 catch (const std::exception &e)
188 {
189 // Handle Warm/Fast reboot scenario
190 SWSS_LOG_NOTICE("Skipping non-existent port channel %s pid...", alias.c_str());
191 continue;
192 }
193
194 try
195 {
196 pid = static_cast<pid_t>(std::stoul(res, nullptr, 10));
197 aliasPidMap[alias] = pid;
198
199 SWSS_LOG_INFO("Read port channel %s pid %d", alias.c_str(), pid);
200 }
201 catch (const std::exception &e)
202 {
203 SWSS_LOG_ERROR("Failed to read port channel %s pid: %s", alias.c_str(), e.what());
204 continue;
205 }
206
207 try
208 {
209 std::stringstream cmd;
210 cmd << "kill -TERM " << pid;
211 EXEC_WITH_ERROR_THROW(cmd.str(), res);
212
213 SWSS_LOG_NOTICE("Sent SIGTERM to port channel %s pid %d", alias.c_str(), pid);
214 }
215 catch (const std::exception &e)
216 {
217 SWSS_LOG_ERROR("Failed to send SIGTERM to port channel %s pid %d: %s", alias.c_str(), pid, e.what());
218 aliasPidMap.erase(alias);
219 }
220 }
221
222 for (const auto& cit: aliasPidMap)
223 {
224 const auto &alias = cit.first;
225 const auto &pid = cit.second;
226
227 std::stringstream cmd;
228 std::string res;
229
230 SWSS_LOG_NOTICE("Waiting for port channel %s pid %d to stop...", alias.c_str(), pid);
231
232 cmd << "tail -f --pid=" << pid << " /dev/null";
233 EXEC_WITH_ERROR_THROW(cmd.str(), res);
234 }
235
236 SWSS_LOG_NOTICE("LAGs cleanup is done");
237}

239void TeamMgr::doLagTask(Consumer &consumer)
240{
241 SWSS_LOG_ENTER();
242
243 auto it = consumer.m_toSync.begin();
244 while (it != consumer.m_toSync.end())
245 {
246 KeyOpFieldsValuesTuple t = it->second;
247
248 string alias = kfvKey(t);
249 string op = kfvOp(t);
250
251 if (op == SET_COMMAND)
252 {
253 int min_links = 0;
254 bool fallback = false;
255 bool fast_rate = false;
256 string admin_status = DEFAULT_ADMIN_STATUS_STR;
257 string mtu = DEFAULT_MTU_STR;
258 string learn_mode;
259 string tpid;
260
261 for (auto i : kfvFieldsValues(t))
262 {
263 // min_links and fallback attributes cannot be changed
264 // after the LAG is created.
265 if (fvField(i) == "min_links")
266 {
267 min_links = stoi(fvValue(i));
268 SWSS_LOG_INFO("Get min_links value %d", min_links);
269 }
270 else if (fvField(i) == "fallback")
271 {
272 fallback = fvValue(i) == "true";
273 SWSS_LOG_INFO("Get fallback option %s",
274 fallback ? "true" : "false");
275 }
276 else if (fvField(i) == "admin_status")
277 {
278 admin_status = fvValue(i);;
279 SWSS_LOG_INFO("Get admin_status %s",
280 admin_status.c_str());
281 }
282 else if (fvField(i) == "mtu")
283 {
284 mtu = fvValue(i);
285 SWSS_LOG_INFO("Get MTU %s", mtu.c_str());
286 }
287 else if (fvField(i) == "learn_mode")
288 {
289 learn_mode = fvValue(i);
290 SWSS_LOG_INFO("Get learn_mode %s",
291 learn_mode.c_str());
292 }
293 else if (fvField(i) == "tpid")
294 {
295 tpid = fvValue(i);
296 SWSS_LOG_INFO("Get TPID %s", tpid.c_str());
297 }
298 else if (fvField(i) == "fast_rate")
299 {
300 fast_rate = fvValue(i) == "true";
301 SWSS_LOG_INFO("Get fast_rate `%s`",
302 fast_rate ? "true" : "false");
303 }
304 }
305
306 if (m_lagList.find(alias) == m_lagList.end())
307 {
308 if (addLag(alias, min_links, fallback, fast_rate) == task_need_retry)
309 {
310 it++;
311 continue;
312 }
313
314 m_lagList.insert(alias);
315 }
316
317 setLagAdminStatus(alias, admin_status);
318 setLagMtu(alias, mtu);
319 if (!learn_mode.empty())
320 {
321 setLagLearnMode(alias, learn_mode);
322 SWSS_LOG_NOTICE("Configure %s MAC learn mode to %s", alias.c_str(), learn_mode.c_str());
323 }
324 if (!tpid.empty())
325 {
326 setLagTpid(alias, tpid);
327 SWSS_LOG_NOTICE("Configure %s TPID to %s", alias.c_str(), tpid.c_str());
328 }
329 }
330 else if (op == DEL_COMMAND)
331 {
332 if (m_lagList.find(alias) != m_lagList.end())
333 {
334 removeLag(alias);
335 m_lagList.erase(alias);
336 }
337 }
338
339 it = consumer.m_toSync.erase(it);
340 }
341}

343void TeamMgr::doLagMemberTask(Consumer &consumer)
344{
345 SWSS_LOG_ENTER();
346
347 auto it = consumer.m_toSync.begin();
348 while (it != consumer.m_toSync.end())
349 {
350 KeyOpFieldsValuesTuple t = it->second;
351
352 auto tokens = tokenize(kfvKey(t), config_db_key_delimiter);
353 auto lag = tokens[0];
354 auto member = tokens[1];
355
356 auto op = kfvOp(t);
357
358 if (op == SET_COMMAND)
359 {
360 if (!isPortStateOk(member) || !isLagStateOk(lag))
361 {
362 it++;
363 continue;
364 }
365 if (isMACsecAttached(member) && !isMACsecIngressSAOk(member))
366 {
367 it++;
368 continue;
369 }
370 if (addLagMember(lag, member) == task_need_retry)
371 {
372 it++;
373 continue;
374 }
375 }
376 else if (op == DEL_COMMAND)
377 {
378 removeLagMember(lag, member);
379 }
380
381 it = consumer.m_toSync.erase(it);
382 }

385bool TeamMgr::checkPortIffUp(const string &port)
386{
387 SWSS_LOG_ENTER();
388
389 struct ifreq ifr;
390 memcpy(ifr.ifr_name, port.c_str(), strlen(port.c_str()));
391 ifr.ifr_name[strlen(port.c_str())] = 0;
392
393 int fd = socket(AF_UNIX, SOCK_DGRAM, 0);
394 if (fd == -1 || ioctl(fd, SIOCGIFFLAGS, &ifr) == -1)
395 {
396 SWSS_LOG_ERROR("Failed to get port %s flags", port.c_str());
397 return false;
398 }
399
400 SWSS_LOG_INFO("Get port %s flags %i", port.c_str(), ifr.ifr_flags);
401
402 return ifr.ifr_flags & IFF_UP;
403}

441void TeamMgr::doPortUpdateTask(Consumer &consumer)
442{
443 SWSS_LOG_ENTER();
444
445 auto it = consumer.m_toSync.begin();

447 {
448 KeyOpFieldsValuesTuple t = it->second;
449
450 auto alias = kfvKey(t);
451 auto op = kfvOp(t);
452
453 if (op == SET_COMMAND)
454 {

456

462 it++;

465 }
466
467 if (addLagMember(lag, alias) == task_need_retry)
468 {
469 it++;
470 continue;
471 }
472 }
473 }
474 else if (op == DEL_COMMAND)
475 {
476 SWSS_LOG_INFO("Received port %s state removal", alias.c_str());
477 }
478
479 it = consumer.m_toSync.erase(it);

563task_process_status TeamMgr::addLag(const string &alias, int min_links, bool fallback, bool fast_rate)
564{
565 SWSS_LOG_ENTER();
566
567 stringstream cmd;
568 string res;
569
570 stringstream conf;
571
572 const string dump_path = "/var/warmboot/teamd/";
573 MacAddress mac_boot = m_mac;
574
575 // set portchannel mac same with mac before warmStart, when warmStart and there
576 // is a file written by teamd.
577 ifstream aliasfile(dump_path + alias);
578 if (WarmStart::isWarmStart() && aliasfile.is_open())
579 {
580 const int partner_system_id_offset = 40;
581 string line;
582
583 while (getline(aliasfile, line))
584 {
585 ifstream memberfile(dump_path + line, ios::binary);
586 uint8_t mac_temp[ETHER_ADDR_LEN] = {0};
587 uint8_t null_mac[ETHER_ADDR_LEN] = {0};
588
589 if (!memberfile.is_open())
590 continue;
591
592 memberfile.seekg(partner_system_id_offset, std::ios::beg);
593 memberfile.read(reinterpret_cast<char*>(mac_temp), ETHER_ADDR_LEN);
594
595 /* During negotiation stage partner info of pdu is empty , skip it */
596 if (memcmp(mac_temp, null_mac, ETHER_ADDR_LEN) == 0)
597 continue;
598
599 mac_boot = MacAddress(mac_temp);
600 break;
601 }
602 }
603
604 conf << "'{\"device\":\"" << alias << "\","
605 << "\"hwaddr\":\"" << mac_boot.to_string() << "\","
606 << "\"runner\":{"
607 << "\"active\":true,"
608 << "\"name\":\"lacp\"";
609
610 if (min_links != 0)
611 {
612 conf << ",\"min_ports\":" << min_links;
613 }
614
615 if (fallback)
616 {
617 conf << ",\"fallback\":true";
618 }
619
620 if (fast_rate)
621 {
622 conf << ",\"fast_rate\":true";
623 }
624
625 conf << "}}'";
626
627 SWSS_LOG_INFO("Port channel %s teamd configuration: %s",
628 alias.c_str(), conf.str().c_str());
629
630 string warmstart_flag = WarmStart::isWarmStart() ? " -w -o " : " -r ";
631
632 cmd << TEAMD_CMD
633 << warmstart_flag
634 << " -t " << alias
635 << " -c " << conf.str()
636 << " -L " << dump_path
637 << " -g -d";
638
639 if (exec(cmd.str(), res) != 0)
640 {
641 SWSS_LOG_INFO("Failed to start port channel %s with teamd, retry...",
642 alias.c_str());
643 return task_need_retry;
644 }
645
646 SWSS_LOG_NOTICE("Start port channel %s with teamd", alias.c_str());
647
648 return task_success;
649}

666// Port-channel names are in the pattern of "PortChannel####"
667//
668// The LACP key could be generated in 3 ways based on the value in config DB:
669// 1. "auto" - LACP key is extracted from the port-channel name and is set to be the number at the end of the port-channel name
670// We are adding 1 at the beginning to avoid LACP key collisions between similar LACP keys e.g. PortChannel10 and PortChannel010.
671// 2. n - LACP key will be n.
672// 3. "" - LACP key will be 0 - exists for backward compatibility.
673uint16_t TeamMgr::generateLacpKey(const string& lag)
674{
675 vector <FieldValueTuple> fvs;
676 m_cfgLagTable.get(lag, fvs);
677
678 auto it = find_if(fvs.begin(), fvs.end(), [](const FieldValueTuple& fv)
679 {
680 return fv.first == "lacp_key";
681 });
682 string lacp_key;
683 if (it != fvs.end())
684 {
685 lacp_key = it->second;
686 if (!lacp_key.empty())
687 {
688 try
689 {
690 if (lacp_key == "auto")
691 {
692 return static_cast<uint16_t>(std::stoul("1" + lag.substr(lag.find_first_of("0123456789"))));
693 }
694 else
695 {
696 return static_cast<uint16_t>(std::stoul(lacp_key));
697 }
698 }
699 catch (const std::exception& e)
700 {
701 SWSS_LOG_THROW("Failed to parse LACP key %s for port channel %s", lacp_key.c_str(), lag.c_str());
702 }
703 }
704 else
705 {
706 return 0;
707 }
708 }
709 return 0;
710}

712// Once a port is enslaved into a port channel, the port's MTU will
713// be inherited from the master's MTU while the port's admin status
714// will still be controlled separately.
715task_process_status TeamMgr::addLagMember(const string &lag, const string &member)
716{
717 SWSS_LOG_ENTER();
718
719 stringstream cmd;
720 string res;
721
722 // If port was already deleted, ignore this operation
723 cmd << IP_CMD << " link show " << shellquote(member);
724 if (exec(cmd.str(), res) != 0)
725 {
726 SWSS_LOG_WARN("Unable to find port %s", member.c_str());
727 return task_ignore;
728 }
729
730 // If port is already enslaved, ignore this operation
731 // TODO: check the current master if it is the same as to be configured
732 if (isPortEnslaved(member))
733 {
734 return task_ignore;
735 }
736
737 uint16_t keyId = generateLacpKey(lag);
738 cmd.str("");
739 cmd.clear();
740
741 // Set admin down LAG member (required by teamd) and enslave it
742 // ip link set dev <member> down;
743 // teamdctl <port_channel_name> port config update <member> { "lacp_key": <lacp_key>, "link_watch": { "name": "ethtool" } };
744 // teamdctl <port_channel_name> port add <member>;
745 cmd << IP_CMD << " link set dev " << shellquote(member) << " down; ";
746 cmd << TEAMDCTL_CMD << " " << shellquote(lag) << " port config update " << shellquote(member)
747 << " '{\"lacp_key\":"
748 << keyId
749 << ",\"link_watch\": {\"name\": \"ethtool\"} }'; ";
750 cmd << TEAMDCTL_CMD << " " << shellquote(lag) << " port add " << shellquote(member);
751
752 if (exec(cmd.str(), res) != 0)
753 {
754 // teamdctl port add command will fail when the member port is not
755 // set to admin status down; it is possible that some other processes
756 // or users (e.g. portmgrd) are executing the command to bring up the
757 // member port while adding this port into the port channel. This piece
758 // of code will check if the port is set to admin status up. If yes,
759 // it will retry to add the port into the port channel.
760 if (checkPortIffUp(member))
761 {
762 SWSS_LOG_INFO("Failed to add %s to port channel %s, retry...",
763 member.c_str(), lag.c_str());
764 return task_need_retry;
765 }
766 else
767 {
768 SWSS_LOG_ERROR("Failed to add %s to port channel %s",
769 member.c_str(), lag.c_str());
770 return task_failed;
771 }
772 }
773
774 vector<FieldValueTuple> fvs;
775 m_cfgPortTable.get(member, fvs);
776
777 // Get the member admin status
778 auto it = find_if(fvs.begin(), fvs.end(), [](const FieldValueTuple &fv) {
779 return fv.first == "admin_status";
780 });
781
782 string admin_status = DEFAULT_ADMIN_STATUS_STR;
783 if (it != fvs.end())
784 {
785 admin_status = it->second;
786 }
787
