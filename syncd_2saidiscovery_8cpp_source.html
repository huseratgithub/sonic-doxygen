37void SaiDiscovery::discover(
38 _In_ sai_object_id_t rid,
39 _Inout_ std::set<sai_object_id_t> &discovered)
40{
41 SWSS_LOG_ENTER();
42
43 /*
44 * NOTE: This method is only good after switch init since we are making
45 * assumptions that there are no ACL after initialization.
46 *
47 * NOTE: Input set could be a map of sets, this way we will also have
48 * dependency on each oid.
49 */
50
51 if (rid == SAI_NULL_OBJECT_ID)
52 {
53 return;
54 }
55
56 if (discovered.find(rid) != discovered.end())
57 {
58 return;
59 }
60
61 sai_object_type_t ot = m_sai->objectTypeQuery(rid);
62
63 if (ot == SAI_OBJECT_TYPE_NULL)
64 {
65 SWSS_LOG_THROW("objectTypeQuery: rid %s returned NULL object type",
66 sai_serialize_object_id(rid).c_str());
67 }
68
69 SWSS_LOG_DEBUG("processing %s: %s",
70 sai_serialize_object_id(rid).c_str(),
71 sai_serialize_object_type(ot).c_str());
72
73 /*
74 * We will ignore STP ports by now, since when removing bridge port, then
75 * associated stp port is automatically removed, and we don't use STP in
76 * out solution. This causing inconsistency with redis ASIC view vs
77 * actual ASIC asic state.
78 *
79 * TODO: This needs to be solved by sending discovered state to sairedis
80 * metadata db for reference count.
81 *
82 * XXX: workaround
83 */
84
85 if (ot != SAI_OBJECT_TYPE_STP_PORT)
86 {
87 discovered.insert(rid);
88 }
89
90 const sai_object_type_info_t *info = sai_metadata_get_object_type_info(ot);
91
92 /*
93 * We will query only oid object types
94 * then we don't need meta key, but we need to add to metadata
95 * pointers to only generic functions.
96 */
97
98 sai_object_meta_key_t mk = { .objecttype = ot, .objectkey = { .key = { .object_id = rid } } };
99
100 for (int idx = 0; info->attrmetadata[idx] != NULL; ++idx)
101 {
102 const sai_attr_metadata_t *md = info->attrmetadata[idx];
103
104 /*
105 * Note that we don't care about ACL object id's since
106 * we assume that there are no ACLs on switch after init.
107 */
108
109 sai_attribute_t attr;
110
111 attr.id = md->attrid;
112
113 if (md->attrvaluetype == SAI_ATTR_VALUE_TYPE_OBJECT_ID)
114 {
115 if (md->defaultvaluetype == SAI_DEFAULT_VALUE_TYPE_CONST)
116 {
117 /*
118 * This means that default value for this object is
119 * SAI_NULL_OBJECT_ID, since this is discovery after
120 * create, we don't need to query this attribute.
121 */
122
123 //continue;
124 }
125
126 if (md->objecttype == SAI_OBJECT_TYPE_STP &&
127 md->attrid == SAI_STP_ATTR_BRIDGE_ID)
128 {
129 // XXX workaround (for mlnx)
130 SWSS_LOG_WARN("skipping since it causes crash: %s", md->attridname);
131 continue;
132 }
133
134 if (md->objecttype == SAI_OBJECT_TYPE_BRIDGE_PORT)
135 {
136 if (md->attrid == SAI_BRIDGE_PORT_ATTR_TUNNEL_ID ||
137 md->attrid == SAI_BRIDGE_PORT_ATTR_RIF_ID)
138 {
139 /*
140 * We know that bridge port is bound on PORT, no need
141 * to query those attributes.
142 */
143
144 continue;
145 }
146 }
147
148 SWSS_LOG_DEBUG("getting %s for %s", md->attridname,
149 sai_serialize_object_id(rid).c_str());
150
151 sai_status_t status = m_sai->get(mk.objecttype, mk.objectkey.key.object_id, 1, &attr);
152
153 if (status != SAI_STATUS_SUCCESS)
154 {
155 /*
156 * We failed to get value, maybe it's not supported ?
157 */
158
159 SWSS_LOG_INFO("%s: %s on %s",
160 md->attridname,
161 sai_serialize_status(status).c_str(),
162 sai_serialize_object_id(rid).c_str());
163
164 continue;
165 }
166
167 m_defaultOidMap[rid][attr.id] = attr.value.oid;
168
169 if (!md->allownullobjectid && attr.value.oid == SAI_NULL_OBJECT_ID)
170 {
171 // SWSS_LOG_WARN("got null on %s, but not allowed", md->attridname);
172 }
173
174 if (attr.value.oid != SAI_NULL_OBJECT_ID)
175 {
176 ot = m_sai->objectTypeQuery(attr.value.oid);
177
178 if (ot == SAI_OBJECT_TYPE_NULL)
179 {
180 SWSS_LOG_THROW("when query %s (on %s RID %s) got value %s objectTypeQuery returned NULL object type",
181 md->attridname,
182 sai_serialize_object_type(md->objecttype).c_str(),
183 sai_serialize_object_id(rid).c_str(),
184 sai_serialize_object_id(attr.value.oid).c_str());
185 }
186 }
187
188 discover(attr.value.oid, discovered); // recursion
189 }
190 else if (md->attrvaluetype == SAI_ATTR_VALUE_TYPE_OBJECT_LIST)
191 {
192 if (md->defaultvaluetype == SAI_DEFAULT_VALUE_TYPE_EMPTY_LIST)
193 {
194 /*
195 * This means that default value for this object is
196 * empty list, since this is discovery after
197 * create, we don't need to query this attribute.
198 */
199
200 //continue;
201 }
202
203 SWSS_LOG_DEBUG("getting %s for %s", md->attridname,
204 sai_serialize_object_id(rid).c_str());
205
206 sai_object_id_t local[SAI_DISCOVERY_LIST_MAX_ELEMENTS];
207
208 attr.value.objlist.count = SAI_DISCOVERY_LIST_MAX_ELEMENTS;
209 attr.value.objlist.list = local;
210
211 sai_status_t status = m_sai->get(mk.objecttype, mk.objectkey.key.object_id, 1, &attr);
212
213 if (status != SAI_STATUS_SUCCESS)
214 {
215 /*
216 * We failed to get value, maybe it's not supported ?
217 */
218
219 SWSS_LOG_INFO("%s: %s on %s",
220 md->attridname,
221 sai_serialize_status(status).c_str(),
222 sai_serialize_object_id(rid).c_str());
223
224 continue;
225 }
226
227 SWSS_LOG_DEBUG("list count %s %u", md->attridname, attr.value.objlist.count);
228
229 for (uint32_t i = 0; i < attr.value.objlist.count; ++i)
230 {
231 sai_object_id_t oid = attr.value.objlist.list[i];
232
233 ot = m_sai->objectTypeQuery(oid);
234
235 if (ot == SAI_OBJECT_TYPE_NULL)
236 {
237 SWSS_LOG_THROW("when query %s (on %s RID %s) got value %s objectTypeQuery returned NULL object type",
238 md->attridname,
239 sai_serialize_object_type(md->objecttype).c_str(),
240 sai_serialize_object_id(rid).c_str(),
241 sai_serialize_object_id(oid).c_str());
242 }
243
244 discover(oid, discovered); // recursion
245 }
246 }
247 }

250std::set<sai_object_id_t> SaiDiscovery::discover(
251 _In_ sai_object_id_t startRid)
252{
253 SWSS_LOG_ENTER();
254
255 /*
256 * Preform discovery on the switch to obtain ASIC view of
257 * objects that are created internally.
258 */
259
260 m_defaultOidMap.clear();
261
262 std::set<sai_object_id_t> discovered_rids;
263
264 {
265 SWSS_LOG_TIMER("discover");
266
267 setApiLogLevel(SAI_LOG_LEVEL_CRITICAL);
268
269 discover(startRid, discovered_rids);
270
271 setApiLogLevel(SAI_LOG_LEVEL_NOTICE);
272 }
273
274 SWSS_LOG_NOTICE("discovered objects count: %zu", discovered_rids.size());
275
276 std::map<sai_object_type_t, int> map;
277
278 for (sai_object_id_t rid: discovered_rids)
279 {
280 /*
281 * We don't need to check for null since saiDiscovery already checked
282 * that.
283 */
284
285 map[m_sai->objectTypeQuery(rid)]++;
286 }
287
288 for (const auto &p: map)
289 {
290 SWSS_LOG_NOTICE("%s: %d", sai_serialize_object_type(p.first).c_str(), p.second);
291 }
292
293 return discovered_rids;
294}
