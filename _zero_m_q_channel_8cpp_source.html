124void ZeroMQChannel::notificationThreadFunction()
125{
126 SWSS_LOG_ENTER();
127
128 SWSS_LOG_NOTICE("start listening for notifications");
129
130 std::vector<uint8_t> buffer;
131
132 buffer.resize(ZMQ_RESPONSE_BUFFER_SIZE);
133
134 while (m_runNotificationThread)
135 {
136 // NOTE: this entire loop internal could be encapsulated into separate class
137 // which will inherit from Selectable class, and name this as ntf receiver
138
139 int rc = zmq_recv(m_ntfSocket, buffer.data(), ZMQ_RESPONSE_BUFFER_SIZE, 0);
140
141 if (!m_runNotificationThread)
142 break;
143
144 if (rc <= 0 && zmq_errno() == ETERM)
145 {
146 SWSS_LOG_NOTICE("zmq_recv interrupted with ETERM, ending thread");
147 break;
148 }
149
150 if (rc < 0)
151 {
152 SWSS_LOG_ERROR("zmq_recv failed, zmqerrno: %d", zmq_errno());
153
154 // at this point we don't know if next zmq_recv will succeed
155
156 continue;
157 }
158
159 if (rc >= ZMQ_RESPONSE_BUFFER_SIZE)
160 {
161 SWSS_LOG_WARN("zmq_recv message was truncated (over %d bytes, received %d), increase buffer size, message DROPPED",
162 ZMQ_RESPONSE_BUFFER_SIZE,
163 rc);
164
165 continue;
166 }
167
168 buffer.at(rc) = 0; // make sure that we end string with zero before parse
169
170 SWSS_LOG_DEBUG("ntf: %s", buffer.data());
171
172 std::vector<swss::FieldValueTuple> values;
173
174 swss::JSon::readJson((char*)buffer.data(), values);
175
176 swss::FieldValueTuple fvt = values.at(0);
177
178 const std::string& op = fvField(fvt);
179 const std::string& data = fvValue(fvt);
180
181 values.erase(values.begin());
182
183 SWSS_LOG_DEBUG("notification: op = %s, data = %s", op.c_str(), data.c_str());
184
185 m_callback(op, data, values);
186 }
187
188 SWSS_LOG_NOTICE("exiting notification thread");
189}

253sai_status_t ZeroMQChannel::wait(
254 _In_ const std::string& command,
255 _Out_ swss::KeyOpFieldsValuesTuple& kco)
256{
257 SWSS_LOG_ENTER();
258
259 SWSS_LOG_INFO("wait for %s response", command.c_str());
260
261 zmq_pollitem_t items [1] = { };
262
263 items[0].socket = m_socket;
264 items[0].events = ZMQ_POLLIN;
265
266 int rc;
267
268 for (int i = 0; true ; ++i)
269 {
270 rc = zmq_poll(items, 1, (int)m_responseTimeoutMs);
271
272 if (rc == 0)
273 {
274 SWSS_LOG_ERROR("zmq_poll timed out for: %s", command.c_str());
275
276 // notice, at this point we could throw, since in REP/REQ pattern
277 // we are forced to use send/recv in that specific order
278
279 return SAI_STATUS_FAILURE;
280 }
281 if (rc < 0 && zmq_errno() == EINTR && i < ZMQ_MAX_RETRY)
282 {
283 continue;
284 }
285 if (rc < 0)
286 {
287 SWSS_LOG_THROW("zmq_poll failed, zmqerrno: %d", zmq_errno());
288 }
289 break;
290 }
291
292 for (int i = 0; true ; ++i)
293 {
294 rc = zmq_recv(m_socket, m_buffer.data(), ZMQ_RESPONSE_BUFFER_SIZE, 0);
295
296 if (rc < 0 && zmq_errno() == EINTR && i < ZMQ_MAX_RETRY)
297 {
298 continue;
299 }
300 if (rc < 0)
301 {
302 SWSS_LOG_THROW("zmq_recv failed, zmqerrno: %d", zmq_errno());
303 }
304 if (rc >= ZMQ_RESPONSE_BUFFER_SIZE)
305 {
306 SWSS_LOG_THROW("zmq_recv message was truncated (over %d bytes, received %d), increase buffer size, message DROPPED",
307 ZMQ_RESPONSE_BUFFER_SIZE,
308 rc);
309 }
310 break;
311 }
312
313 m_buffer.at(rc) = 0; // make sure that we end string with zero before parse
314
315 SWSS_LOG_DEBUG("response: %s", m_buffer.data());
316
317 std::vector<swss::FieldValueTuple> values;
318
319 swss::JSon::readJson((char*)m_buffer.data(), values);
320
321 swss::FieldValueTuple fvt = values.at(0);
322
323 const std::string& opkey = fvField(fvt);
324 const std::string& op= fvValue(fvt);
325
326 values.erase(values.begin());
327
328 kfvFieldsValues(kco) = values;
329 kfvOp(kco) = op;
330 kfvKey(kco) = opkey;
331
332 SWSS_LOG_INFO("response: op = %s, key = %s", opkey.c_str(), op.c_str());
333
334 if (op != command)
335 {
336 // we can hit this place if there were some timeouts
337 // as well, if there will be multiple "GET" messages, then
338 // we can receive response from not the expected GET
339
340 SWSS_LOG_THROW("got not expected response: %s:%s, expected: %s", opkey.c_str(), op.c_str(), command.c_str());
341 }
342
343 sai_status_t status;
344 sai_deserialize_status(opkey, status);
345
346 SWSS_LOG_DEBUG("%s status: %s", command.c_str(), opkey.c_str());
347
348 return status;
