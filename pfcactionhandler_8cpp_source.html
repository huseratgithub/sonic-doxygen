49void PfcWdActionHandler::initCounters(void)
50{
51 SWSS_LOG_ENTER();
52
53 if (!getHwCounters(m_hwStats))
54 {
55 return;
56 }
57
58 auto wdQueueStats = getQueueStats(m_countersTable, sai_serialize_object_id(m_queue));
59 // initCounters() is called when the event channel receives
60 // a storm signal. This can happen when there is a true new storm or
61 // when there is an existing storm ongoing before warm-reboot. In the latter case,
62 // we treat the storm as an old storm. In particular,
63 // we do not increment the detectCount so as to clamp the
64 // gap between detectCount and restoreCount by 1 at maximum
65 if (!(wdQueueStats.detectCount > wdQueueStats.restoreCount))
66 {
67 wdQueueStats.detectCount++;
68
69 wdQueueStats.txPktLast = 0;
70 wdQueueStats.txDropPktLast = 0;
71 wdQueueStats.rxPktLast = 0;
72 wdQueueStats.rxDropPktLast = 0;
73 }
74 wdQueueStats.operational = false;
75
76 updateWdCounters(sai_serialize_object_id(m_queue), wdQueueStats);
77}

79void PfcWdActionHandler::commitCounters(bool periodic /* = false */)
80{
81 SWSS_LOG_ENTER();
82
83 PfcWdHwStats hwStats;
84
85 if (!getHwCounters(hwStats))
86 {
87 return;
88 }
89
90 auto finalStats = getQueueStats(m_countersTable, sai_serialize_object_id(m_queue));
91
92 if (!periodic)
93 {
94 finalStats.restoreCount++;
95 }
96 finalStats.operational = !periodic;
97
98 finalStats.txPktLast += hwStats.txPkt - m_hwStats.txPkt;
99 finalStats.txDropPktLast += hwStats.txDropPkt - m_hwStats.txDropPkt;
100 finalStats.rxPktLast += hwStats.rxPkt - m_hwStats.rxPkt;
101 finalStats.rxDropPktLast += hwStats.rxDropPkt - m_hwStats.rxDropPkt;
102
103 finalStats.txPkt += hwStats.txPkt - m_hwStats.txPkt;
104 finalStats.txDropPkt += hwStats.txDropPkt - m_hwStats.txDropPkt;
105 finalStats.rxPkt += hwStats.rxPkt - m_hwStats.rxPkt;
106 finalStats.rxDropPkt += hwStats.rxDropPkt - m_hwStats.rxDropPkt;
107
108 m_hwStats = hwStats;
109
110 updateWdCounters(sai_serialize_object_id(m_queue), finalStats);
111}

113PfcWdActionHandler::PfcWdQueueStats PfcWdActionHandler::getQueueStats(shared_ptr<Table> countersTable, const string &queueIdStr)
114{
115 SWSS_LOG_ENTER();
116
117 PfcWdQueueStats stats;
118 memset(&stats, 0, sizeof(PfcWdQueueStats));
119 stats.operational = true;
120 vector<FieldValueTuple> fieldValues;
121
122 if (!countersTable->get(queueIdStr, fieldValues))
123 {
124 return stats;
125 }
126
127 for (const auto& fv : fieldValues)
128 {
129 const auto field = fvField(fv);
130 const auto value = fvValue(fv);
131
132 if (field == PFC_WD_QUEUE_STATS_DEADLOCK_DETECTED)
133 {
134 stats.detectCount = stoul(value);
135 }
136 else if (field == PFC_WD_QUEUE_STATS_DEADLOCK_RESTORED)
137 {
138 stats.restoreCount = stoul(value);
139 }
140 else if (field == PFC_WD_QUEUE_STATUS)
141 {
142 stats.operational = value == PFC_WD_QUEUE_STATUS_OPERATIONAL ? true : false;
143 }
144 else if (field == PFC_WD_QUEUE_STATS_TX_PACKETS)
145 {
146 stats.txPkt = stoul(value);
147 }
148 else if (field == PFC_WD_QUEUE_STATS_TX_DROPPED_PACKETS)
149 {
150 stats.txDropPkt = stoul(value);
151 }
152 else if (field == PFC_WD_QUEUE_STATS_RX_PACKETS)
153 {
154 stats.rxPkt = stoul(value);
155 }
156 else if (field == PFC_WD_QUEUE_STATS_RX_DROPPED_PACKETS)
157 {
158 stats.rxDropPkt = stoul(value);
159 }
160 else if (field == PFC_WD_QUEUE_STATS_TX_PACKETS_LAST)
161 {
162 stats.txPktLast = stoul(value);
163 }
164 else if (field == PFC_WD_QUEUE_STATS_TX_DROPPED_PACKETS_LAST)
165 {
166 stats.txDropPktLast = stoul(value);
167 }
168 else if (field == PFC_WD_QUEUE_STATS_RX_PACKETS_LAST)
169 {
170 stats.rxPktLast = stoul(value);
171 }
172 else if (field == PFC_WD_QUEUE_STATS_RX_DROPPED_PACKETS_LAST)
173 {
174 stats.rxDropPktLast = stoul(value);
175 }
176 }
177
178 return stats;

308PfcWdAclHandler::PfcWdAclHandler(sai_object_id_t port, sai_object_id_t queue,
309 uint8_t queueId, shared_ptr<Table> countersTable):
310 PfcWdLossyHandler(port, queue, queueId, countersTable)
311{
312 SWSS_LOG_ENTER();
313
314 string table_type;
315
316 string queuestr = to_string(queueId);
317 m_strRule = "Rule_PfcWdAclHandler_" + queuestr;
318
319 // Ingress table/rule creation
320 table_type = TABLE_TYPE_DROP;
321 m_strIngressTable = INGRESS_TABLE_DROP;
322 auto found = m_aclTables.find(m_strIngressTable);
323 if (found == m_aclTables.end())
324 {
325 // First time of handling PFC for this queue, create ACL table, and bind
326 createPfcAclTable(port, m_strIngressTable, true);
327 shared_ptr<AclRulePacket> newRule = make_shared<AclRulePacket>(gAclOrch, m_strRule, m_strIngressTable);
328 createPfcAclRule(newRule, queueId, m_strIngressTable, port);
329 }
330 else
331 {
332 AclRule* rule = gAclOrch->getAclRule(m_strIngressTable, m_strRule);
333 if (rule == nullptr)
334 {
335 shared_ptr<AclRulePacket> newRule = make_shared<AclRulePacket>(gAclOrch, m_strRule, m_strIngressTable);
336 createPfcAclRule(newRule, queueId, m_strIngressTable, port);
337 }
338 else
339 {
340 gAclOrch->updateAclRule(m_strIngressTable, m_strRule, MATCH_IN_PORTS, &port, RULE_OPER_ADD);
341 }
342 }
343
344 // Egress table/rule creation
345 table_type = TABLE_TYPE_PFCWD;
346 m_strEgressTable = "EgressTable_PfcWdAclHandler_" + queuestr;
347 found = m_aclTables.find(m_strEgressTable);
348 if (found == m_aclTables.end())
349 {
350 // First time of handling PFC for this queue, create ACL table, and bind
351 createPfcAclTable(port, m_strEgressTable, false);
352 shared_ptr<AclRulePacket> newRule = make_shared<AclRulePacket>(gAclOrch, m_strRule, m_strEgressTable);
353 createPfcAclRule(newRule, queueId, m_strEgressTable, port);
354 }
355 else
356 {
357 // Otherwise just bind ACL table with the port
358 found->second.bind(port);
359 }
360}

399void PfcWdAclHandler::createPfcAclTable(sai_object_id_t port, string strTable, bool ingress)
400{
401 SWSS_LOG_ENTER();
402
403 auto inserted = m_aclTables.emplace(piecewise_construct,
404 std::forward_as_tuple(strTable),
405 std::forward_as_tuple(gAclOrch, strTable));
406
407 assert(inserted.second);
408
409 AclTable& aclTable = inserted.first->second;
410
411 sai_object_id_t table_oid = gAclOrch->getTableById(strTable);
412 if (ingress && table_oid != SAI_NULL_OBJECT_ID)
413 {
414 // DROP ACL table is already created
415 SWSS_LOG_NOTICE("ACL table %s exists, reuse the same", strTable.c_str());
416 aclTable = *(gAclOrch->getTableByOid(table_oid));
417 return;
418 }
419
420 aclTable.link(port);
421
422 if (ingress)
423 {
424 auto dropType = gAclOrch->getAclTableType(TABLE_TYPE_DROP);
425 assert(dropType);
426 aclTable.validateAddType(*dropType);
427 aclTable.stage = ACL_STAGE_INGRESS;
428 }
429 else
430 {
431 auto pfcwdType = gAclOrch->getAclTableType(TABLE_TYPE_PFCWD);
432 assert(pfcwdType);
433 aclTable.validateAddType(*pfcwdType);
434 aclTable.stage = ACL_STAGE_EGRESS;
435 }
436
437 gAclOrch->addAclTable(aclTable);
438}

440void PfcWdAclHandler::createPfcAclRule(shared_ptr<AclRulePacket> rule, uint8_t queueId, string strTable, sai_object_id_t portOid)
441{
442 SWSS_LOG_ENTER();
443
444 string attr_name, attr_value;
445
446 attr_name = RULE_PRIORITY;
447 attr_value = "999";
448 rule->validateAddPriority(attr_name, attr_value);
449
450 attr_name = MATCH_TC;
451 attr_value = to_string(queueId);
452 rule->validateAddMatch(attr_name, attr_value);
453
454 // Add MATCH_IN_PORTS as match criteria for ingress table
455 if (strTable == INGRESS_TABLE_DROP)
456 {
457 Port p;
458 attr_name = MATCH_IN_PORTS;
459
460 if (!gPortsOrch->getPort(portOid, p))
461 {
462 SWSS_LOG_ERROR("Failed to get port structure from port oid 0x%" PRIx64, portOid);
463 return;
464 }
465
466 attr_value = p.m_alias;
467 rule->validateAddMatch(attr_name, attr_value);
468 }
469
470 attr_name = ACTION_PACKET_ACTION;
471 attr_value = PACKET_ACTION_DROP;
472 rule->validateAddAction(attr_name, attr_value);
473
474 gAclOrch->addAclRule(rule, strTable);
475}

535bool PfcWdLossyHandler::getHwCounters(PfcWdHwStats& counters)
536{
537 SWSS_LOG_ENTER();
538
539 static const vector<sai_stat_id_t> queueStatIds =
540 {
541 SAI_QUEUE_STAT_PACKETS,
542 SAI_QUEUE_STAT_DROPPED_PACKETS,
543 };
544
545 static const vector<sai_stat_id_t> pgStatIds =
546 {
547 SAI_INGRESS_PRIORITY_GROUP_STAT_PACKETS,
548 SAI_INGRESS_PRIORITY_GROUP_STAT_DROPPED_PACKETS,
549 };
550
551 vector<uint64_t> queueStats;
552 queueStats.resize(queueStatIds.size());
553
554 sai_status_t status = sai_queue_api->get_queue_stats(
555 getQueue(),
556 static_cast<uint32_t>(queueStatIds.size()),
557 queueStatIds.data(),
558 queueStats.data());
559
560 if (status != SAI_STATUS_SUCCESS)
561 {
562 SWSS_LOG_ERROR("Failed to fetch queue 0x%" PRIx64 " stats: %d", getQueue(), status);
563 return false;
564 }
565
566 // PG counters not yet supported in Mellanox platform
567 Port portInstance;
568 if (!gPortsOrch->getPort(getPort(), portInstance))
569 {
570 SWSS_LOG_ERROR("Cannot get port by ID 0x%" PRIx64, getPort());
571 return false;
572 }
573
574 sai_object_id_t pg = portInstance.m_priority_group_ids[static_cast <size_t> (getQueueId())];
575 vector<uint64_t> pgStats;
576 pgStats.resize(pgStatIds.size());
577
578 status = sai_buffer_api->get_ingress_priority_group_stats(
579 pg,
580 static_cast<uint32_t>(pgStatIds.size()),
581 pgStatIds.data(),
582 pgStats.data());
583
584 if (status != SAI_STATUS_SUCCESS)
585 {
586 SWSS_LOG_ERROR("Failed to fetch pg 0x%" PRIx64 " stats: %d", pg, status);
587 return false;
588 }
589
590 counters.txPkt = queueStats[0];
591 counters.txDropPkt = queueStats[1];
592 counters.rxPkt = pgStats[0];
593 counters.rxDropPkt = pgStats[1];
594
595 return true;
596}
