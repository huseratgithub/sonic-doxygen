111void GreTunnelManager::drain()
112{
113 SWSS_LOG_ENTER();
114
115 for (const auto &key_op_fvs_tuple : m_entries)
116 {
117 std::string table_name;
118 std::string key;
119 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &key);
120 const std::vector<swss::FieldValueTuple> &attributes = kfvFieldsValues(key_op_fvs_tuple);
121
122 const std::string &operation = kfvOp(key_op_fvs_tuple);
123
124 ReturnCode status;
125 auto app_db_entry_or = deserializeP4GreTunnelAppDbEntry(key, attributes);
126 if (!app_db_entry_or.ok())
127 {
128 status = app_db_entry_or.status();
129 SWSS_LOG_ERROR("Unable to deserialize GRE Tunnel APP DB entry with key %s: %s",
130 QuotedVar(kfvKey(key_op_fvs_tuple)).c_str(), status.message().c_str());
131 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
132 status,
133 /*replace=*/true);
134 continue;
135 }
136 auto &app_db_entry = *app_db_entry_or;
137
138 const std::string tunnel_key = KeyGenerator::generateTunnelKey(app_db_entry.tunnel_id);
139
140 // Fulfill the operation.
141 if (operation == SET_COMMAND)
142 {
143 status = validateGreTunnelAppDbEntry(app_db_entry);
144 if (!status.ok())
145 {
146 SWSS_LOG_ERROR("Validation failed for GRE Tunnel APP DB entry with key %s: %s",
147 QuotedVar(kfvKey(key_op_fvs_tuple)).c_str(), status.message().c_str());
148 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple),
149 status,
150 /*replace=*/true);
151 continue;
152 }
153 auto *gre_tunnel_entry = getGreTunnelEntry(tunnel_key);
154 if (gre_tunnel_entry == nullptr)
155 {
156 // Create new GRE tunnel.
157 status = processAddRequest(app_db_entry);
158 }
159 else
160 {
161 // Modify existing GRE tunnel.
162 status = processUpdateRequest(app_db_entry, gre_tunnel_entry);
163 }
164 }
165 else if (operation == DEL_COMMAND)
166 {
167 // Delete GRE tunnel.
168 status = processDeleteRequest(tunnel_key);
169 }
170 else
171 {
172 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Unknown operation type " << QuotedVar(operation);
173 SWSS_LOG_ERROR("%s", status.message().c_str());
174 }
175 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple), status,
176 /*replace=*/true);
177 }
178 m_entries.clear();
179}

213ReturnCodeOr<P4GreTunnelAppDbEntry> GreTunnelManager::deserializeP4GreTunnelAppDbEntry(
214 const std::string &key, const std::vector<swss::FieldValueTuple> &attributes)
215{
216 SWSS_LOG_ENTER();
217
218 P4GreTunnelAppDbEntry app_db_entry = {};
219 app_db_entry.encap_src_ip = swss::IpAddress("0.0.0.0");
220 app_db_entry.encap_dst_ip = swss::IpAddress("0.0.0.0");
221
222 try
223 {
224 nlohmann::json j = nlohmann::json::parse(key);
225 app_db_entry.tunnel_id = j[prependMatchField(p4orch::kTunnelId)];
226 }
227 catch (std::exception &ex)
228 {
229 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to deserialize GRE tunnel id";
230 }
231
232 for (const auto &it : attributes)
233 {
234 const auto &field = fvField(it);
235 const auto &value = fvValue(it);
236 if (field == prependParamField(p4orch::kRouterInterfaceId))
237 {
238 app_db_entry.router_interface_id = value;
239 }
240 else if (field == prependParamField(p4orch::kEncapSrcIp))
241 {
242 try
243 {
244 app_db_entry.encap_src_ip = swss::IpAddress(value);
245 }
246 catch (std::exception &ex)
247 {
248 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
249 << "Invalid IP address " << QuotedVar(value) << " of field " << QuotedVar(field);
250 }
251 }
252 else if (field == prependParamField(p4orch::kEncapDstIp))
253 {
254 try
255 {
256 app_db_entry.encap_dst_ip = swss::IpAddress(value);
257 }
258 catch (std::exception &ex)
259 {
260 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
261 << "Invalid IP address " << QuotedVar(value) << " of field " << QuotedVar(field);
262 }
263 }
264 else if (field == p4orch::kAction)
265 {
266 app_db_entry.action_str = value;
267 }
268 else if (field != p4orch::kControllerMetadata)
269 {
270 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
271 << "Unexpected field " << QuotedVar(field) << " in table entry";
272 }
273 }
274
275 return app_db_entry;
276}

292ReturnCode GreTunnelManager::createGreTunnel(P4GreTunnelEntry &gre_tunnel_entry)
293{
294 SWSS_LOG_ENTER();
295
296 // Check the existence of the GRE tunnel in GRE tunnel manager and centralized
297 // mapper.
298 if (getGreTunnelEntry(gre_tunnel_entry.tunnel_key) != nullptr)
299 {
300 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_EXISTS)
301 << "GRE tunnel with key " << QuotedVar(gre_tunnel_entry.tunnel_key)
302 << " already exists in GRE tunnel manager");
303 }
304 if (m_p4OidMapper->existsOID(SAI_OBJECT_TYPE_TUNNEL, gre_tunnel_entry.tunnel_key))
305 {
306 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("GRE tunnel with key " << QuotedVar(gre_tunnel_entry.tunnel_key)
307 << " already exists in centralized mapper");
308 }
309
310 // From centralized mapper, get OID of router interface that GRE tunnel
311 // depends on.
312 const auto router_interface_key = KeyGenerator::generateRouterInterfaceKey(gre_tunnel_entry.router_interface_id);
313 if (!m_p4OidMapper->getOID(SAI_OBJECT_TYPE_ROUTER_INTERFACE, router_interface_key,
314 &gre_tunnel_entry.underlay_if_oid))
315 {
316 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
317 << "Router intf " << QuotedVar(gre_tunnel_entry.router_interface_id) << " does not exist");
318 }
319
320 std::vector<sai_attribute_t> overlay_intf_attrs;
321
322 sai_attribute_t overlay_intf_attr;
323 overlay_intf_attr.id = SAI_ROUTER_INTERFACE_ATTR_VIRTUAL_ROUTER_ID;
324 overlay_intf_attr.value.oid = gVirtualRouterId;
325 overlay_intf_attrs.push_back(overlay_intf_attr);
326
327 overlay_intf_attr.id = SAI_ROUTER_INTERFACE_ATTR_TYPE;
328 overlay_intf_attr.value.s32 = SAI_ROUTER_INTERFACE_TYPE_LOOPBACK;
329 overlay_intf_attrs.push_back(overlay_intf_attr);
330
331 // Call SAI API.
332 CHECK_ERROR_AND_LOG_AND_RETURN(
333 sai_router_intfs_api->create_router_interface(&gre_tunnel_entry.overlay_if_oid, gSwitchId,
334 (uint32_t)overlay_intf_attrs.size(), overlay_intf_attrs.data()),
335 "Failed to create the Loopback router interface for GRE tunnel "
336 "SAI_TUNNEL_ATTR_OVERLAY_INTERFACE attribute"
337 << QuotedVar(gre_tunnel_entry.tunnel_key));
338
339 // Prepare attributes for the SAI creation call.
340 std::vector<sai_attribute_t> tunnel_attrs = getSaiAttrs(gre_tunnel_entry);
341
342 // Call SAI API.
343 auto sai_status = sai_tunnel_api->create_tunnel(&gre_tunnel_entry.tunnel_oid, gSwitchId,
344 (uint32_t)tunnel_attrs.size(), tunnel_attrs.data());
345 if (sai_status != SAI_STATUS_SUCCESS)
346 {
347 auto status = ReturnCode(sai_status) << "Failed to create GRE tunnel " << QuotedVar(gre_tunnel_entry.tunnel_key)
348 << " on rif " << QuotedVar(gre_tunnel_entry.router_interface_id);
349 SWSS_LOG_ERROR("%s", status.message().c_str());
350 auto recovery_status = sai_router_intfs_api->remove_router_interface(gre_tunnel_entry.overlay_if_oid);
351 if (recovery_status != SAI_STATUS_SUCCESS)
352 {
353 auto rc = ReturnCode(recovery_status) << "Failed to recover overlay router interface due to SAI call "
354 "failure: Failed to remove loopback router interface "
355 << QuotedVar(sai_serialize_object_id(gre_tunnel_entry.overlay_if_oid))
356 << " while clean up dependencies.";
357 SWSS_LOG_ERROR("%s", rc.message().c_str());
358 SWSS_RAISE_CRITICAL_STATE(rc.message());
359 }
360 return status;
361 }
362
363 // On successful creation, increment ref count.
364 m_p4OidMapper->increaseRefCount(SAI_OBJECT_TYPE_ROUTER_INTERFACE, router_interface_key);
365
366 // Add created entry to internal table.
367 m_greTunnelTable.emplace(gre_tunnel_entry.tunnel_key, gre_tunnel_entry);
368
369 // Add the key to OID map to centralized mapper.
370 m_p4OidMapper->setOID(SAI_OBJECT_TYPE_TUNNEL, gre_tunnel_entry.tunnel_key, gre_tunnel_entry.tunnel_oid);
371
372 return ReturnCode();
373}

400ReturnCode GreTunnelManager::removeGreTunnel(const std::string &tunnel_key)
401{
402 SWSS_LOG_ENTER();
403
404 // Check the existence of the GRE tunnel in GRE tunnel manager and centralized
405 // mapper.
406 auto *gre_tunnel_entry = getGreTunnelEntry(tunnel_key);
407 if (gre_tunnel_entry == nullptr)
408 {
409 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
410 << "GRE tunnel with key " << QuotedVar(tunnel_key)
411 << " does not exist in GRE tunnel manager");
412 }
413
414 // Check if there is anything referring to the GRE tunnel before deletion.
415 uint32_t ref_count;
416 if (!m_p4OidMapper->getRefCount(SAI_OBJECT_TYPE_TUNNEL, tunnel_key, &ref_count))
417 {
418 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Failed to get reference count for GRE tunnel "
419 << QuotedVar(tunnel_key));
420 }
421 if (ref_count > 0)
422 {
423 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
424 << "GRE tunnel " << QuotedVar(gre_tunnel_entry->tunnel_key)
425 << " referenced by other objects (ref_count = " << ref_count);
426 }
427
428 // Call SAI API.
429 CHECK_ERROR_AND_LOG_AND_RETURN(sai_tunnel_api->remove_tunnel(gre_tunnel_entry->tunnel_oid),
430 "Failed to remove GRE tunnel " << QuotedVar(gre_tunnel_entry->tunnel_key));
431
432 auto sai_status = sai_router_intfs_api->remove_router_interface(gre_tunnel_entry->overlay_if_oid);
433 if (sai_status != SAI_STATUS_SUCCESS)
434 {
435 auto status = ReturnCode(sai_status) << "Failed to remove loopback router interface "
436 << QuotedVar(sai_serialize_object_id(gre_tunnel_entry->overlay_if_oid))
437 << " when removing GRE tunnel " << QuotedVar(gre_tunnel_entry->tunnel_key);
438 SWSS_LOG_ERROR("%s", status.message().c_str());
439
440 // Try to recreate the GRE tunnel
441 std::vector<sai_attribute_t> tunnel_attrs = getSaiAttrs(*gre_tunnel_entry);
442
443 // Call SAI API.
444 auto recovery_status = sai_tunnel_api->create_tunnel(&gre_tunnel_entry->tunnel_oid, gSwitchId,
445 (uint32_t)tunnel_attrs.size(), tunnel_attrs.data());
446 if (recovery_status != SAI_STATUS_SUCCESS)
447 {
448 auto rc = ReturnCode(recovery_status) << "Failed to recover the GRE tunnel due to SAI call failure : "
449 "Failed to create GRE tunnel "
450 << QuotedVar(gre_tunnel_entry->tunnel_key) << " on rif "
451 << QuotedVar(gre_tunnel_entry->router_interface_id);
452 SWSS_LOG_ERROR("%s", rc.message().c_str());
453 SWSS_RAISE_CRITICAL_STATE(rc.message());
454 }
455 return status;
456 }
457
458 // On successful deletion, decrement ref count.
459 m_p4OidMapper->decreaseRefCount(SAI_OBJECT_TYPE_ROUTER_INTERFACE,
460 KeyGenerator::generateRouterInterfaceKey(gre_tunnel_entry->router_interface_id));
461
462 // Remove the key to OID map to centralized mapper.
463 m_p4OidMapper->eraseOID(SAI_OBJECT_TYPE_TUNNEL, tunnel_key);
464
465 // Remove the entry from internal table.
466 m_greTunnelTable.erase(tunnel_key);
467
468 return ReturnCode();
469}

471std::string GreTunnelManager::verifyState(const std::string &key, const std::vector<swss::FieldValueTuple> &tuple)
472{
473 SWSS_LOG_ENTER();
474
475 auto pos = key.find_first_of(kTableKeyDelimiter);
476 if (pos == std::string::npos)
477 {
478 return std::string("Invalid key: ") + key;
479 }
480 std::string p4rt_table = key.substr(0, pos);
481 std::string p4rt_key = key.substr(pos + 1);
482 if (p4rt_table != APP_P4RT_TABLE_NAME)
483 {
484 return std::string("Invalid key: ") + key;
485 }
486 std::string table_name;
487 std::string key_content;
488 parseP4RTKey(p4rt_key, &table_name, &key_content);
489 if (table_name != APP_P4RT_TUNNEL_TABLE_NAME)
490 {
491 return std::string("Invalid key: ") + key;
492 }
493
494 ReturnCode status;
495 auto app_db_entry_or = deserializeP4GreTunnelAppDbEntry(key_content, tuple);
496 if (!app_db_entry_or.ok())
497 {
498 status = app_db_entry_or.status();
499 std::stringstream msg;
500 msg << "Unable to deserialize key " << QuotedVar(key) << ": " << status.message();
501 return msg.str();
502 }
503 auto &app_db_entry = *app_db_entry_or;
504 const std::string tunnel_key = KeyGenerator::generateTunnelKey(app_db_entry.tunnel_id);
505 auto *gre_tunnel_entry = getGreTunnelEntry(tunnel_key);
506 if (gre_tunnel_entry == nullptr)
507 {
508 std::stringstream msg;
509 msg << "No entry found with key " << QuotedVar(key);
510 return msg.str();
511 }
512
513 std::string cache_result = verifyStateCache(app_db_entry, gre_tunnel_entry);
514 std::string asic_db_result = verifyStateAsicDb(gre_tunnel_entry);
515 if (cache_result.empty())
516 {
517 return asic_db_result;
518 }
519 if (asic_db_result.empty())
520 {
521 return cache_result;
522 }
523 return cache_result + "; " + asic_db_result;
524}
