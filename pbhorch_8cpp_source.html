31template<typename K, typename V>
32static inline std::set<K> uMapToKeySet(const umap_t<K, V> &uMap)
33{
34 std::set<K> s;
35
36 std::transform(
37 uMap.cbegin(),
38 uMap.cend(),
39 std::inserter(s, s.begin()),
40 [](const std::pair<K, V> &p) {
41 return p.first;
42 }
43 );
44
45 return s;
46}

48template<typename K, typename V>
49static inline std::vector<K> uMapDiffByKey(const umap_t<K, V> &uMap1, const umap_t<K, V> &uMap2)
50{
51 std::vector<K> v;
52
53 const auto &s1 = uMapToKeySet(uMap1);
54 const auto &s2 = uMapToKeySet(uMap2);
55
56 std::set_difference(
57 s1.cbegin(),
58 s1.cend(),
59 s2.cbegin(),
60 s2.cend(),
61 std::back_inserter(v)
62 );
63
64 return v;
65}

78 s2.cbegin(),
79 s2.cend(),
80 std::back_inserter(v)
81 );

83 return v;

406{
407 SWSS_LOG_ENTER();
408
409 auto &map = this->pbhHlpr.tableTask.pendingSetupMap;
410 auto it = map.begin();
411
412 while (it != map.end())
413 {
414 auto &key = it->first;
415 auto &table = it->second;
416
417 PbhTable tObj;
418
419 if (!this->pbhHlpr.getPbhTable(tObj, key))
420 {
421 if (!this->createPbhTable(table))
422 {
423 SWSS_LOG_ERROR("Failed to create PBH table(%s): ASIC and CONFIG DB are diverged", key.c_str());
424 }
425 }
426 else
427 {
428 if (!this->updatePbhTable(table))
429 {
430 SWSS_LOG_ERROR("Failed to update PBH table(%s): ASIC and CONFIG DB are diverged", key.c_str());
431 }
432 }
433
434 it = map.erase(it);
435 }
