99void TeamSync::onMsg(int nlmsg_type, struct nl_object *obj)
100{
101 struct rtnl_link *link = (struct rtnl_link *)obj;
102 if ((nlmsg_type != RTM_NEWLINK) && (nlmsg_type != RTM_DELLINK))
103 return;
104
105 string lagName = rtnl_link_get_name(link);
106
107 /* Listens to LAG messages */
108 char *type = rtnl_link_get_type(link);
109 if (!type || (strcmp(type, TEAM_DRV_NAME) != 0))
110 return;
111
112 unsigned int flags = rtnl_link_get_flags(link);
113 bool admin = flags & IFF_UP;
114 bool oper = flags & IFF_LOWER_UP;
115 unsigned int ifindex = rtnl_link_get_ifindex(link);
116
117 if (type)
118 {
119 SWSS_LOG_INFO(" nlmsg type:%d key:%s admin:%d oper:%d ifindex:%d type:%s",
120 nlmsg_type, lagName.c_str(), admin, oper, ifindex, type);
121 }
122 else
123 {
124 SWSS_LOG_INFO(" nlmsg type:%d key:%s admin:%d oper:%d ifindex:%d",
125 nlmsg_type, lagName.c_str(), admin, oper, ifindex);
126 }
127
128 if (nlmsg_type == RTM_DELLINK)
129 {
130 if (m_teamSelectables.find(lagName) != m_teamSelectables.end())
131 {
132 /* Remove LAG ports and delete LAG */
133 removeLag(lagName);
134 }
135 return;
136 }
137
138 unsigned int mtu = rtnl_link_get_mtu(link);
139 addLag(lagName, rtnl_link_get_ifindex(link),
140 rtnl_link_get_flags(link) & IFF_UP,
141 rtnl_link_get_flags(link) & IFF_LOWER_UP, mtu);
142}

144void TeamSync::addLag(const string &lagName, int ifindex, bool admin_state,
145 bool oper_state, unsigned int mtu)
146{
147 /* Set the LAG */
148 std::vector<FieldValueTuple> fvVector;
149 FieldValueTuple a("admin_status", admin_state ? "up" : "down");
150 FieldValueTuple o("oper_status", oper_state ? "up" : "down");
151 FieldValueTuple m("mtu", std::to_string(mtu));
152 fvVector.push_back(a);
153 fvVector.push_back(o);
154 fvVector.push_back(m);
155 m_lagTable.set(lagName, fvVector);
156
157 SWSS_LOG_INFO("Add %s admin_status:%s oper_status:%s, mtu: %d",
158 lagName.c_str(), admin_state ? "up" : "down", oper_state ? "up" : "down", mtu);
159
160 bool lag_update = true;
161 /* Return when the team instance has already been tracked */
162 if (m_teamSelectables.find(lagName) != m_teamSelectables.end())
163 {
164 auto tsync = m_teamSelectables[lagName];
165 if (tsync->admin_state == admin_state && tsync->mtu == mtu)
166 return;
167 tsync->admin_state = admin_state;
168 tsync->mtu = mtu;
169 lag_update = false;
170 }
171
172 FieldValueTuple s("state", "ok");
173 fvVector.push_back(s);
174 if (m_warmstart)
175 {
176 m_stateLagTablePreserved[lagName] = fvVector;
177 }
178 else
179 {
180 m_stateLagTable.set(lagName, fvVector);
181 }
182
183 if (lag_update)
184 {
185 /* Create the team instance */
186 auto sync = make_shared<TeamPortSync>(lagName, ifindex, &m_lagMemberTable);
187 m_teamSelectables[lagName] = sync;
188 m_selectablesToAdd.insert(lagName);
189 }
190}

192void TeamSync::removeLag(const string &lagName)
193{
194 /* Delete all members */
195 auto selectable = m_teamSelectables[lagName];
196 for (auto it : selectable->m_lagMembers)
197 {
198 m_lagMemberTable.del(lagName + ":" + it.first);
199
200 SWSS_LOG_INFO("Remove member %s before removing LAG %s",
201 it.first.c_str(), lagName.c_str());
202 }
203
204 /* Delete the LAG */
205 m_lagTable.del(lagName);
206
207 SWSS_LOG_INFO("Remove LAG %s", lagName.c_str());
208
209 /* Return when the team instance hasn't been tracked before */
210 if (m_teamSelectables.find(lagName) == m_teamSelectables.end())
211 return;
212
213 if (m_warmstart)
214 {
215 m_stateLagTablePreserved.erase(lagName);
216 }
217 else
218 {
219 m_stateLagTable.del(lagName);
220 }
221
222 m_selectablesToRemove.insert(lagName);
223}

243TeamSync::TeamPortSync::TeamPortSync(const string &lagName, int ifindex,
244 ProducerStateTable *lagMemberTable) :
245 m_lagMemberTable(lagMemberTable),
246 m_lagName(lagName),
247 m_ifindex(ifindex)
248{
249 int count = 0;
250 int max_retries = 3;
251
252 while (true)
253 {
254 try
255 {
256 m_team = team_alloc();
257 if (!m_team)
258 {
259 throw system_error(make_error_code(errc::address_not_available),
260 "Unable to allocate team socket");
261 }
262
263 int err = team_init(m_team, ifindex);
264 if (err)
265 {
266 team_free(m_team);
267 m_team = NULL;
268 throw system_error(make_error_code(errc::address_not_available),
269 "Unable to initialize team socket");
270 }
271
272 err = team_change_handler_register(m_team, &gPortChangeHandler, this);
273 if (err)
274 {
275 team_free(m_team);
276 m_team = NULL;
277 throw system_error(make_error_code(errc::address_not_available),
278 "Unable to register port change event");
279 }
280
281 break;
282 }
283 catch (const system_error& e)
284 {
285 if (++count == max_retries)
286 {
287 throw;
288 }
289 else
290 {
291 SWSS_LOG_WARN("Failed to initialize team handler. LAG=%s error=%d:%s, attempt=%d",
292 lagName.c_str(), e.code().value(), e.what(), count);
293 }
294
295 sleep(1);
296 }
297 }
298
299 /* Sync LAG at first */
300 onChange();
301}

312int TeamSync::TeamPortSync::onChange()
313{
314 struct team_port *port;
315 map<string, bool> tmp_lag_members;
316
317 /* Check each port */
318 team_for_each_port(port, m_team)
319 {
320 uint32_t ifindex;
321 char ifname[MAX_IFNAME + 1] = {0};
322 bool enabled;
323
324 ifindex = team_get_port_ifindex(port);
325
326 /* Skip if interface is not found */
327 if (!team_ifindex2ifname(m_team, ifindex, ifname, MAX_IFNAME))
328 {
329 SWSS_LOG_INFO("Interface ifindex(%u) is not found", ifindex);
330 continue;
331 }
332
333 /* Skip the member that is removed from the LAG */
334 if (team_is_port_removed(port))
335 {
336 continue;
337 }
338
339 team_get_port_enabled(m_team, ifindex, &enabled);
340 tmp_lag_members[string(ifname)] = enabled;
341 }
342
343 /* Compare old and new LAG members and set/del accordingly */
344 for (auto it : tmp_lag_members)
345 {
346 if (m_lagMembers.find(it.first) == m_lagMembers.end() || it.second != m_lagMembers[it.first])
347 {
348 string key = m_lagName + ":" + it.first;
349 vector<FieldValueTuple> v;
350 FieldValueTuple l("status", it.second ? "enabled" : "disabled");
351 v.push_back(l);
352 m_lagMemberTable->set(key, v);
353
354 SWSS_LOG_INFO("Set LAG %s member %s with status %s",
355 m_lagName.c_str(), it.first.c_str(), it.second ? "enabled" : "disabled");
356 }
357 }
358
359 for (auto it : m_lagMembers)
360 {
361 if (tmp_lag_members.find(it.first) == tmp_lag_members.end())
362 {
363 string key = m_lagName + ":" + it.first;
364 m_lagMemberTable->del(key);
365
366 SWSS_LOG_INFO("Remove member %s from LAG %s",
367 it.first.c_str(), m_lagName.c_str());
368 }
369 }
370
371 /* Replace the old LAG members with the new ones */
372 m_lagMembers = tmp_lag_members;
373 return 0;
374}
