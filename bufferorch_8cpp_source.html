77void BufferOrch::initBufferReadyLists(DBConnector *applDb, DBConnector *confDb)
78{
79 /*
80 Map m_ready_list and m_port_ready_list_ref are designed to track whether the ports are "ready" from buffer's POV
81 by testing whether all configured buffer PGs and queues have been applied to SAI. The idea is:
82 - bufferorch read initial configuration and put them into m_port_ready_list_ref.
83 - A buffer pg or queue item will be put into m_ready_list once it is applied to SAI.
84 The rest of port initialization won't be started before the port being ready.
85
86 However, the items won't be applied to admin down ports in dynamic buffer model, which means the admin down ports won't be "ready"
87 The solution is:
88 - buffermgr to notify bufferorch explicitly to remove the PG and queue items configured on admin down ports
89 - bufferorch to add the items to m_ready_list on receiving notifications, which is an existing logic
90
91 Theoretically, the initial configuration should come from CONFIG_DB but APPL_DB is used for warm reboot, because:
92 - For cold/fast start, buffermgr is responsible for injecting items to APPL_DB
93 There is no guarantee that items in APPL_DB are ready when orchagent starts
94 - For warm reboot, APPL_DB is restored from the previous boot, which means they are ready when orchagent starts
95 In addition, bufferorch won't be notified removal of items on admin down ports during warm reboot
96 because buffermgrd hasn't been started yet.
97 Using APPL_DB means items of admin down ports won't be inserted into m_port_ready_list_ref
98 and guarantees the admin down ports always be ready in dynamic buffer model
99 */
100 SWSS_LOG_ENTER();
101
102 if (WarmStart::isWarmStart())
103 {
104 Table pg_table(applDb, APP_BUFFER_PG_TABLE_NAME);
105 initBufferReadyList(pg_table, false);
106
107 if(gMySwitchType == "voq")
108 {
109 Table queue_table(applDb, APP_BUFFER_QUEUE_TABLE_NAME);
110 initVoqBufferReadyList(queue_table, false);
111 }
112 else
113 {
114 Table queue_table(applDb, APP_BUFFER_QUEUE_TABLE_NAME);
115 initBufferReadyList(queue_table, false);
116 }
117 }
118 else
119 {
120 Table pg_table(confDb, CFG_BUFFER_PG_TABLE_NAME);
121 initBufferReadyList(pg_table, true);
122
123 if(gMySwitchType == "voq")
124 {
125 Table queue_table(confDb, CFG_BUFFER_QUEUE_TABLE_NAME);
126 initVoqBufferReadyList(queue_table, true);
127 }
128 else
129 {
130 Table queue_table(confDb, CFG_BUFFER_QUEUE_TABLE_NAME);
131 initBufferReadyList(queue_table, true);

276void BufferOrch::generateBufferPoolWatermarkCounterIdList(void)
277{
278 // This function will be called in FlexCounterOrch when field:value tuple "FLEX_COUNTER_STATUS":"enable"
279 // is received on buffer pool watermark key under table "FLEX_COUNTER_GROUP_TABLE"
280 // Because the SubscriberStateTable listens to the entire keyspace of "BUFFER_POOL_WATERMARK", any update
281 // to field value tuples under key "BUFFER_POOL_WATERMARK" will cause this tuple to be heard again
282 // To avoid resync the counter ID list a second time, we introduce a data member variable to mark whether
283 // this operation has already been done or not yet
284 if (m_isBufferPoolWatermarkCounterIdListGenerated)
285 {
286 return;
287 }
288
289 // Detokenize the SAI watermark stats to a string, separated by comma
290 string statList;
291 for (const auto &it : bufferPoolWatermarkStatIds)
292 {
293 statList += (sai_serialize_buffer_pool_stat(it) + list_item_delimiter);
294 }
295 if (!statList.empty())
296 {
297 statList.pop_back();
298 }
299
300 // Some platforms do not support buffer pool watermark clear operation on a particular pool
301 // Invoke the SAI clear_stats API per pool to query the capability from the API call return status
302 // We use bit mask to mark the clear watermark capability of each buffer pool. We use an unsigned int to place hold
303 // these bits. This assumes the total number of buffer pools to be no greater than 32, which should satisfy all use cases.
304 unsigned int noWmClrCapability = 0;
305 unsigned int bitMask = 1;
306 for (const auto &it : *(m_buffer_type_maps[APP_BUFFER_POOL_TABLE_NAME]))
307 {
308 sai_status_t status = sai_buffer_api->clear_buffer_pool_stats(
309 it.second.m_saiObjectId,
310 static_cast<uint32_t>(bufferPoolWatermarkStatIds.size()),
311 reinterpret_cast<const sai_stat_id_t *>(bufferPoolWatermarkStatIds.data()));
312 if (status == SAI_STATUS_NOT_SUPPORTED || status == SAI_STATUS_NOT_IMPLEMENTED)
313 {
314 SWSS_LOG_NOTICE("Clear watermark failed on %s, rv: %s", it.first.c_str(), sai_serialize_status(status).c_str());
315 noWmClrCapability |= bitMask;
316 }
317
318 bitMask <<= 1;
319 }
320
321 if (!noWmClrCapability)
322 {
323 vector<FieldValueTuple> fvs;
324
325 fvs.emplace_back(STATS_MODE_FIELD, STATS_MODE_READ_AND_CLEAR);
326 m_flexCounterGroupTable->set(BUFFER_POOL_WATERMARK_STAT_COUNTER_FLEX_COUNTER_GROUP, fvs);
327 }
328
329 // Push buffer pool watermark COUNTER_ID_LIST to FLEX_COUNTER_TABLE on a per buffer pool basis
330 vector<FieldValueTuple> fvTuples;
331 fvTuples.emplace_back(BUFFER_POOL_COUNTER_ID_LIST, statList);
332 bitMask = 1;
333 for (const auto &it : *(m_buffer_type_maps[APP_BUFFER_POOL_TABLE_NAME]))
334 {
335 string key = BUFFER_POOL_WATERMARK_STAT_COUNTER_FLEX_COUNTER_GROUP ":" + sai_serialize_object_id(it.second.m_saiObjectId);
336
337 if (noWmClrCapability)
338 {
339 string stats_mode = STATS_MODE_READ_AND_CLEAR;
340 if (noWmClrCapability & bitMask)
341 {
342 stats_mode = STATS_MODE_READ;
343 }
344 fvTuples.emplace_back(STATS_MODE_FIELD, stats_mode);
345
346 m_flexCounterTable->set(key, fvTuples);
347 fvTuples.pop_back();
348 bitMask <<= 1;
349 }
350 else
351 {
352 m_flexCounterTable->set(key, fvTuples);
353 }
354 }
355
356 m_isBufferPoolWatermarkCounterIdListGenerated = true;
357}

367task_process_status BufferOrch::processBufferPool(KeyOpFieldsValuesTuple &tuple)
368{
369 SWSS_LOG_ENTER();
370 sai_status_t sai_status;
371 sai_object_id_t sai_object = SAI_NULL_OBJECT_ID;
372 string map_type_name = APP_BUFFER_POOL_TABLE_NAME;
373 string object_name = kfvKey(tuple);
374 string op = kfvOp(tuple);
375
376 SWSS_LOG_DEBUG("object name:%s", object_name.c_str());
377 if (m_buffer_type_maps[map_type_name]->find(object_name) != m_buffer_type_maps[map_type_name]->end())
378 {
379 sai_object = (*(m_buffer_type_maps[map_type_name]))[object_name].m_saiObjectId;
380 SWSS_LOG_DEBUG("found existing object:%s of type:%s", object_name.c_str(), map_type_name.c_str());
381 if ((*(m_buffer_type_maps[map_type_name]))[object_name].m_pendingRemove && op == SET_COMMAND)
382 {
383 SWSS_LOG_NOTICE("Entry %s %s is pending remove, need retry", map_type_name.c_str(), object_name.c_str());
384 return task_process_status::task_need_retry;
385 }
386 }
387 SWSS_LOG_DEBUG("processing command:%s", op.c_str());
388
389 if (op == SET_COMMAND)
390 {
391 vector<sai_attribute_t> attribs;
392 for (auto i = kfvFieldsValues(tuple).begin(); i != kfvFieldsValues(tuple).end(); i++)
393 {
394 string field = fvField(*i);
395 string value = fvValue(*i);
396
397 SWSS_LOG_DEBUG("field:%s, value:%s", field.c_str(), value.c_str());
398 sai_attribute_t attr;
399 if (field == buffer_size_field_name)
400 {
401 attr.id = SAI_BUFFER_POOL_ATTR_SIZE;
402 attr.value.u64 = (uint64_t)stoul(value);
403 attribs.push_back(attr);
404 }
405 else if (field == buffer_pool_type_field_name)
406 {
407 string type = value;
408
409 if (SAI_NULL_OBJECT_ID != sai_object)
410 {
411 // We should skip the pool type when setting a pool's attribute because it's create only
412 // when setting a pool's attribute.
413 SWSS_LOG_INFO("Skip setting buffer pool type %s for pool %s", type.c_str(), object_name.c_str());
414 continue;
415 }
416
417 if (type == buffer_value_ingress)
418 {
419 attr.value.u32 = SAI_BUFFER_POOL_TYPE_INGRESS;
420 }
421 else if (type == buffer_value_egress)
422 {
423 attr.value.u32 = SAI_BUFFER_POOL_TYPE_EGRESS;
424 }
425 else if (type == buffer_value_both)
426 {
427 attr.value.u32 = SAI_BUFFER_POOL_TYPE_BOTH;
428 }
429 else
430 {
431 SWSS_LOG_ERROR("Unknown pool type specified:%s", type.c_str());
432 return task_process_status::task_invalid_entry;
433 }
434 attr.id = SAI_BUFFER_POOL_ATTR_TYPE;
435 attribs.push_back(attr);
436 }
437 else if (field == buffer_pool_mode_field_name)
438 {
439 string mode = value;
440
441 if (SAI_NULL_OBJECT_ID != sai_object)
442 {
443 // We should skip the pool mode when setting a pool's attribute because it's create only.
444 SWSS_LOG_INFO("Skip setting buffer pool mode %s for pool %s", mode.c_str(), object_name.c_str());
445 continue;
446 }
447
448 if (mode == buffer_pool_mode_dynamic_value)
449 {
450 attr.value.u32 = SAI_BUFFER_POOL_THRESHOLD_MODE_DYNAMIC;
451 }
452 else if (mode == buffer_pool_mode_static_value)
453 {
454 attr.value.u32 = SAI_BUFFER_POOL_THRESHOLD_MODE_STATIC;
455 }
456 else
457 {
458 SWSS_LOG_ERROR("Unknown pool mode specified:%s", mode.c_str());
459 return task_process_status::task_invalid_entry;
460 }
461 attr.id = SAI_BUFFER_POOL_ATTR_THRESHOLD_MODE;
462 attribs.push_back(attr);
463 }
464 else if (field == buffer_pool_xoff_field_name)
465 {
466 attr.value.u64 = (uint64_t)stoul(value);
467 attr.id = SAI_BUFFER_POOL_ATTR_XOFF_SIZE;
468 attribs.push_back(attr);
469 }
470 else
471 {
472 SWSS_LOG_ERROR("Unknown pool field specified:%s, ignoring", field.c_str());
473 continue;
474 }
475 }
476 if (SAI_NULL_OBJECT_ID != sai_object)
477 {
478 for (auto &attribute : attribs)
479 {
480 sai_status = sai_buffer_api->set_buffer_pool_attribute(sai_object, &attribute);
481 if (SAI_STATUS_ATTR_NOT_IMPLEMENTED_0 == sai_status)
482 {
483 SWSS_LOG_NOTICE("Buffer pool SET for name:%s, sai object:%" PRIx64 ", not implemented. status:%d. Ignoring it", object_name.c_str(), sai_object, sai_status);
484 return task_process_status::task_ignore;
485 }
486 else if (SAI_STATUS_SUCCESS != sai_status)
487 {
488 SWSS_LOG_ERROR("Failed to modify buffer pool, name:%s, sai object:%" PRIx64 ", status:%d", object_name.c_str(), sai_object, sai_status);
489 task_process_status handle_status = handleSaiSetStatus(SAI_API_BUFFER, sai_status);
490 if (handle_status != task_process_status::task_success)
491 {
492 return handle_status;
493 }
494 }
495 }
496 SWSS_LOG_DEBUG("Modified existing pool:%" PRIx64 ", type:%s name:%s ", sai_object, map_type_name.c_str(), object_name.c_str());
497 }
498 else
499 {
500 sai_status = sai_buffer_api->create_buffer_pool(&sai_object, gSwitchId, (uint32_t)attribs.size(), attribs.data());
501 if (SAI_STATUS_SUCCESS != sai_status)
502 {
503 SWSS_LOG_ERROR("Failed to create buffer pool %s with type %s, rv:%d", object_name.c_str(), map_type_name.c_str(), sai_status);
504 task_process_status handle_status = handleSaiCreateStatus(SAI_API_BUFFER, sai_status);
505 if (handle_status != task_process_status::task_success)
506 {
507 return handle_status;
508 }
509 }
510
511 (*(m_buffer_type_maps[map_type_name]))[object_name].m_saiObjectId = sai_object;
512 (*(m_buffer_type_maps[map_type_name]))[object_name].m_pendingRemove = false;
513 SWSS_LOG_NOTICE("Created buffer pool %s with type %s", object_name.c_str(), map_type_name.c_str());
514 // Here we take the PFC watchdog approach to update the COUNTERS_DB metadata (e.g., PFC_WD_DETECTION_TIME per queue)
515 // at initialization (creation and registration phase)
516 // Specifically, we push the buffer pool name to oid mapping upon the creation of the oid
517 // In pg and queue case, this mapping installment is deferred to FlexCounterOrch at a reception of field
518 // "FLEX_COUNTER_STATUS"
519 m_countersDb->hset(COUNTERS_BUFFER_POOL_NAME_MAP, object_name, sai_serialize_object_id(sai_object));
520 }
521 }
522 else if (op == DEL_COMMAND)
523 {
524 if (isObjectBeingReferenced(m_buffer_type_maps, map_type_name, object_name))
525 {
526 auto hint = objectReferenceInfo(m_buffer_type_maps, map_type_name, object_name);
527 SWSS_LOG_NOTICE("Can't remove object %s due to being referenced (%s)", object_name.c_str(), hint.c_str());
528 (*(m_buffer_type_maps[map_type_name]))[object_name].m_pendingRemove = true;
529
530 return task_process_status::task_need_retry;
531 }
532
533 if (SAI_NULL_OBJECT_ID != sai_object)
534 {
535 clearBufferPoolWatermarkCounterIdList(sai_object);
536 sai_status = sai_buffer_api->remove_buffer_pool(sai_object);
537 if (SAI_STATUS_SUCCESS != sai_status)
538 {
539 SWSS_LOG_ERROR("Failed to remove buffer pool %s with type %s, rv:%d", object_name.c_str(), map_type_name.c_str(), sai_status);
540 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_BUFFER, sai_status);
541 if (handle_status != task_process_status::task_success)
542 {
543 return handle_status;
544 }
545 }
546 SWSS_LOG_NOTICE("Removed buffer pool %s with type %s", object_name.c_str(), map_type_name.c_str());
547 }
548 auto it_to_delete = (m_buffer_type_maps[map_type_name])->find(object_name);
549 (m_buffer_type_maps[map_type_name])->erase(it_to_delete);
550 m_countersDb->hdel(COUNTERS_BUFFER_POOL_NAME_MAP, object_name);
551 }
552 else
553 {
554 SWSS_LOG_ERROR("Unknown operation type %s", op.c_str());
555 return task_process_status::task_invalid_entry;
556 }
557 return task_process_status::task_success;
558}

763/*
764 Input sample "BUFFER_QUEUE|Ethernet4,Ethernet45|10-15" or
765 "BUFFER_QUEUE|STG01-0101-0400-01T2-LC6|ASIC0|Ethernet4|10-15"
766*/
767task_process_status BufferOrch::processQueue(KeyOpFieldsValuesTuple &tuple)
768{
769 SWSS_LOG_ENTER();
770 sai_object_id_t sai_buffer_profile;
771 string buffer_profile_name;
772 const string key = kfvKey(tuple);
773 string op = kfvOp(tuple);
774 vector<string> tokens;
775 sai_uint32_t range_low, range_high;
776 bool need_update_sai = true;
777
778 SWSS_LOG_DEBUG("Processing:%s", key.c_str());
779 tokens = tokenize(key, delimiter);
780
781 vector<string> port_names;
782 if (gMySwitchType == "voq")
783 {
784 if (tokens.size() != 4)
785 {
786 SWSS_LOG_ERROR("malformed key:%s. Must contain 4 tokens", key.c_str());
787 return task_process_status::task_invalid_entry;
788 }
789
790 port_names = tokenize(tokens[0] + config_db_key_delimiter + tokens[1] + config_db_key_delimiter + tokens[2], list_item_delimiter);
791 if (!parseIndexRange(tokens[3], range_low, range_high))
792 {
793 return task_process_status::task_invalid_entry;
794 }
795 }
796 else
797 {
798 if (tokens.size() != 2)
799 {
800 SWSS_LOG_ERROR("malformed key:%s. Must contain 2 tokens", key.c_str());
801 return task_process_status::task_invalid_entry;
802 }
803
804 port_names = tokenize(tokens[0], list_item_delimiter);
805 if (!parseIndexRange(tokens[1], range_low, range_high))
806 {
807 return task_process_status::task_invalid_entry;
808 }
809 }
810
811 if (op == SET_COMMAND)
812 {
813 ref_resolve_status resolve_result = resolveFieldRefValue(m_buffer_type_maps, buffer_profile_field_name,
814 buffer_to_ref_table_map.at(buffer_profile_field_name), tuple,
815 sai_buffer_profile, buffer_profile_name);
816 if (ref_resolve_status::success != resolve_result)
817 {
818 if (ref_resolve_status::not_resolved == resolve_result)
819 {
820 SWSS_LOG_INFO("Missing or invalid queue buffer profile reference specified");
821 return task_process_status::task_need_retry;
822 }
823
824 SWSS_LOG_ERROR("Resolving queue profile reference failed");
825 return task_process_status::task_failed;
826 }
827
828 SWSS_LOG_NOTICE("Set buffer queue %s to %s", key.c_str(), buffer_profile_name.c_str());
829
830 setObjectReference(m_buffer_type_maps, APP_BUFFER_QUEUE_TABLE_NAME, key, buffer_profile_field_name, buffer_profile_name);
831 }
832 else if (op == DEL_COMMAND)
833 {
834 auto &typemap = (*m_buffer_type_maps[APP_BUFFER_QUEUE_TABLE_NAME]);
835 if (typemap.find(key) == typemap.end())
836 {
837 SWSS_LOG_INFO("%s doesn't not exist, don't need to notfiy SAI", key.c_str());
838 need_update_sai = false;
839 }
840 sai_buffer_profile = SAI_NULL_OBJECT_ID;
841 SWSS_LOG_NOTICE("Remove buffer queue %s", key.c_str());
842 removeObject(m_buffer_type_maps, APP_BUFFER_QUEUE_TABLE_NAME, key);
843 }
844 else
845 {
846 SWSS_LOG_ERROR("Unknown operation type %s", op.c_str());
847 return task_process_status::task_invalid_entry;
848 }
849
850 sai_attribute_t attr;
851 attr.id = SAI_QUEUE_ATTR_BUFFER_PROFILE_ID;
852 attr.value.oid = sai_buffer_profile;
853 for (string port_name : port_names)
854 {
855 Port port;
856 SWSS_LOG_DEBUG("processing port:%s", port_name.c_str());
857 if (!gPortsOrch->getPort(port_name, port))
858 {
859 SWSS_LOG_ERROR("Port with alias:%s not found", port_name.c_str());
860 return task_process_status::task_invalid_entry;
861 }
862 for (size_t ind = range_low; ind <= range_high; ind++)
863 {
864 SWSS_LOG_DEBUG("processing queue:%zd", ind);
865 sai_object_id_t queue_id;
866
867 if (gMySwitchType == "voq")
868 {
869 std :: vector<sai_object_id_t> queue_ids = gPortsOrch->getPortVoQIds(port);
870 if (queue_ids.size() <= ind)
871 {
872 SWSS_LOG_ERROR("Invalid voq index specified:%zd", ind);
873 return task_process_status::task_invalid_entry;
874 }
875 queue_id = queue_ids[ind];
876 }
877 else
878 {
879 if (port.m_queue_ids.size() <= ind)
880 {
881 SWSS_LOG_ERROR("Invalid queue index specified:%zd", ind);
882 return task_process_status::task_invalid_entry;
883 }
884 if (port.m_queue_lock[ind])
885 {
886 SWSS_LOG_WARN("Queue %zd on port %s is locked, will retry", ind, port_name.c_str());
887 return task_process_status::task_need_retry;
888 }
889 queue_id = port.m_queue_ids[ind];
890 }
891
892 if (need_update_sai)
893 {
894 SWSS_LOG_DEBUG("Applying buffer profile:0x%" PRIx64 " to queue index:%zd, queue sai_id:0x%" PRIx64, sai_buffer_profile, ind, queue_id);
895 sai_status_t sai_status = sai_queue_api->set_queue_attribute(queue_id, &attr);
896 if (sai_status != SAI_STATUS_SUCCESS)
897 {
898 SWSS_LOG_ERROR("Failed to set queue's buffer profile attribute, status:%d", sai_status);
899 task_process_status handle_status = handleSaiSetStatus(SAI_API_QUEUE, sai_status);
900 if (handle_status != task_process_status::task_success)
901 {
902 return handle_status;
903 }
904 }
905 // create/remove a port queue counter for the queue buffer
906 else
907 {
908 auto flexCounterOrch = gDirectory.get<FlexCounterOrch*>();
909 auto queues = tokens[1];
910 if (op == SET_COMMAND &&
911 (flexCounterOrch->getQueueCountersState() || flexCounterOrch->getQueueWatermarkCountersState()))
912 {
913 gPortsOrch->createPortBufferQueueCounters(port, queues);
914 }
915 else if (op == DEL_COMMAND &&
916 (flexCounterOrch->getQueueCountersState() || flexCounterOrch->getQueueWatermarkCountersState()))
917 {
918 gPortsOrch->removePortBufferQueueCounters(port, queues);
919 }
920 }
921 }
922
923 /* when we apply buffer configuration we need to increase the ref counter of this port
924 * or decrease the ref counter for this port when we remove buffer cfg
925 * so for each priority cfg in each port we will increase/decrease the ref counter
926 * also we need to know when the set command is for creating a buffer cfg or modifying buffer cfg -
927 * we need to increase ref counter only on create flow.
928 * so we added a map that will help us to know what was the last command for this port and priority -
929 * if the last command was set command then it is a modify command and we dont need to increase the buffer counter
930 * all other cases (no last command exist or del command was the last command) it means that we need to increase the ref counter */
931 if (op == SET_COMMAND)
932 {
933 if (queue_port_flags[port_name][ind] != SET_COMMAND)
934 {
935 /* if the last operation was not "set" then it's create and not modify - need to increase ref counter */
936 gPortsOrch->increasePortRefCount(port_name);
937 }
938 }
939 else if (op == DEL_COMMAND)
940 {
941 if (queue_port_flags[port_name][ind] == SET_COMMAND)
942 {
943 /* we need to decrease ref counter only if the last operation was "SET_COMMAND" */
944 gPortsOrch->decreasePortRefCount(port_name);
945 }
946 }
947 else
948 {
949 SWSS_LOG_ERROR("operation value is not SET or DEL (op = %s)", op.c_str());
950 return task_process_status::task_invalid_entry;
951 }
952 /* save the last command (set or delete) */
953 queue_port_flags[port_name][ind] = op;
954
955 }
956 }
957
958 if (m_ready_list.find(key) != m_ready_list.end())
959 {
960 m_ready_list[key] = true;
961 }
962 else
963 {
964 // If a buffer queue profile is not in the initial CONFIG_DB BUFFER_QUEUE table
965 // at BufferOrch object instantiation, it is considered being applied
966 // at run time, and, in this case, is not tracked in the m_ready_list. It is up to
967 // the application to guarantee the set order that the buffer queue profile
968 // should be applied to a physical port before the physical port is brought up to
969 // carry traffic. Here, we alert to application through syslog when such a wrong
970 // set order is detected.
971 for (const auto &port_name : port_names)
972 {
973 if (gPortsOrch->isPortAdminUp(port_name)) {
974 SWSS_LOG_WARN("Queue profile '%s' applied after port %s is up", key.c_str(), port_name.c_str());
975 }
976 }
977 }
978
979 return task_process_status::task_success;
980}
