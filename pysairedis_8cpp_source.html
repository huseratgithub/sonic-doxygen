24static int profile_get_next_value (

209sai_pointer_t sai_get_notification_pointer(
210 sai_attr_id_t id,
211 PyObject*callback)
212{
213 if (!PyCallable_Check(callback))
214 {
215 SWSS_LOG_THROW("second parameter must be python callable object");
216 }
217
218 Py_XINCREF(callback); // inc ref on current callback
219
220 switch(id)
221 {
222 case SAI_SWITCH_ATTR_SWITCH_STATE_CHANGE_NOTIFY:
223 Py_XDECREF(py_switch_state_change_notification);
224 py_switch_state_change_notification = callback;
225 return (void*)&sai_switch_state_change_notification;
226
227 case SAI_SWITCH_ATTR_SWITCH_SHUTDOWN_REQUEST_NOTIFY:
228 Py_XDECREF(py_switch_shutdown_request_notification);
229 py_switch_shutdown_request_notification = callback;
230 return (void*)&sai_switch_shutdown_request_notification;
231
232 case SAI_SWITCH_ATTR_FDB_EVENT_NOTIFY:
233 Py_XDECREF(py_fdb_event_notification);
234 py_fdb_event_notification = callback;
235 return (void*)&sai_fdb_event_notification;
236
237 case SAI_SWITCH_ATTR_PORT_STATE_CHANGE_NOTIFY:
238 Py_XDECREF(py_port_state_change_notification);
239 py_port_state_change_notification = callback;
240 return (void*)&sai_port_state_change_notification;
241
242 case SAI_SWITCH_ATTR_QUEUE_PFC_DEADLOCK_NOTIFY:
243 Py_XDECREF(py_queue_pfc_deadlock_notification);
244 py_queue_pfc_deadlock_notification = callback;
245 return (void*)&sai_queue_pfc_deadlock_notification;
