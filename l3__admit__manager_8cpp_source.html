77void L3AdmitManager::drain()
78{
79 SWSS_LOG_ENTER();
80
81 for (const auto &key_op_fvs_tuple : m_entries)
82 {
83 std::string table_name;
84 std::string key;
85 parseP4RTKey(kfvKey(key_op_fvs_tuple), &table_name, &key);

87
88 ReturnCode status;
89 auto app_db_entry_or = deserializeP4L3AdmitAppDbEntry(key, attributes);

99 }
100 auto &app_db_entry = *app_db_entry_or;
101

104 app_db_entry.port_name, app_db_entry.priority);
105
106 // Fulfill the operation.
107 const std::string &operation = kfvOp(key_op_fvs_tuple);
108 if (operation == SET_COMMAND)
109 {
110 auto *l3_admit_entry = getL3AdmitEntry(l3_admit_key);
111 if (l3_admit_entry == nullptr)
112 {
113 // Create new l3 admit.

119 status = ReturnCode(StatusCode::SWSS_RC_SUCCESS)
120 << "L3 Admit entry with the same key received: " << QuotedVar(l3_admit_key);
121 }
122 }
123 else if (operation == DEL_COMMAND)

125 // Delete l3 admit.
126 status = processDeleteRequest(l3_admit_key);
127 }
128 else
129 {
130 status = ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Unknown operation type " << QuotedVar(operation);
131 SWSS_LOG_ERROR("%s", status.message().c_str());

133 m_publisher->publish(APP_P4RT_TABLE_NAME, kfvKey(key_op_fvs_tuple), kfvFieldsValues(key_op_fvs_tuple), status,

155ReturnCodeOr<P4L3AdmitAppDbEntry> L3AdmitManager::deserializeP4L3AdmitAppDbEntry(
156 const std::string &key, const std::vector<swss::FieldValueTuple> &attributes)
157{
158 SWSS_LOG_ENTER();
159
160 P4L3AdmitAppDbEntry app_db_entry = {};
161
162 try
163 {
164 nlohmann::json j = nlohmann::json::parse(key);
165 // "match/dst_mac":"00:02:03:04:00:00&ff:ff:ff:ff:00:00"
166 if (j.find(prependMatchField(p4orch::kDstMac)) != j.end())
167 {
168 std::string dst_mac_data_and_mask = j[prependMatchField(p4orch::kDstMac)];
169 const auto &data_and_mask = swss::tokenize(dst_mac_data_and_mask, p4orch::kDataMaskDelimiter);
170 app_db_entry.mac_address_data = swss::MacAddress(trim(data_and_mask[0]));
171 if (data_and_mask.size() > 1)
172 {
173 app_db_entry.mac_address_mask = swss::MacAddress(trim(data_and_mask[1]));
174 }
175 else
176 {
177 app_db_entry.mac_address_mask = swss::MacAddress("ff:ff:ff:ff:ff:ff");
178 }
179 }
180 else
181 {
182 // P4RT set "don't care" value for dst_mac - mask should be all 0
183 app_db_entry.mac_address_data = swss::MacAddress("00:00:00:00:00:00");
184 app_db_entry.mac_address_mask = swss::MacAddress("00:00:00:00:00:00");
185 }
186
187 // "priority":2030
188 auto priority_j = j[p4orch::kPriority];
189 if (!priority_j.is_number_unsigned())
190 {
191 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
192 << "Invalid l3 admit entry priority type: should be uint32_t";
193 }
194 app_db_entry.priority = static_cast<uint32_t>(priority_j);
195
196 // "match/in_port":"Ethernet0"
197 if (j.find(prependMatchField(p4orch::kInPort)) != j.end())
198 {
199 app_db_entry.port_name = j[prependMatchField(p4orch::kInPort)];
200 }
201 }
202 catch (std::exception &ex)
203 {
204 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to deserialize l3 admit key";
205 }
206
207 for (const auto &it : attributes)
208 {
209 const auto &field = fvField(it);
210 const auto &value = fvValue(it);
211 // "action": "admit_to_l3"
212 if (field == p4orch::kAction)
213 {
214 if (value != p4orch::kL3AdmitAction)
215 {
216 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
217 << "Unexpected action " << QuotedVar(value) << " in L3 Admit table entry";
218 }
219 }
220 else if (field != p4orch::kControllerMetadata)
221 {
222 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
223 << "Unexpected field " << QuotedVar(field) << " in L3 Admit table entry";
224 }
225 }
226
227 return app_db_entry;
228}

230ReturnCode L3AdmitManager::processAddRequest(const P4L3AdmitAppDbEntry &app_db_entry, const std::string &l3_admit_key)
231{
232 SWSS_LOG_ENTER();
233
234 // Check the existence of the l3 admit in l3 admit manager and centralized
235 // mapper.
236 if (getL3AdmitEntry(l3_admit_key) != nullptr)

249 auto status = createL3Admit(l3_admit_entry);
250 if (!status.ok())
251 {
252 SWSS_LOG_ERROR("Failed to create l3 admit with key %s", QuotedVar(l3_admit_key).c_str());
253 return status;
254 }
255 // Increase reference count to port
256 if (!l3_admit_entry.port_name.empty())

259 }
260 // Add created entry to internal table.
261 m_l3AdmitTable.emplace(l3_admit_key, l3_admit_entry);
262
263 // Add the key to OID map to centralized mapper.
264 m_p4OidMapper->setOID(SAI_OBJECT_TYPE_MY_MAC, l3_admit_key, l3_admit_entry.l3_admit_oid);
265 return status;

285ReturnCode L3AdmitManager::processDeleteRequest(const std::string &l3_admit_key)
286{
287 SWSS_LOG_ENTER();
288
289 // Check the existence of the l3 admit in l3 admit manager and centralized
290 // mapper.
291 auto *l3_admit_entry = getL3AdmitEntry(l3_admit_key);
292 if (l3_admit_entry == nullptr)
293 {
294 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_NOT_FOUND)
295 << "l3 admit with key " << QuotedVar(l3_admit_key)
296 << " does not exist in l3 admit manager");
297 }
298
299 // Check if there is anything referring to the l3 admit before deletion.
300 uint32_t ref_count;
301 if (!m_p4OidMapper->getRefCount(SAI_OBJECT_TYPE_MY_MAC, l3_admit_key, &ref_count))
302 {
303 RETURN_INTERNAL_ERROR_AND_RAISE_CRITICAL("Failed to get reference count for l3 admit "
304 << QuotedVar(l3_admit_key));
305 }
306 if (ref_count > 0)
307 {
308 LOG_ERROR_AND_RETURN(ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
309 << "l3 admit " << QuotedVar(l3_admit_key)
310 << " referenced by other objects (ref_count = " << ref_count);
311 }
312
313 // Call SAI API
314 auto status = removeL3Admit(l3_admit_key);
315 if (!status.ok())
316 {
317 SWSS_LOG_ERROR("Failed to remove l3 admit with key %s", QuotedVar(l3_admit_key).c_str());
318 return status;
319 }
320
321 // Decrease reference count to port
322 if (!l3_admit_entry->port_name.empty())
323 {
324 gPortsOrch->decreasePortRefCount(l3_admit_entry->port_name);
325 }
326 // Remove the key to OID map to centralized mapper.
327 m_p4OidMapper->eraseOID(SAI_OBJECT_TYPE_MY_MAC, l3_admit_key);
328
329 // Remove the entry from internal table.
330 m_l3AdmitTable.erase(l3_admit_key);
331 return status;
332}

345std::string L3AdmitManager::verifyState(const std::string &key, const std::vector<swss::FieldValueTuple> &tuple)
346{
347 SWSS_LOG_ENTER();
348
349 auto pos = key.find_first_of(kTableKeyDelimiter);
350 if (pos == std::string::npos)
351 {
352 return std::string("Invalid key: ") + key;
353 }
354 std::string p4rt_table = key.substr(0, pos);
355 std::string p4rt_key = key.substr(pos + 1);
356 if (p4rt_table != APP_P4RT_TABLE_NAME)
357 {
358 return std::string("Invalid key: ") + key;
359 }
360 std::string table_name;
361 std::string key_content;
362 parseP4RTKey(p4rt_key, &table_name, &key_content);
363 if (table_name != APP_P4RT_L3_ADMIT_TABLE_NAME)
364 {
365 return std::string("Invalid key: ") + key;
366 }
367
368 ReturnCode status;
369 auto app_db_entry_or = deserializeP4L3AdmitAppDbEntry(key_content, tuple);
370 if (!app_db_entry_or.ok())
371 {
372 status = app_db_entry_or.status();
373 std::stringstream msg;
374 msg << "Unable to deserialize key " << QuotedVar(key) << ": " << status.message();
375 return msg.str();
376 }
377 auto &app_db_entry = *app_db_entry_or;
378 const std::string l3_admit_key = KeyGenerator::generateL3AdmitKey(
379 app_db_entry.mac_address_data, app_db_entry.mac_address_mask, app_db_entry.port_name, app_db_entry.priority);
380 auto *l3_admit_entry = getL3AdmitEntry(l3_admit_key);
381 if (l3_admit_entry == nullptr)
382 {
383 std::stringstream msg;
384 msg << "No entry found with key " << QuotedVar(key);
385 return msg.str();
386 }
387
388 std::string cache_result = verifyStateCache(app_db_entry, l3_admit_entry);
389 std::string asic_db_result = verifyStateAsicDb(l3_admit_entry);
390 if (cache_result.empty())
391 {
392 return asic_db_result;
393 }
394 if (asic_db_result.empty())
395 {
396 return cache_result;
397 }
398 return cache_result + "; " + asic_db_result;
