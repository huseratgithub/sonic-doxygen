57void Request::parseKey(const KeyOpFieldsValuesTuple& request)
58{
59 full_key_ = kfvKey(request);
60
61 // split the key by separator
62 std::vector<std::string> key_items;
63 size_t key_item_start = 0;
64 size_t key_item_end = full_key_.find(key_separator_);
65 while (key_item_end != std::string::npos)
66 {
67 key_items.push_back(full_key_.substr(key_item_start, key_item_end - key_item_start));
68 key_item_start = key_item_end + 1;
69 key_item_end = full_key_.find(key_separator_, key_item_start);
70 }
71 key_items.push_back(full_key_.substr(key_item_start, full_key_.length()));
72
73 /*
74 * Attempt to parse an IPv6 address only if the following conditions are met:
75 * - The key separator is ":"
76 * - The above logic will already correctly parse IPv6 addresses using other key separators
77 * - Special consideration is only needed for ":" key separators since IPv6 addresses also use ":" as the field separator
78 * - The number of parsed key items exceeds the number of expected key items
79 * - If we have too many key items and the last key item is supposed to be an IP or prefix, there is a chance that it was an
80 * IPv6 address that got segmented during parsing
81 * - The above logic will always break an IPv6 address into at least 2 parts, so if an IPv6 address has been parsed incorrectly it will
82 * always increase the number of key items
83 * - The last key item is an IP address or prefix
84 * - This runs under the assumption that an IPv6 address, if present, will always be the last key item
85 */
86 if (key_separator_ == ':' and
87 key_items.size() > number_of_key_items_ and
88 (request_description_.key_item_types.back() == REQ_T_IP or request_description_.key_item_types.back() == REQ_T_IP_PREFIX))
89 {
90 // Remove key_items so that key_items.size() is correct, then assemble the removed items into an IPv6 address
91 std::vector<std::string> ip_addr_groups(--key_items.begin() + number_of_key_items_, key_items.end());
92 key_items.erase(--key_items.begin() + number_of_key_items_, key_items.end());
93
94 std::string ip_string;
95
96 for (const auto &i : ip_addr_groups)
97 {
98 ip_string += i + ":";
99 }
100 ip_string.pop_back(); // remove extra ":" from end of string
101
102 key_items.push_back(ip_string);
103 }
104 if (key_items.size() != number_of_key_items_)
105 {
106 throw std::invalid_argument(std::string("Wrong number of key items. Expected ")
107 + std::to_string(number_of_key_items_)
108 + std::string(" item(s). Key: '")
109 + full_key_
110 + std::string("'"));
111 }
112
113 // check types of the key items
114 for (int i = 0; i < static_cast<int>(number_of_key_items_); i++)
115 {
116 switch(request_description_.key_item_types[i])
117 {
118 case REQ_T_STRING:
119 key_item_strings_[i] = key_items[i];
120 break;
121 case REQ_T_MAC_ADDRESS:
122 key_item_mac_addresses_[i] = parseMacAddress(key_items[i]);
123 break;
124 case REQ_T_IP:
125 key_item_ip_addresses_[i] = parseIpAddress(key_items[i]);
126 break;
127 case REQ_T_IP_PREFIX:
128 key_item_ip_prefix_[i] = parseIpPrefix(key_items[i]);
129 break;
130 case REQ_T_UINT:
131 key_item_uint_[i] = parseUint(key_items[i]);
132 break;
133 default:
134 throw std::logic_error(std::string("Not implemented key type parser. Key '")
135 + full_key_
136 + std::string("'. Key item:")
137 + key_items[i]);
138 }
139 }

142void Request::parseAttrs(const KeyOpFieldsValuesTuple& request)
143{
144 const auto not_found = std::end(request_description_.attr_item_types);
145
146 for (auto i = kfvFieldsValues(request).begin();
147 i != kfvFieldsValues(request).end(); i++)
148 {
149 if (fvField(*i) == "empty" || fvField(*i) == "NULL")
150 {
151 // if name of the attribute is 'empty' or 'NULL', just skip it.
152 // it's used when we don't have any attributes, but we have to provide one for redis
153 continue;
154 }
155 const auto item = request_description_.attr_item_types.find(fvField(*i));
156 if (item == not_found)
157 {
158 throw std::invalid_argument(std::string("Unknown attribute name: ") + fvField(*i));
159 }
160 attr_names_.insert(fvField(*i));
161 switch(item->second)
162 {
163 case REQ_T_STRING:
164 attr_item_strings_[fvField(*i)] = fvValue(*i);
165 break;
166 case REQ_T_BOOL:
167 attr_item_bools_[fvField(*i)] = parseBool(fvValue(*i));
168 break;
169 case REQ_T_MAC_ADDRESS:
170 attr_item_mac_addresses_[fvField(*i)] = parseMacAddress(fvValue(*i));
171 break;
172 case REQ_T_PACKET_ACTION:
173 attr_item_packet_actions_[fvField(*i)] = parsePacketAction(fvValue(*i));
174 break;
175 case REQ_T_VLAN:
176 attr_item_vlan_[fvField(*i)] = parseVlan(fvValue(*i));
177 break;
178 case REQ_T_IP:
179 attr_item_ip_[fvField(*i)] = parseIpAddress(fvValue(*i));
180 break;
181 case REQ_T_IP_PREFIX:
182 attr_item_ip_prefix_[fvField(*i)] = parseIpPrefix(fvValue(*i));
183 break;
184 case REQ_T_UINT:
185 attr_item_uint_[fvField(*i)] = parseUint(fvValue(*i));
186 break;
187 case REQ_T_SET:
188 attr_item_set_[fvField(*i)] = parseSet(fvValue(*i));
189 break;
190 case REQ_T_MAC_ADDRESS_LIST:
191 attr_item_mac_addresses_list_[fvField(*i)] = parseMacAddressList(fvValue(*i));
192 break;
193 case REQ_T_IP_LIST:
194 attr_item_ip_list_[fvField(*i)] = parseIpAddressList(fvValue(*i));
195 break;
196 case REQ_T_UINT_LIST:
197 attr_item_uint_list_[fvField(*i)] = parseUintList(fvValue(*i));
198 break;
199 default:
200 throw std::logic_error(std::string("Not implemented attribute type parser for attribute:") + fvField(*i));
201 }
202 }
203
204 if (operation_ == DEL_COMMAND && attr_names_.size() > 0)
205 {
206 throw std::invalid_argument("Delete operation request contains attributes");
207 }
208
209 if (operation_ == SET_COMMAND)
210 {
211 for (const auto& attr: request_description_.mandatory_attr_items)
212 {
213 if (attr_names_.find(attr) == std::end(attr_names_))
214 {
215 throw std::invalid_argument(std::string("Mandatory attribute '") + attr + std::string("' not found"));
216 }
217 }
218 }

310uint16_t Request::parseVlan(const std::string& str)
311{
312 uint16_t ret = 0;
313
314 const auto vlan_prefix = std::string("Vlan");
315 const auto prefix_len = vlan_prefix.length();
316
317 if (str.substr(0, prefix_len) != vlan_prefix)
318 {
319 throw std::invalid_argument(std::string("Invalid vlan interface: ") + str);
320 }
321
322 try
323 {
324 ret = static_cast<uint16_t>(std::stoul(str.substr(prefix_len)));
325 }
326 catch(std::invalid_argument& _)
327 {
328 throw std::invalid_argument(std::string("Invalid vlan id: ") + str);
329 }
330 catch(std::out_of_range& _)
331 {
332 throw std::invalid_argument(std::string("Out of range vlan id: ") + str);
333 }
334
335 if (ret == 0 || ret > 4094)
336 {
337 throw std::invalid_argument(std::string("Out of range vlan id: ") + str);
338 }
339
340 return ret;
341}
