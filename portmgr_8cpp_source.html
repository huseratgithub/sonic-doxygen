23bool PortMgr::setPortMtu(const string &alias, const string &mtu)
24{
25 stringstream cmd;
26 string res, cmd_str;
27
28 // ip link set dev <port_name> mtu <mtu>
29 cmd << IP_CMD << " link set dev " << shellquote(alias) << " mtu " << shellquote(mtu);
30 cmd_str = cmd.str();
31 int ret = swss::exec(cmd_str, res);
32 if (!ret)
33 {
34 // Set the port MTU in application database to update both
35 // the port MTU and possibly the port based router interface MTU
36 return writeConfigToAppDb(alias, "mtu", mtu);
37 }
38 else if (!isPortStateOk(alias))
39 {
40 // Can happen when a DEL notification is sent by portmgrd immediately followed by a new SET notif
41 SWSS_LOG_WARN("Setting mtu to alias:%s netdev failed with cmd:%s, rc:%d, error:%s", alias.c_str(), cmd_str.c_str(), ret, res.c_str());
42 return false;
43 }
44 else
45 {
46 throw runtime_error(cmd_str + " : " + res);
47 }
48 return true;
49}

96void PortMgr::doTask(Consumer &consumer)
97{
98 SWSS_LOG_ENTER();
99
100 auto table = consumer.getTableName();
101
102 auto it = consumer.m_toSync.begin();
103 while (it != consumer.m_toSync.end())
104 {
105 KeyOpFieldsValuesTuple t = it->second;
106
107 string alias = kfvKey(t);
108 string op = kfvOp(t);
109
110 if (op == SET_COMMAND)
111 {
112 /* portOk=true indicates that the port has been created in kernel.
113 * We should not call any ip command if portOk=false. However, it is
114 * valid to put port configuration to APP DB which will trigger port creation in kernel.
115 */
116 bool portOk = isPortStateOk(alias);
117
118 string admin_status, mtu;
119 std::vector<FieldValueTuple> field_values;
120
121 bool configured = (m_portList.find(alias) != m_portList.end());
122
123 /* If this is the first time we set port settings
124 * assign default admin status and mtu
125 */
126 if (!configured)
127 {
128 admin_status = DEFAULT_ADMIN_STATUS_STR;
129 mtu = DEFAULT_MTU_STR;
130
131 m_portList.insert(alias);
132 }
133 else if (!portOk)
134 {
135 it++;
136 continue;
137 }
138
139 for (auto i : kfvFieldsValues(t))
140 {
141 if (fvField(i) == "mtu")
142 {
143 mtu = fvValue(i);
144 }
145 else if (fvField(i) == "admin_status")
146 {
147 admin_status = fvValue(i);
148 }
149 else
150 {
151 field_values.emplace_back(i);
152 }
153 }
154
155 if (field_values.size())
156 {
157 writeConfigToAppDb(alias, field_values);
158 }
159
160 if (!portOk)
161 {
162 SWSS_LOG_INFO("Port %s is not ready, pending...", alias.c_str());
163
164 writeConfigToAppDb(alias, "mtu", mtu);
165 writeConfigToAppDb(alias, "admin_status", admin_status);
166 /* Retry setting these params after the netdev is created */
167 field_values.clear();
168 field_values.emplace_back("mtu", mtu);
169 field_values.emplace_back("admin_status", admin_status);
170 it->second = KeyOpFieldsValuesTuple{alias, SET_COMMAND, field_values};
171 it++;
172 continue;
173 }
174
175 if (!mtu.empty())
176 {
177 setPortMtu(alias, mtu);
178 SWSS_LOG_NOTICE("Configure %s MTU to %s", alias.c_str(), mtu.c_str());
179 }
180
181 if (!admin_status.empty())
182 {
183 setPortAdminStatus(alias, admin_status == "up");
184 SWSS_LOG_NOTICE("Configure %s admin status to %s", alias.c_str(), admin_status.c_str());
185 }
186 }
187 else if (op == DEL_COMMAND)
188 {
189 SWSS_LOG_NOTICE("Delete Port: %s", alias.c_str());
190 m_appPortTable.del(alias);
191 m_portList.erase(alias);
192 }
193
194 it = consumer.m_toSync.erase(it);
195 }
196}
