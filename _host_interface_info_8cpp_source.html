127void HostInterfaceInfo::veth2tap_fun()
128{
129 SWSS_LOG_ENTER();
130
131 unsigned char buffer[ETH_FRAME_BUFFER_SIZE];
132
133 swss::Select s;
134 SelectableFd fd(m_packet_socket);
135
136 s.addSelectable(&m_e2tEvent);
137 s.addSelectable(&fd);
138
139 while (m_run_thread)
140 {
141 struct msghdr msg;
142 memset(&msg, 0, sizeof(struct msghdr));
143
144 struct sockaddr_storage src_addr;
145
146 struct iovec iov[1];
147
148 iov[0].iov_base = buffer; // buffer for message
149 iov[0].iov_len = sizeof(buffer);
150
151 char control[CONTROL_MESSAGE_BUFFER_SIZE]; // buffer for control messages
152
153 msg.msg_name = &src_addr;
154 msg.msg_namelen = sizeof(src_addr);
155 msg.msg_iov = iov;
156 msg.msg_iovlen = 1;
157 msg.msg_control = control;
158 msg.msg_controllen = sizeof(control);
159
160 swss::Selectable *sel = NULL;
161
162 int result = s.select(&sel);
163
164 if (result != swss::Select::OBJECT)
165 {
166 SWSS_LOG_ERROR("selectable failed: %d, ending thread for %s", result, m_name.c_str());
167 return;
168 }
169
170 if (sel == &m_e2tEvent) // thread end event
171 break;
172
173 ssize_t size = recvmsg(m_packet_socket, &msg, 0);
174
175 if (size < 0)
176 {
177 if (errno != ENETDOWN)
178 {
179 SWSS_LOG_ERROR("failed to read from socket fd %d, errno(%d): %s",
180 m_packet_socket, errno, strerror(errno));
181 }
182
183 continue;
184 }
185
186 if (size < (ssize_t)sizeof(ethhdr))
187 {
188 SWSS_LOG_ERROR("invalid ethernet frame length: %zu", msg.msg_controllen);
189 continue;
190 }
191
192 // Buffer include the ingress packets
193 // MACsec scenario: EAPOL packets and encrypted packets
194 size_t length = static_cast<size_t>(size);
195 auto ret = m_e2tFilters.execute(buffer, length);
196
197 if (ret == TrafficFilter::TERMINATE)
198 {
199 continue;
200 }
201 else if (ret == TrafficFilter::ERROR)
202 {
203 // Error log should be recorded in filter
204 return;
205 }
206
207 addVlanTag(buffer, length, msg);
208
209 async_process_packet_for_fdb_event(buffer, length);
210
211 if (!sendTo(m_tapfd, buffer, length))
212 {
213 break;
214 }
215 }
216
217 SWSS_LOG_NOTICE("ending thread proc for %s", m_name.c_str());
218}
