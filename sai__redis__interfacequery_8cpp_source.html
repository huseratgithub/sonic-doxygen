88sai_status_t sai_api_query(
89 _In_ sai_api_t sai_api_id,
90 _Out_ void** api_method_table)
91{
92 SWSS_LOG_ENTER();
93
94 if (api_method_table == NULL)
95 {
96 SWSS_LOG_ERROR("NULL method table passed to SAI API initialize");
97
98 return SAI_STATUS_INVALID_PARAMETER;
99 }
100
101 if (sai_api_id == SAI_API_UNSPECIFIED)
102 {
103 SWSS_LOG_ERROR("api ID is unspecified api");
104
105 return SAI_STATUS_INVALID_PARAMETER;
106 }
107
108 if (sai_metadata_get_enum_value_name(&sai_metadata_enum_sai_api_t, sai_api_id))
109 {
110 *api_method_table = ((void**)&redis_apis)[sai_api_id - 1];
111 return SAI_STATUS_SUCCESS;
112 }
113
114 SWSS_LOG_ERROR("Invalid API type %d", sai_api_id);
115
116 return SAI_STATUS_INVALID_PARAMETER;
117}
