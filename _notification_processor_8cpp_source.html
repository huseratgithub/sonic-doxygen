99void NotificationProcessor::redisPutFdbEntryToAsicView(
100 _In_ const sai_fdb_event_notification_data_t *fdb)
101{
102 SWSS_LOG_ENTER();
103
104 // NOTE: this fdb entry already contains translated RID to VID
105
106 std::vector<swss::FieldValueTuple> entry;
107
108 entry = SaiAttributeList::serialize_attr_list(
109 SAI_OBJECT_TYPE_FDB_ENTRY,
110 fdb->attr_count,
111 fdb->attr,
112 false);
113
114 sai_object_meta_key_t metaKey;
115
116 metaKey.objecttype = SAI_OBJECT_TYPE_FDB_ENTRY;
117 metaKey.objectkey.key.fdb_entry = fdb->fdb_entry;
118
119 std::string strFdbEntry = sai_serialize_fdb_entry(fdb->fdb_entry);
120
121 if ((fdb->fdb_entry.switch_id == SAI_NULL_OBJECT_ID ||
122 fdb->fdb_entry.bv_id == SAI_NULL_OBJECT_ID) &&
123 (fdb->event_type != SAI_FDB_EVENT_FLUSHED))
124 {
125 SWSS_LOG_WARN("skipped to put int db: %s", strFdbEntry.c_str());
126 return;
127 }
128
129 if (fdb->event_type == SAI_FDB_EVENT_AGED)
130 {
131 SWSS_LOG_DEBUG("remove fdb entry %s for SAI_FDB_EVENT_AGED",
132 sai_serialize_object_meta_key(metaKey).c_str());
133
134 m_client->removeAsicObject(metaKey);
135 return;
136 }
137
138 if (fdb->event_type == SAI_FDB_EVENT_FLUSHED)
139 {
140 sai_object_id_t bv_id = fdb->fdb_entry.bv_id;
141 sai_object_id_t port_oid = 0;
142
143 sai_fdb_flush_entry_type_t type = SAI_FDB_FLUSH_ENTRY_TYPE_DYNAMIC;
144
145 for (uint32_t i = 0; i < fdb->attr_count; i++)
146 {
147 if (fdb->attr[i].id == SAI_FDB_ENTRY_ATTR_BRIDGE_PORT_ID)
148 {
149 port_oid = fdb->attr[i].value.oid;
150 }
151 else if (fdb->attr[i].id == SAI_FDB_ENTRY_ATTR_TYPE)
152 {
153 type = (fdb->attr[i].value.s32 == SAI_FDB_ENTRY_TYPE_STATIC)
154 ? SAI_FDB_FLUSH_ENTRY_TYPE_STATIC
155 : SAI_FDB_FLUSH_ENTRY_TYPE_DYNAMIC;
156 }
157 }
158
159 m_client->processFlushEvent(fdb->fdb_entry.switch_id, port_oid, bv_id, type);
160 return;
161 }
162
163 if (fdb->event_type == SAI_FDB_EVENT_LEARNED || fdb->event_type == SAI_FDB_EVENT_MOVE)
164 {
165 // currently we need to add type manually since fdb event don't contain type
166 sai_attribute_t attr;
167
168 attr.id = SAI_FDB_ENTRY_ATTR_TYPE;
169 attr.value.s32 = SAI_FDB_ENTRY_TYPE_DYNAMIC;
170
171 auto objectType = SAI_OBJECT_TYPE_FDB_ENTRY;
172
173 auto meta = sai_metadata_get_attr_metadata(objectType, attr.id);
174
175 if (meta == NULL)
176 {
177 SWSS_LOG_THROW("unable to get metadata for object type %s, attribute %d",
178 sai_serialize_object_type(objectType).c_str(),
179 attr.id);
180 /*
181 * TODO We should notify orch agent here. And also this probably should
182 * not be here, but on redis side, getting through metadata.
183 */
184 }
185
186 std::string strAttrId = sai_serialize_attr_id(*meta);
187 std::string strAttrValue = sai_serialize_attr_value(*meta, attr);
188
189 entry.emplace_back(strAttrId, strAttrValue);
190
191 m_client->createAsicObject(metaKey, entry);
192 return;
193 }
194
195 SWSS_LOG_ERROR("event type %s not supported, FIXME",
196 sai_serialize_fdb_event(fdb->event_type).c_str());
197}

213bool NotificationProcessor::check_fdb_event_notification_data(
214 _In_ const sai_fdb_event_notification_data_t& data)
215{
216 SWSS_LOG_ENTER();
217
218 /*
219 * Any new RID value spotted in fdb notification can happen for 2 reasons:
220 *
221 * - a bug is present on the vendor SAI, all RID's are already in local or
222 * REDIS ASIC DB but vendor SAI returned new or invalid RID
223 *
224 * - orch agent didn't query yet bridge ID/vlan ID and already
225 * started to receive fdb notifications in which case warn message
226 * could be ignored.
227 *
228 * If vendor SAI will return invalid RID, then this will later on lead to
229 * inconsistent DB state and possible failure on apply view after cold or
230 * warm boot.
231 *
232 * On switch init we do discover phase, and we should discover all objects
233 * so we should not get any of those messages if SAI is in consistent
234 * state.
235 */
236
237 bool result = true;
238
239 if (!m_translator->checkRidExists(data.fdb_entry.bv_id, true))
240 {
241 SWSS_LOG_ERROR("bv_id RID 0x%" PRIx64 " is not present on local ASIC DB: %s", data.fdb_entry.bv_id,
242 sai_serialize_fdb_entry(data.fdb_entry).c_str());
243
244 result = false;
245 }
246
247 if (!m_translator->checkRidExists(data.fdb_entry.switch_id) || data.fdb_entry.switch_id == SAI_NULL_OBJECT_ID)
248 {
249 SWSS_LOG_ERROR("switch_id RID 0x%" PRIx64 " is not present on local ASIC DB: %s", data.fdb_entry.switch_id,
250 sai_serialize_fdb_entry(data.fdb_entry).c_str());
251
252 result = false;
253 }
254
255 for (uint32_t i = 0; i < data.attr_count; i++)
256 {
257 const sai_attribute_t& attr = data.attr[i];
258
259 auto meta = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_FDB_ENTRY, attr.id);
260
261 if (meta == NULL)
262 {
263 SWSS_LOG_ERROR("unable to get metadata for fdb_entry attr.id = %d", attr.id);
264 continue;
265 }
266
267 // skip non oid attributes
268 if (meta->attrvaluetype != SAI_ATTR_VALUE_TYPE_OBJECT_ID)
269 continue;
270
271 if (!m_translator->checkRidExists(attr.value.oid, true))
272 {
273 SWSS_LOG_WARN("RID 0x%" PRIx64 " on %s is not present on local ASIC DB", attr.value.oid, meta->attridname);
274
275 result = false;
276 }
277 }
278
279 return result;
280}

299void NotificationProcessor::process_on_fdb_event(
300 _In_ uint32_t count,
301 _In_ sai_fdb_event_notification_data_t *data)
302{
303 SWSS_LOG_ENTER();
304
305 SWSS_LOG_INFO("fdb event count: %u", count);
306
307 bool sendntf = true;
308
309 for (uint32_t i = 0; i < count; i++)
310 {
311 sai_fdb_event_notification_data_t *fdb = &data[i];
312
313 sendntf &= check_fdb_event_notification_data(*fdb);
314
315 if (!sendntf)
316 {
317 SWSS_LOG_ERROR("invalid OIDs in fdb notifications, NOT translating and NOT storing in ASIC DB");
318 continue;
319 }
320
321 SWSS_LOG_DEBUG("fdb %u: type: %d", i, fdb->event_type);
322
323 fdb->fdb_entry.switch_id = m_translator->translateRidToVid(fdb->fdb_entry.switch_id, SAI_NULL_OBJECT_ID);
324
325 fdb->fdb_entry.bv_id = m_translator->translateRidToVid(fdb->fdb_entry.bv_id, fdb->fdb_entry.switch_id, true);
326
327 m_translator->translateRidToVid(SAI_OBJECT_TYPE_FDB_ENTRY, fdb->fdb_entry.switch_id, fdb->attr_count, fdb->attr, true);
328
329 /*
330 * Currently because of brcm bug, we need to install fdb entries in
331 * asic view and currently this event don't have fdb type which is
332 * required on creation.
333 */
334
335 redisPutFdbEntryToAsicView(fdb);
336 }
337
338 if (sendntf)
339 {
340 std::string s = sai_serialize_fdb_event_ntf(count, data);
341
342 sendNotification(SAI_SWITCH_NOTIFICATION_NAME_FDB_EVENT, s);
343 }
344 else
345 {
346 SWSS_LOG_ERROR("FDB notification was not sent since it contain invalid OIDs, bug?");
347 }
348}

657void NotificationProcessor::syncProcessNotification(
658 _In_ const swss::KeyOpFieldsValuesTuple& item)
659{
660 SWSS_LOG_ENTER();
661
662 std::string notification = kfvKey(item);
663 std::string data = kfvOp(item);
664
665 if (notification == SAI_SWITCH_NOTIFICATION_NAME_SWITCH_STATE_CHANGE)
666 {
667 handle_switch_state_change(data);
668 }
669 else if (notification == SAI_SWITCH_NOTIFICATION_NAME_FDB_EVENT)
670 {
671 handle_fdb_event(data);
672 }
673 else if (notification == SAI_SWITCH_NOTIFICATION_NAME_NAT_EVENT)
674 {
675 handle_nat_event(data);
676 }
677 else if (notification == SAI_SWITCH_NOTIFICATION_NAME_PORT_STATE_CHANGE)
678 {
679 handle_port_state_change(data);
680 }
681 else if (notification == SAI_SWITCH_NOTIFICATION_NAME_SWITCH_SHUTDOWN_REQUEST)
682 {
683 handle_switch_shutdown_request(data);
684 }
685 else if (notification == SAI_SWITCH_NOTIFICATION_NAME_QUEUE_PFC_DEADLOCK)
686 {
687 handle_queue_deadlock(data);
688 }
689 else if (notification == SAI_SWITCH_NOTIFICATION_NAME_BFD_SESSION_STATE_CHANGE)
690 {
691 handle_bfd_session_state_change(data);
692 }
693 else
694 {
695 SWSS_LOG_ERROR("unknown notification: %s", notification.c_str());
696 }

699void NotificationProcessor::ntf_process_function()
700{
701 SWSS_LOG_ENTER();
702
703 std::mutex ntf_mutex;
704
705 std::unique_lock<std::mutex> ulock(ntf_mutex);
706
707 while (m_runThread)
708 {
709 m_cv.wait(ulock);
710
711 // this is notifications processing thread context, which is different
712 // from SAI notifications context, we can safe use syncd mutex here,
713 // processing each notification is under same mutex as processing main
714 // events, counters and reinit
715
716 swss::KeyOpFieldsValuesTuple item;
717
718 while (m_notificationQueue->tryDequeue(item))
719 {
720 processNotification(item);
721 }
722 }
723}
