67void write_to_state_db(shared_ptr<vector<KeyOpFieldsValuesTuple>> db_items_ptr)
68{
69 DBConnector db("STATE_DB", 0, true);
70 auto &db_items = *db_items_ptr;
71 for (auto &db_item : db_items)
72 {
73 dump_db_item(db_item);
74
75 string key = kfvKey(db_item);
76 size_t pos = key.find(":");
77 if ((string::npos == pos) || ((key.size() - 1) == pos))
78 {
79 SWSS_LOG_ERROR("Invalid formatted hash:%s\n", key.c_str());
80 return;
81 }
82 string table_name = key.substr(0, pos);
83 string key_name = key.substr(pos + 1);
84 Table stateTable(&db, table_name);

86 stateTable.set(key_name, kfvFieldsValues(db_item), SET_COMMAND);
