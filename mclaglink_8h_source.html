105 struct mclag_fdb
106 {
107 std::string mac;
108 unsigned int vid;
109 std::string port_name;
110 std::string type;/*dynamic or static*/
111
112 mclag_fdb(std::string val_mac, unsigned int val_vid, std::string val_pname,
113 std::string val_type) : mac(val_mac), vid(val_vid), port_name(val_pname), type(val_type)
114 {
115 }
116 mclag_fdb()
117 {
118 }
119
120 bool operator <(const mclag_fdb &fdb) const
121 {
122 if (mac != fdb.mac)
123 return mac < fdb.mac;
124 else if (vid != fdb.vid)
125 return vid < fdb.vid;
126 else
127 return port_name < fdb.port_name;
128 //else if (port_name != fdb.port_name) return port_name < fdb.port_name;
129 //else return type <fdb.type;
130 }
131
132 bool operator ==(const mclag_fdb &fdb) const
133 {
134 if (mac != fdb.mac)
135 return 0;
136 if (vid != fdb.vid)
137 return 0;
138 return 1;
139 }

165 struct mclagDomainData
166 {
167 std::string source_ip;
168 std::string peer_ip;
169 std::string peer_link;
170 int keepalive_interval;
171 int session_timeout;
172
173 mclagDomainData()
174 {
175 keepalive_interval = -1;
176 session_timeout = -1;
177 }
178
179 bool mandatoryFieldsPresent() const
180 {
181 return (!source_ip.empty() && !peer_ip.empty());
182 }
183
184 bool allFieldsEmpty() const
185 {
186 return (source_ip.empty() && peer_ip.empty() && peer_link.empty() && keepalive_interval == -1 && session_timeout == -1);
