27bool VRFOrch::addOperation(const Request& request)
28{
29 SWSS_LOG_ENTER();
30 uint32_t vni = 0;
31 bool error = true;
32
33 sai_attribute_t attr;
34 vector<sai_attribute_t> attrs;
35
36 for (const auto& name: request.getAttrFieldNames())
37 {
38 if (name == "v4")
39 {
40 attr.id = SAI_VIRTUAL_ROUTER_ATTR_ADMIN_V4_STATE;
41 attr.value.booldata = request.getAttrBool("v4");
42 }
43 else if (name == "v6")
44 {
45 attr.id = SAI_VIRTUAL_ROUTER_ATTR_ADMIN_V6_STATE;
46 attr.value.booldata = request.getAttrBool("v6");
47 }
48 else if (name == "src_mac")
49 {
50 const auto& mac = request.getAttrMacAddress("src_mac");
51 attr.id = SAI_VIRTUAL_ROUTER_ATTR_SRC_MAC_ADDRESS;
52 memcpy(attr.value.mac, mac.getMac(), sizeof(sai_mac_t));
53 }
54 else if (name == "ttl_action")
55 {
56 attr.id = SAI_VIRTUAL_ROUTER_ATTR_VIOLATION_TTL1_PACKET_ACTION;
57 attr.value.s32 = request.getAttrPacketAction("ttl_action");
58 }
59 else if (name == "ip_opt_action")
60 {
61 attr.id = SAI_VIRTUAL_ROUTER_ATTR_VIOLATION_IP_OPTIONS_PACKET_ACTION;
62 attr.value.s32 = request.getAttrPacketAction("ip_opt_action");
63 }
64 else if (name == "l3_mc_action")
65 {
66 attr.id = SAI_VIRTUAL_ROUTER_ATTR_UNKNOWN_L3_MULTICAST_PACKET_ACTION;
67 attr.value.s32 = request.getAttrPacketAction("l3_mc_action");
68 }
69 else if (name == "vni")
70 {
71 vni = static_cast<uint32_t>(request.getAttrUint(name));
72 continue;
73 }
74 else if ((name == "mgmtVrfEnabled") || (name == "in_band_mgmt_enabled"))
75 {
76 SWSS_LOG_INFO("MGMT VRF field: %s ignored", name.c_str());
77 continue;
78 }
79 else
80 {
81 SWSS_LOG_ERROR("Logic error: Unknown attribute: %s", name.c_str());
82 continue;
83 }
84 attrs.push_back(attr);
85 }
86
87 const std::string& vrf_name = request.getKeyString(0);
88 auto it = vrf_table_.find(vrf_name);
89 if (it == std::end(vrf_table_))
90 {
91 // Create a new vrf
92 sai_object_id_t router_id;
93 sai_status_t status = sai_virtual_router_api->create_virtual_router(&router_id,
94 gSwitchId,
95 static_cast<uint32_t>(attrs.size()),
96 attrs.data());
97 if (status != SAI_STATUS_SUCCESS)
98 {
99 SWSS_LOG_ERROR("Failed to create virtual router name: %s, rv: %d", vrf_name.c_str(), status);
100 task_process_status handle_status = handleSaiCreateStatus(SAI_API_VIRTUAL_ROUTER, status);
101 if (handle_status != task_success)
102 {
103 return parseHandleSaiStatusFailure(handle_status);
104 }
105 }
106
107 vrf_table_[vrf_name].vrf_id = router_id;
108 vrf_table_[vrf_name].ref_count = 0;
109 vrf_id_table_[router_id] = vrf_name;
110 gFlowCounterRouteOrch->onAddVR(router_id);
111 if (vni != 0)
112 {
113 SWSS_LOG_INFO("VRF '%s' vni %d add", vrf_name.c_str(), vni);
114 error = updateVrfVNIMap(vrf_name, vni);
115 if (error == false)
116 {
117 return false;
118 }
119 }
120 m_stateVrfObjectTable.hset(vrf_name, "state", "ok");
121 SWSS_LOG_NOTICE("VRF '%s' was added", vrf_name.c_str());
122 }
123 else
124 {
125 // Update an existing vrf
126
127 sai_object_id_t router_id = it->second.vrf_id;
128
129 for (const auto& attr: attrs)
130 {
131 sai_status_t status = sai_virtual_router_api->set_virtual_router_attribute(router_id, &attr);
132 if (status != SAI_STATUS_SUCCESS)
133 {
134 SWSS_LOG_ERROR("Failed to update virtual router attribute. vrf name: %s, rv: %d", vrf_name.c_str(), status);
135 task_process_status handle_status = handleSaiSetStatus(SAI_API_VIRTUAL_ROUTER, status);
136 if (handle_status != task_success)
137 {
138 return parseHandleSaiStatusFailure(handle_status);
139 }
140 }
141 }
142
143 SWSS_LOG_INFO("VRF '%s' vni %d modify", vrf_name.c_str(), vni);
144 error = updateVrfVNIMap(vrf_name, vni);
145 if (error == false)
146 {
147 return false;
148 }
149
150 SWSS_LOG_NOTICE("VRF '%s' was updated", vrf_name.c_str());
151 }
152
153 return true;
154}

156bool VRFOrch::delOperation(const Request& request)
157{
158 SWSS_LOG_ENTER();
159 bool error = true;
160
161 const std::string& vrf_name = request.getKeyString(0);
162 if (vrf_table_.find(vrf_name) == std::end(vrf_table_))
163 {
164 SWSS_LOG_ERROR("VRF '%s' doesn't exist", vrf_name.c_str());
165 return true;
166 }
167
168 if (vrf_table_[vrf_name].ref_count)
169 return false;
170
171 sai_object_id_t router_id = vrf_table_[vrf_name].vrf_id;
172 sai_status_t status = sai_virtual_router_api->remove_virtual_router(router_id);
173 if (status != SAI_STATUS_SUCCESS)
174 {
175 SWSS_LOG_ERROR("Failed to remove virtual router name: %s, rv:%d", vrf_name.c_str(), status);
176 task_process_status handle_status = handleSaiRemoveStatus(SAI_API_VIRTUAL_ROUTER, status);
177 if (handle_status != task_success)
178 {
179 return parseHandleSaiStatusFailure(handle_status);
180 }
181 }
182
183 gFlowCounterRouteOrch->onRemoveVR(router_id);
184
185 vrf_table_.erase(vrf_name);
186 vrf_id_table_.erase(router_id);
187 error = delVrfVNIMap(vrf_name, 0);
188 if (error == false)
189 {
190 return false;
191 }
192 m_stateVrfObjectTable.del(vrf_name);
193
194 SWSS_LOG_NOTICE("VRF '%s' was removed", vrf_name.c_str());
195
196 return true;
197}
