452 sai_status_t flush_removing_entries(
453 _Inout_ std::vector<Te> &rs)
454 {
455 if (rs.empty())
456 {
457 return SAI_STATUS_SUCCESS;
458 }
459 size_t count = rs.size();
460 std::vector<sai_status_t> statuses(count);
461 sai_status_t status = (*remove_entries)((uint32_t)count, rs.data(), SAI_BULK_OP_ERROR_MODE_IGNORE_ERROR, statuses.data());
462 if (status == SAI_STATUS_SUCCESS)
463 {
464 SWSS_LOG_INFO("EntityBulker.flush removing_entries %zu\n", count);
465 }
466 else
467 {
468 SWSS_LOG_ERROR("EntityBulker.flush remove entries failed, number of entries to remove: %zu, status: %s",
469 count, sai_serialize_status(status).c_str());
470 }
471
472 for (size_t ir = 0; ir < count; ir++)
473 {
474 auto& entry = rs[ir];
475 sai_status_t *object_status = removing_entries[entry];
476 if (object_status)
477 {
478 *object_status = statuses[ir];
479 }
480 }
481
482 rs.clear();
483
484 return status;
485 }
