137{

140
141 attr.id = SAI_TUNNEL_ATTR_TYPE;
142 attr.value.s32 = SAI_TUNNEL_TYPE_NVGRE;
143 tunnel_attrs.push_back(attr);
144
145 attr.id = SAI_TUNNEL_ATTR_UNDERLAY_INTERFACE;
146 attr.value.oid = underlay_rif;
147 tunnel_attrs.push_back(attr);
148
149 sai_object_id_t decap_map_list[MAP_T_MAX];

156 }
157
158 attr.id = SAI_TUNNEL_ATTR_DECAP_MAPPERS;
159 attr.value.objlist.count = num_decap_map;
160 attr.value.objlist.list = decap_map_list;
161 tunnel_attrs.push_back(attr);
162
163 sai_object_id_t encap_map_list[MAP_T_MAX];
164 uint8_t num_encap_map = 0;

172 attr.id = SAI_TUNNEL_ATTR_ENCAP_MAPPERS;
173 attr.value.objlist.count = num_encap_map;
174 attr.value.objlist.list = encap_map_list;
175 tunnel_attrs.push_back(attr);

177 attr.id = SAI_TUNNEL_ATTR_ENCAP_SRC_IP;
178 attr.value.ipaddr = src_ip;
179 tunnel_attrs.push_back(attr);

182 sai_status_t status = sai_tunnel_api->create_tunnel(

186 tunnel_attrs.data()
