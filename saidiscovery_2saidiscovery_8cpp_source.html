400static void handleProfileMap(
401 _In_ const std::string& profileMapFile)
402{
403 SWSS_LOG_ENTER();
404
405 if (profileMapFile.size() == 0)
406 {
407 return;
408 }
409
410 std::ifstream profile(profileMapFile);
411
412 if (!profile.is_open())
413 {
414 SWSS_LOG_ERROR("failed to open profile map file: '%s' : %s",
415 profileMapFile.c_str(), strerror(errno));
416
417 exit(EXIT_FAILURE);
418 }
419
420 std::string line;
421
422 while (getline(profile, line))
423 {
424 if (line.size() > 0 && (line[0] == '#' || line[0] == ';'))
425 {
426 continue;
427 }
428
429 size_t pos = line.find("=");
430
431 if (pos == std::string::npos)
432 {
433 SWSS_LOG_WARN("not found '=' in line %s", line.c_str());
434 continue;
435 }
436
437 std::string key = line.substr(0, pos);
438 std::string value = line.substr(pos + 1);
439
440 gProfileMap[key] = value;
441
442 SWSS_LOG_INFO("inserted: %s:%s", key.c_str(), value.c_str());
443 }
