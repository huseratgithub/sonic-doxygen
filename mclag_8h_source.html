158/*
159 * mclag_msg_hdr_ok
160 *
161 * Returns TRUE if a message header looks well-formed.
162 */
163static inline int
164mclag_msg_hdr_ok (const mclag_msg_hdr_t *hdr)
165{
166 size_t msg_len;
167
168 if (hdr->msg_type == MCLAG_MSG_TYPE_NONE)
169 return 0;
170
171 msg_len = mclag_msg_len (hdr);
172
173 if (msg_len < MCLAG_MSG_HDR_LEN || msg_len > MCLAG_MAX_MSG_LEN)
174 return 0;
175
176 return 1;
177}
178
179/*
180 * mclag_msg_ok
181 *
182 * Returns TRUE if a message looks well-formed.
183 *
184 * @param len The length in bytes from 'hdr' to the end of the buffer.
185 */
186static inline int
187mclag_msg_ok (const mclag_msg_hdr_t *hdr, size_t len)
188{
189 if (len < MCLAG_MSG_HDR_LEN)
190 return 0;
191
192 if (!mclag_msg_hdr_ok (hdr))
193 return 0;
194
195 if (mclag_msg_len (hdr) > len)
196 return 0;
197
198 return 1;
199}
