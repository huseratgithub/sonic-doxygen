226/*
227 * Zero buffer pools and profiles are introduced for reclaiming reserved buffer on unused ports.
228 *
229 * They are loaded into buffer manager through a json file provided from CLI on a per-platform basis
230 * and will be applied to APPL_DB and STATE_DB on the ports once they are admin down.
231 * They are loaded into APPL_DB in an order in which they occur in the json file, which means
232 * it's vendor's responsibility to guarantee the order reflects the dependency among zero pools and profiles.
233 * They are removed from APPL_DB once all ports are admin up.
234 * The zero profiles are removed first and then the zero pools. This is to respect the dependency between them.
235 *
236 * The keys can be in format of:
237 * - <TABLE NAME>|<object name>: represents a zero buffer object, like a zero buffer pool or zero buffer profile
238 * All necessary fields of the object should be provided in the json file according to the type of the object.
239 * For the buffer profiles, if the buffer pools referenced are the normal pools, like {ingress|egress}_{lossless|lossy}_pool,
240 * the zero profile name will be stored in the referenced pool's "zero_profile_name" field for the purpose of
241 * - constructing the zero profile list or providing the zero profiles for PGs or queues
242 * - fetching the zero profile for a certain pool when applying the zero profile for configured items
243 *
244 * - control_fields: represents the ids required for reclaiming unused buffers, including:
245 * - pgs_to_apply_zero_profile, represents the PGs on which the zero profiles will be applied for reclaiming unused buffers
246 * If it is not provided, zero profiles will be applied on all PGs.
247 * - ingress_zero_profile, represents the zero buffer profille which will be applied on PGs for reclaiming unused buffer.
248 * Typically, it is provided along with pgs_to_apply_zero_profile.
249 * In case pgs_to_apply_zero_profile is defined but ingress_zero_profile is not provided, the first zero profile on the ingress side in json file will be used.
250 * - queues_to_apply_zero_profile, represents the queues on which the zero profiles will be applied
251 * If it is not provided, zero profiles will be applied on all queues.
252 * - egress_zero_profile, represents the zero buffer profille which will be applied on queues for reclaiming unused buffer.
253 * Typically, it is provided along with queues_to_apply_zero_profile.
254 * In case queues_to_apply_zero_profile is defined but egress_zero_profile is not provided, the first zero profile on the egress side in json file will be used.
255 * - support_removing_buffer_items, represents whether the buffer items are supported to be removed.
256 * The number of queues and PGs are pushed into BUFFER_MAX_PARAM table in STATE_DB at the beginning of ports orchagent
257 * and will be learnt by buffer manager when it's starting.
258 */
259void BufferMgrDynamic::loadZeroPoolAndProfiles()
260{
261 for (auto &kfv : m_zeroPoolAndProfileInfo)
262 {
263 auto &table_key = kfvKey(kfv);
264
265 if (table_key == "control_fields")
266 {
267 auto &fvs = kfvFieldsValues(kfv);
268 for (auto &fv : fvs)
269 {
270 if (fvField(fv) == "pgs_to_apply_zero_profile")
271 {
272 m_bufferObjectIdsToZero[BUFFER_PG] = fvValue(fv);
273 }
274 else if (fvField(fv) == "ingress_zero_profile")
275 {
276 m_bufferZeroProfileName[BUFFER_PG] = fvValue(fv);
277 }
278 else if (fvField(fv) == "queues_to_apply_zero_profile")
279 {
280 m_bufferObjectIdsToZero[BUFFER_QUEUE] = fvValue(fv);
281 }
282 else if (fvField(fv) == "egress_zero_profile")
283 {
284 m_bufferZeroProfileName[BUFFER_QUEUE] = fvValue(fv);
285 }
286 else if (fvField(fv) == "support_removing_buffer_items")
287 {
288 m_supportRemoving = (fvValue(fv) == "yes");
289 }
290 }
291
292 continue;
293 }
294
295 auto const &table = parseObjectNameFromKey(table_key, 0);
296 auto const &key = parseObjectNameFromKey(table_key, 1);
297
298 if (table.empty() || key.empty())
299 {
300 SWSS_LOG_ERROR("Invalid format of key %s for zero profile info, won't initialize it",
301 kfvKey(kfv).c_str());
302 return;
303 }
304
305 if (table == APP_BUFFER_POOL_TABLE_NAME)
306 {
307 m_applBufferPoolTable.set(key, kfvFieldsValues(kfv));
308 m_stateBufferPoolTable.set(key, kfvFieldsValues(kfv));
309 SWSS_LOG_NOTICE("Loaded zero buffer pool %s", key.c_str());
310 m_zeroPoolNameSet.insert(key);
311 }
312 else if (table == APP_BUFFER_PROFILE_TABLE_NAME)
313 {
314 auto &fvs = kfvFieldsValues(kfv);
315 bool poolNotFound = false;
316 for (auto &fv : fvs)
317 {
318 if (fvField(fv) == "pool")
319 {
320 const auto &poolName = fvValue(fv);
321 auto poolSearchRef = m_bufferPoolLookup.find(poolName);
322 if (poolSearchRef != m_bufferPoolLookup.end())
323 {
324 auto &poolObj = poolSearchRef->second;
325 if (poolObj.zero_profile_name.empty())
326 {
327 poolObj.zero_profile_name = key;
328 if (m_bufferZeroProfileName[poolObj.direction].empty())
329 m_bufferZeroProfileName[poolObj.direction] = poolObj.zero_profile_name;
330 }
331 else
332 {
333 SWSS_LOG_ERROR("Multiple zero profiles (%s, %s) detected for pool %s, takes the former and ignores the latter",
334 poolObj.zero_profile_name.c_str(),
335 key.c_str(),
336 fvValue(fv).c_str());
337 }
338 }
339 else if (m_zeroPoolNameSet.find(poolName) == m_zeroPoolNameSet.end())
340 {
341 SWSS_LOG_WARN("Profile %s is not loaded as the referenced pool %s is not defined",
342 key.c_str(),
343 fvValue(fv).c_str());
344 poolNotFound = true;
345 break;
346 }
347
348 m_zeroProfiles.emplace_back(key, poolName);
349 }
350 }
351 if (poolNotFound)
352 {
353 continue;
354 }
355 m_applBufferProfileTable.set(key, fvs);
356 m_stateBufferProfileTable.set(key, fvs);
357 SWSS_LOG_NOTICE("Loaded zero buffer profile %s", key.c_str());
358 }
359 else
360 {
361 SWSS_LOG_ERROR("Unknown keys %s with zero table name %s isn't loaded to APPL_DB", key.c_str(), table.c_str());
362 continue;
363 }
364 }
365
366 // Consistency checking
367 // 1. For any buffer pool, if there is no zero profile provided, removing buffer items must be supported
368 // because the reserved buffer will be reclaimed by removing buffer items
369 // 2. If pgs_to_apply_zero_profile or queues_to_apply_zero_profile is provided, removing buffer items must be supported
370 // because the PGs or queues that are not in the ID list will be removed
371 bool noReclaiming = false;
372 if (!m_supportRemoving)
373 {
374 for (auto &poolRef: m_bufferPoolLookup)
375 {
376 if (poolRef.second.zero_profile_name.empty())
377 {
378 // For any buffer pool, zero profile must be provided
379 SWSS_LOG_ERROR("Zero profile is not provided for pool %s while removing buffer items is not supported, reserved buffer can not be reclaimed correctly", poolRef.first.c_str());
380 noReclaiming = true;
381 }
382 }
383
384 if (!m_bufferObjectIdsToZero[BUFFER_PG].empty() || !m_bufferObjectIdsToZero[BUFFER_QUEUE].empty())
385 {
386 SWSS_LOG_ERROR("Unified IDs of queues or priority groups specified while removing buffer items is not supported, reserved buffer can not be reclaimed correctly");
387 noReclaiming = true;
388 }
389 }
390
391 if (noReclaiming)
392 {
393 unloadZeroPoolAndProfiles();
394 m_zeroPoolAndProfileInfo.clear();
395 }
396 else
397 {
398 m_zeroProfilesLoaded = true;
399 }

503 ((m_model_number / 1000 == 5) && (speed != "800000"))))
504 {
505 // On Mellanox platform, ports with 8 lanes have different(double) xon value then other ports
506 // For ports at speed other than 400G can have
507 // - 8 lanes, double xon
508 // - other number of lanes, normal xon
509 // So they can not share the same buffer profiles.
510 // An extra "_8lane" is added to the name of buffer profiles to distinguish both scenarios
511 // Eg.
512 // - A 100G port with 8 lanes will use buffer profile "pg_profile_100000_5m_8lane_profile"
513 // - A 100G port with 4 lanes will use buffer profile "pg_profile_100000_5m_profile"
514 // Currently, for 4xxx models, 400G ports can only have 8 lanes,
515 // and for 5xxx models, 800G ports can only have 8 lanes. So we don't add this to the profile.
516 buffer_profile_key = buffer_profile_key + "_8lane";
517 }
518 }
519
520 return buffer_profile_key + "_profile";

544// Conduct the effective speed and compare the new value against the old one
545// Return true if they differ and false otherwise, meaning the headroom should be updated accordingly.
546//
547// The way to conduct the effective speed for headroom calculating
548// If auto_neg enabled on the port
549// if adv_speed configured
550// max(adv_speed) => new effective speed
551// elif sup_speed learnt
552// max(sup_speed) => new effective speed
553// elif PORT READY
554// INITIALIZING => port.state
555// else
556// port speed will be the new effective speed
557// return whether effective speed has been updated
558bool BufferMgrDynamic::needRefreshPortDueToEffectiveSpeed(port_info_t &portInfo, string &portName)
559{
560 string newEffectiveSpeed;
561
562 if (portInfo.auto_neg)
563 {
564 if (isNonZero(portInfo.adv_speeds))
565 {
566 newEffectiveSpeed = move(getMaxSpeedFromList(portInfo.adv_speeds));
567 SWSS_LOG_INFO("Port %s: maximum configured advertised speed (%s from %s) is taken as the effective speed",
568 portName.c_str(), portInfo.effective_speed.c_str(), portInfo.adv_speeds.c_str());
569 }
570 else if (isNonZero(portInfo.supported_speeds))
571 {
572 newEffectiveSpeed = move(getMaxSpeedFromList(portInfo.supported_speeds));
573 SWSS_LOG_INFO("Port %s: maximum supported speed (%s from %s) is taken as the effective speed",
574 portName.c_str(), portInfo.effective_speed.c_str(), portInfo.supported_speeds.c_str());
575 }
576 else if (portInfo.state == PORT_READY)
577 {
578 portInfo.state = PORT_INITIALIZING;
579 SWSS_LOG_NOTICE("Port %s: unable to deduct the effective speed because auto negotiation is enabled but neither configured advertised speed nor supported speed is available", portName.c_str());
580 }
581 }
582 else
583 {
584 newEffectiveSpeed = portInfo.speed;
585 SWSS_LOG_INFO("Port %s: speed (%s) is taken as the effective speed", portName.c_str(), portInfo.effective_speed.c_str());
586 }
587
588 bool effectiveSpeedChanged = (newEffectiveSpeed != portInfo.effective_speed);

597// Meta flows which are called by main flows
598void BufferMgrDynamic::calculateHeadroomSize(buffer_profile_t &headroom)
599{
600 // Call vendor-specific lua plugin to calculate the xon, xoff, xon_offset, size and threshold
601 vector<string> keys = {};
602 vector<string> argv = {};
603
604 keys.emplace_back(headroom.name);
605 argv.emplace_back(headroom.speed);
606 argv.emplace_back(headroom.cable_length);
607 argv.emplace_back(headroom.port_mtu);
608 argv.emplace_back(m_identifyGearboxDelay);
609 argv.emplace_back(to_string(headroom.lane_count));
610
611 try
612 {
613 auto ret = swss::runRedisScript(*m_applDb, m_headroomSha, keys, argv);
614
615 if (ret.empty())
616 {
617 SWSS_LOG_WARN("Failed to calculate headroom for %s", headroom.name.c_str());
618 return;
619 }
620
621 // The format of the result:
622 // a list of strings containing key, value pairs with colon as separator
623 // each is a field of the profile
624 // "xon:18432"
625 // "xoff:18432"
626 // "size:36864"
627
628 for ( auto i : ret)
629 {
630 auto pairs = tokenize(i, ':');
631 if (pairs[0] == "xon")
632 headroom.xon = pairs[1];
633 if (pairs[0] == "xoff")
634 headroom.xoff = pairs[1];
635 if (pairs[0] == "size")
636 headroom.size = pairs[1];
637 if (pairs[0] == "xon_offset")
638 headroom.xon_offset = pairs[1];
639 }
640 }
641 catch (...)
642 {
643 SWSS_LOG_WARN("Lua scripts for headroom calculation were not executed successfully");
644 }
645}
646
647// This function is designed to fetch the sizes of shared buffer pool and shared headroom pool
648// and programe them to APPL_DB if they differ from the current value.
649// The function is called periodically:
650// 1. Fetch the sizes by calling lug plugin
651// - For each of the pools, it checks the size of shared buffer pool.
652// - For ingress_lossless_pool, it checks the size of the shared headroom pool (field xoff of the pool) as well.
653// 2. Compare the fetched value and the previous value
654// 3. Program to APPL_DB.BUFFER_POOL_TABLE only if its sizes differ from the stored value
655void BufferMgrDynamic::recalculateSharedBufferPool()
656{
657 try
658 {
659 vector<string> keys = {};
660 vector<string> argv = {};
661
662 if (!m_bufferPoolReady)
663 {
664 // In case all buffer pools have a configured size,
665 // The m_bufferPoolReady will be set to true
666 // It can happen on vs.
667 bool hasDynamicSizePool = false, hasBufferPool = false;
668 for (auto &poolRef : m_bufferPoolLookup)
669 {
670 hasBufferPool = true;
671 if (poolRef.second.dynamic_size)
672 {
673 hasDynamicSizePool = true;
674 }
675 }
676
677 if (!hasBufferPool)
678 {
679 SWSS_LOG_INFO("No shared buffer pool configured, skip calculating shared buffer pool size");
680 return;
681 }
682
683 if (hasBufferPool && !hasDynamicSizePool)
684 {
685 m_bufferPoolReady = true;
686 SWSS_LOG_NOTICE("No pool requires calculating size dynamically. All buffer pools are ready");
687 }
688 }
689
690 auto ret = runRedisScript(*m_applDb, m_bufferpoolSha, keys, argv);
691
692 // The format of the result:
693 // a list of lines containing key, value pairs with colon as separator
694 // each is the size of a buffer pool
695 // possible format of each line:
696 // 1. shared buffer pool only:
697 // <pool name>:<pool size>
698 // eg: "egress_lossless_pool:12800000"
699 // 2. shared buffer pool and shared headroom pool, for ingress_lossless_pool only:
700 // ingress_lossless_pool:<pool size>:<shared headroom pool size>,
701 // eg: "ingress_lossless_pool:3200000:1024000"
702 // 3. debug information:
703 // debug:<debug info>
704
705 if (ret.empty())
706 {
707 SWSS_LOG_WARN("Failed to recalculate the shared buffer pool size");
708 return;
709 }
710
711 for ( auto i : ret)
712 {
713 auto pairs = tokenize(i, ':');
714 auto poolName = pairs[0];
715
716 if ("debug" != poolName)
717 {
718 // We will handle the sizes of buffer pool update here.
719 // For the ingress_lossless_pool, there are some dedicated steps for shared headroom pool
720 // - The sizes of both the shared headroom pool and the shared buffer pool should be taken into consideration
721 // - In case the shared headroom pool size is statically configured, as it is programmed to APPL_DB during buffer pool handling,
722 // - any change from lua plugin will be ignored.
723 // - will handle ingress_lossless_pool in the way all other pools are handled in this case
724 const auto &poolRef = m_bufferPoolLookup.find(poolName);
725 if (poolRef == m_bufferPoolLookup.end())
726 {
727 SWSS_LOG_WARN("Unconfigured buffer pool %s got from lua plugin", poolName.c_str());
728 continue;
729 }
730 auto &pool = poolRef->second;
731 auto &poolSizeStr = pairs[1];
732 auto old_xoff = pool.xoff;
733 bool xoff_updated = false;
734
735 if (poolName == INGRESS_LOSSLESS_PG_POOL_NAME && !isNonZero(m_configuredSharedHeadroomPoolSize))
736 {
737 // Shared headroom pool size is treated as "updated" if either of the following conditions satisfied:
738 // - It is legal and differs from the stored value.
739 // - The lua plugin doesn't return the shared headroom pool size but there is a non-zero value stored
740 // This indicates the shared headroom pool was enabled by over subscribe ratio and is disabled.
741 // In this case a "0" will programmed to APPL_DB, indicating the shared headroom pool is disabled.
742 SWSS_LOG_DEBUG("Buffer pool ingress_lossless_pool xoff: %s, size %s", pool.xoff.c_str(), pool.total_size.c_str());
743
744 if (pairs.size() > 2)
745 {
746 auto &xoffStr = pairs[2];
747 if (pool.xoff != xoffStr)
748 {
749 unsigned long xoffNum = atol(xoffStr.c_str());
750 if (m_mmuSizeNumber > 0 && m_mmuSizeNumber < xoffNum)
751 {
752 SWSS_LOG_ERROR("Buffer pool %s: Invalid xoff %s, exceeding the mmu size %s, ignored xoff but the pool size will be updated",
753 poolName.c_str(), xoffStr.c_str(), m_mmuSize.c_str());
754 }
755 else
756 {
757 pool.xoff = xoffStr;
758 xoff_updated = true;
759 }
760 }
761 }
762 else
763 {
764 if (isNonZero(pool.xoff))
765 {
766 xoff_updated = true;
767 }
768 pool.xoff = "0";
769 }
770 }
771
772 // In general, the APPL_DB should be updated in case any of the following conditions satisfied
773 // 1. Shared headroom pool size has been updated
774 // This indicates the shared headroom pool is enabled by configuring over subscribe ratio,
775 // which means the shared headroom pool size has updated by lua plugin
776 // 2. The size of the shared buffer pool isn't configured and has been updated by lua plugin
777 if ((pool.total_size == poolSizeStr || !pool.dynamic_size) && !xoff_updated)
778 continue;
779
780 unsigned long poolSizeNum = atol(poolSizeStr.c_str());
781 if (m_mmuSizeNumber > 0 && m_mmuSizeNumber < poolSizeNum)
782 {
783 SWSS_LOG_ERROR("Buffer pool %s: Invalid size %s, exceeding the mmu size %s",
784 poolName.c_str(), poolSizeStr.c_str(), m_mmuSize.c_str());
785 continue;
786 }
787
788 auto old_size = pool.total_size;
789 pool.total_size = poolSizeStr;
790 updateBufferPoolToDb(poolName, pool);
791
792 if (!pool.xoff.empty())
793 {
794 SWSS_LOG_NOTICE("Buffer pool %s has been updated: size from [%s] to [%s], xoff from [%s] to [%s]",
795 poolName.c_str(), old_size.c_str(), pool.total_size.c_str(), old_xoff.c_str(), pool.xoff.c_str());
796 }
797 else
798 {
799 SWSS_LOG_NOTICE("Buffer pool %s has been updated: size from [%s] to [%s]", poolName.c_str(), old_size.c_str(), pool.total_size.c_str());
800 }
801 }
802 else
803 {
804 SWSS_LOG_INFO("Buffer pool debug info %s", i.c_str());
805 }
806 }
807 }
808 catch (...)
809 {
810 SWSS_LOG_WARN("Lua scripts for buffer calculation were not executed successfully");
811 }
812
813 if (!m_bufferPoolReady)
814 m_bufferPoolReady = true;
815}

817void BufferMgrDynamic::checkSharedBufferPoolSize(bool force_update_during_initialization = false)
818{
819 // PortInitDone indicates all steps of port initialization has been done
820 // Only after that does the buffer pool size update starts
821 if (!m_portInitDone && !force_update_during_initialization)
822 {
823 SWSS_LOG_INFO("Skip buffer pool updating during initialization");
824 return;
825 }
826
827 if (!m_portInitDone)
828 {
829 vector<FieldValueTuple> values;
830 if (m_applPortTable.get("PortInitDone", values))
831 {
832 SWSS_LOG_NOTICE("Buffer pools start to be updated");
833 m_portInitDone = true;
834 }
835 else
836 {
837 if (!m_bufferPoolReady)
838 {
839 // It's something like a placeholder especially for warm reboot flow
840 // without all buffer pools created, buffer profiles are unable to be created,
841 // which in turn causes buffer pgs and buffer queues unable to be created,
842 // which prevents the port from being ready and eventually fails the warm reboot
843 // After the buffer pools are created for the first time, we won't touch it
844 // until portInitDone
845 // Eventually, the correct values will pushed to APPL_DB and then ASIC_DB
846 recalculateSharedBufferPool();
847 SWSS_LOG_NOTICE("Buffer pool update deferred because port is still under initialization, start polling timer");
848 }
849
850 return;
851 }
852 }
853
854 if (!m_mmuSize.empty())
855 recalculateSharedBufferPool();
856}

1120 * The following functions are defnied for reclaiming reserved buffers
1121 * - constructZeroProfileListFromNormalProfileList
1122 * - reclaimReservedBufferForPort
1123 * - removeSupportedButNotConfiguredItemsOnPort
1124 * - applyNormalBufferObjectsOnPort, handling queues and profile lists.
1125 * The priority groups are handle by refreshPgsForPort
1126 *
1127 * The overall logic of reclaiming reserved buffers is handled in handlePortTable:
1128 * Shutdown flow (to reclaim unused buffer):
1129 * 1. reclaimReservedBufferForPort
1130 * Start up flow (to reserved buffer for the port)
1131 * 1. removeSupportedButNotConfiguredItemsOnPort
1132 * 2. applyNormalBufferObjectsOnPort
1133 * 3. refreshPgsForPort
1134 */

1136/*
1137 * constructZeroProfileListFromNormalProfileList
1138 * Tool function for constructing zero profile list from normal profile list.
1139 * There should be one buffer profile for each buffer pool on each side (ingress/egress) in the profile_list
1140 * in BUFFER_PORT_INGRESS_PROFILE_LIST and BUFFER_PORT_EGRESS_PROFILE_LIST table.
1141 * For example, on ingress side, there are ingress_lossless_profile and ingress_lossy_profile in the profile list
1142 * for buffer pools ingress_lossless_pool and ingress_lossy_pool respectively.
1143 * There should be one zero profile for each pool in the profile_list when reclaiming buffer on a port as well.
1144 * The arguments is the normal profile list and the port.
1145 * The logic is:
1146 * For each buffer profile in the list
1147 * 1. Fetch the buffer pool referenced by the profile
1148 * 2. Fetch the zero buffer profile of the buffer pool. The pool is skipped in case there is no zero buffer profile defined for it.
1149 * 3. Construct the zero buffer profile by joining all zero buffer profiles, separating by ","
1150 */
1151string BufferMgrDynamic::constructZeroProfileListFromNormalProfileList(const string &normalProfileList, const string &port)
1152{
1153 string zeroProfileNameList;
1154
1155 auto profileRefs = tokenize(normalProfileList, ',');
1156 for (auto &profileName : profileRefs)
1157 {
1158 const auto &zeroProfile = fetchZeroProfileFromNormalProfile(profileName);
1159 if (!zeroProfile.empty())
1160 {
1161 zeroProfileNameList += zeroProfile + ",";
1162 }
1163 else
1164 {
1165 SWSS_LOG_WARN("Unable to apply zero profile for profile %s on port %s because no zero profile configured for the pool",
1166 profileName.c_str(),
1167 port.c_str());
1168 }
1169 }
1170
1171 if (!zeroProfileNameList.empty())
1172 zeroProfileNameList.pop_back();
1173
1174 return zeroProfileNameList;
1175}

1265/*
1266 * reclaimReservedBufferForPort
1267 * Called when a port is admin down
1268 * Reserved buffer is reclaimed by applying zero profiles on the buffer item or just removing them.
1269 *
1270 * Parameters:
1271 * - port: the name of the port
1272 * Return:
1273 * - task_process_status
1274 *
1275 * Purpose:
1276 * 1. Reclaim all the priority groups and queues of the port from APPL_DB
1277 * 2. Remove all the buffer profiles that are dynamically calculated and no longer referenced
1278 * 3. Reclaim the ingress/egress profile list of the port form APPL_DB
1279 *
1280 * The flow:
1281 * 1. Load the zero pools and profiles into APPL_DB if they have been provided but not loaded.
1282 * 2. Handle priority group, and queues:
1283 * Two modes for them to be reclaimed:
1284 * - Apply zero buffer profiles on all configured and supported-but-not-configured items.
1285 * Eg.
1286 * - 16 queues supported, 0-2, 5-6 are configured as lossy and 3-4 are configured as lossless
1287 * - Zero profiles will be applied on
1288 * - 0-2, 5-6, 7-15: egress_lossy_zero_profile
1289 * - 3-4: egress_lossless_zero_profile
1290 * - Apply zero buffer profiles on items specified by vendor and remove all other items.
1291 * Eg.
1292 * - 8 PGs supported, 0 is configured as lossy and 3-4 are configured as lossless
1293 * - PGs to be applied zero profile on is 0, 3-4 will be removed
1294 * Queues and priority groups share the common logic.
1295 * 3. Handle ingress and egress buffer profile list, which shared the common logic:
1296 * - Construct the zero buffer profile list from the normal buffer profile list.
1297 * - Apply the zero buffer profile list on the port.
1298 */
1299template<class T> task_process_status BufferMgrDynamic::reclaimReservedBufferForPort(const string &port, T &portObjectLookup, buffer_direction_t dir)
1300{
1301 auto &portBufferObjects = portObjectLookup[port];
1302 auto &portInfo = m_portInfoLookup[port];
1303 set<string> profilesToBeReleased;
1304 const string &portKeyPrefix = port + delimiter;
1305 const string &objectTypeName = m_bufferObjectNames[dir];
1306
1307 if (!m_zeroPoolAndProfileInfo.empty() && (!m_zeroProfilesLoaded))
1308 {
1309 loadZeroPoolAndProfiles();
1310 }
1311
1312 bool applyZeroProfileOnSpecifiedObjects = !m_bufferObjectIdsToZero[dir].empty();
1313
1314 SWSS_LOG_NOTICE("Reclaiming buffer reserved for all %s(s) from port %s", objectTypeName.c_str(), port.c_str());
1315
1316 unsigned long objectsMap = (1 << portInfo.maximum_buffer_objects[dir]) - 1;
1317 for (auto &it : portBufferObjects)
1318 {
1319 auto &key = it.first;
1320 auto &portObject = it.second;
1321 const string &runningProfile = portObject.running_profile_name;
1322 const string &zeroProfile = fetchZeroProfileFromNormalProfile(runningProfile);
1323
1324 SWSS_LOG_INFO("Reclaiming buffer reserved for %s %s", objectTypeName.c_str(), key.c_str());
1325
1326 if (!applyZeroProfileOnSpecifiedObjects)
1327 {
1328 // Apply zero profiles on each configured buffer objects and supported-but-not-configured buffer objects
1329 // Fetch the zero profile of the pool referenced by the configured profile
1330 if (!zeroProfile.empty())
1331 {
1332 SWSS_LOG_INFO("Applying zero profile %s on %s %s", zeroProfile.c_str(), objectTypeName.c_str(), key.c_str());
1333 updateBufferObjectToDb(key, zeroProfile, true, dir);
1334 }
1335 else
1336 {
1337 // No zero profile defined for the pool. Reclaim buffer by removing the buffer object
1338 SWSS_LOG_INFO("Zero profile is not defined for profile %s, reclaim reserved buffer by removing %s %s", runningProfile.c_str(), objectTypeName.c_str(), key.c_str());
1339 updateBufferObjectToDb(key, runningProfile, false, dir);
1340 }
1341
1342 objectsMap ^= generateBitMapFromIdsStr(parseObjectNameFromKey(key, 1));
1343 }
1344 else
1345 {
1346 // Apply the zero profile on specified buffer objects only.
1347 // Remove each buffer object first
1348 // In this case, removing must be supported (checked when the json was loaded)
1349 SWSS_LOG_INFO("Zero profile will be applied on %s. Remove %s %s first", m_bufferObjectIdsToZero[dir].c_str(), objectTypeName.c_str(), key.c_str());
1350 updateBufferObjectToDb(key, runningProfile, false, dir);
1351 }
1352
1353 if (dir == BUFFER_PG && !runningProfile.empty())
1354 {
1355 m_bufferProfileLookup[runningProfile].port_pgs.erase(key);
1356 profilesToBeReleased.insert(runningProfile);
1357 portObject.running_profile_name.clear();
1358 }
1359 }
1360
1361 if (applyZeroProfileOnSpecifiedObjects)
1362 {
1363 updateBufferObjectToDb(portKeyPrefix + m_bufferObjectIdsToZero[dir], m_bufferZeroProfileName[dir], true, dir);
1364 }
1365 else if (!m_bufferZeroProfileName[dir].empty())
1366 {
1367 // Apply zero profiles on supported-but-not-configured buffer objects
1368 portInfo.supported_but_not_configured_buffer_objects[dir] = generateIdListFromMap(objectsMap, portInfo.maximum_buffer_objects[dir]);
1369 if (m_waitApplyAdditionalZeroProfiles == 0)
1370 {
1371 for(auto &it: portInfo.supported_but_not_configured_buffer_objects[dir])
1372 {
1373 updateBufferObjectToDb(portKeyPrefix + it, m_bufferZeroProfileName[dir], true, dir);
1374 }
1375 }
1376 else
1377 {
1378 m_pendingSupportedButNotConfiguredPorts[dir].insert(port);
1379 }
1380 }
1381
1382 // Remove the old profile which is probably not referenced anymore.
1383 if (!profilesToBeReleased.empty())
1384 {
1385 for (auto &oldProfile : profilesToBeReleased)
1386 {
1387 releaseProfile(oldProfile);
1388 }
1389 }
1390
1391 vector<FieldValueTuple> fvVector;
1392
1393 SWSS_LOG_NOTICE("Reclaiming buffer reserved for ingress profile list from port %s", port.c_str());
1394 const auto &profileList = m_portProfileListLookups[dir][port];
1395 if (!profileList.empty())
1396 {
1397 const string &zeroIngressProfileNameList = constructZeroProfileListFromNormalProfileList(profileList, port);
1398 fvVector.emplace_back(buffer_profile_list_field_name, zeroIngressProfileNameList);

1400 }
1401
1402 return task_process_status::task_success;

1405// Called when speed/cable length updated from CONFIG_DB
1406// Update buffer profile of a certain PG of a port or all PGs of the port according to its speed, cable_length and mtu
1407// Called when
1408// - port's speed, cable_length or mtu updated
1409// - one buffer pg of port's is set to dynamic calculation
1410// Args
1411// port - port name
1412// speed, cable_length, mtu - port info
1413// exactly_matched_key - representing a port,pg. when provided, only profile of this key is updated
1414// Flow
1415// 1. For each PGs in the port
1416// a. skip non-dynamically-calculated or non-exactly-matched PGs
1417// b. allocate/reuse profile according to speed/cable length/mtu
1418// c. check accumulative headroom size, fail if exceeding per-port limit
1419// d. profile reference: remove reference to old profile and add reference to new profile
1420// e. put the old profile to to-be-released profile set
1421// f. update BUFFER_PG database
1422// 2. Update port's info: speed, cable length and mtu
1423// 3. If any of the PGs is updated, recalculate pool size
1424// 4. try release each profile in to-be-released profile set
1425task_process_status BufferMgrDynamic::refreshPgsForPort(const string &port, const string &speed, const string &cable_length, const string &mtu, const string &exactly_matched_key = "")
1426{
1427 port_info_t &portInfo = m_portInfoLookup[port];
1428 string &gearbox_model = portInfo.gearbox_model;
1429 long laneCount = portInfo.lane_count;
1430 bool isHeadroomUpdated = false;
1431 buffer_pg_lookup_t &portPgs = m_portPgLookup[port];
1432 set<string> profilesToBeReleased;
1433
1434 if (portInfo.state == PORT_ADMIN_DOWN)
1435 {
1436 SWSS_LOG_INFO("Nothing to be done since the port %s is administratively down", port.c_str());
1437 return task_process_status::task_success;
1438 }
1439
1440 if (!m_bufferPoolReady || m_defaultThreshold.empty())
1441 {
1442 SWSS_LOG_INFO("Nothing to be done since either the buffer pool or default threshold is not ready");
1443 m_bufferObjectsPending = true;
1444 return task_process_status::task_success;
1445 }
1446
1447 SWSS_LOG_NOTICE("Refresh priority groups for port %s", port.c_str());
1448
1449 // Iterate all the lossless PGs configured on this port
1450 for (auto it = portPgs.begin(); it != portPgs.end(); ++it)
1451 {
1452 auto &key = it->first;
1453 if (!exactly_matched_key.empty() && key != exactly_matched_key)
1454 {
1455 SWSS_LOG_DEBUG("Update buffer PGs: key %s doesn't match %s, skipped ", key.c_str(), exactly_matched_key.c_str());
1456 continue;
1457 }
1458 auto &portPg = it->second;
1459 string newProfile, oldProfile;
1460
1461 oldProfile = portPg.running_profile_name;
1462
1463 if (portPg.dynamic_calculated)
1464 {
1465 if (portInfo.state != PORT_READY)
1466 {
1467 SWSS_LOG_INFO("Nothing to be done for %s since port is not ready", key.c_str());
1468 continue;
1469 }
1470
1471 string threshold;
1472 // Calculate new headroom size
1473 if (portPg.static_configured)
1474 {
1475 // static_configured but dynamic_calculated means non-default threshold value
1476 auto &profile = m_bufferProfileLookup[portPg.configured_profile_name];
1477 threshold = profile.threshold;
1478 }
1479 else
1480 {
1481 threshold = m_defaultThreshold;
1482 }
1483 newProfile = getDynamicProfileName(speed, cable_length, mtu, threshold, gearbox_model, laneCount);
1484 auto rc = allocateProfile(speed, cable_length, mtu, threshold, gearbox_model, laneCount, newProfile);
1485 if (task_process_status::task_success != rc)
1486 return rc;
1487
1488 SWSS_LOG_DEBUG("Handling PG %s port %s, for dynamically calculated profile %s", key.c_str(), port.c_str(), newProfile.c_str());
1489 }
1490 else
1491 {
1492 newProfile = portPg.configured_profile_name;
1493
1494 SWSS_LOG_DEBUG("Handling PG %s port %s, for static profile %s", key.c_str(), port.c_str(), newProfile.c_str());
1495 }
1496
1497 // Calculate whether accumulative headroom size exceeds the maximum value
1498 // Abort if it does
1499 if (!isHeadroomResourceValid(port, m_bufferProfileLookup[newProfile], exactly_matched_key))
1500 {
1501 SWSS_LOG_ERROR("Update speed (%s) and cable length (%s) for port %s failed, accumulative headroom size exceeds the limit",
1502 speed.c_str(), cable_length.c_str(), port.c_str());
1503
1504 releaseProfile(newProfile);
1505
1506 return task_process_status::task_failed;
1507 }
1508
1509 if (newProfile != oldProfile)
1510 {
1511 // Need to remove the reference to the old profile
1512 // and create the reference to the new one
1513 m_bufferProfileLookup[newProfile].port_pgs.insert(key);
1514 SWSS_LOG_INFO("Move profile reference for %s from [%s] to [%s]", key.c_str(), oldProfile.c_str(), newProfile.c_str());
1515
1516 // Add the old profile to "to be removed" set
1517 if (!oldProfile.empty())
1518 {
1519 profilesToBeReleased.insert(oldProfile);
1520 m_bufferProfileLookup[oldProfile].port_pgs.erase(key);
1521 }
1522
1523 // buffer pg needs to be updated as well
1524 portPg.running_profile_name = newProfile;
1525 }
1526
1527 // appl_db Database operation: set item BUFFER_PG|<port>|<pg>
1528 updateBufferObjectToDb(key, newProfile, true);
1529 isHeadroomUpdated = true;
1530 }
1531
1532 if (isHeadroomUpdated)
1533 {
1534 checkSharedBufferPoolSize();
1535 }
1536 else
1537 {
1538 SWSS_LOG_DEBUG("Nothing to do for port %s since no PG configured on it", port.c_str());
1539 }
1540
1541 // Remove the old profile which is probably not referenced anymore.
1542 if (!profilesToBeReleased.empty())
1543 {
1544 for (auto &oldProfile : profilesToBeReleased)
1545 {
1546 releaseProfile(oldProfile);
1547 }
1548 }
1549
1550 return task_process_status::task_success;

1553void BufferMgrDynamic::refreshSharedHeadroomPool(bool enable_state_updated_by_ratio, bool enable_state_updated_by_size)
1554{
1555 // The lossless profiles need to be refreshed only if system is switched between SHP and non-SHP
1556 bool need_refresh_profiles = false;
1557 bool shp_enabled_by_size = isNonZero(m_configuredSharedHeadroomPoolSize);
1558 bool shp_enabled_by_ratio = isNonZero(m_overSubscribeRatio);
1559
1560 /*
1561 * STATE EVENT ACTION
1562 * enabled by size -> config ratio: no action
1563 * enabled by size -> remove ratio: no action
1564 * enabled by size -> remove size: shared headroom pool disabled
1565 * SHP size will be set here (zero) and programmed to APPL_DB in handleBufferPoolTable
1566 * enabled by ratio -> config size: SHP size will be set here (non zero) and programmed to APPL_DB in handleBufferPoolTable
1567 * enabled by ratio -> remove size: SHP size will be set and programmed to APPL_DB during buffer pool size update
1568 * enabled by ratio -> remove ratio: shared headroom pool disabled
1569 * dynamic size: SHP size will be handled and programmed to APPL_DB during buffer pool size update
1570 * static size: SHP size will be set (zero) and programmed to APPL_DB here
1571 * disabled -> config ratio: shared headroom pool enabled. all lossless profiles refreshed
1572 * SHP size will be handled during buffer pool size update
1573 * disabled -> config size: shared headroom pool enabled. all lossless profiles refreshed
1574 * SHP size will be handled here and programmed to APPL_DB during buffer pool table handling
1575 */
1576
1577 auto &ingressLosslessPool = m_bufferPoolLookup[INGRESS_LOSSLESS_PG_POOL_NAME];
1578 if (enable_state_updated_by_ratio)
1579 {
1580 if (shp_enabled_by_size)
1581 {
1582 // enabled by size -> config or remove ratio, no action
1583 SWSS_LOG_INFO("SHP: No need to refresh lossless profiles even if enable state updated by over subscribe ratio, already enabled by SHP size");
1584 }
1585 else
1586 {
1587 // enabled by ratio -> remove ratio
1588 // disabled -> config ratio
1589 need_refresh_profiles = true;
1590 }
1591 }
1592
1593 if (enable_state_updated_by_size)
1594 {
1595 if (shp_enabled_by_ratio)
1596 {
1597 // enabled by ratio -> config size, size will be updated (later in this function)
1598 // enabled by ratio -> remove size, no action here, will be handled during buffer pool size update
1599 SWSS_LOG_INFO("SHP: No need to refresh lossless profiles even if enable state updated by SHP size, already enabled by over subscribe ratio");
1600 }
1601 else
1602 {
1603 // disabled -> config size
1604 // enabled by size -> remove size
1605 need_refresh_profiles = true;
1606 }
1607 }
1608
1609 if (need_refresh_profiles)
1610 {
1611 SWSS_LOG_NOTICE("Updating dynamic buffer profiles due to shared headroom pool state updated");
1612
1613 for (auto it = m_bufferProfileLookup.begin(); it != m_bufferProfileLookup.end(); ++it)
1614 {
1615 auto &name = it->first;
1616 auto &profile = it->second;
1617 if (profile.static_configured)
1618 {
1619 SWSS_LOG_INFO("Non dynamic profile %s skipped", name.c_str());
1620 continue;
1621 }
1622 SWSS_LOG_INFO("Updating profile %s with speed %s cable length %s mtu %s gearbox model %s",
1623 name.c_str(),
1624 profile.speed.c_str(), profile.cable_length.c_str(), profile.port_mtu.c_str(), profile.gearbox_model.c_str());
1625 // recalculate the headroom size
1626 calculateHeadroomSize(profile);
1627 if (task_process_status::task_success != doUpdateBufferProfileForSize(profile, false))
1628 {
1629 SWSS_LOG_ERROR("Failed to update buffer profile %s when toggle shared headroom pool. See previous message for detail. Please adjust the configuration manually", name.c_str());
1630 }
1631 }
1632 SWSS_LOG_NOTICE("Updating dynamic buffer profiles finished");
1633 }
1634
1635 if (shp_enabled_by_size)
1636 {
1637 ingressLosslessPool.xoff = m_configuredSharedHeadroomPoolSize;
1638 if (isNonZero(ingressLosslessPool.total_size))
1639 updateBufferPoolToDb(INGRESS_LOSSLESS_PG_POOL_NAME, ingressLosslessPool);
1640 }
1641 else if (!shp_enabled_by_ratio && enable_state_updated_by_ratio)
1642 {
1643 // shared headroom pool is enabled by ratio and will be disabled
1644 // need to program APPL_DB because nobody else will take care of it
1645 ingressLosslessPool.xoff = "0";
1646 if (isNonZero(ingressLosslessPool.total_size))
1647 updateBufferPoolToDb(INGRESS_LOSSLESS_PG_POOL_NAME, ingressLosslessPool);
1648 }
1649
1650 if (m_portInitDone)
1651 {
1652 checkSharedBufferPoolSize();
1653 }
1654}

1816/*
1817 * handleBufferMaxParam, handles the BUFFER_MAX_PARAMETER table which contains some thresholds related to buffer.
1818 * The available fields depend on the key of the item:
1819 * 1. "global", available keys:
1820 * - mmu_size, represents the maximum buffer size of the chip
1821 * 2. "<port name>", available keys:
1822 * - max_priority_groups, represents the maximum number of priority groups of the port.
1823 * It is pushed into STATE_DB by ports orchagent when it starts and will be used to generate the default priority groups to reclaim.
1824 * When reserved buffer is reclaimed for a port, and no priority group IDs specified in the "control_fields" in json file,
1825 * the default priority groups will be used to apply zero buffer profiles on all priority groups.
1826 * - max_queues, represents the maximum number of queues of the port.
1827 * It is used in the same way as max_priority_groups.
1828 * - max_headroom_size, represents the maximum headroom size of the port.
1829 * It is used for checking whether the accumulative headroom of the port exceeds the port's threshold
1830 * before applying a new priority group on a port or changing an existing buffer profile.
1831 * It is referenced by lua plugin "check headroom size" only and will not be handled in this function.
1832 */
1833task_process_status BufferMgrDynamic::handleBufferMaxParam(KeyOpFieldsValuesTuple &tuple)
1834{
1835 string &op = kfvOp(tuple);
1836 string &key = kfvKey(tuple);
1837
1838 if (op == SET_COMMAND)
1839 {
1840 if (key != "global")
1841 {
1842 const auto &portSearchRef = m_portInfoLookup.find(key);
1843 if (portSearchRef == m_portInfoLookup.end())

1972task_process_status BufferMgrDynamic::handleCableLenTable(KeyOpFieldsValuesTuple &tuple)
1973{
1974 string op = kfvOp(tuple);
1975
1976 task_process_status task_status = task_process_status::task_success;
1977 int failed_item_count = 0;
1978 if (op == SET_COMMAND)
1979 {
1980 m_cableLengths.clear();
1981 for (auto i : kfvFieldsValues(tuple))
1982 {
1983 // receive and cache cable length table
1984 auto &port = fvField(i);
1985 auto &cable_length = fvValue(i);
1986 port_info_t &portInfo = m_portInfoLookup[port];
1987 string &effectiveSpeed = portInfo.effective_speed;
1988 string &mtu = portInfo.mtu;
1989
1990 SWSS_LOG_DEBUG("Handling CABLE_LENGTH table field %s length %s", port.c_str(), cable_length.c_str());
1991 SWSS_LOG_DEBUG("Port Info for %s before handling %s %s %s",
1992 port.c_str(),
1993 portInfo.effective_speed.c_str(), portInfo.cable_length.c_str(), portInfo.gearbox_model.c_str());
1994
1995 m_cableLengths[port] = cable_length;
1996
1997 if (portInfo.cable_length == cable_length)
1998 {
1999 continue;
2000 }
2001
2002 portInfo.cable_length = cable_length;
2003 if (effectiveSpeed.empty())
2004 {
2005 SWSS_LOG_WARN("Speed for %s hasn't been configured yet, unable to calculate headroom", port.c_str());
2006 // We don't retry here because it doesn't make sense until the speed is configured.
2007 continue;
2008 }
2009
2010 if (mtu.empty())
2011 {
2012 // During initialization, the flow can be
2013 // 1. mtu, cable_length, speed
2014 // 2. cable_length, speed, mtu, or speed, cable_length, mtu
2015 // 3. cable_length, speed, but no mtu specified, or speed, cable_length but no mtu
2016 // it's ok for the 1st case
2017 // but for the 2nd case, we can't wait mtu for calculating the headrooms
2018 // because we can't distinguish case 2 from case 3 which is also legal.
2019 // So once we have speed updated, let's try to calculate the headroom with default mtu.
2020 // The cost is that if the mtu is provided in the following iteration
2021 // the current calculation is of no value and will be replaced.
2022 mtu = DEFAULT_MTU_STR;
2023 }

2028 switch (portInfo.state)
2029 {
2030 case PORT_INITIALIZING:
2031 portInfo.state = PORT_READY;
2032 task_status = refreshPgsForPort(port, effectiveSpeed, cable_length, mtu);
2033 break;
2034
2035 case PORT_READY:
2036 task_status = refreshPgsForPort(port, effectiveSpeed, cable_length, mtu);
2037 break;
2038
2039 case PORT_ADMIN_DOWN:
2040 // Nothing to be done here
2041 SWSS_LOG_INFO("Nothing to be done when port %s's cable length updated", port.c_str());
2042 task_status = task_process_status::task_success;
2043 break;
2044 }
2045
2046 switch (task_status)
2047 {
2048 case task_process_status::task_need_retry:
2049 return task_status;
2050 case task_process_status::task_failed:
2051 // We shouldn't return directly here. Because doing so will cause the following items lost
2052 failed_item_count++;
2053 break;
2054 default:
2055 break;
2056 }
2057
2058 SWSS_LOG_DEBUG("Port Info for %s after handling speed %s cable %s gb %s",
2059 port.c_str(),
2060 portInfo.effective_speed.c_str(), portInfo.cable_length.c_str(), portInfo.gearbox_model.c_str());
2061 }
2062 }
2063
2064 if (failed_item_count > 0)
2065 {
2066 return task_process_status::task_failed;
2067 }
2068
2069 return task_process_status::task_success;

2364 SWSS_LOG_DEBUG("Processing command:%s table BUFFER_POOL key %s", op.c_str(), pool.c_str());
2365 if (op == SET_COMMAND)

2371 string newSHPSize = "0";
2372
2373 bufferPool.dynamic_size = true;
2374 for (auto i = kfvFieldsValues(tuple).begin(); i != kfvFieldsValues(tuple).end(); i++)
2375 {
2376 string &field = fvField(*i);
2377 string &value = fvValue(*i);

2386 newSHPSize = value;

2388 else if (field == buffer_pool_mode_field_name)

2392 else if (field == buffer_pool_type_field_name)

2395 bufferPool.direction = BUFFER_INGRESS;

2397 bufferPool.direction = BUFFER_EGRESS;

2400 SWSS_LOG_INFO("Inserting BUFFER_POOL table field %s value %s", field.c_str(), value.c_str());

2404 if (pool == INGRESS_LOSSLESS_PG_POOL_NAME)

2408 * Don't update | pool size | SHP enabled by size | SHP enabled by over subscribe ratio
2409 * True | Dynamic | Any | Any
2410 * False | Static | True | Any
2411 * True | Static | False | True
2412 * False | Static | False | False
2413 */
2414 bool willSHPBeEnabledBySize = isNonZero(newSHPSize);

2418
2419 m_configuredSharedHeadroomPoolSize = newSHPSize;
2420 refreshSharedHeadroomPool(false, isSHPEnabledBySize != willSHPBeEnabledBySize);
2421 }

2425 }

2428 {
2429 // Don't need to program APPL_DB if shared headroom pool is enabled
2430 dontUpdatePoolToDb |= isNonZero(m_overSubscribeRatio);
2431 }
2432 }
2433 else if (isNonZero(newSHPSize))
2434 {
2435 SWSS_LOG_ERROR("Field xoff is supported for %s only, but got for %s, ignored", INGRESS_LOSSLESS_PG_POOL_NAME, pool.c_str());
2436 }

2439 {
2440 m_applBufferPoolTable.set(pool, fvVector);
2441 m_stateBufferPoolTable.set(pool, fvVector);
2442 }
2443 }
2444 else if (op == DEL_COMMAND)
2445 {

2447 m_applBufferPoolTable.del(pool);
2448 m_stateBufferPoolTable.del(pool);
2449 m_bufferPoolLookup.erase(pool);
2450 if (pool == INGRESS_LOSSLESS_PG_POOL_NAME)

2694void BufferMgrDynamic::handleSetSingleBufferObjectOnAdminDownPort(buffer_direction_t direction, const string &port, const string &key, const string &profile)
2695{
2696 auto &idsToZero = m_bufferObjectIdsToZero[direction];
2697 auto &table = m_applBufferObjectTables[direction];
2698 auto const &objectName = m_bufferObjectNames[direction];
2699 auto &portInfo = m_portInfoLookup[port];
2700
2701 if (m_portInitDone)
2702 {
2703 if (idsToZero.empty())
2704 {
2705 // Happens only after "config qos reload"
2706 if (!m_zeroProfilesLoaded)
2707 {
2708 loadZeroPoolAndProfiles();
2709 }
2710
2711 // If initialization finished, no extra handle required.
2712 // Check whether the key overlaps with supported but not configured map
2713 auto const &idsToAdd = parseObjectNameFromKey(key, 1);
2714 auto idsToAddBitmap = generateBitMapFromIdsStr(idsToAdd);
2715 auto &supportedButNotConfiguredItems = portInfo.supported_but_not_configured_buffer_objects[direction];
2716 unsigned long overlappedUnconfiguredIdsMap = 0;
2717 string overlappedUnconfiguredIdsStr;
2718 const string &portPrefix = port + delimiter;
2719 for (auto const &ids : supportedButNotConfiguredItems)
2720 {
2721 /*
2722 * An exmaple to explain the logic
2723 * Item no 01234567
2724 * Supported but not
2725 * configured items: -**-****
2726 * Items to be added: -----**-
2727 * Slice to be removed
2728 * in the next iternation: ----****
2729 * Legend: *: valid -: empty
2730 * Currently, supported but not configured items: 1-2, 4-7
2731 * Items to be added: 5-6
2732 * It will result in
2733 * - items 4-7 to be removed from APPL_DB
2734 * - items 4, 5-6, 7 to be added to APPL_DB
2735 */
2736 auto idsBitmap = generateBitMapFromIdsStr(ids);
2737 if ((idsToAddBitmap & idsBitmap) == idsToAddBitmap)
2738 {
2739 // Ids to add overlaps with this slice of unconfigured IDs
2740 // Remove the slice and regenerate them
2741 // It should be slice 4-7 in the above example
2742 auto const &keyToRemove = portPrefix + ids;
2743 SWSS_LOG_INFO("Buffer %s %s overlapped with existing zero item %s, remove the latter first",
2744 objectName.c_str(), key.c_str(), keyToRemove.c_str());
2745 table.del(keyToRemove);
2746 overlappedUnconfiguredIdsMap = (idsBitmap ^ idsToAddBitmap);
2747 overlappedUnconfiguredIdsStr = ids;
2748 break;
2749 }
2750 /*
2751 * Result after iteration:
2752 * overlappedUnconfiguredIdsMap: ----*--* (Items to be added has been removed from this map)
2753 * overlappedUnconfiguredIdsStr: ----****
2754 */
2755 }
2756 // Regenerate unconfigured IDs
2757 supportedButNotConfiguredItems.erase(overlappedUnconfiguredIdsStr);
2758 auto const &splitIds = generateIdListFromMap(overlappedUnconfiguredIdsMap, portInfo.maximum_buffer_objects[direction]);
2759 /*
2760 * Slices returned: ----*--*
2761 * Readd zero profiles for each slice in the set
2762 */
2763 for (auto const &splitId : splitIds)
2764 {
2765 auto const &keyToAdd = portPrefix + splitId;
2766 updateBufferObjectToDb(keyToAdd, m_bufferZeroProfileName[direction], true, direction);
2767 supportedButNotConfiguredItems.insert(splitId);
2768 SWSS_LOG_INFO("Add %s zero item %s", objectName.c_str(), keyToAdd.c_str());
2769 }
2770
2771 auto const &zeroProfile = fetchZeroProfileFromNormalProfile(profile);
2772 if (!zeroProfile.empty())
2773 {
2774 updateBufferObjectToDb(key, zeroProfile, true, direction);
2775 SWSS_LOG_INFO("Add configured %s item %s as zero profile", objectName.c_str(), key.c_str());
2776 }
2777 else
2778 {
2779 updateBufferObjectToDb(key, zeroProfile, false, direction);
2780 SWSS_LOG_INFO("No zero profile defined for %s, reclaiming buffer by removing buffer %s %s",
2781 profile.c_str(), objectName.c_str(), key.c_str());
2782 }
2783 }
2784 return;
2785 }
2786
2787 if (!idsToZero.empty())
2788 {
2789 if (port + delimiter + idsToZero != key)
2790 {
2791 // For admin-down ports, if zero profiles have been applied on specified items,
2792 // notify APPL_DB to remove only if the item ID doesn't equal specified items
2793 // and the system is during initialization
2794 // This is to guarantee the port can be "ready" in orchagent
2795 // In case the port is admin down during initialization, the PG will be removed from the port,
2796 // which effectively notifies bufferOrch to add the item to the m_ready_list
2797 table.del(key);
2798 }
2799 }
2800 else
2801 {
2802 // Notify APPL_DB to add zero profile to items
2803 // An side effect is the items will be ready in orchagent (added to m_ready_list)
2804 string &zeroProfile = fetchZeroProfileFromNormalProfile(profile);
2805 if (!zeroProfile.empty())
2806 {
2807 SWSS_LOG_INFO("Reclaim reserved buffer for %s %s by applying zero profile %s", objectName.c_str(), key.c_str(), zeroProfile.c_str());
2808 updateBufferObjectToDb(key, zeroProfile, true, direction);
2809 }
2810 else
2811 {
2812 SWSS_LOG_INFO("Reclaim reserved buffer for %s %s by removing it due to empty zero profile", objectName.c_str(), key.c_str());
2813 updateBufferObjectToDb(key, zeroProfile, false, direction);
2814 }
2815 }

3407 * We do not apply any buffer profiles and objects until the buffer pools are ready.
3408 * This is to prevent buffer orchagent from keeping retrying if buffer profiles and objects are inserted to APPL_DB while buffer pools are not.
3409 *
3410 * Originally, all buffer profiles and objects were applied to APPL_DB as soon as they were learnt from CONFIG_DB.
3411 * The buffer profiles, and objects will be available as soon as they are received from CONFIG_DB when the system is starting
3412 * but it takes more seconds for the buffer pools to be applied to APPL_DB
3413 * because the sizes need to be calculated first and can be calculated only after some other data available.
3414 * This causes buffer orchagent receives buffer profiles and objects first and then buffer pools.
3415 * The buffer orchagent has to keep retrying in such case, which wastes time and incurs error messages.
3416 *
3417 * Now, we will guarantee the buffer profiles and objects to be applied only after buffer pools are ready.
3418 * This is achieved by the following steps:
3419 * - m_bufferPoolReady indicates whether the buffer pools are ready.
3420 * - When the buffer profiles and objects are received from CONFIG_DB,
3421 * they will be handled and stored in internal data struct
3422 * (but not applied to APPL_DB until the buffer pools are ready, AKA. m_bufferPoolReady == true)
3423 * - Once buffer pools are ready, all stored buffer profiles and objects will be applied to APPL_DB by this function.
3424 *
3425 * Applying additional zero profiles, which is supported but not configured profiles, if any, will be deferred in non-warmreboot.
3426 * This is to accelerate the start flow especially for fast reboot.
3427 *
3428 * The progress of pushing buffer pools into APPL_DB has also been accelerated by the lua plugin for calculating buffer sizes.
3429 * Originally, if the data are not available for calculating buffer sizes, it just return an empty vectors and buffer manager will retry

3431 * - If there are buffer sizes in APPL_DB, return the sizes in APPL_DB.

3434 * The sizes will be updated to non zero when the data for calculating it are ready.
