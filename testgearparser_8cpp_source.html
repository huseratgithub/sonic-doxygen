87int unitTestMain()
88{
89 CU_pSuite pSuite = NULL;
90 CU_pTest pTest = NULL;
91 int ret = 0;
92
93 if (CUE_SUCCESS != CU_initialize_registry())
94 {
95 printf("%s: cunit failed to initialize registry\n", __FUNCTION__);
96 ret = CU_get_error();
97 }
98 else
99 {
100 CU_basic_set_mode(CU_BRM_VERBOSE);
101
102 // Run the tests and show the run summary
103 pSuite = CU_add_suite("gearbox_config_suite", 0, 0);
104 if (NULL == pSuite)
105 {
106 printf("%s: cunit failed to create suite\n", __FUNCTION__);
107 CU_cleanup_registry();
108 ret = CU_get_error();
109 }
110 }
111
112 // Add the test to the suite
113 if (ret == 0)
114 {
115 pTest = CU_add_test(pSuite, "gearbox_positive_config_test", positiveConfigTest);
116 if (pTest != NULL)
117 {
118 pTest = CU_add_test(pSuite, "gearbox_negative_config_test", negativeConfigTest);
119 }
120
121 if (pTest == NULL)
122 {
123 CU_cleanup_registry();
124 printf("%s: cunit failed to add test\n", __FUNCTION__);
125 ret = CU_get_error();
126 }
127 }
128
129 if (ret == 0)
130 {
131 CU_basic_run_tests();
132 }
133 return ret;
134}
