86sai_status_t sai_api_query(
87 _In_ sai_api_t sai_api_id,
88 _Out_ void** api_method_table)
89{
90 SWSS_LOG_ENTER();
91
92 if (api_method_table == NULL)
93 {
94 SWSS_LOG_ERROR("NULL method table passed to SAI API initialize");
95 return SAI_STATUS_INVALID_PARAMETER;
96 }
97
98 if (sai_api_id == SAI_API_UNSPECIFIED)
99 {
100 SWSS_LOG_ERROR("api ID is unspecified api");
101
102 return SAI_STATUS_INVALID_PARAMETER;
103 }
104
105 if (sai_metadata_get_enum_value_name(&sai_metadata_enum_sai_api_t, sai_api_id))
106 {
107 *api_method_table = ((void**)&vs_apis)[sai_api_id - 1];
108 return SAI_STATUS_SUCCESS;
109 }
110
111 SWSS_LOG_ERROR("Invalid API type %d", sai_api_id);
112
113 return SAI_STATUS_INVALID_PARAMETER;
114}
