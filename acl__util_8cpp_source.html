13bool parseAclTableAppDbActionField(const std::string &aggr_actions_str, std::vector<P4ActionParamName> *action_list,
14 std::vector<P4PacketActionWithColor> *action_color_list)
15{
16 try
17 {
18 const auto &j = nlohmann::json::parse(aggr_actions_str);
19 if (!j.is_array())
20 {
21 SWSS_LOG_ERROR("Invalid ACL table definition action %s, expecting an array.\n", aggr_actions_str.c_str());
22 return false;
23 }
24 P4ActionParamName action_with_param;
25 for (auto &action_item : j)
26 {
27 auto sai_action_it = action_item.find(kAction);
28 if (sai_action_it == action_item.end())
29 {
30 SWSS_LOG_ERROR("Invalid ACL table definition action %s, missing 'action':\n", aggr_actions_str.c_str());
31 return false;
32 }
33 if (aclPacketActionLookup.find(sai_action_it.value()) == aclPacketActionLookup.end())
34 {
35 action_with_param.sai_action = sai_action_it.value();
36 auto action_param_it = action_item.find(kActionParamPrefix);
37 if (action_param_it != action_item.end() && !action_param_it.value().is_null())
38 {
39 action_with_param.p4_param_name = action_param_it.value();
40 }
41 action_list->push_back(action_with_param);
42 }
43 else
44 {
45 auto packet_color_it = action_item.find(kPacketColor);
46 P4PacketActionWithColor packet_action_with_color;
47 packet_action_with_color.packet_action = sai_action_it.value();
48 if (packet_color_it != action_item.end() && !packet_color_it.value().is_null())
49 {
50 packet_action_with_color.packet_color = packet_color_it.value();
51 }
52 action_color_list->push_back(packet_action_with_color);
53 }
54 }
55 return true;
56 }
57 catch (std::exception &ex)
58 {
59 SWSS_LOG_ERROR("Failed to deserialize ACL table definition action fields: %s (%s)", aggr_actions_str.c_str(),
60 ex.what());
61 return false;
62 }

308 << "ACL table match field {" << p4_match << ": " << aggr_match_str
309 << "} is an invalid ACL table attribute: " << kAclMatchFieldFormat

327 (*udf_fields_lookup)[p4_match];

382 return rc;
383 }

387 if (sai_field_it != composite_sai_match_fields_lookup->end() && udf_field_it != udf_fields_lookup->end())

389 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)

391 << "} is an invalid ACL table attribute: composite element "
392 "'kind' should be consistent within all elements.";
393 }

395 // in composite fields

409 sai_field_it->second[1].table_attr == SAI_ACL_TABLE_ATTR_FIELD_DST_IPV6_WORD2) ||
410 (sai_field_it->second[0].table_attr == SAI_ACL_TABLE_ATTR_FIELD_SRC_IPV6_WORD3 &&
411 sai_field_it->second[1].table_attr == SAI_ACL_TABLE_ATTR_FIELD_SRC_IPV6_WORD2)))
412 {
413 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
414 << "ACL table match field {" << p4_match << ": " << aggr_match_str
415 << "} is an invalid ACL table attribute: For composite match "
416 "field "
417 "with element.kind == sai_field, the SAI match field "
418 "in elements list should be either pair {"
419 << P4_MATCH_DST_IPV6_WORD3 << ", " << P4_MATCH_DST_IPV6_WORD2 << "} or pair {"
420 << P4_MATCH_SRC_IPV6_WORD3 << ", " << P4_MATCH_SRC_IPV6_WORD2 << "} with the correct sequence";
421 }
422 total_bitwidth = sai_field_it->second[0].bitwidth + sai_field_it->second[1].bitwidth;
423 }
424 if (udf_field_it != udf_fields_lookup->end())
425 {
426 for (const auto &udf_field : udf_field_it->second)
427 {
428 total_bitwidth += (uint32_t)udf_field.length * BYTE_BITWIDTH;
429 }
430 }
431 if (total_bitwidth != composite_bitwidth)
432 {
433 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
434 << "ACL table match field {" << p4_match << ": " << aggr_match_str
435 << "} is an invalid ACL table attribute: composite bitwidth "
436 "does not equal with the sum of elements bitwidth.";
437 }
438 return ReturnCode();

676ReturnCode setCompositeSaiMatchValue(const acl_entry_attr_union_t attr_name, const std::string &attr_value,
677 sai_attribute_value_t *value)
678{
679 try
680 {
681 const auto &tokenized_ip = swss::tokenize(attr_value, kDataMaskDelimiter);
682 swss::IpAddress ip_data;
683 swss::IpAddress ip_mask;
684 if (tokenized_ip.size() == 2)
685 {
686 // data & mask
687 ip_data = swss::IpAddress(trim(tokenized_ip[0]));
688 if (ip_data.isV4())
689 {
690 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
691 << "IP data type should be v6 type: " << attr_value;
692 }
693 ip_mask = swss::IpAddress(trim(tokenized_ip[1]));
694 if (ip_mask.isV4())
695 {
696 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
697 << "IP mask type should be v6 type: " << attr_value;
698 }
699 }
700 else
701 {
702 // LPM annotated value
703 swss::IpPrefix ip_prefix(trim(attr_value));
704 if (ip_prefix.isV4())
705 {
706 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "IP type should be v6 type: " << attr_value;
707 }
708 ip_data = ip_prefix.getIp();
709 ip_mask = ip_prefix.getMask();
710 }
711 switch (attr_name)
712 {
713 case SAI_ACL_TABLE_ATTR_FIELD_DST_IPV6_WORD3:
714 case SAI_ACL_TABLE_ATTR_FIELD_SRC_IPV6_WORD3: {
715 // IPv6 Address 127:96 32 bits
716 memcpy(&value->aclfield.data.ip6[0], &ip_data.getV6Addr()[0], IPV6_SINGLE_WORD_BYTES_LENGTH);
717 memcpy(&value->aclfield.mask.ip6[0], &ip_mask.getV6Addr()[0], IPV6_SINGLE_WORD_BYTES_LENGTH);
718 break;
719 }
720 case SAI_ACL_TABLE_ATTR_FIELD_DST_IPV6_WORD2:
721 case SAI_ACL_TABLE_ATTR_FIELD_SRC_IPV6_WORD2: {
722 // IPv6 Address 95:64 32 bits
723 memcpy(&value->aclfield.data.ip6[IPV6_SINGLE_WORD_BYTES_LENGTH],
724 &ip_data.getV6Addr()[IPV6_SINGLE_WORD_BYTES_LENGTH], IPV6_SINGLE_WORD_BYTES_LENGTH);
725 memcpy(&value->aclfield.mask.ip6[IPV6_SINGLE_WORD_BYTES_LENGTH],
726 &ip_mask.getV6Addr()[IPV6_SINGLE_WORD_BYTES_LENGTH], IPV6_SINGLE_WORD_BYTES_LENGTH);
727 break;
728 }
729 default: {
730 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
731 << "ACL match field " << attr_name << " is not supported in composite match field sai_field type.";
732 }
733 }
734 }
735 catch (std::exception &e)
736 {
737 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM) << "Failed to parse match attribute " << attr_name
738 << " (value: " << attr_value << "). Error:" << e.what();
739 }
740 value->aclfield.enable = true;
741 return ReturnCode();

744ReturnCode setUdfMatchValue(const P4UdfField &udf_field, const std::string &attr_value, sai_attribute_value_t *value,
745 P4UdfDataMask *udf_data_mask, uint16_t bytes_offset)
746{
747 if (!udf_data_mask->data.empty() || !udf_data_mask->mask.empty())
748 {
749 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
750 << "Failed to set UDF match field " << udf_field.udf_id << " with value " << attr_value
751 << " in ACL rule: the UDF: duplicated UDF value found for the same "
752 "UDF field.";
753 }
754 try
755 {
756 // Extract UDF field values by length(in bytes) and offset(in bytes)
757 const std::vector<std::string> &value_and_mask = swss::tokenize(attr_value, kDataMaskDelimiter);
758 uint32_t data_str_offset = bytes_offset * 2, mask_str_offset = bytes_offset * 2;
759 const auto &data = trim(value_and_mask[0]);
760 if (data.size() > 2 && data[0] == '0' && (data[1] == 'x' || data[1] == 'X'))
761 {
762 data_str_offset += 2;
763 }
764 std::string mask = EMPTY_STRING;
765 if (value_and_mask.size() > 1)
766 {
767 mask = trim(value_and_mask[1]);
768 if (mask.size() > 2 && mask[0] == '0' && (mask[1] == 'x' || mask[1] == 'X'))
769 {
770 mask_str_offset += 2;
771 }
772 }
773 for (uint16_t i = 0; i < udf_field.length; i++)
774 {
775 // Add to udf_data uint8_t list
776 udf_data_mask->data.push_back(std::stoul(data.substr(data_str_offset, 2), nullptr, 16) & 0xFF);
777 data_str_offset += 2;
778 if (value_and_mask.size() > 1)
779 {
780 // Add to udf_mask uint8_t list
781 udf_data_mask->mask.push_back((std::stoul(mask.substr(mask_str_offset, 2), nullptr, 16)) & 0xFF);
782 mask_str_offset += 2;
783 }
784 else
785 {
786 udf_data_mask->mask.push_back(0xFF);
787 }
788 }
789 value->aclfield.data.u8list.count = udf_field.length;
790 value->aclfield.data.u8list.list = udf_data_mask->data.data();
791 value->aclfield.mask.u8list.count = udf_field.length;
792 value->aclfield.mask.u8list.list = udf_data_mask->mask.data();
793 }
794 catch (std::exception &ex)
795 {
796 return ReturnCode(StatusCode::SWSS_RC_INVALID_PARAM)
797 << "Failed to set UDF match field " << udf_field.udf_id << " with value " << attr_value
798 << " in ACL rule: " << ex.what();
799 }
800 value->aclfield.enable = true;
801 return ReturnCode();
802}
