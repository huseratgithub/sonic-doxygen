231 * Remove the group, releasing the SAI ID.
232 */
233 virtual bool remove()
234 {
235 SWSS_LOG_ENTER();
236
237 /*
238 * If the group is already removed, there is nothing to be done.
239 */
240 if (!isSynced())
241 {
242 return true;
243 }
244
245 /*
246 * Remove the group members.
247 */
248 set<MbrKey> members;
249
250 for (const auto &member : m_members)
251 {
252 members.insert(member.first);
253 }
254
255 if (!removeMembers(members))
256 {
257 SWSS_LOG_ERROR("Failed to remove next hop group %s members",
258 to_string().c_str());
259 return false;
260 }
261
262 /*
263 * Remove the NHG over SAI.
264 */
265 auto status = sai_next_hop_group_api->remove_next_hop_group(m_id);
266
267 if (status != SAI_STATUS_SUCCESS)
268 {
269 SWSS_LOG_ERROR("Failed to remove next hop group %s, rv: %d",
270 to_string().c_str(), status);
271 return false;
272 }
273
274 /*
275 * Decrease the number of programmed NHGs.
276 */
277 gCrmOrch->decCrmResUsedCounter(CrmResourceType::CRM_NEXTHOP_GROUP);
278 decSyncedCount();
279
280 /*
281 * Reset the group ID.
282 */
283 m_id = SAI_NULL_OBJECT_ID;
284
285 return true;
286 }

309 /*
310 * Remove the given members from the group.
311 */
312 virtual bool removeMembers(const set<MbrKey> &member_keys)
313 {
314 SWSS_LOG_ENTER();
315
316 /*
317 * Remove all the given members from the group.
318 */
319 ObjectBulker<sai_next_hop_group_api_t> bulker(sai_next_hop_group_api,
320 gSwitchId,
321 gMaxBulkSize);
322 map<MbrKey, sai_status_t> statuses;
323
324 for (const auto &key : member_keys)
325 {
326 const auto &nhgm = m_members.at(key);
327
328 if (nhgm.isSynced())
329 {
330 bulker.remove_entry(&statuses[key], nhgm.getId());
331 }
332 }
333
334 /*
335 * Flush the bulker to remove the members.
336 */
337 bulker.flush();
338
339 /*
340 * Iterate over the returned statuses and check if the removal was
341 * successful. If it was, remove the member, otherwise log an error
342 * message.
343 */
344 bool success = true;
345
346 for (const auto &status : statuses)
347 {
348 auto &member = m_members.at(status.first);
349
350 if (status.second == SAI_STATUS_SUCCESS)
351 {
352 member.remove();
353 }
354 else
355 {
356 SWSS_LOG_ERROR("Failed to remove next hop group member %s, rv: %d",
357 member.to_string().c_str(),
358 status.second);
359 success = false;
360 }
361 }
362
363 return success;
364 }

373/*
374 * Structure describing a next hop group which NhgOrch owns. Beside having a
375 * unique pointer to that next hop group, we also want to keep a ref count so
376 * NhgOrch knows how many other objects reference the next hop group in order
377 * not to remove them while still being referenced.
378 */
379template <typename NhgClass>
380struct NhgEntry
381{
382 /* Pointer to the next hop group. NhgOrch is the sole owner of it. */
383 std::unique_ptr<NhgClass> nhg;
384
385 /* Number of external objects referencing this next hop group. */
386 unsigned ref_count;
387
388 NhgEntry() = default;
389 explicit NhgEntry(std::unique_ptr<NhgClass>&& _nhg,
390 unsigned int _ref_count = 0) :
391 nhg(std::move(_nhg)), ref_count(_ref_count) {}
392};
