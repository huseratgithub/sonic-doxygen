92void SwitchStateBase::findBridgeVlanForPortVlan(
93 _In_ sai_object_id_t port_id,
94 _In_ sai_vlan_id_t vlan_id,
95 _Inout_ sai_object_id_t &bv_id,
96 _Inout_ sai_object_id_t &bridge_port_id)
97{
98 SWSS_LOG_ENTER();
99
100 bv_id = SAI_NULL_OBJECT_ID;
101 bridge_port_id = SAI_NULL_OBJECT_ID;
102
103 sai_object_id_t bridge_id;
104
105 /*
106 * The bridge port lookup process is two steps:
107 *
108 * - use (vlan_id, physical port_id) to match any .1D bridge port created.
109 * If there is match, then quit, found=true
110 *
111 * - use (physical port_id) to match any .1Q bridge created. if there is a
112 * match, the quite, found=true.
113 *
114 * If found==true, generate fdb learn event on the .1D or .1Q bridge port.
115 * If not found, then do not generate fdb event. It means the packet is not
116 * received on the bridge port.
117 *
118 * XXX this is not whats happening here, we are just looking for any
119 * bridge id (as in our case this is shortcut, we will remove all bridge ports
120 * when we will use router interface based port/lag and no bridge
121 * will be found.
122 */
123
124 auto &objectHash = m_objectHash.at(SAI_OBJECT_TYPE_BRIDGE_PORT);
125
126 // iterate via all bridge ports to find match on port id
127
128 sai_object_id_t lag_id = SAI_NULL_OBJECT_ID;
129
130 if (getLagFromPort(port_id,lag_id))
131 {
132 SWSS_LOG_INFO("got lag %s for port %s",
133 sai_serialize_object_id(lag_id).c_str(),
134 sai_serialize_object_id(port_id).c_str());
135 }
136
137 bool bv_id_set = false;
138
139 for (auto it = objectHash.begin(); it != objectHash.end(); ++it)
140 {
141 sai_object_id_t bpid;
142
143 sai_deserialize_object_id(it->first, bpid);
144
145 sai_attribute_t attrs[2];
146
147 attrs[0].id = SAI_BRIDGE_PORT_ATTR_PORT_ID;
148 attrs[1].id = SAI_BRIDGE_PORT_ATTR_TYPE;
149
150 sai_status_t status = get(SAI_OBJECT_TYPE_BRIDGE_PORT, bpid, (uint32_t)(sizeof(attrs)/sizeof(attrs[0])), attrs);
151
152 if (status != SAI_STATUS_SUCCESS)
153 {
154 SWSS_LOG_WARN("failed to get attr PORT_ID and TYPE for bridge port %s",
155 sai_serialize_object_id(bpid).c_str());
156 continue;
157 }
158

161 // if port is member of lag, we should check if port_id is that LAG
162

171 continue;
172 }

175 {
176 // this is not expected port
177 continue;
178 }
179 }
180 else if (port_id != attrs[0].value.oid)
181 {
182 // this is not expected port
183 continue;
184 }
185
186 bridge_port_id = bpid;
187
188 // get the 1D bridge id if the bridge port type is subport
189 auto bp_type = attrs[1].value.s32;
190
191 SWSS_LOG_DEBUG("found bridge port %s of type %d",
192 sai_serialize_object_id(bridge_port_id).c_str(),
193 bp_type);
194
195 if (bp_type == SAI_BRIDGE_PORT_TYPE_SUB_PORT)
196 {
197 sai_attribute_t attr;
198 attr.id = SAI_BRIDGE_PORT_ATTR_BRIDGE_ID;
199
200 status = get(SAI_OBJECT_TYPE_BRIDGE_PORT, bridge_port_id, 1, &attr);
201
202 if (status != SAI_STATUS_SUCCESS)
203 {
204 break;
205 }
206
207 bridge_id = attr.value.oid;
208
209 SWSS_LOG_DEBUG("found bridge %s for port %s",
210 sai_serialize_object_id(bridge_id).c_str(),
211 sai_serialize_object_id(port_id).c_str());
212
213 attr.id = SAI_BRIDGE_ATTR_TYPE;
214

216
217 if (status != SAI_STATUS_SUCCESS)
218 {
219 break;
220 }
221
222 SWSS_LOG_DEBUG("bridge %s type is %d",
223 sai_serialize_object_id(bridge_id).c_str(),
224 attr.value.s32);
225 bv_id = bridge_id;
226 bv_id_set = true;
227 }
228 else
229 {

233

267 sai_serialize_object_id(port_id).c_str());

387void SwitchStateBase::process_packet_for_fdb_event(
388 _In_ sai_object_id_t portId,
389 _In_ const std::string& name,
390 _In_ const uint8_t *buffer,
391 _In_ size_t size)
392{
393 SWSS_LOG_ENTER();
394
395 // we would need hostif info here and maybe interface index, then we can
396 // find host info from index
397
398 uint32_t frametime = (uint32_t)time(NULL);
399
400 /*
401 * We add +2 in case if frame contains 1Q VLAN tag.
402 */
403
404 if (size < (sizeof(ethhdr) + 2))
405 {
406 SWSS_LOG_WARN("ethernet frame is too small: %zu", size);
407 return;
408 }
409
410 const ethhdr *eh = (const ethhdr*)buffer;
411
412 uint16_t proto = htons(eh->h_proto);
413
414 uint16_t vlan_id = DEFAULT_VLAN_NUMBER;
415
416 bool tagged = (proto == ETH_P_8021Q);
417
418 if (tagged)
419 {
420 // this is tagged frame, get vlan id from frame
421
422 uint16_t tci = htons(((const uint16_t*)&eh->h_proto)[1]); // tag is after h_proto field
423
424 vlan_id = tci & 0xfff;
425
426 if (vlan_id == 0xfff)
427 {
428 SWSS_LOG_WARN("invalid vlan id %u in ethernet frame on %s", vlan_id, name.c_str());
429 return;
430 }
431
432 if (vlan_id == 0)
433 {
434 // priority packet, frame should be treated as non tagged
435 tagged = false;
436 }
437 }
438
439 if (tagged == false)
440 {
441 // untagged ethernet frame
442
443 sai_attribute_t attr;
444
445#ifdef SAI_LAG_ATTR_PORT_VLAN_ID
446
447 sai_object_id_t lag_id;
448
449 if (getLagFromPort(portid, lag_id))
450 {
451 // if port belongs to lag we need to get SAI_LAG_ATTR_PORT_VLAN_ID
452
453 attr.id = SAI_LAG_ATTR_PORT_VLAN_ID
454
455 sai_status_t status = get(SAI_OBJECT_TYPE_LAG, lag_id, 1, &attr);
456
457 if (status != SAI_STATUS_SUCCESS)
458 {
459 SWSS_LOG_WARN("failed to get lag vlan id from lag %s",
460 sai_serialize_object_id(lag_id).c_str());
461 return;
462 }
463
464 vlan_id = attr.value.u16;
465
466 if (isLagOrPortRifBased(lag_id))
467 {
468 // this lag is router interface based, skip mac learning
469 return;
470 }
471 }
472 else
473#endif
474 {
475 attr.id = SAI_PORT_ATTR_PORT_VLAN_ID;
476
477 sai_status_t status = get(SAI_OBJECT_TYPE_PORT, portId, 1, &attr);
478
479 if (status != SAI_STATUS_SUCCESS)
480 {
481 SWSS_LOG_WARN("failed to get port vlan id from port %s",
482 sai_serialize_object_id(portId).c_str());
483 return;
484 }
485
486 // untagged port vlan (default is 1, but may change setting port attr)
487 vlan_id = attr.value.u16;
488 }
489 }
490
491 sai_object_id_t lag_id;
492 if (getLagFromPort(portId, lag_id) && isLagOrPortRifBased(lag_id))
493 {
494 SWSS_LOG_DEBUG("lag %s is rif based, skip mac learning for port %s",
495 sai_serialize_object_id(lag_id).c_str(),
496 sai_serialize_object_id(portId).c_str());
497 return;
498 }
499
500 if (isLagOrPortRifBased(portId))
501 {
502 SWSS_LOG_DEBUG("port %s is rif based, skip mac learning",
503 sai_serialize_object_id(portId).c_str());
504 return;
505 }
506
507 // we have vlan and mac address which is KEY, so just see if that is already defined
508
509 FdbInfo fi;
510
511 fi.setPortId((lag_id != SAI_NULL_OBJECT_ID) ? lag_id : portId);
512
513 fi.setVlanId(vlan_id);
514
515 memcpy(fi.m_fdbEntry.mac_address, eh->h_source, sizeof(sai_mac_t));
516
517 std::set<FdbInfo>::iterator it = m_fdb_info_set.find(fi);
518
519 if (it != m_fdb_info_set.end())
520 {
521 // this key was found, update timestamp
522 // and since iterator is const we need to reinsert
523
524 fi = *it;
525
526 fi.setTimestamp(frametime);
527
528 m_fdb_info_set.insert(fi);
529
530 return;
531 }
532
533 // key was not found, get additional information
534
535 fi.setTimestamp(frametime);
536
537 fi.m_fdbEntry.switch_id = m_switch_id;
538
539 findBridgeVlanForPortVlan(portId, vlan_id, fi.m_fdbEntry.bv_id, fi.m_bridgePortId);
540
541 if (fi.getFdbEntry().bv_id == SAI_NULL_OBJECT_ID)
542 {
543 SWSS_LOG_WARN("skipping mac learn for %s, since BV_ID was not found for mac",
544 sai_serialize_fdb_entry(fi.getFdbEntry()).c_str());
545
546 // bridge was not found, skip mac learning
547 return;
548 }
549
550 sai_attribute_t attr;
551
552 attr.id = SAI_BRIDGE_PORT_ATTR_FDB_LEARNING_MODE;
553
554 sai_status_t status = get(SAI_OBJECT_TYPE_BRIDGE_PORT, fi.getBridgePortId(), 1, &attr);
555
556 if (status == SAI_STATUS_SUCCESS)
557 {
558 if (attr.value.s32 == SAI_BRIDGE_PORT_FDB_LEARNING_MODE_HW)
559 {
560 SWSS_LOG_INFO("inserting to fdb_info set: %s, vlan id: %d",
561 sai_serialize_fdb_entry(fi.getFdbEntry()).c_str(),
562 fi.getVlanId());
563
564 m_fdb_info_set.insert(fi);
565
566 processFdbInfo(fi, SAI_FDB_EVENT_LEARNED);
567 }
568 else if (attr.value.s32 == SAI_BRIDGE_PORT_FDB_LEARNING_MODE_DISABLE)
569 {
570 // do not learn, actually linux kernel will learn that MAC
571 }
572 else
573 {
574 SWSS_LOG_WARN("not supported SAI_BRIDGE_PORT_ATTR_FDB_LEARNING_MODE: %d, for %s",
575 attr.value.s32,
576 sai_serialize_fdb_entry(fi.getFdbEntry()).c_str());
577 }
578 }
579 else
580 {
581 SWSS_LOG_ERROR("failed to get SAI_BRIDGE_PORT_ATTR_FDB_LEARNING_MODE for %s: %s",
582 sai_serialize_object_id(fi.getBridgePortId()).c_str(),
583 sai_serialize_status(status).c_str());
584 }
