45void NotificationHandler::updateNotificationsPointers(
46 _In_ uint32_t attr_count,
47 _In_ sai_attribute_t *attr_list) const
48{
49 SWSS_LOG_ENTER();
50
51 /*
52 * This function should only be called on CREATE/SET api when object is
53 * SWITCH.
54 *
55 * Notifications pointers needs to be corrected since those we receive from
56 * sairedis are in sairedis memory space and here we are using those ones
57 * we declared in syncd memory space.
58 *
59 * Also notice that we are using the same pointers for ALL switches.
60 */
61
62 for (uint32_t index = 0; index < attr_count; ++index)
63 {
64 sai_attribute_t &attr = attr_list[index];
65
66 auto meta = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_SWITCH, attr.id);
67
68 if (meta->attrvaluetype != SAI_ATTR_VALUE_TYPE_POINTER)
69 {
70 continue;
71 }
72
73 /*
74 * Does not matter if pointer is valid or not, we just want the
75 * previous value.
76 */
77
78 sai_pointer_t prev = attr.value.ptr;
79
80 if (prev == NULL)
81 {
82 /*
83 * If pointer is NULL, then fine, let it be.
84 */
85
86 continue;
87 }
88
89 switch (attr.id)
90 {
91 case SAI_SWITCH_ATTR_SWITCH_STATE_CHANGE_NOTIFY:
92 attr.value.ptr = (void*)m_switchNotifications.on_switch_state_change;
93 break;
94
95 case SAI_SWITCH_ATTR_SHUTDOWN_REQUEST_NOTIFY:
96 attr.value.ptr = (void*)m_switchNotifications.on_switch_shutdown_request;
97 break;
98
99 case SAI_SWITCH_ATTR_FDB_EVENT_NOTIFY:
100 attr.value.ptr = (void*)m_switchNotifications.on_fdb_event;
101 break;
102
103 case SAI_SWITCH_ATTR_NAT_EVENT_NOTIFY:
104 attr.value.ptr = (void*)m_switchNotifications.on_nat_event;
105 break;
106
107 case SAI_SWITCH_ATTR_PORT_STATE_CHANGE_NOTIFY:
108 attr.value.ptr = (void*)m_switchNotifications.on_port_state_change;
109 break;
110
111 case SAI_SWITCH_ATTR_QUEUE_PFC_DEADLOCK_NOTIFY:
112 attr.value.ptr = (void*)m_switchNotifications.on_queue_pfc_deadlock;
113 break;
114
115 case SAI_SWITCH_ATTR_BFD_SESSION_STATE_CHANGE_NOTIFY:
116 attr.value.ptr = (void*)m_switchNotifications.on_bfd_session_state_change;
117 break;
118
119 default:
120
121 SWSS_LOG_ERROR("pointer for %s is not handled, FIXME!", meta->attridname);
122 continue;
123 }
124
125 // Here we translated pointer, just log it.
126
127 SWSS_LOG_INFO("%s: 0x%" PRIx64 " (orch) => 0x%" PRIx64 " (syncd)", meta->attridname, (uint64_t)prev, (uint64_t)attr.value.ptr);
128 }
129}
