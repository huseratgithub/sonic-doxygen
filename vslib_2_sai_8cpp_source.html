60sai_status_t Sai::initialize(
61 _In_ uint64_t flags,
62 _In_ const sai_service_method_table_t *service_method_table)
63{
64 MUTEX();
65
66 SWSS_LOG_ENTER();
67
68 if (m_apiInitialized)
69 {
70 SWSS_LOG_ERROR("%s: api already initialized", __PRETTY_FUNCTION__);
71
72 return SAI_STATUS_FAILURE;
73 }
74
75 if (flags != 0)
76 {
77 SWSS_LOG_ERROR("invalid flags passed to SAI API initialize");
78
79 return SAI_STATUS_INVALID_PARAMETER;
80 }
81
82 if ((service_method_table == NULL) ||
83 (service_method_table->profile_get_next_value == NULL) ||
84 (service_method_table->profile_get_value == NULL))
85 {
86 SWSS_LOG_ERROR("invalid service_method_table handle passed to SAI API initialize");
87
88 return SAI_STATUS_INVALID_PARAMETER;
89 }
90
91 memcpy(&m_service_method_table, service_method_table, sizeof(m_service_method_table));
92
93 auto switch_type = service_method_table->profile_get_value(0, SAI_KEY_VS_SWITCH_TYPE);
94
95 if (switch_type == NULL)
96 {
97 SWSS_LOG_ERROR("failed to obtain service method table value: %s", SAI_KEY_VS_SWITCH_TYPE);
98
99 return SAI_STATUS_FAILURE;
100 }
101
102 auto sai_switch_type = service_method_table->profile_get_value(0, SAI_KEY_VS_SAI_SWITCH_TYPE);
103 sai_switch_type_t saiSwitchType;
104
105 if (sai_switch_type == NULL)
106 {
107 SWSS_LOG_NOTICE("failed to obtain service method table value: %s", SAI_KEY_VS_SAI_SWITCH_TYPE);
108 saiSwitchType = SAI_SWITCH_TYPE_NPU;
109 }
110 else if (!SwitchConfig::parseSaiSwitchType(sai_switch_type, saiSwitchType))
111 {
112 return SAI_STATUS_FAILURE;
113 }
114
115 auto *laneMapFile = service_method_table->profile_get_value(0, SAI_KEY_VS_INTERFACE_LANE_MAP_FILE);
116
117 m_laneMapContainer = LaneMapFileParser::parseLaneMapFile(laneMapFile);
118
119 auto *fabricLaneMapFile = service_method_table->profile_get_value(0, SAI_KEY_VS_INTERFACE_FABRIC_LANE_MAP_FILE);
120 if (fabricLaneMapFile)
121 {
122 m_fabricLaneMapContainer = LaneMapFileParser::parseLaneMapFile(fabricLaneMapFile);
123 }
124
125 auto *corePortIndexMapFile = service_method_table->profile_get_value(0, SAI_KEY_VS_CORE_PORT_INDEX_MAP_FILE);
126
127 m_corePortIndexMapContainer = CorePortIndexMapFileParser::parseCorePortIndexMapFile(corePortIndexMapFile);
128
129 auto *resourceLimiterFile = service_method_table->profile_get_value(0, SAI_KEY_VS_RESOURCE_LIMITER_FILE);
130
131 m_resourceLimiterContainer = ResourceLimiterParser::parseFromFile(resourceLimiterFile);
132
133 auto boot_type = service_method_table->profile_get_value(0, SAI_KEY_BOOT_TYPE);
134 m_warm_boot_read_file = service_method_table->profile_get_value(0, SAI_KEY_WARM_BOOT_READ_FILE);
135 m_warm_boot_write_file = service_method_table->profile_get_value(0, SAI_KEY_WARM_BOOT_WRITE_FILE);
136
137 sai_vs_boot_type_t bootType;
138
139 if (!SwitchConfig::parseBootType(boot_type, bootType))
140 {
141 return SAI_STATUS_FAILURE;
142 }
143
144 sai_vs_switch_type_t switchType;
145
146 if (!SwitchConfig::parseSwitchType(switch_type, switchType))
147 {
148 return SAI_STATUS_FAILURE;
149 }
150
151 const char *use_tap_dev = service_method_table->profile_get_value(0, SAI_KEY_VS_HOSTIF_USE_TAP_DEVICE);
152
153 auto useTapDevice = SwitchConfig::parseUseTapDevice(use_tap_dev);
154
155 SWSS_LOG_NOTICE("hostif use TAP device: %s", (useTapDevice ? "true" : "false"));
156
157 auto cstrGlobalContext = service_method_table->profile_get_value(0, SAI_KEY_VS_GLOBAL_CONTEXT);
158
159 m_globalContext = 0;
160
161 if (cstrGlobalContext != nullptr)
162 {
163 if (sscanf(cstrGlobalContext, "%u", &m_globalContext) != 1)
164 {
165 SWSS_LOG_WARN("failed to parse '%s' as uint32 using default globalContext", cstrGlobalContext);
166
167 m_globalContext = 0;
168 }
169 }
170
171 SWSS_LOG_NOTICE("using globalContext = %u", m_globalContext);
172
173 auto cstrContextConfig = service_method_table->profile_get_value(0, SAI_KEY_VS_CONTEXT_CONFIG);
174
175 auto ccc = ContextConfigContainer::loadFromFile(cstrContextConfig);
176
177 for (auto& cc: ccc->getAllContextConfigs())
178 {
179 auto context = std::make_shared<Context>(cc);
180
181 m_contextMap[cc->m_guid] = context;
182 }
183
184 auto context = getContext(m_globalContext);
185
186 if (context == nullptr)
187 {
188 SWSS_LOG_ERROR("no context defined for global context %u", m_globalContext);
189
190 return SAI_STATUS_FAILURE;
191 }
192
193 auto contextConfig = context->getContextConfig();
194
195 auto scc = contextConfig->m_scc;
196
197 if (scc->getSwitchConfigs().size() == 0)
198 {
199 SWSS_LOG_WARN("no switch configs defined, using default switch config");
200
201 auto sc = std::make_shared<SwitchConfig>(0, "");
202
203 scc->insert(sc);
204 }
205
206 // TODO currently switch configuration will share signal and event queue
207 // but it should be moved to Context class
208
209 m_signal = std::make_shared<Signal>();
210
211 m_eventQueue = std::make_shared<EventQueue>(m_signal);
212
213 for (auto& sc: scc->getSwitchConfigs())
214 {
215 // NOTE: switch index and hardware info is already populated
216
217 sc->m_saiSwitchType = saiSwitchType;
218 sc->m_switchType = switchType;
219 sc->m_bootType = bootType;
220 sc->m_useTapDevice = useTapDevice;
221 sc->m_laneMap = m_laneMapContainer->getLaneMap(sc->m_switchIndex);
222
223 if (sc->m_laneMap == nullptr)
224 {
225 SWSS_LOG_WARN("lane map for switch index %u is empty, loading default map (may have ifname conflict)", sc->m_switchIndex);
226
227 sc->m_laneMap = LaneMap::getDefaultLaneMap(sc->m_switchIndex);
228 }
229
230 if (m_fabricLaneMapContainer)
231 {
232 sc->m_fabricLaneMap = m_fabricLaneMapContainer->getLaneMap(sc->m_switchIndex);
233 }
234
235 sc->m_eventQueue = m_eventQueue;
236 sc->m_resourceLimiter = m_resourceLimiterContainer->getResourceLimiter(sc->m_switchIndex);
237 sc->m_corePortIndexMap = m_corePortIndexMapContainer->getCorePortIndexMap(sc->m_switchIndex);
238 }
239
240 // most important
241
242 // TODO move to Context class
243
244 m_vsSai = std::make_shared<VirtualSwitchSaiInterface>(contextConfig);
245
246 m_meta = std::make_shared<saimeta::Meta>(m_vsSai);
247
248 m_vsSai->setMeta(m_meta);
249
250 if (bootType == SAI_VS_BOOT_TYPE_WARM)
251 {
252 if (!m_vsSai->readWarmBootFile(m_warm_boot_read_file))
253 {
254 SWSS_LOG_WARN("failed to read warm boot read file, switching to COLD BOOT");
255
256 for (auto& sc: scc->getSwitchConfigs())
257 {
258 sc->m_bootType = SAI_VS_BOOT_TYPE_COLD;
259 }
260 }
261 }
262
263 startEventQueueThread();
264
265 startUnittestThread();
266
267 if (saiSwitchType == SAI_SWITCH_TYPE_NPU)
268 {
269 startFdbAgingThread();
270 }
271
272 m_apiInitialized = true;
273
274 return SAI_STATUS_SUCCESS;
275}

277sai_status_t Sai::uninitialize(void)
278{
279 SWSS_LOG_ENTER();
280 VS_CHECK_API_INITIALIZED();
281
282 SWSS_LOG_NOTICE("begin");
283
284 // no mutex on uninitialized to prevent deadlock
285 // if some thread would try to gram api mutex
286 // so threads must be stopped first
287
288 SWSS_LOG_NOTICE("stopping threads");
289
290 stopUnittestThread();
291
292 stopFdbAgingThread();
293
294 stopEventQueueThread();
295
296 // at this point packets may still arrive on hostif but event queue thread
297 // ended so they will not be processed
298
299 // NOTE: at this point we can have MUTEX();
300
301 // clear state after ending all threads
302
303 m_vsSai->writeWarmBootFile(m_warm_boot_write_file);
304
305 m_vsSai = nullptr;
306 m_meta = nullptr;
307
308 m_apiInitialized = false;
309
310 SWSS_LOG_NOTICE("end");
311
312 return SAI_STATUS_SUCCESS;
313}

347sai_status_t Sai::set(
348 _In_ sai_object_type_t objectType,
349 _In_ sai_object_id_t objectId,
350 _In_ const sai_attribute_t *attr)
351{
352 MUTEX();
353 SWSS_LOG_ENTER();
354 VS_CHECK_API_INITIALIZED();
355
356 if (objectType == SAI_OBJECT_TYPE_SWITCH)
357 {
358 if (attr)
359 {
360 if (attr->id == SAI_VS_SWITCH_ATTR_META_ENABLE_UNITTESTS)
361 {
362 m_meta->meta_unittests_enable(attr->value.booldata);
363 return SAI_STATUS_SUCCESS;
364 }
365
366 if (attr->id == SAI_VS_SWITCH_ATTR_META_ALLOW_READ_ONLY_ONCE)
367 {
368 return m_meta->meta_unittests_allow_readonly_set_once(SAI_OBJECT_TYPE_SWITCH, attr->value.s32);
369 }
370 }
371 }
372
373 return m_meta->set(objectType, objectId, attr);
374}
