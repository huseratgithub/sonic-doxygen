191void VirtualOidTranslator::translateRidToVid(
192 _In_ sai_object_type_t objectType,
193 _In_ sai_object_id_t switchVid,
194 _In_ uint32_t attr_count,
195 _Inout_ sai_attribute_t *attrList,
196 _In_ bool translateRemoved)
197{
198 SWSS_LOG_ENTER();
199
200 /*
201 * We receive real id's here, if they are new then create new VIDs for them
202 * and put in db, if entry exists in db, use it.
203 *
204 * NOTE: switch_id is VID of switch on which those RIDs are provided.
205 */
206
207 for (uint32_t i = 0; i < attr_count; i++)
208 {
209 sai_attribute_t &attr = attrList[i];
210
211 auto meta = sai_metadata_get_attr_metadata(objectType, attr.id);
212
213 if (meta == NULL)
214 {
215 SWSS_LOG_THROW("unable to get metadata for object type %x, attribute %d", objectType, attr.id);
216 }
217
218 /*
219 * TODO: Many times we do switch for list of attributes to perform some
220 * operation on each oid from that attribute, we should provide clever
221 * way via sai metadata utils to get that.
222 */
223
224 switch (meta->attrvaluetype)
225 {
226 case SAI_ATTR_VALUE_TYPE_OBJECT_ID:
227 attr.value.oid = translateRidToVid(attr.value.oid, switchVid, translateRemoved);
228 break;
229
230 case SAI_ATTR_VALUE_TYPE_OBJECT_LIST:
231 translateRidToVid(attr.value.objlist, switchVid, translateRemoved);
232 break;
233
234 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_ID:
235 if (attr.value.aclfield.enable)
236 attr.value.aclfield.data.oid = translateRidToVid(attr.value.aclfield.data.oid, switchVid, translateRemoved);
237 break;
238
239 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_LIST:
240 if (attr.value.aclfield.enable)
241 translateRidToVid(attr.value.aclfield.data.objlist, switchVid, translateRemoved);
242 break;
243
244 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_ID:
245 if (attr.value.aclaction.enable)
246 attr.value.aclaction.parameter.oid = translateRidToVid(attr.value.aclaction.parameter.oid, switchVid, translateRemoved);
247 break;
248
249 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_LIST:
250 if (attr.value.aclaction.enable)
251 translateRidToVid(attr.value.aclaction.parameter.objlist, switchVid, translateRemoved);
252 break;
253
254 default:
255
256 /*
257 * If in future new attribute with object id will be added this
258 * will make sure that we will need to add handler here.
259 */
260
261 if (meta->isoidattribute)
262 {
263 SWSS_LOG_THROW("attribute %s is object id, but not processed, FIXME", meta->attridname);
264 }
265
266 break;
267 }
268 }

271sai_object_id_t VirtualOidTranslator::translateVidToRid(
272 _In_ sai_object_id_t vid)
273{
274 SWSS_LOG_ENTER();
275
276 std::lock_guard<std::mutex> lock(m_mutex);
277
278 if (vid == SAI_NULL_OBJECT_ID)
279 {
280 SWSS_LOG_DEBUG("translated VID null to RID null");
281
282 return SAI_NULL_OBJECT_ID;
283 }
284
285 auto it = m_vid2rid.find(vid);
286
287 if (it != m_vid2rid.end())
288 {
289 return it->second;
290 }
291
292 auto rid = m_client->getRidForVid(vid);
293
294 if (rid == SAI_NULL_OBJECT_ID)
295 {
296 /*
297 * If user created object that is object id, then it should not
298 * query attributes of this object in init view mode, because he
299 * knows all attributes passed to that object.
300 *
301 * NOTE: This may be a problem for some objects in init view mode.
302 * We will need to revisit this after checking with real SAI
303 * implementation. Problem here may be that user will create some
304 * object and actually will need to to query some of it's values,
305 * like buffer limitations etc, mostly probably this will happen on
306 * SWITCH object.
307 */
308
309 // SWSS_LOG_THROW("can't get RID in init view mode - don't query created objects");
310
311 SWSS_LOG_THROW("unable to get RID for VID %s",
312 sai_serialize_object_id(vid).c_str());
313 }
314
315 /*
316 * We got this RID from redis db, so put it also to local db so it will be
317 * faster to retrieve it late on.
318 */
319
320 m_vid2rid[vid] = rid;
321
322 SWSS_LOG_DEBUG("translated VID %s to RID %s",
323 sai_serialize_object_id(vid).c_str(),
324 sai_serialize_object_id(rid).c_str());
325
326 return rid;

384{
385 SWSS_LOG_ENTER();
386
387 auto info = sai_metadata_get_object_type_info(metaKey.objecttype);

390 {
391 return tryTranslateVidToRid(
392 metaKey.objectkey.key.object_id,

395
396 for (size_t j = 0; j < info->structmemberscount; ++j)

408 m->setoid(&metaKey, rid);

413 }
414 }
