124{
125 SWSS_LOG_ENTER();

128
129 uint32_t count = 0;
130
131 const sai_object_id_t *objectIdList = NULL;
132
133 /*
134 * For ACL fields and actions we need to use enable flag as
135 * indicator, since when attribute is disabled then parameter can
136 * be garbage.
137 */
138
139 switch (m_meta->attrvaluetype)
140 {
141 case SAI_ATTR_VALUE_TYPE_OBJECT_ID:
142
143 count = 1;
144 objectIdList = &attr.value.oid;
145
146 break;
147
148 case SAI_ATTR_VALUE_TYPE_OBJECT_LIST:
149
150 count = attr.value.objlist.count;
151 objectIdList = attr.value.objlist.list;
152
153 break;
154
155 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_ID:
156
157 if (attr.value.aclfield.enable)
158 {
159 count = 1;
160 objectIdList = &attr.value.aclfield.data.oid;
161 }
162
163 break;
164
165 case SAI_ATTR_VALUE_TYPE_ACL_FIELD_DATA_OBJECT_LIST:
166
167 if (attr.value.aclfield.enable)
168 {
169 count = attr.value.aclfield.data.objlist.count;
170 objectIdList = attr.value.aclfield.data.objlist.list;
171 }
172
173 break;
174
175 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_ID:
176
177 if (attr.value.aclaction.enable)
178 {
179 count = 1;
180 objectIdList = &attr.value.aclaction.parameter.oid;
181 }
182
183 break;
184
185 case SAI_ATTR_VALUE_TYPE_ACL_ACTION_DATA_OBJECT_LIST:
186
187 if (attr.value.aclaction.enable)
188 {
189 count = attr.value.aclaction.parameter.objlist.count;
190 objectIdList = attr.value.aclaction.parameter.objlist.list;
191 }
192
193 break;
194
195 default:
196
197 if (m_meta->isoidattribute)
198 {
199 SWSS_LOG_THROW("attribute %s is oid attribute but not handled", m_meta->attridname);
200 }
201
202 // Attribute not contain any object ids.
203
204 break;
205 }

207 std::vector<sai_object_id_t> result(objectIdList, objectIdList + count);
