59{
60 SWSS_LOG_ENTER();
61
62 /*
63 * This function should only be called on CREATE/SET

65 */
66
67 for (uint32_t index = 0; index < attrCount; ++index)
68 {
69 auto &attr = attrList[index];
70
71 auto meta = sai_metadata_get_attr_metadata(SAI_OBJECT_TYPE_SWITCH, attr.id);
72
73 if (meta == NULL)
74 SWSS_LOG_THROW("failed to find metadata for switch attr %d", attr.id);
75
76 if (meta->attrvaluetype != SAI_ATTR_VALUE_TYPE_POINTER)
77 continue;
78
79 switch (attr.id)
80 {
81 case SAI_SWITCH_ATTR_SWITCH_STATE_CHANGE_NOTIFY:
82 m_switchNotifications.on_switch_state_change =
