47bool Recorder::setRecordingOutputDirectory(
48 _In_ const sai_attribute_t &attr)
49{
50 SWSS_LOG_ENTER();
51
52 if (attr.value.s8list.count == 0)
53 {
54 m_recordingOutputDirectory = ".";
55
56 SWSS_LOG_NOTICE("setting recording directory to: %s", m_recordingOutputDirectory.c_str());
57
58 requestLogRotate();
59
60 return true;
61 }
62
63 if (attr.value.s8list.list == NULL)
64 {
65 SWSS_LOG_ERROR("list pointer is NULL");
66
67 return false;
68 }
69
70 size_t len = strnlen((const char *)attr.value.s8list.list, attr.value.s8list.count);
71
72 if (len != (size_t)attr.value.s8list.count)
73 {
74 SWSS_LOG_ERROR("count (%u) is different than strnlen (%zu)", attr.value.s8list.count, len);
75
76 return false;
77 }
78
79 std::string dir((const char*)attr.value.s8list.list, len);
80
81 int result = access(dir.c_str(), W_OK);
82
83 if (result != 0)
84 {
85 SWSS_LOG_ERROR("can't access dir '%s' for writing", dir.c_str());
86
87 return false;
88 }
89
90 m_recordingOutputDirectory = dir;
91
92 // perform log rotate when log directory gets changed
93
94 requestLogRotate();
95
96 return true;
97}

258std::string Recorder::getTimestamp()
259{
260 SWSS_LOG_ENTER();
261
262 char buffer[64];
263 struct timeval tv;
264
265 gettimeofday(&tv, NULL);
266
267 struct tm now;
268 localtime_r(&tv.tv_sec, &now);
269
270 size_t size = strftime(buffer, 32, "%Y-%m-%d.%T.", &now);
271
272 snprintf(&buffer[size], 32, "%06ld", tv.tv_usec);
273
274 return std::string(buffer);
275}
