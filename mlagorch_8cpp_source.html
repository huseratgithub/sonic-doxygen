45void MlagOrch::doTask(Consumer &consumer)
46{
47 SWSS_LOG_ENTER();
48
49 if (!gPortsOrch->allPortsReady())
50 {
51 return;
52 }
53 string table_name = consumer.getTableName();
54 if (table_name == CFG_MCLAG_TABLE_NAME)
55 {
56 doMlagDomainTask(consumer);
57 }
58 else if (table_name == CFG_MCLAG_INTF_TABLE_NAME)
59 {
60 doMlagInterfaceTask(consumer);
61 }
62 else
63 {
64 SWSS_LOG_ERROR("MLAG receives invalid table %s", table_name.c_str());
65 }

68//Only interest in peer-link info from MLAG domain table
69void MlagOrch::doMlagDomainTask(Consumer &consumer)
70{
71 SWSS_LOG_ENTER();
72
73 string peer_link;
74
75 auto it = consumer.m_toSync.begin();
76 while (it != consumer.m_toSync.end())
77 {
78 KeyOpFieldsValuesTuple t = it->second;
79 string op = kfvOp(t);
80
81 if (op == SET_COMMAND)
82 {
83 for (auto i : kfvFieldsValues(t))
84 {
85 if (fvField(i) == "peer_link")
86 {
87 peer_link = fvValue(i);
88 break;
89 }
90 }
91 if (!peer_link.empty())
92 {
93 if (addIslInterface(peer_link))
94 it = consumer.m_toSync.erase(it);
95 else
96 it++;
97 }
98 else
99 it = consumer.m_toSync.erase(it);
100 }
101 else if (op == DEL_COMMAND)
102 {
103 if (delIslInterface())
104 it = consumer.m_toSync.erase(it);
105 else
106 it++;
107 }
108 else
109 {
110 SWSS_LOG_ERROR("MLAG receives unknown operation type %s", op.c_str());
111 it = consumer.m_toSync.erase(it);
112 }
113 }
114}

116//MLAG interface table key format: MCLAG_INTF_TABLE|mclag<id>|ifname
117void MlagOrch::doMlagInterfaceTask(Consumer &consumer)
118{
119 SWSS_LOG_ENTER();
120
121 size_t delimiter_pos;
122 string mlag_if_name;
123
124 auto it = consumer.m_toSync.begin();
125 while (it != consumer.m_toSync.end())
126 {
127 KeyOpFieldsValuesTuple t = it->second;
128 string op = kfvOp(t);
129 string key = kfvKey(t);
130
131 delimiter_pos = key.find_first_of("|");
132 mlag_if_name = key.substr(delimiter_pos+1);
133
134 if (op == SET_COMMAND)
135 {
136 if (addMlagInterface(mlag_if_name))
137 it = consumer.m_toSync.erase(it);
138 else
139 it++;
140 }
141 else if (op == DEL_COMMAND)
142 {
143 if (delMlagInterface(mlag_if_name))
144 it = consumer.m_toSync.erase(it);
145 else
146 it++;
147 }
148 else
149 {
150 SWSS_LOG_ERROR("MLAG receives unknown operation type %s", op.c_str());
151 it = consumer.m_toSync.erase(it);
152 }
153 }
154}

156bool MlagOrch::addIslInterface(string isl_name)
157{
158 Port isl_port;
159 MlagIslUpdate update;
160
161 //No change

163 return true;
164

166
167 //Update observers

215bool MlagOrch::delMlagInterface(string if_name)
216{
217 MlagIfUpdate update;
218
219 //Delete an unknown MLAG interface

222 SWSS_LOG_ERROR("MLAG deletes unknown MLAG interface %s", if_name.c_str());
223 }
224 else
225 {
226 m_mlagIntfs.erase(if_name);
227
228 //Notify observers
229 update.if_name = if_name;
230 update.is_add = false;
231 notify(SUBJECT_TYPE_MLAG_INTF_CHANGE, static_cast<void *>(&update));
