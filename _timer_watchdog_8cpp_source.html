120void TimerWatchdog::threadFunction()
121{
122 SWSS_LOG_ENTER();
123
124 SWSS_LOG_NOTICE("starting timer watchdog thread");
125
126 while (m_run)
127 {
128 std::this_thread::sleep_for(std::chrono::seconds(1));
129
130 MUTEX; // to protect event name
131
132 // we make local copies, since executing functions can be so fast that
133 // when we will read second time start timestamp it can be different
134 // than previous one
135
136 int64_t start = m_startTimestamp;
137 int64_t end = m_endTimestamp;
138 int64_t now = getTimeSinceEpoch(); // now needs to be obtained after obtaining start
139
140 int64_t span = end - start;
141
142 if (span < 0 && start > m_lastCheckTimestamp)
143 {
144 // this means start > end, so new function is currently executing,
145 // or that function hanged, so see how long that function is
146 // executing, this negative span can be arbitrary long even hours,
147 // and that is fine, since we don't know when OA makes next
148 // function call
149
150 span = now - start; // this must be always non negative
151
152 SWSS_LOG_NOTICE("time span %ld ms for '%s'", span/1000, m_eventName.c_str()); // TODO remove this
153
154 if (span < 0)
155 SWSS_LOG_THROW("negative span 'now - start': %ld - %ld", now, start);
156
157 if (span > m_warnTimespan)
158 {
159 m_lastCheckTimestamp = start;
160
161 // function probably hanged

163 SWSS_LOG_ERROR("time span WD exceeded %ld ms for %s", span/1000, m_eventName.c_str());
