52/*
53 * Structure describing the next hop group used by a route. As the next hop
54 * groups can either be owned by RouteOrch or by NhgOrch, we have to keep track
55 * of the next hop group index, as it is the one telling us which one owns it.
56 */
57struct RouteNhg
58{
59 NextHopGroupKey nhg_key;
60
61 /*
62 * Index of the next hop group used. Filled only if referencing a
63 * NhgOrch's owned next hop group.
64 */
65 std::string nhg_index;
66
67 RouteNhg() = default;
68 RouteNhg(const NextHopGroupKey& key, const std::string& index) :
69 nhg_key(key), nhg_index(index) {}
70
71 bool operator==(const RouteNhg& rnhg)
72 { return ((nhg_key == rnhg.nhg_key) && (nhg_index == rnhg.nhg_index)); }
73 bool operator!=(const RouteNhg& rnhg) { return !(*this == rnhg); }
74};
