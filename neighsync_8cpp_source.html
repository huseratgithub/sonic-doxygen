58void NeighSync::onMsg(int nlmsg_type, struct nl_object *obj)
59{
60 char ipStr[MAX_ADDR_SIZE + 1] = {0};
61 char macStr[MAX_ADDR_SIZE + 1] = {0};
62 struct rtnl_neigh *neigh = (struct rtnl_neigh *)obj;
63 string key;
64 string family;
65 string intfName;
66 std::vector<std::string> peerSwitchKeys;
67 m_cfgPeerSwitchTable.getKeys(peerSwitchKeys);
68 bool is_dualtor = peerSwitchKeys.size() > 0;
69
70 if ((nlmsg_type != RTM_NEWNEIGH) && (nlmsg_type != RTM_GETNEIGH) &&
71 (nlmsg_type != RTM_DELNEIGH))
72 return;
73
74 if (rtnl_neigh_get_family(neigh) == AF_INET)
75 family = IPV4_NAME;
76 else if (rtnl_neigh_get_family(neigh) == AF_INET6)
77 family = IPV6_NAME;
78 else
79 return;
80
81 key+= LinkCache::getInstance().ifindexToName(rtnl_neigh_get_ifindex(neigh));
82 intfName = key;
83 key+= ":";
84
85 nl_addr2str(rtnl_neigh_get_dst(neigh), ipStr, MAX_ADDR_SIZE);
86
87 /* Ignore IPv4 link-local addresses as neighbors if subtype is dualtor */
88 IpAddress ipAddress(ipStr);
89 if (family == IPV4_NAME && ipAddress.getAddrScope() == IpAddress::AddrScope::LINK_SCOPE && is_dualtor)
90 {
91 SWSS_LOG_INFO("Link Local address received on dualtor, ignoring for %s", ipStr);
92 return;
93 }
94
95
96 /* Ignore IPv6 link-local addresses as neighbors, if ipv6 link local mode is disabled */
97 if (family == IPV6_NAME && IN6_IS_ADDR_LINKLOCAL(nl_addr_get_binary_addr(rtnl_neigh_get_dst(neigh))))
98 {
99 if ((isLinkLocalEnabled(intfName) == false) && (nlmsg_type != RTM_DELNEIGH))
100 {
101 return;
102 }
103 }
104 /* Ignore IPv6 multicast link-local addresses as neighbors */
105 if (family == IPV6_NAME && IN6_IS_ADDR_MC_LINKLOCAL(nl_addr_get_binary_addr(rtnl_neigh_get_dst(neigh))))
106 return;
107 key+= ipStr;
108
109 int state = rtnl_neigh_get_state(neigh);
110 if (state == NUD_NOARP)
111 {
112 return;
113 }
114
115 bool delete_key = false;
116 bool use_zero_mac = false;
117 if (is_dualtor && (state == NUD_INCOMPLETE || state == NUD_FAILED))
118 {
119 SWSS_LOG_INFO("Unable to resolve %s, setting zero MAC", key.c_str());
120 use_zero_mac = true;
121
122 // Unresolved neighbor deletion on dual ToR devices must be handled
123 // separately, otherwise delete_key is never set to true
124 // and neighorch is never able to remove the neighbor
125 if (nlmsg_type == RTM_DELNEIGH)
126 {
127 delete_key = true;
128 }
129 }
130 else if ((nlmsg_type == RTM_DELNEIGH) ||
131 (state == NUD_INCOMPLETE) || (state == NUD_FAILED))
132 {
133 delete_key = true;
134 }
135
136 if (use_zero_mac)
137 {
138 std::string zero_mac = "00:00:00:00:00:00";
139 strncpy(macStr, zero_mac.c_str(), zero_mac.length());
140 }
141 else
142 {
143 nl_addr2str(rtnl_neigh_get_lladdr(neigh), macStr, MAX_ADDR_SIZE);
144 }
145
146 /* Ignore neighbor entries with Broadcast Mac - Trigger for directed broadcast */
147 if (!delete_key && (MacAddress(macStr) == MacAddress("ff:ff:ff:ff:ff:ff")))
148 {
149 SWSS_LOG_INFO("Broadcast Mac received, ignoring for %s", ipStr);
150 return;
151 }
152
153 std::vector<FieldValueTuple> fvVector;
154 FieldValueTuple f("family", family);
155 FieldValueTuple nh("neigh", macStr);
156 fvVector.push_back(nh);
157 fvVector.push_back(f);
158
159 // If warmstart is in progress, we take all netlink changes into the cache map
160 if (m_AppRestartAssist->isWarmStartInProgress())
161 {
162 m_AppRestartAssist->insertToMap(APP_NEIGH_TABLE_NAME, key, fvVector, delete_key);
163 }
164 else
165 {
166 if (delete_key == true)
167 {
168 m_neighTable.del(key);
169 return;
170 }
171 m_neighTable.set(key, fvVector);
172 }
173}

176bool NeighSync::isLinkLocalEnabled(const string &port)
177{
178 vector<FieldValueTuple> values;
179
180 if (!port.compare(0, strlen("Vlan"), "Vlan"))
181 {
182 if (!m_cfgVlanInterfaceTable.get(port, values))
183 {
184 SWSS_LOG_INFO("IPv6 Link local is not enabled on %s", port.c_str());
185 return false;
186 }
187 }
188 else if (!port.compare(0, strlen("PortChannel"), "PortChannel"))
189 {
190 if (!m_cfgLagInterfaceTable.get(port, values))
191 {
192 SWSS_LOG_INFO("IPv6 Link local is not enabled on %s", port.c_str());
193 return false;
194 }
195 }
196 else if (!port.compare(0, strlen("Ethernet"), "Ethernet"))
197 {
198 if (!m_cfgInterfaceTable.get(port, values))
199 {
200 SWSS_LOG_INFO("IPv6 Link local is not enabled on %s", port.c_str());
201 return false;
202 }
203 }
204 else
205 {
206 SWSS_LOG_INFO("IPv6 Link local is not supported for %s ", port.c_str());
207 return false;
208 }
209
210 auto it = std::find_if(values.begin(), values.end(), [](const FieldValueTuple& t){ return t.first == "ipv6_use_link_local_only";});
211 if (it != values.end())
212 {
213 if (it->second == "enable")
214 {
215 SWSS_LOG_INFO("IPv6 Link local is enabled on %s", port.c_str());
216 return true;
217 }
218 }
219
220 SWSS_LOG_INFO("IPv6 Link local is not enabled on %s", port.c_str());
221 return false;
