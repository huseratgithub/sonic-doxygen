205{
206 SWSS_LOG_ENTER();
207
208 const string & table_name = consumer.getTableName();
209 auto it = consumer.m_toSync.begin();
210 while (it != consumer.m_toSync.end())
211 {
212 bool task_result = false;
213 auto t = it->second;
214 const std::string & op = kfvOp(t);
215
216 if (op == SET_COMMAND)
217 {
218 if (table_name == CFG_VNET_TABLE_NAME)
219 {
220 task_result = doVxlanCreateTask(t);
221 }
222 else if (table_name == CFG_VXLAN_TUNNEL_TABLE_NAME)
223 {
224 task_result = doVxlanTunnelCreateTask(t);
225 }
226 else if (table_name == CFG_VXLAN_TUNNEL_MAP_TABLE_NAME)

228 task_result = doVxlanTunnelMapCreateTask(t);
229 }
230 else if (table_name == CFG_VXLAN_EVPN_NVO_TABLE_NAME)
231 {
232 task_result = doVxlanEvpnNvoCreateTask(t);
233 }
234 else
235 {
236 SWSS_LOG_ERROR("Unknown table : %s", table_name.c_str());
237 }
238 }
239 else if (op == DEL_COMMAND)
240 {
241 if (table_name == CFG_VNET_TABLE_NAME)
242 {
243 task_result = doVxlanDeleteTask(t);
244 }
245 else if (table_name == CFG_VXLAN_TUNNEL_TABLE_NAME)
246 {
247 task_result = doVxlanTunnelDeleteTask(t);

254 {
255 task_result = doVxlanEvpnNvoDeleteTask(t);
256 }
257 else
258 {
259 SWSS_LOG_ERROR("Unknown table : %s", table_name.c_str());
260 }
261 }
262 else
263 {
264 SWSS_LOG_ERROR("Unknown command : %s", op.c_str());
265 }
266
267 if (task_result == true)
268 {
269 it = consumer.m_toSync.erase(it);
270 }
271 else

274 }

278bool VxlanMgr::doVxlanCreateTask(const KeyOpFieldsValuesTuple & t)
279{
280 SWSS_LOG_ENTER();
281
282 VxlanInfo info;
283 info.m_vnet = kfvKey(t);
284 for (auto i : kfvFieldsValues(t))
285 {
286 const std::string & field = fvField(i);
287 const std::string & value = fvValue(i);
288 if (field == VXLAN_TUNNEL)
289 {
290 info.m_vxlanTunnel = value;
291 }
292 else if (field == VNI)
293 {
294 info.m_vni = value;
295 }
296 }
297
298 // If all information of vnet has been set
299 if (info.m_vxlanTunnel.empty()
300 || info.m_vni.empty())
301 {
302 SWSS_LOG_DEBUG("Vnet %s information is incomplete", info.m_vnet.c_str());
303 // if the information is incomplete, just ignore this message
304 // because all information will be sent if the information was
305 // completely set.
306 return true;
307 }
308
309 // If the vxlan tunnel has been created
310 auto it = m_vxlanTunnelCache.find(info.m_vxlanTunnel);
311 if (it == m_vxlanTunnelCache.end())
312 {
313 SWSS_LOG_DEBUG("Vxlan tunnel %s has not been created", info.m_vxlanTunnel.c_str());
314 // Suspend this message until the vxlan tunnel is created
315 return false;
316 }
317
318 // If the VRF(Vnet is a special VRF) has been created
319 if (!isVrfStateOk(info.m_vnet))
320 {
321 SWSS_LOG_DEBUG("Vrf %s has not been created", info.m_vnet.c_str());
322 // Suspend this message until the vrf is created
323 return false;
324 }
325
326 // If the mac address has been set
327 auto macAddress = getVxlanRouterMacAddress();
328 if (!macAddress.first)
329 {
330 SWSS_LOG_DEBUG("Mac address is not ready");
331 // Suspend this message until the mac address is set
332 return false;
333 }
334 info.m_macAddress = macAddress.second;
335
336 auto sourceIp = std::find_if(
337 it->second.fvt.begin(),
338 it->second.fvt.end(),
339 [](const FieldValueTuple & fvt){ return fvt.first == SOURCE_IP; });
340 if (sourceIp != it->second.fvt.end())
341 {
342 info.m_sourceIp = sourceIp->second;
343 }
344 info.m_vxlan = getVxlanName(info);
345 info.m_vxlanIf = getVxlanIfName(info);
346
347 // If this vxlan has been created
348 if (isVxlanStateOk(info.m_vxlan))
349 {
350 // Because the vxlan has been create, so this message is to update
351 // the information of vxlan.
352 // This program just delete the old vxlan and create a new one
353 // according to this message.
354 doVxlanDeleteTask(t);
355 }
356
357 if (!createVxlan(info))
358 {
359 SWSS_LOG_ERROR("Cannot create vxlan %s", info.m_vxlan.c_str());
360 return true;
361 }
362
363 m_vnetCache[info.m_vnet] = info;
364 SWSS_LOG_INFO("Create vxlan %s", info.m_vxlan.c_str());
365
366 return true;

836 }
837
