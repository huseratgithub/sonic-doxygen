52void WatermarkOrch::doTask(Consumer &consumer)
53{
54 SWSS_LOG_ENTER();
55
56 if (!gPortsOrch->allPortsReady())
57 {
58 return;
59 }
60
61 auto it = consumer.m_toSync.begin();
62 while (it != consumer.m_toSync.end())
63 {
64 KeyOpFieldsValuesTuple t = it->second;
65
66 string key = kfvKey(t);
67 string op = kfvOp(t);
68 std::vector<FieldValueTuple> fvt = kfvFieldsValues(t);
69
70 if (op == SET_COMMAND)
71 {
72 if (consumer.getTableName() == CFG_WATERMARK_TABLE_NAME)
73 {
74 handleWmConfigUpdate(key, fvt);
75 }
76 else if (consumer.getTableName() == CFG_FLEX_COUNTER_TABLE_NAME)
77 {
78 handleFcConfigUpdate(key, fvt);
79 }
80 }
81 else if (op == DEL_COMMAND)
82 {
83 SWSS_LOG_WARN("Unsupported op %s", op.c_str());
84 }
85 else
86 {
87 SWSS_LOG_ERROR("Unknown operation type %s\n", op.c_str());
88 }
89
90 consumer.m_toSync.erase(it++);
91 }
92}

95{

116void WatermarkOrch::handleFcConfigUpdate(const std::string &key, const std::vector<FieldValueTuple> &fvt)
117{
118 SWSS_LOG_ENTER();
119 uint8_t prevStatus = m_wmStatus;
120 if (key == "QUEUE_WATERMARK" || key == "PG_WATERMARK")
121 {
122 for (std::pair<std::basic_string<char>, std::basic_string<char> > i: fvt)
123 {
124 if (i.first == "FLEX_COUNTER_STATUS")
125 {
126 if (i.second == "enable")
127 {
128 m_wmStatus = (uint8_t) (m_wmStatus | groupToMask.at(key));
129 }
130 else if (i.second == "disable")
131 {
132 m_wmStatus = (uint8_t) (m_wmStatus & ~(groupToMask.at(key)));
133 }
134 }
135 }
136 if (!prevStatus && m_wmStatus)
137 {
138 m_telemetryTimer->start();
139 }
140 SWSS_LOG_DEBUG("Status of WMs: %u", m_wmStatus);
141 }
