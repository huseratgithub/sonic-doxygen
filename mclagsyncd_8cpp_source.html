73

77 mclag.processStateFdb((SubscriberStateTable *)temps);
78 }
79 else if ( temps == (Selectable *)&mclag_cfg_tbl ) //Reading MCLAG Domain Config Table
80 {
81 SWSS_LOG_DEBUG("MCLAGSYNCD processing mclag_cfg_tbl notifications");
82 std::deque<KeyOpFieldsValuesTuple> entries;

87 {
88 SWSS_LOG_DEBUG("MCLAGSYNCD processing mclag_intf_cfg_tbl notifications");
89 std::deque<KeyOpFieldsValuesTuple> entries;
90 mclag.getMclagIntfCfgTable()->pops(entries);
91 mclag.mclagsyncdSendMclagIfaceCfg(entries);
92 }
93 else if (temps == (Selectable *)mclag.getMclagUniqueCfgTable()) //Reading MCLAG Unique IP Config Table
94 {
95 SWSS_LOG_DEBUG("MCLAGSYNCD processing mclag_unique_ip_cfg_tbl notifications");
96 std::deque<KeyOpFieldsValuesTuple> entries;
97 mclag.getMclagUniqueCfgTable()->pops(entries);
98 mclag.mclagsyncdSendMclagUniqueIpCfg(entries);
99 }
100 else if (temps == (Selectable *)mclag.getStateVlanMemberTable())
101 {
102 SWSS_LOG_INFO(" MCLAGSYNCD Matching vlan Member selectable");
103 mclag.processStateVlanMember((SubscriberStateTable *)temps);
104 }
105 else
106 {
107 pipeline.flush();
108 SWSS_LOG_DEBUG("Pipeline flushed");
109 }
110 }
111 }
112 catch (MclagLink::MclagConnectionClosedException &e)
113 {
114 cout << "Connection lost, reconnecting..." << endl;
115 }
116 catch (const exception& e)
117 {

119 return 0;
120 }
121 }
122
