55void FlowCounterRouteOrch::doTask(Consumer &consumer)
56{
57 SWSS_LOG_ENTER();
58 if (!gRouteOrch || !mRouteFlowCounterSupported)
59 {
60 return;
61 }
62
63 auto it = consumer.m_toSync.begin();
64 while (it != consumer.m_toSync.end())
65 {
66 KeyOpFieldsValuesTuple t = it->second;
67
68 const auto &key = kfvKey(t);
69 const auto &op = kfvOp(t);
70 const auto &data = kfvFieldsValues(t);
71 if (op == SET_COMMAND)
72 {
73 size_t maxMatchCount = ROUTE_PATTERN_DEFAULT_MAX_MATCH_COUNT;
74 for (auto valuePair : data)
75 {
76 const auto &field = fvField(valuePair);
77 const auto &value = fvValue(valuePair);
78 if (field == ROUTE_PATTERN_MAX_MATCH_COUNT_FIELD)
79 {
80 maxMatchCount = (size_t)std::stoul(value);
81 if (maxMatchCount == 0)
82 {
83 SWSS_LOG_WARN("Max match count for route pattern cannot be 0, set it to default value 30");
84 maxMatchCount = ROUTE_PATTERN_DEFAULT_MAX_MATCH_COUNT;
85 }
86 }
87 }
88
89 addRoutePattern(key, maxMatchCount);
90 }
91 else if (op == DEL_COMMAND)
92 {
93 removeRoutePattern(key);
94 }
95 consumer.m_toSync.erase(it++);
96 }

454bool FlowCounterRouteOrch::bindFlowCounter(const RoutePattern &route_pattern, sai_object_id_t vrf_id, const IpPrefix& ip_prefix)
455{
456 SWSS_LOG_ENTER();
457
458 SWSS_LOG_NOTICE("Binding route entry vrf=%s prefix=%s to flow counter", route_pattern.vrf_name.c_str(), ip_prefix.to_string().c_str());
459
460 sai_object_id_t counter_oid;

470 copy(route_entry.destination, ip_prefix);

473 attr.value.oid = counter_oid;
474
475 auto status = sai_route_api->set_route_entry_attribute(&route_entry, &attr);
476 if (status != SAI_STATUS_SUCCESS)
477 {
478 FlowCounterHandler::removeGenericCounter(counter_oid);
479 SWSS_LOG_WARN("Failed to bind route entry vrf=%s prefix=%s to flow counter", route_pattern.vrf_name.c_str(), ip_prefix.to_string().c_str());
480 return false;
481 }
482
483 pendingUpdateFlexDb(route_pattern, ip_prefix, counter_oid);
484 return true;

510bool FlowCounterRouteOrch::removeRouteFlowCounter(const RoutePattern &route_pattern, sai_object_id_t vrf_id, const IpPrefix& ip_prefix)
511{
512 SWSS_LOG_ENTER();
513
514 SWSS_LOG_NOTICE("Removing route entry vrf=%s prefix=%s from flow counter", route_pattern.vrf_name.c_str(), ip_prefix.to_string().c_str());
515
516 // Check if the entry is in mPendingAddToFlexCntr
517 sai_object_id_t counter_oid = SAI_NULL_OBJECT_ID;
518 auto pending_iter = mPendingAddToFlexCntr.find(route_pattern);
519 if (pending_iter != mPendingAddToFlexCntr.end())
520 {
521 auto iter_prefix = pending_iter->second.find(ip_prefix);
522 if (iter_prefix != pending_iter->second.end())
523 {
524 counter_oid = iter_prefix->second;
525 pending_iter->second.erase(iter_prefix);
526 if (pending_iter->second.empty())
527 {
528 mPendingAddToFlexCntr.erase(pending_iter);
529 }
530 }
531 }
532
533 if (counter_oid == SAI_NULL_OBJECT_ID)
534 {
535 // Check if the entry is in mBoundRouteCounters
536 auto cache_iter = mBoundRouteCounters.find(route_pattern);
537 if (cache_iter != mBoundRouteCounters.end())
538 {
539 auto iter_prefix = cache_iter->second.find(ip_prefix);
540 if (iter_prefix != cache_iter->second.end())
541 {
542 counter_oid = iter_prefix->second;
543 removeRouteFlowCounterFromDB(vrf_id, ip_prefix, counter_oid);
544 cache_iter->second.erase(iter_prefix);
545 if (cache_iter->second.empty())
546 {
547 mBoundRouteCounters.erase(cache_iter);
548 }
549 }
550 }
551 }
552
553 // No need unbind because the route entry has been removed, just remove the generic counter
554 if (counter_oid != SAI_NULL_OBJECT_ID)
555 {
556 FlowCounterHandler::removeGenericCounter(counter_oid);
557 return true;
558 }
559
560 return false;
561}

619void FlowCounterRouteOrch::createRouteFlowCounterByPattern(const RoutePattern &route_pattern, size_t current_bound_count)
620{
621 SWSS_LOG_ENTER();
622 if (!isRouteFlowCounterEnabled())
623 {
624 return;
625 }
626
627 auto &syncdRoutes = gRouteOrch->getSyncdRoutes();
628 auto iter = syncdRoutes.find(route_pattern.vrf_id);
629 if (iter != syncdRoutes.end())
630 {
631 SWSS_LOG_NOTICE("Creating route flow counter for pattern %s", route_pattern.to_string().c_str());
632
633 for (auto &entry : iter->second)
634 {
635 if (current_bound_count == route_pattern.max_match_count)
636 {
637 return;
638 }
639
640 if (route_pattern.is_match(route_pattern.vrf_id, entry.first))
641 {
642 if (isRouteAlreadyBound(route_pattern, entry.first))
643 {
644 continue;
645 }
646
647 if (bindFlowCounter(route_pattern, route_pattern.vrf_id, entry.first))
648 {
649 ++current_bound_count;
650 }
651 }
652 }
653 }
654
655 createRouteFlowCounterFromVnetRoutes(route_pattern, current_bound_count);
656
657 auto misc_iter = mMiscRoutes.find(route_pattern.vrf_id);
658 if (misc_iter != mMiscRoutes.end())
659 {
660 SWSS_LOG_NOTICE("Creating route flow counter for pattern %s for other type route entries", route_pattern.to_string().c_str());
661
662 for (auto ip_prefix : misc_iter->second)
663 {
664 if (current_bound_count == route_pattern.max_match_count)
665 {
666 return;
667 }
668
669 if (route_pattern.is_match(route_pattern.vrf_id, ip_prefix))
670 {
671 if (isRouteAlreadyBound(route_pattern, ip_prefix))
672 {
673 continue;
674 }
675
676 if (bindFlowCounter(route_pattern, route_pattern.vrf_id, ip_prefix))
677 {
678 ++current_bound_count;
679 }
680 }
681 }
682 }

686{
687 SWSS_LOG_ENTER();
688
689 auto *vnet_orch = gDirectory.get<VNetOrch*>();
690 assert(vnet_orch); // VnetOrch instance is created before RouteOrch
691
692 if (!vnet_orch->isVnetExists(route_pattern.vrf_name))
693 {
694 return;
695 }
696
697 SWSS_LOG_NOTICE("Creating route flow counter for pattern %s for VNET route entries", route_pattern.to_string().c_str());
698
699 auto *vrf_obj = vnet_orch->getTypePtr<VNetVrfObject>(route_pattern.vrf_name);
700 const auto &route_map = vrf_obj->getRouteMap();
701 for (const auto &entry : route_map)
702 {
703 if (current_bound_count == route_pattern.max_match_count)
704 {
705 return;
706 }
707
708 if (route_pattern.is_match(route_pattern.vrf_id, entry.first))
709 {
710 if (isRouteAlreadyBound(route_pattern, entry.first))
711 {
712 continue;
713 }
714
715 if (bindFlowCounter(route_pattern, route_pattern.vrf_id, entry.first))
716 {
717 ++current_bound_count;
718 }
719 }
720 }
721
722 const auto &tunnel_routes = vrf_obj->getTunnelRoutes();
723 for (const auto &entry : tunnel_routes)
724 {
725 if (current_bound_count == route_pattern.max_match_count)
726 {
727 return;
728 }
729 if (route_pattern.is_match(route_pattern.vrf_id, entry.first))
730 {
731 if (isRouteAlreadyBound(route_pattern, entry.first))
732 {
733 continue;
734 }
735
736 if (bindFlowCounter(route_pattern, route_pattern.vrf_id, entry.first))
737 {
738 ++current_bound_count;
739 }
740 }

744void FlowCounterRouteOrch::reapRouteFlowCounterByPattern(const RoutePattern &route_pattern, size_t current_bound_count)
745{
746 SWSS_LOG_ENTER();
747
748 auto pending_iter = mPendingAddToFlexCntr.find(route_pattern);
749 auto iter = mBoundRouteCounters.find(route_pattern);
750 if (iter == mBoundRouteCounters.end() && pending_iter == mPendingAddToFlexCntr.end())
751 {
752 return;
753 }
754
755 // Remove from pending cache first
756 if (pending_iter != mPendingAddToFlexCntr.end())
757 {
758 while(current_bound_count > route_pattern.max_match_count)
759 {
760 auto bound_iter = pending_iter->second.begin();
761 if (bound_iter == pending_iter->second.end())
762 {
763 break;
764 }
765 unbindFlowCounter(route_pattern, route_pattern.vrf_id, bound_iter->first, bound_iter->second);
766 pending_iter->second.erase(bound_iter);
767 --current_bound_count;
768 }
769 }
770
771 // Remove from bound cache
772 if (iter != mBoundRouteCounters.end())
773 {
774 while(current_bound_count > route_pattern.max_match_count)
775 {
776 auto bound_iter = iter->second.begin();
777 if (bound_iter == iter->second.end())
778 {
779 break;
780 }
781
782 removeRouteFlowCounterFromDB(route_pattern.vrf_id, bound_iter->first, bound_iter->second);
783 unbindFlowCounter(route_pattern, route_pattern.vrf_id, bound_iter->first, bound_iter->second);
784 iter->second.erase(bound_iter);
785 --current_bound_count;
786 }
787 }
788}

790void FlowCounterRouteOrch::onRoutePatternMaxMatchCountChange(RoutePattern &route_pattern, size_t new_max_match_count)
791{
792 SWSS_LOG_ENTER();
793
794 if (route_pattern.max_match_count != new_max_match_count)
795 {
796 auto old_max_match_count = route_pattern.max_match_count;
797 route_pattern.max_match_count = new_max_match_count;
798
799 if (!isRouteFlowCounterEnabled())
800 {
801 return;
802 }
803
804 auto current_bound_count = getRouteFlowCounterSizeByPattern(route_pattern);
805 SWSS_LOG_NOTICE("Current bound route flow counter count is %zu, new limit is %zu, old limit is %zu", current_bound_count, new_max_match_count, old_max_match_count);
806 if (new_max_match_count > old_max_match_count)
807 {
808 if (current_bound_count == old_max_match_count)
809 {
810 createRouteFlowCounterByPattern(route_pattern, current_bound_count);
811 }
812 }
813 else
814 {
815 if (current_bound_count > new_max_match_count)
816 {
817 reapRouteFlowCounterByPattern(route_pattern, current_bound_count);
818 }
819 }
820 }
